304double GetExactSolutionValue (const std::vector < double >& x) {
305 double pi = acos (-1.);
306 if (x.size() == 1) {
307 return cos (pi * x[0]);
308 }
309 else {
310 return cos (pi * x[0]) * cos (pi * x[1]);
311 }

315void GetExactSolutionGradient (const std::vector < double >& x, vector < double >& solGrad) {
316 double pi = acos (-1.);
317 if (x.size() == 1) {
318 solGrad[0] = -pi * sin (pi * x[0]);
319 }
320 else {
321 solGrad[0] = -pi * sin (pi * x[0]) * cos (pi * x[1]);
322 solGrad[1] = -pi * cos (pi * x[0]) * sin (pi * x[1]);
323 }
324};

327double GetExactSolutionLaplace (const std::vector < double >& x) {
328 double pi = acos (-1.);
329 if (x.size() == 1) {
330 return -pi * pi * cos (pi * x[0]); // - pi*pi*cos(pi*x[0])*cos(pi*x[1]);
331 }
332 else {
333 return -2.*pi * pi * cos (pi * x[0]) * cos (pi * x[1]); // - pi*pi*cos(pi*x[0])*cos(pi*x[1]);
334 }
335};

338std::pair < double, double > GetErrorNormWithProjection (MultiLevelSolution* mlSol) {
339 unsigned level = mlSol->_mlMesh->GetNumberOfLevels() - 1u;
340 // extract pointers to the several objects that we are going to use
341 Mesh* msh = mlSol->_mlMesh->GetLevel (level); // pointer to the mesh (level) object
342 elem* el = msh->el; // pointer to the elem object in msh (level)
343 Solution* sol = mlSol->GetSolutionLevel (level); // pointer to the solution (level) object
344
345 const unsigned dim = msh->GetDimension(); // get the domain dimension of the problem
346 unsigned iproc = msh->processor_id(); // get the process_id (for parallel computation)
347
348//solution variable
349
350
351 std::string Uxname[3] = {"ux", "uy", "uz"};
352 std::vector < unsigned > soluIndex (dim + 1);
353 for (unsigned k = 0; k < dim; k++) {
354 soluIndex[k] = mlSol->GetIndex (Uxname[k].c_str());
355 }
356
357 soluIndex[dim] = mlSol->GetIndex ("u"); // get the position of "u" in the ml_sol object
358 unsigned soluType = mlSol->GetSolutionType (soluIndex[dim]); // get the finite element type for "u"
359
360 std::vector < std::vector < double > > solu (dim + 1); // local solution
361
362 vector < vector < double > > x (dim); // local coordinates
363 unsigned xType = 2; // get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC)
364
365 vector <double> phi; // local test function
366 vector <double> phi_x; // local test function first order partial derivatives
367 double weight; // gauss point weight
368
369 // reserve memory for the local standar vectors
370
371 double seminorm = 0.;
372 double l2norm = 0.;
373
374// element loop: each process loops only on the elements that owns
375 for (int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
376
377
378 short unsigned ielGeom = msh->GetElementType (iel);
379 unsigned nDofs = msh->GetElementDofNumber (iel, soluType); // number of solution element dofs
380
381 // resize local arrays
382 for (int k = 0; k < dim; k++) {
383 x[k].resize (nDofs);
384 solu[k].resize (nDofs);
385 }
386 solu[dim].resize (nDofs);
387
388 // local storage of global mapping and solution
389 for (unsigned i = 0; i < nDofs; i++) {
390 unsigned solDof = msh->GetSolutionDof (i, iel, soluType); // global to global mapping between solution node and solution dof
391 for (unsigned k = 0; k < dim + 1; k++)
392 solu[k][i] = (*sol->_Sol[soluIndex[k]]) (solDof); // global extraction and local storage for the solution
393 }
394
395 // local storage of coordinates
396 for (unsigned i = 0; i < nDofs; i++) {
397 unsigned xDof = msh->GetSolutionDof (i, iel, xType); // global to global mapping between coordinates node and coordinate dof
398
399 for (unsigned k = 0; k < dim; k++) {
400 x[k][i] = (*msh->_topology->_Sol[k]) (xDof); // global extraction and local storage for the element coordinates
401 }
402 }
403
404 // *** Gauss point loop ***
405 for (unsigned ig = 0; ig < msh->_finiteElement[ielGeom][soluType]->GetGaussPointNumber(); ig++) {
406 // *** get gauss point weight, test function and test function partial derivatives ***
407 msh->_finiteElement[ielGeom][soluType]->Jacobian (x, ig, weight, phi, phi_x);
408
409 // evaluate the solution, the solution derivatives and the coordinates in the gauss point
410 double soluGauss = 0;
411 vector < double > soluGauss_x (dim, 0.);
412 vector < double > xGauss (dim, 0.);
413
414 for (unsigned i = 0; i < nDofs; i++) {
415 soluGauss += phi[i] * solu[dim][i];
416
417 for (unsigned k = 0; k < dim; k++) {
418 soluGauss_x[k] += phi[i] * solu[k][i];
419 xGauss[k] += x[k][i] * phi[i];
420 }
421 }
422
423 vector <double> solGrad (dim);
424 GetExactSolutionGradient (xGauss, solGrad);
425
426 for (unsigned j = 0; j < dim ; j++) {
427 seminorm += ( (soluGauss_x[j] - solGrad[j]) * (soluGauss_x[j] - solGrad[j])) * weight;
428 }
429
430 double exactSol = GetExactSolutionValue (xGauss);
431 l2norm += (exactSol - soluGauss) * (exactSol - soluGauss) * weight;
432 } // end gauss point loop
433 } //end element loop for each process
434
435// add the norms of all processes
436 NumericVector* norm_vec;
437 norm_vec = NumericVector::build().release();
438 norm_vec->init (msh->n_processors(), 1 , false, AUTOMATIC);
439
440 norm_vec->set (iproc, l2norm);
441 norm_vec->close();
442 l2norm = norm_vec->l1_norm();
443
444 norm_vec->set (iproc, seminorm);
445 norm_vec->close();
446 seminorm = norm_vec->l1_norm();
447
448 delete norm_vec;
449
450 std::pair < double, double > norm;
451 norm.first = sqrt (l2norm);
452 norm.second = sqrt (seminorm);
453
454 return norm;
455
456}

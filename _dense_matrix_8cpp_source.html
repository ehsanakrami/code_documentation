38void DenseMatrix::left_multiply (const DenseMatrixBase& M2) {
39 if (this->use_blas)
40 this->_multiply_blas(M2, LEFT_MULTIPLY);
41 else {
42 // (*this) <- M2 * (*this) Where: (*this) = (m x n), M2 = (m x p), M3 = (p x n)
43 // M3 is a copy of *this before it gets resize()d
44 DenseMatrix M3(*this);
45 // Resize *this so that the result can fit
46 this->resize (M2.m(), M3.n());
47 // Call the multiply function in the base class
48 this->multiply(*this, M2, M3);
49 }
50}

53void DenseMatrix::left_multiply_transpose(const DenseMatrix& A) {
54 if (this->use_blas) this->_multiply_blas(A, LEFT_MULTIPLY_TRANSPOSE);
55 else {
56 //Check to see if we are doing (A^T)*A
57 if (this == &A) {
58 //libmesh_here();
59 DenseMatrix B(*this);
60 // Simple but inefficient way
61 // return this->left_multiply_transpose(B);
62
63 // More efficient, but more code way
64 // If A is mxn, the result will be a square matrix of Size n x n.
65 const int m = A.m();
66 const int n = A.n();
67
68 // resize() *this and also zero out all entries.
69 this->resize(n,n);
70
71 // Compute the lower-triangular part
72 for (int i=0; i<n; ++i)
73 for (int j=0; j<=i; ++j)
74 for (int k=0; k<m; ++k) // inner products are over m
75 (*this)(i,j) += B(k,i)*B(k,j);
76
77 // Copy lower-triangular part into upper-triangular part
78 for (int i=0; i<n; ++i)
79 for (int j=i+1; j<n; ++j)
80 (*this)(i,j) = (*this)(j,i);
81 }
82
83 else {
84 DenseMatrix B(*this);
85 this->resize (A.n(), B.n());
86
87 assert (A.m() == B.m());
88 assert (this->m() == A.n());
89 assert (this->n() == B.n());
90
91 const int m_s = A.n();
92 const int p_s = A.m();
93 const int n_s = this->n();
94
95 // Do it this way because there is a
96 // decent chance (at least for constraint matrices)
97 // that A.transpose(i,k) = 0.
98 for (int i=0; i<m_s; i++)
99 for (int k=0; k<p_s; k++)
100 if (A.transpose(i,k) != 0.)
101 for (int j=0; j<n_s; j++)
102 (*this)(i,j) += A.transpose(i,k)*B(k,j);
103 }
104 }

199

202 assert(n_rows == (int)dest.size());

205 DenseMatrix A(*this);
206

267void DenseMatrix::_lu_back_substitute (DenseVector& b,
268 DenseVector& x,
269 const bool ) const {
270 const int
271 n = this->n();
272 assert (this->m() == n);
273 assert (this->m() == (int)b.size());
274
275 x.resize (n);
276 // A convenient reference to *this
277 const DenseMatrix& A = *this;
278
279 // Transform the RHS vector
280 for (int i=0; i<(n-1); i++) {
281 // Get the diagonal entry and take its inverse
282 const double diag = A(i,i);
283 assert (fabs(diag) < 1.e-20);
284 const double diag_inv = 1./diag;
285 // Get the entry b(i) and store it
286 const double bi = b(i);
287 for (int j=(i+1); j<n; j++) b(j) -= bi*A(j,i)*diag_inv;
288 }
289
290
291 // Perform back-substitution
292 {
293 x(n-1) = b(n-1)/A(n-1,n-1);
294 for (int i=0; i<=(n-1); i++) {
295 const int ib = (n-1)-i;
296 // Get the diagonal and take its inverse
297 const double diag = A(ib,ib);
298 assert (fabs(diag) < 1.e-20);
299 const double diag_inv = 1./diag;
300
301 for (int j=(ib+1); j<n; j++) {
302 b(ib) -= A(ib,j)*x(j);
303 x(ib) = b(ib)*diag_inv;
304 }
305
306 }
307 }
308 return;
309}

312// ===========================================================
313void DenseMatrix::_lu_decompose (const bool partial_pivot) {
314 // If this function was called, there better not be any
315 // previous decomposition of the matrix.
316 assert(this->_decomposition_type == NONE);
317 // Get the matrix size and make sure it is square
318 const int
319 m = this->m();
320 assert (m == this->n());
321 // A convenient reference to *this
322 DenseMatrix& A = *this;
323
324 // Straight, vanilla LU factorization without pivoting
325 if (!partial_pivot) {
326 // For each row in the matrix
327 for (int i=0; i<m; i++) {
328 // Get the diagonal entry and take its inverse
329 const double diag = A(i,i);
330 assert (fabs(diag) < 1.e-20);
331 const double diag_inv = 1./diag;
332
333 // For each row in the submatrix
334 for (int j=i+1; j<m; j++) {
335 // Get the scale factor for this row
336 const double fact = A(j,i)*diag_inv;
337 // For each column in the subrow scale it by the factor
338 for (int k=i+1; k<m; k++) A(j,k) -= fact*A(i,k);
339 }
340 }
341 } else {
342 // Do partial pivoting.
343 exit(0);
344 }
345 // Set the flag for LU decomposition
346 this->_decomposition_type = LU;
347}

349// ========================================
350double DenseMatrix::det () {
351 // First LU decompose the matrix (without partial pivoting).
352 // Note that the lu_decompose routine will check to see if the
353 // matrix is square so we don't worry about it.
354 if (this->_decomposition_type == NONE)
355 this->_lu_decompose(false);
356 else if (this->_decomposition_type != LU) {
357 std::cerr << "Error! Can't compute the determinant under "
358 << "the current decomposition."
359 << std::endl;
360 exit(0);
361 }
362
363 // A variable to keep track of the running product of diagonal terms.
364 double determinant = 1.;
365 // Loop over diagonal terms, computing the product.
366 for (int i=0; i<this->m(); i++) determinant *= (*this)(i,i);
367 // Return the determinant
368 return determinant;
369}

377void DenseMatrix::cholesky_solve (DenseVector& b,
378 DenseVector& x) {

380 switch (this->_decomposition_type) {

385

397 }

406// This algorithm is based on the Cholesky decomposition in
407// the Numerical Recipes in C book.
408
409void DenseMatrix::_cholesky_decompose () {
410 // If we called this function, there better not be any
411 // previous decomposition of the matrix.
412 assert(this->_decomposition_type == NONE);
413
414 // Shorthand notation for number of rows and columns.
415 const int
416 m = this->m(),
417 n = this->n();
418
419 // Just to be really sure...
420 assert(m==n);
421
422 // A convenient reference to *this
423 DenseMatrix& A = *this;
424
425 for (int i=0; i<m; ++i) {
426 for (int j=i; j<n; ++j) {
427 for (int k=0; k<i; ++k)
428 A(i,j) -= A(i,k) * A(j,k);
429
430 if (i == j) {
431#ifndef LIBMESH_USE_COMPLEX_NUMBERS
432 if (A(i,j) <= 0.0) {
433 std::cerr << "Error! Can only use Cholesky decomposition "
434 << "with symmetric positive definite matrices."
435 << std::endl;
436 exit(0);
437 }
438#endif
439
440 A(i,i) = std::sqrt(A(i,j));
441 } else
442 A(j,i) = A(i,j) / A(i,i);
443 }
444 }
445
446 // Set the flag for CHOLESKY decomposition
447 this->_decomposition_type = CHOLESKY;
448}

452void DenseMatrix::_cholesky_back_substitute (DenseVector& b,
453 DenseVector& x) const {
454 // Shorthand notation for number of rows and columns.
455 const int
456 m = this->m(),
457 n = this->n();
458
459 // Just to be really sure...
460 assert(m==n);
461
462 // A convenient reference to *this
463 const DenseMatrix& A = *this;
464
465 // Now compute the solution to Ax =b using the factorization.
466 x.resize(m);
467
468 // Solve for Ly=b
469 for (int i=0; i<n; ++i) {
470 for (int k=0; k<i; ++k)
471 b(i) -= A(i,k)*x(k);
472
473 x(i) = b(i) / A(i,i);
474 }
475
476 // Solve for L^T x = y
477 for (int i=0; i<n; ++i) {
478 const int ib = (n-1)-i;
479
480 for (int k=(ib+1); k<n; ++k)
481 x(ib) -= A(k,ib) * x(k);
482
483 x(ib) /= A(ib,ib);
484 }
485}

494void DenseMatrix::_multiply_blas(const DenseMatrixBase& other,
495 _BLAS_Multiply_Flag flag) {
496 int result_size = 0;
497
498 // For each case, determine the size of the final result make sure
499 // that the inner dimensions match
500 switch (flag) {
501 case LEFT_MULTIPLY: {
502 result_size = other.m() * this->n();
503 if (other.n() == this->m())
504 break;
505 }
506 case RIGHT_MULTIPLY: {
507 result_size = other.n() * this->m();
508 if (other.m() == this->n())
509 break;
510 }
511 case LEFT_MULTIPLY_TRANSPOSE: {
512 result_size = other.n() * this->n();
513 if (other.m() == this->m())
514 break;
515 }
516 case RIGHT_MULTIPLY_TRANSPOSE: {
517 result_size = other.m() * this->m();
518 if (other.n() == this->n())
519 break;
520 }
521 default: {
522 std::cout << "Unknown flag selected or matrices are ";
523 std::cout << "incompatible for multiplication." << std::endl;
524 exit(0);
525 }
526 }
527
528 // For this to work, the passed arg. must actually be a DenseMatrix
529 const DenseMatrix* const_that = dynamic_cast< const DenseMatrix* >(&other);
530 if (!const_that) {
531 std::cerr << "Unable to cast input matrix to usable type." << std::endl;
532 exit(0);
533 }
534
535 // Also, although 'that' is logically const in this BLAS routine,
536 // the PETSc BLAS interface does not specify that any of the inputs are
537 // const. To use it, I must cast away const-ness.
538 DenseMatrix* that = const_cast< DenseMatrix* > (const_that);
539
540 // Initialize A, B pointers for LEFT_MULTIPLY* cases
541 DenseMatrix
542 *A = this,
543 *B = that;
544
545 // For RIGHT_MULTIPLY* cases, swap the meaning of A and B.
546 // Here is a full table of combinations we can pass to BLASgemm, and what the answer is when finished:
547 // pass A B -> (Fortran) -> A^T B^T -> (C++) -> (A^T B^T)^T -> (identity) -> B A "lt multiply"
548 // pass B A -> (Fortran) -> B^T A^T -> (C++) -> (B^T A^T)^T -> (identity) -> A B "rt multiply"
549 // pass A B^T -> (Fortran) -> A^T B -> (C++) -> (A^T B)^T -> (identity) -> B^T A "lt multiply t"
550 // pass B^T A -> (Fortran) -> B A^T -> (C++) -> (B A^T)^T -> (identity) -> A B^T "rt multiply t"
551 if (flag==RIGHT_MULTIPLY || flag==RIGHT_MULTIPLY_TRANSPOSE)
552 std::swap(A,B);
553
554 // transa, transb values to pass to blas
555 char
556 transa[] = "n",
557 transb[] = "n";
558
559 // Integer values to pass to BLAS:
560 //
561 // M
562 // In Fortran, the number of rows of op(A),
563 // In the BLAS documentation, typically known as 'M'.
564 //
565 // In C/C++, we set:
566 // M = n_cols(A) if (transa='n')
567 // n_rows(A) if (transa='t')
568 int M = static_cast<int>( A->n() );
569
570 // N
571 // In Fortran, the number of cols of op(B), and also the number of cols of C.
572 // In the BLAS documentation, typically known as 'N'.
573 //
574 // In C/C++, we set: N = n_rows(B) if (transb='n') n_cols(B) if (transb='t')
575 int N = static_cast<int>( B->m() );
576
577 // K
578 // In Fortran, the number of cols of op(A), and also
579 // the number of rows of op(B). In the BLAS documentation,
580 // typically known as 'K'.
581 //
582 // In C/C++, we set: K = n_rows(A) if (transa='n') n_cols(A) if (transa='t')
583 int K = static_cast<int>( A->m() );
584 // LDA (leading dimension of A). In our cases LDA is always the number of columns of A.
585 int LDA = static_cast<int>( A->n() );
586 // LDB (leading dimension of B). In our cases, LDB is always the number of columns of B.
587 int LDB = static_cast<int>( B->n() );
588
589 if (flag == LEFT_MULTIPLY_TRANSPOSE) {
590 transb[0] = 't';
591 N = static_cast<int>( B->n() );
592 }
593
594 else if (flag == RIGHT_MULTIPLY_TRANSPOSE) {
595 transa[0] = 't';
596 std::swap(M,K);
597 }
598
599 // LDC (leading dimension of C). LDC is the number of columns in the solution matrix.
600 int LDC = M;
601
602 // Scalar values to pass to BLAS //
603 // scalar multiplying the whole product AB
604 double alpha = 1.;
605 // scalar multiplying C, which is the original matrix.
606 double beta = 0.;
607 // Storage for the result
608 std::vector <double> result (result_size);
609 // Finally ready to call the BLAS
610 BLASgemm_(transa, transb, &M, &N, &K, &alpha, &(A->_val[0]), &LDA, &(B->_val[0]), &LDB, &beta, &result[0], &LDC);
611
612 // Update the relevant dimension for this matrix.
613 switch (flag) {
614 case LEFT_MULTIPLY:
615 this->_m = other.m();
616 break;
617 case RIGHT_MULTIPLY:
618 this->_n = other.n();
619 break;
620 case LEFT_MULTIPLY_TRANSPOSE:
621 this->_m = other.n();
622 break;
623 case RIGHT_MULTIPLY_TRANSPOSE:
624 this->_n = other.m();
625 break;
626 default:
627 std::cout << "Unknown flag selected." << std::endl;
628 exit(0);
629 }
630
631 // Swap my data vector with the result
632 this->_val.swap(result);
633}

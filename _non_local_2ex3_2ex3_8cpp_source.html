268void GetL2Norm (MultiLevelSolution & mlSol, MultiLevelSolution & mlSolGlobal) {
269
270 //using the same mesh for FETI and global solutions
271 const unsigned level = mlSol._mlMesh->GetNumberOfLevels() - 1;
272 Mesh* msh = mlSol._mlMesh->GetLevel (level);
273
274 Solution* sol = mlSol.GetSolutionLevel (level);
275 Solution* solGlobal = mlSolGlobal.GetSolutionLevel (level);
276
277 const unsigned dim = msh->GetDimension();
278
279 unsigned xType = 2; // get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC)
280
281 double L2_error_FETI_global = 0.;
282
283 unsigned solu1Index;
284 solu1Index = mlSol.GetIndex ("u1");
285 unsigned solu1Type = mlSol.GetSolutionType (solu1Index);
286
287 unsigned solu2Index;
288 solu2Index = mlSol.GetIndex ("u2");
289 unsigned solu2Type = mlSol.GetSolutionType (solu2Index);
290
291 unsigned soluIndex;
292 soluIndex = mlSolGlobal.GetIndex ("u");
293 unsigned soluType = mlSolGlobal.GetSolutionType (soluIndex);
294
295 unsigned iproc = msh->processor_id();
296 unsigned nprocs = msh->n_processors();
297
298 for (int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
299
300 short unsigned ielGeom = msh->GetElementType (iel);
301 short unsigned ielGroup = msh->GetElementGroup (iel);
302 unsigned nDofu = msh->GetElementDofNumber (iel, soluType);
303 unsigned nDofx = msh->GetElementDofNumber (iel, xType);
304
305 vector < vector < double > > x1 (dim);
306
307 for (int i = 0; i < dim; i++) {
308 x1[i].resize (nDofx);
309 }
310
311 vector < double > solu (nDofu);
312 vector < double > solu1 (nDofu);
313 vector < double > solu2 (nDofu);
314
315 for (unsigned i = 0; i < nDofu; i++) {
316 unsigned solDofu = msh->GetSolutionDof (i, iel, soluType);
317 unsigned solDofu1 = msh->GetSolutionDof (i, iel, solu1Type);
318 unsigned solDofu2 = msh->GetSolutionDof (i, iel, solu2Type);
319 solu[i] = (*solGlobal->_Sol[soluIndex]) (solDofu);
320 solu1[i] = (*sol->_Sol[solu1Index]) (solDofu1);
321 solu2[i] = (*sol->_Sol[solu2Index]) (solDofu2);
322 }
323
324 for (unsigned i = 0; i < nDofx; i++) {
325 unsigned xDof = msh->GetSolutionDof (i, iel, xType);
326
327 for (unsigned jdim = 0; jdim < dim; jdim++) {
328 x1[jdim][i] = (*msh->_topology->_Sol[jdim]) (xDof);
329 }
330 }
331
332 vector <double> phi;
333 vector <double> phi_x;
334 double weight;
335
336 for (unsigned ig = 0; ig < msh->_finiteElement[ielGeom][soluType]->GetGaussPointNumber(); ig++) {
337
338 msh->_finiteElement[ielGeom][soluType]->Jacobian (x1, ig, weight, phi, phi_x);
339
340 double solu_gss = 0.;
341 double solu1_gss = 0.;
342 double solu2_gss = 0.;
343 double x_gss = 0.;
344 double y_gss = 0.;
345
346 for (unsigned i = 0; i < nDofu; i++) {
347 solu_gss += phi[i] * solu[i];
348 solu1_gss += phi[i] * solu1[i];
349 solu2_gss += phi[i] * solu2[i];
350 x_gss += phi[i] * x1[0][i]; // this is x at the Gauss point
351 y_gss += phi[i] * x1[1][i]; // this is y at the Gauss point
352 }
353
354 double uFETI_gss;
355 if (ielGroup == 5 || ielGroup == 6 || ielGroup == 8 || ielGroup == 9) uFETI_gss = solu1_gss;
356 else uFETI_gss = solu2_gss;
357
358 L2_error_FETI_global += (solu_gss - uFETI_gss) * (solu_gss - uFETI_gss) * weight;
359
360 }
361 }
362
363 double norm2 = 0.;
364 MPI_Allreduce (&L2_error_FETI_global, &norm2, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
365 double norm = sqrt (norm2);
366 std::cout.precision (16);
367 std::cout << "L2 norm of ERROR: Global - FETI = " << norm << std::endl;
368
369
370}

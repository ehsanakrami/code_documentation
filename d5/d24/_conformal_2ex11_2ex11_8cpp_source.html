231void ProjectSolution(MultiLevelSolution& mlSol) {
232 unsigned level = mlSol.GetMLMesh()->GetNumberOfLevels() - 1u;
233
234 Solution* sol = mlSol.GetSolutionLevel(level);
235 Mesh* msh = mlSol.GetMLMesh()->GetLevel(level);
236 elem* el = msh->el;
237
238 //unsigned dim = msh->GetDimension();
239 unsigned dim = 2;
240 unsigned DIM = 3;
241 std::vector < unsigned > indexDx(DIM);
242 indexDx[0] = mlSol.GetIndex("Dx1");
243 indexDx[1] = mlSol.GetIndex("Dx2");
244 indexDx[2] = mlSol.GetIndex("Dx3");
245 unsigned solType = mlSol.GetSolutionType(indexDx[0]);
246
247 unsigned iproc = msh->processor_id();
248
249 for(int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
250
251 unsigned nDofs = msh->GetElementDofNumber(iel, solType);
252
253 for(unsigned i = 0; i < nDofs; i++) {
254 unsigned idof = msh->GetSolutionDof(i, iel, solType);
255 unsigned xdof = msh->GetSolutionDof(i, iel, 2);
256
257 double Yhat = (*msh->GetTopology()->_Sol[1])(xdof);
258 double Zhat = (*msh->GetTopology()->_Sol[2])(xdof);
259
260 double DY = (*sol->_Sol[indexDx[1]])(idof);
261 double DZ = (*sol->_Sol[indexDx[2]])(idof);
262
263 double theta = atan2(Zhat + DZ, Yhat + DY);
264
265 sol->_Sol[indexDx[1]]->set(idof, 0.5 * cos(theta) - Yhat);
266 sol->_Sol[indexDx[2]]->set(idof, 0.5 * sin(theta) - Zhat);
267 }
268 }
269
270 sol->_Sol[indexDx[1]]->close();
271 sol->_Sol[indexDx[2]]->close();
272
273}

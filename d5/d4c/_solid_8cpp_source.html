49Solid::Solid(Parameter& par,
50 const double young_module,
51 const double poisson_coeff,
52 const double density,
53 const char model[],
54 const double k,
55 const double cp,
56 const double alpha) : Material(par, density, k, cp, alpha) {
57
58 _young_module = young_module;
59
60
61 if (!strcmp(model,"Linear_elastic") || !strcmp(model,"Saint-Venant")) {
62 _model = 0;
63 _penalty = false;
64 _mass_penalty = false;
65 }
66 else if (!strcmp(model,"Saint-Venant-Penalty")) {
67 _model = 0;
68 _penalty = true;
69 _mass_penalty = false;
70 }
71 else if (!strcmp(model,"Neo-Hookean") || !strcmp(model,"Neo-Hookean-MassPenalty")) {
72 _model = 1;
73 _penalty = false;
74 _mass_penalty = !strcmp(model,"Neo-Hookean-MassPenalty") ? true : false;
75 }
76 else if (!strcmp(model,"Neo-Hookean-BW") || !strcmp(model,"Neo-Hookean-BW-MassPenalty") ) { //Bonet-Wood
77 _model = 2;
78 _penalty = false;
79 _mass_penalty = !strcmp(model,"Neo-Hookean-BW-MassPenalty") ? true : false;
80 }
81 else if (!strcmp(model,"Neo-Hookean-BW-Penalty")) {
82 _model = 3;
83 _penalty = true;
84 _mass_penalty = false;
85 }
86 else if (!strcmp(model,"Neo-Hookean-AB-Penalty")) { //Allan-Bower
87 _model = 4;
88 _penalty = true;
89 _mass_penalty = false;
90 }
91 else if (!strcmp(model,"Mooney-Rivlin") || !strcmp(model,"Mooney-Rivlin-MassPenalty")) {
92 _model = 5;
93 _penalty = false;
94 _mass_penalty = (!strcmp(model,"Mooney-Rivlin-MassPenalty")) ? true : false;
95 }
96 else {
97 cout << "Error! This solid model is not implemented " << endl;
98 abort();
99 }
100
101 if (poisson_coeff <= 0.5 && poisson_coeff >= 0) {
102 _poisson_coeff = poisson_coeff;
103 }
104 else {
105 cout << "Error: the value for the Poisson coeffcient must be greater than 0 and less equal than 0.5!" << endl;
106 abort();
107 }
108
109
110 if(poisson_coeff<0.5){
111 _lambda_lame = (_young_module*_poisson_coeff)/((1.+_poisson_coeff)*(1.-2.*_poisson_coeff));
112 }
113 else if (true == _penalty){
114 std::cout << "Error this solid model requires a Poisson coeffcient strictly less than 0.5"<<endl;
115 abort();
116 }
117 else{
118 cout << "Warning: the value for the Poisson coeffcient is 0.5, the material is incompressible"<<endl
119 << "The Lame constant is infinity and it has been set equal to 1.0e100" << endl;
120 _lambda_lame = 1.0e100;
121 }
122 _mu_lame = _young_module/(2.*(1.+_poisson_coeff));
123
124// cout << endl << "SOLID properties: " << endl;
125// cout << "Density [kg/m3]: " << _density << endl;
126// cout << "Young Module [Pa/m2]: " << _young_module << endl;
127// cout << "Poisson Coefficient: " << _poisson_coeff << endl;
128// cout << "Lambda (Lame first parameter): " << _lambda_lame << endl;
129// cout << "Shear Modulus (Lame second parameter): " << _mu_lame << endl << endl;
130
131}

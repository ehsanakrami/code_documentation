165void AssemblePoisson(MultiLevelProblem& ml_prob) {
166 // ml_prob is the global object from/to where get/set all the data
167 // level is the level of the PDE system to be assembled
168 // levelMax is the Maximum level of the MultiLevelProblem
169 // assembleMatrix is a flag that tells if only the residual or also the matrix should be assembled
170
171 // extract pointers to the several objects that we are going to use
172 LinearImplicitSystem* mlPdeSys = &ml_prob.get_system<LinearImplicitSystem> ("Poisson"); // pointer to the linear implicit system named "Poisson"
173 const unsigned level = mlPdeSys->GetLevelToAssemble();
174
175 Mesh* msh = ml_prob._ml_msh->GetLevel(level); // pointer to the mesh (level) object
176 elem* el = msh->el; // pointer to the elem object in msh (level)
177
178 MultiLevelSolution* mlSol = ml_prob._ml_sol; // pointer to the multilevel solution object
179 Solution* sol = ml_prob._ml_sol->GetSolutionLevel(level); // pointer to the solution (level) object
180
181 LinearEquationSolver* pdeSys = mlPdeSys->_LinSolver[level]; // pointer to the equation (level) object
182 bool assembleMatrix = mlPdeSys->GetAssembleMatrix();
183
184 // call the adept stack object
185 SparseMatrix* KK = pdeSys->_KK; // pointer to the global stifness matrix object in pdeSys (level)
186
187 NumericVector* RES = pdeSys->_RES; // pointer to the global residual vector object in pdeSys (level)
188
189 const unsigned dim = msh->GetDimension(); // get the domain dimension of the problem
190 unsigned dim2 = (3 * (dim - 1) + !(dim - 1)); // dim2 is the number of second order partial derivatives (1,3,6 depending on the dimension)
191 unsigned iproc = msh->processor_id(); // get the process_id (for parallel computation)
192
193 // reserve memory for the local standar vectors
194 const unsigned maxSize = static_cast< unsigned >(ceil(pow(3, dim))); // conservative: based on line3, quad9, hex27
195
196 //solution variable
197 unsigned solUIndex;
198 solUIndex = mlSol->GetIndex("U"); // get the position of "U" in the ml_sol object
199 unsigned solUType = mlSol->GetSolutionType(solUIndex); // get the finite element type for "U"
200
201 unsigned solUPdeIndex;
202 solUPdeIndex = mlPdeSys->GetSolPdeIndex("U"); // get the position of "T" in the pdeSys object
203
204 std::vector < double > solU; // local solution
205 std::vector < std::vector < double > > coordX(dim); // local coordinates
206 unsigned coordXType = 2; // get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC)
207
208 solU.reserve(maxSize);
209 for(unsigned k = 0; k < dim; k++) {
210 coordX[k].reserve(maxSize);
211 }
212
213 //RHS vectors
214 std::vector < double > fU; // local solution
215 fU.reserve(maxSize);
216
217 std::vector <double> phiU; // local test function
218 std::vector <double> phiU_x; // local test function first order partial derivatives
219 std::vector <double> phiU_xx; // local test function second order partial derivatives
220
221 phiU.reserve(maxSize);
222 phiU_x.reserve(maxSize * dim);
223 phiU_xx.reserve(maxSize * dim2);
224
225 double weight; // gauss point weight
226
227 std::vector < int > sysDof; // local to global pdeSys dofs
228 sysDof.reserve((dim + 2) *maxSize);
229
230 std::vector < double > Res; // local redidual vector
231 Res.reserve((dim + 2) *maxSize);
232
233 std::vector < double > Jac;
234 Jac.reserve((dim + 2) *maxSize * (dim + 2) *maxSize);
235
236 if(assembleMatrix) KK->zero(); // Set to zero all the entries of the Global Matrix
237
238 //BEGIN element loop
239 for(int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
240
241 //BEGIN local dof number extraction
242 unsigned nDofsU = msh->GetElementDofNumber(iel, solUType); //velocity
243 unsigned nDofsX = msh->GetElementDofNumber(iel, coordXType); // coordinates
244 //END local dof number extraction
245
246 //BEGIN memory allocation
247 Res.resize(nDofsU);
248 std::fill(Res.begin(), Res.end(), 0);
249 Jac.resize(nDofsU * nDofsU);
250 std::fill(Jac.begin(), Jac.end(), 0);
251 sysDof.resize(nDofsU);
252 solU.resize(nDofsU);
253
254 //END memory allocation
255 fU.assign(nDofsU,1.);
256
257 //BEGIN global to local extraction
258 for(unsigned i = 0; i < nDofsU; i++) { //velocity
259 unsigned solUDof = msh->GetSolutionDof(i, iel, solUType); //local to global solution dof
260 solU[i] = (*sol->_Sol[solUIndex])(solUDof); //global to local solution value
261 sysDof[i] = pdeSys->GetSystemDof(solUIndex, solUPdeIndex, i, iel); //local to global system dof
262 }
263
264 for (unsigned i = 0; i < dim; i++){
265 coordX[i].resize(nDofsX);
266 }
267
268 for(unsigned i = 0; i < nDofsX; i++) { //coordinates
269 unsigned coordXDof = msh->GetSolutionDof(i, iel, coordXType); //local to global coordinate dof
270 for(unsigned k = 0; k < dim; k++) {
271 coordX[k][i] = (*msh->_topology->_Sol[k])(coordXDof); //global to local coordinate value
272 }
273 }
274
275 //END global to local extraction
276 //BEGIN Gauss point loop
277 short unsigned ielGeom = msh->GetElementType(iel);
278 for(unsigned ig = 0; ig < msh->_finiteElement[ielGeom][solUType]->GetGaussPointNumber(); ig++) {
279 // *** get gauss point weight, test function and test function partial derivatives ***
280 msh->_finiteElement[ielGeom][solUType]->Jacobian(coordX, ig, weight, phiU, phiU_x, phiU_xx);
281
282 // evaluate the solution, the solution derivatives and the coordinates in the gauss point
283 double solU_gss = 0;
284 std::vector < double > gradSolU_gss(dim, 0.);
285 std::vector < double > x_gss(dim, 0.);
286
287 for(unsigned i = 0; i < nDofsU; i++) {
288 solU_gss += phiU[i] * solU[i];
289 for(unsigned j = 0; j < dim; j++) {
290 gradSolU_gss[j] += phiU_x[i * dim + j] * solU[i];
291 x_gss[j] += coordX[j][i] * phiU[i];
292 }
293 }
294
295 //BEGIN phiU_i loop
296 for(unsigned i = 0; i < nDofsU; i++) {
297 unsigned irow = i;
298 double srcTerm = - GetExactSolutionLaplace(x_gss);
299// Res[irow] += phiU[i] * fU[i] * weight;
300 Res[i] += srcTerm * phiU[i] * weight;
301
302 for(unsigned k = 0; k < dim; k++) {
303 Res[irow] += - phiU_x[i * dim + k] * gradSolU_gss[k] * weight;
304 if(assembleMatrix) {
305 unsigned irowMat = irow * nDofsU;
306 for(unsigned j = 0; j < nDofsU; j++) {
307 Jac[ irowMat + j ] += phiU_x[i * dim + k] * phiU_x[j * dim + k] * weight;
308 }
309 }
310 }
311 }
312 //END phiU_i loop
313 }
314
315 //END Gauss point loop
316 //BEGIN local to global Matrix/Vector assembly
317 RES->add_vector_blocked(Res, sysDof);
318 if(assembleMatrix) {
319 KK->add_matrix_blocked(Jac, sysDof, sysDof);
320 }
321 //END local to global Matrix/Vector assembly
322 }
323
324 //END element loop
325 RES->close();
326 if(assembleMatrix) {
327 KK->close();
328 }
329
330 // ***************** END ASSEMBLY *******************
331}

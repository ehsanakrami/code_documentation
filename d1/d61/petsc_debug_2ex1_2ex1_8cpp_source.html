151void AssemblePoissonProblem_AD(MultiLevelProblem& ml_prob)
152{
153 // ml_prob is the global object from/to where get/set all the data
154 // level is the level of the PDE system to be assembled
155 // levelMax is the Maximum level of the MultiLevelProblem
156 // assembleMatrix is a flag that tells if only the residual or also the matrix should be assembled
157
158 // call the adept stack object
159
160
161 adept::Stack& s = FemusInit::_adeptStack;
162
163 // extract pointers to the several objects that we are going to use
164
165 LinearImplicitSystem* mlPdeSys = &ml_prob.get_system<LinearImplicitSystem> ("Poisson"); // pointer to the linear implicit system named "Poisson"
166 const unsigned level = mlPdeSys->GetLevelToAssemble();
167
168 Mesh* msh = ml_prob._ml_msh->GetLevel(level); // pointer to the mesh (level) object
169 elem* el = msh->el; // pointer to the elem object in msh (level)
170
171 MultiLevelSolution* mlSol = ml_prob._ml_sol; // pointer to the multilevel solution object
172 Solution* sol = ml_prob._ml_sol->GetSolutionLevel(level); // pointer to the solution (level) object
173
174 LinearEquationSolver* pdeSys = mlPdeSys->_LinSolver[level]; // pointer to the equation (level) object
175 SparseMatrix* KK = pdeSys->_KK; // pointer to the global stifness matrix object in pdeSys (level)
176 NumericVector* RES = pdeSys->_RES; // pointer to the global residual vector object in pdeSys (level)
177
178 const unsigned dim = msh->GetDimension(); // get the domain dimension of the problem
179 unsigned dim2 = (3 * (dim - 1) + !(dim - 1)); // dim2 is the number of second order partial derivatives (1,3,6 depending on the dimension)
180 const unsigned maxSize = static_cast< unsigned >(ceil(pow(3, dim))); // conservative: based on line3, quad9, hex27
181
182 unsigned iproc = msh->processor_id(); // get the process_id (for parallel computation)
183
184 //solution variable
185 unsigned soluIndex;
186 soluIndex = mlSol->GetIndex("u"); // get the position of "u" in the ml_sol object
187 unsigned soluType = mlSol->GetSolutionType(soluIndex); // get the finite element type for "u"
188
189 unsigned soluPdeIndex;
190 soluPdeIndex = mlPdeSys->GetSolPdeIndex("u"); // get the position of "u" in the pdeSys object
191
192 vector < adept::adouble > solu; // local solution
193 solu.reserve(maxSize);
194
195 vector < vector < double > > x(dim); // local coordinates
196 unsigned xType = 2; // get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC)
197
198 for(unsigned i = 0; i < dim; i++) {
199 x[i].reserve(maxSize);
200 }
201
202 vector <double> phi; // local test function
203 vector <double> phi_x; // local test function first order partial derivatives
204 vector <double> phi_xx; // local test function second order partial derivatives
205 double weight; // gauss point weight
206
207 phi.reserve(maxSize);
208 phi_x.reserve(maxSize * dim);
209 phi_xx.reserve(maxSize * dim2);
210
211 vector< adept::adouble > aRes; // local redidual vector
212 aRes.reserve(maxSize);
213
214 vector< int > l2GMap; // local to global mapping
215 l2GMap.reserve(maxSize);
216 vector< double > Res; // local redidual vector
217 Res.reserve(maxSize);
218 vector < double > Jac;
219 Jac.reserve(maxSize * maxSize);
220
221 KK->zero(); // Set to zero all the entries of the Global Matrix
222
223 // element loop: each process loops only on the elements that owns
224 for(int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
225
226 short unsigned ielGeom = msh->GetElementType(iel);
227 unsigned nDofu = msh->GetElementDofNumber(iel, soluType); // number of solution element dofs
228 unsigned nDofx = msh->GetElementDofNumber(iel, xType); // number of coordinate element dofs
229
230 // resize local arrays
231 l2GMap.resize(nDofu);
232 solu.resize(nDofu);
233
234 for(int i = 0; i < dim; i++) {
235 x[i].resize(nDofx);
236 }
237
238 aRes.resize(nDofu); //resize
239 std::fill(aRes.begin(), aRes.end(), 0); //set aRes to zero
240
241 // local storage of global mapping and solution
242 for(unsigned i = 0; i < nDofu; i++) {
243 unsigned solDof = msh->GetSolutionDof(i, iel, soluType); // global to global mapping between solution node and solution dof
244 solu[i] = (*sol->_Sol[soluIndex])(solDof); // global extraction and local storage for the solution
245 l2GMap[i] = pdeSys->GetSystemDof(soluIndex, soluPdeIndex, i, iel); // global to global mapping between solution node and pdeSys dof
246 }
247
248 // local storage of coordinates
249 for(unsigned i = 0; i < nDofx; i++) {
250 unsigned xDof = msh->GetSolutionDof(i, iel, xType); // global to global mapping between coordinates node and coordinate dof
251
252 for(unsigned jdim = 0; jdim < dim; jdim++) {
253 x[jdim][i] = (*msh->_topology->_Sol[jdim])(xDof); // global extraction and local storage for the element coordinates
254 }
255 }
256
257
258 // start a new recording of all the operations involving adept::adouble variables
259 s.new_recording();
260
261 // *** Gauss point loop ***
262 for(unsigned ig = 0; ig < msh->_finiteElement[ielGeom][soluType]->GetGaussPointNumber(); ig++) {
263 // *** get gauss point weight, test function and test function partial derivatives ***
264 msh->_finiteElement[ielGeom][soluType]->Jacobian(x, ig, weight, phi, phi_x, phi_xx);
265
266 // evaluate the solution, the solution derivatives and the coordinates in the gauss point
267 adept::adouble solu_gss = 0;
268 vector < adept::adouble > gradSolu_gss(dim, 0.);
269 vector < double > x_gss(dim, 0.);
270
271 for(unsigned i = 0; i < nDofu; i++) {
272 solu_gss += phi[i] * solu[i];
273
274 for(unsigned jdim = 0; jdim < dim; jdim++) {
275 gradSolu_gss[jdim] += phi_x[i * dim + jdim] * solu[i];
276 x_gss[jdim] += x[jdim][i] * phi[i];
277 }
278 }
279
280 // *** phi_i loop ***
281 for(unsigned i = 0; i < nDofu; i++) {
282
283 adept::adouble laplace = 0.;
284
285 for(unsigned jdim = 0; jdim < dim; jdim++) {
286 laplace += phi_x[i * dim + jdim] * gradSolu_gss[jdim];
287 }
288
289 double srcTerm = - GetExactSolutionLaplace(x_gss);
290 aRes[i] += (srcTerm * phi[i] - laplace) * weight;
291
292 } // end phi_i loop
293 } // end gauss point loop
294
295 //--------------------------------------------------------------------------------------------------------
296 // Add the local Matrix/Vector into the global Matrix/Vector
297
298 //copy the value of the adept::adoube aRes in double Res and store
299 Res.resize(nDofu); //resize
300
301 for(int i = 0; i < nDofu; i++) {
302 Res[i] = - aRes[i].value();
303 }
304
305 RES->add_vector_blocked(Res, l2GMap);
306
307
308
309 // define the dependent variables
310 s.dependent(&aRes[0], nDofu);
311
312 // define the independent variables
313 s.independent(&solu[0], nDofu);
314
315 // get the jacobian matrix (ordered by row major )
316 Jac.resize(nDofu * nDofu); //resize
317 s.jacobian(&Jac[0], true);
318
319 //store K in the global matrix KK
320 KK->add_matrix_blocked(Jac, l2GMap, l2GMap);
321
322 s.clear_independents();
323 s.clear_dependents();
324
325 } //end element loop for each process
326
327 RES->close();
328
329 KK->close();
330
331 // ***************** END ASSEMBLY *******************
332}

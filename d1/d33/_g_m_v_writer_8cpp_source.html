44 void GMVWriter::Write( const std::string output_path,
45 const std::string order,
46 const std::vector<std::string>& vars,
47 const unsigned time_step ) {
48
49 // ********** linear -> index==0 *** quadratic -> index==1 **********
50 unsigned index = ( strcmp( order.c_str(), fe_fams_for_files[ FILES_CONTINUOUS_LINEAR ].c_str() ) ) ? FILES_CONTINUOUS_QUADRATIC : FILES_CONTINUOUS_LINEAR;
51
52 const std::string filename_prefix = get_filename_prefix();
53
54
55 std::ostringstream filename;
56 filename << output_path << "/" << filename_prefix << ".level" << _gridn << "." << time_step << "." << order << ".gmv";
57
58 std::ofstream fout;
59
60 if( _iproc != 0 ) {
61 fout.rdbuf(); //redirect to dev_null
62 }
63 else {
64 fout.open( filename.str().c_str() );
65 if( fout.is_open() ) {
66 std::cout << std::endl << " The output is printed to file " << filename.str() << " in GMV format" << std::endl;
67 }
68 else {
69 std::cout << std::endl << " The output file " << filename.str() << " cannot be opened.\n";
70 abort();
71 }
72 }
73
74 const Mesh* mesh = _ml_mesh->GetLevel( _gridn - 1 );
75 const Solution* solution = _ml_sol->GetSolutionLevel( _gridn - 1 );
76
77 unsigned nvt = mesh->GetTotalNumberOfDofs( index );
78 unsigned nel = mesh->GetNumberOfElements();
79 unsigned dim = mesh->GetDimension();
80
81 std::vector < double > vector1;
82 vector1.reserve( nvt );
83
84 std::vector < double > vector2;
85 if( nvt > ( dim + 1 ) *nel )
86 vector2.reserve( nvt );
87 else
88 vector2.reserve( ( dim + 1 ) *nel );
89
90 NumericVector* numVector;
91 numVector = NumericVector::build().release();
92 numVector->init( mesh->dofmap_get_dof_offset(index, _nprocs), mesh->dofmap_get_own_size(index, _iproc), true, AUTOMATIC );
93
94
95 //BEGIN GMV FILE PRINT
96 char* buffer = new char[10];
97 sprintf( buffer, "%s", "gmvinput" );
98 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
99 sprintf( buffer, "%s", "ieeei4r8" );
100 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
101
102
103 //BEGIN COORDINATES
104 sprintf( buffer, "%s", "nodes" );
105 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
106 fout.write( ( char* ) &nvt, sizeof( unsigned ) );
107
108 for( int i = 0; i < 3; i++ ) {
109 if( !_surface ) {
110 numVector->matrix_mult( *mesh->_topology->_Sol[i], * _fe_proj_matrices.GetQitoQjProjection( index, 2, * mesh) );
111 if( _graph && i == 2 ) {
112 const unsigned indGraphVar = _ml_sol->GetIndex( _graphVariable.c_str() );
113 numVector->matrix_mult( *solution->_Sol[indGraphVar], * _fe_proj_matrices.GetQitoQjProjection( index, _ml_sol->GetSolutionType( indGraphVar ), * mesh ) );
114 }
115 }
116 else {
117 const unsigned indSurfVar = _ml_sol->GetIndex( _surfaceVariables[i].c_str() );
118 numVector->matrix_mult( *solution->_Sol[indSurfVar], * _fe_proj_matrices.GetQitoQjProjection( index, _ml_sol->GetSolutionType( indSurfVar ), * mesh ) );
119 }
120
121 numVector->localize_to_one( vector1, 0 );
122 if( _ml_sol != NULL && _moving_mesh && dim > i ) {
123 const unsigned indDXDYDZ = _ml_sol->GetIndex( _moving_vars[i].c_str() );
124 numVector->matrix_mult( *solution->_Sol[indDXDYDZ], * _fe_proj_matrices.GetQitoQjProjection( index, _ml_sol->GetSolutionType( indDXDYDZ ), * mesh ) );
125 numVector->localize_to_one( vector2, 0 );
126 if( _iproc == 0 ) {
127 for( unsigned i = 0; i < nvt; i++ )
128 vector1[i] += vector2[i];
129 }
130 }
131 fout.write( ( char* ) &vector1[0], nvt * sizeof( double ) );
132
133 }
134 //END COORDINATES
135
136 //BEGIN CONNETTIVITY
137 const int eltp[2][6] = {{8, 4, 6, 4, 3, 2}, {20, 10, 15, 8, 6, 3}};
138 sprintf( buffer, "%s", "cells" );
139 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
140
141 fout.write( ( char* ) &nel, sizeof( unsigned ) );
142
143 unsigned topology[27];
144
145 //mesh->_topology->_Sol[mesh->GetTypeIndex()]->localize_to_one( vector1, 0 );
146
147 for( unsigned isdom = 0; isdom < _nprocs; isdom++ ) {
148 mesh->el->LocalizeElementDof( isdom );
149 mesh->el->LocalizeElement_Level_Type_Group_Material(isdom);
150 if( _iproc == 0 ) {
151 for( unsigned ii = mesh->_elementOffset[isdom]; ii < mesh->_elementOffset[isdom + 1]; ii++ ) {
152 //short unsigned ielt = static_cast < short unsigned >( vector1[ii] + 0.25 );
153 short unsigned ielt = mesh->GetElementType(ii);
154 if( ielt == 0 )
155 sprintf( buffer, "phex%d", eltp[index][0] );
156 else if( ielt == 1 )
157 sprintf( buffer, "ptet%d", eltp[index][1] );
158 else if( ielt == 2 )
159 sprintf( buffer, "pprism%d", eltp[index][2] );
160 else if( ielt == 3 ) {
161 if( eltp[index][3] == 8 )
162 sprintf( buffer, "%dquad", eltp[index][3] );
163 else
164 sprintf( buffer, "quad" );
165 }
166 else if( ielt == 4 ) {
167 if( eltp[index][4] == 6 )
168 sprintf( buffer, "%dtri", eltp[index][4] );
169 else
170 sprintf( buffer, "tri" );
171 }
172 else if( ielt == 5 ) {
173 if( eltp[index][5] == 3 )
174 sprintf( buffer, "%dline", eltp[index][5] );
175 else
176 sprintf( buffer, "line" );
177 }
178 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
179 const unsigned nvertices = mesh->GetMeshElements()->GetNVE(ielt, index);
180 fout.write( ( char* ) & nvertices, sizeof( unsigned ) );
181 for( unsigned j = 0; j < nvertices; j++ ) {
182 unsigned jnode_Metis = mesh->GetSolutionDof( j, ii, index );
183 topology[j] = jnode_Metis + 1;
184 }
185 fout.write( ( char* ) topology, sizeof( unsigned ) * nvertices );
186 }
187 }
188 mesh->el->FreeLocalizedElementDof();
189 mesh->el->FreeLocalizedElement_Level_Type_Group_Material();
190 }
191
192 //END CONNETTIVITY
193
194 //BEGIN VARIABLES
195 const unsigned zero = 0u;
196 const unsigned one = 1u;
197 sprintf( buffer, "%s", "variable" );
198 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
199
200 //BEGIN METIS PARTITIONING
201 strcpy( buffer, "METIS_DD" );
202 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
203 fout.write( ( char* ) &zero, sizeof( unsigned ) );
204
205 vector1.resize( nel );
206 unsigned icount = 0;
207 for( int isdom = 0; isdom < _nprocs; isdom++ ) {
208 for( unsigned ii = mesh->_elementOffset[isdom]; ii < mesh->_elementOffset[isdom + 1]; ii++ ) {
209 vector1[icount] = isdom;
210 icount++;
211 }
212 }
213 fout.write( ( char* ) &vector1[0], nel * sizeof( double ) );
214 //END METIS PARTITIONING
215
216 //BEGIN SOLUTION
217 if( _ml_sol != NULL ) {
218
219 bool printAll = 0;
220 for( unsigned ivar = 0; ivar < vars.size(); ivar++ ) {
221 printAll += !( vars[ivar].compare( "All" ) ) + !( vars[ivar].compare( "all" ) ) + !( vars[ivar].compare( "ALL" ) );
222 }
223
224 for( unsigned ivar = 0; ivar < !printAll * vars.size() + printAll * _ml_sol->GetSolutionSize(); ivar++ ) {
225 unsigned i = ( printAll == 0 ) ? _ml_sol->GetIndex( vars[ivar].c_str() ) : ivar;
226
227 for( int name = 0; name < 4; name++ ) {
228
229 const std::string printName = print_sol_bdc_res_eps_name( _ml_sol->GetSolName_from_index( i ) , name);
230
231
232 if( name == _index_sol || ( _debugOutput && solution->is_unknown_of_system(i) ) ) {
233
234
235 //BEGIN LAGRANGIAN Fem SOLUTION
236 if( _ml_sol->GetSolutionType( i ) < NFE_FAMS_C_ZERO_LAGRANGE ) { // ********** on the nodes **********
237 fout.write( ( char* ) printName.c_str(), sizeof( char ) * 8 );
238 fout.write( ( char* ) &one, sizeof( unsigned ) );
239 if( name == _index_sol ) {
240 numVector->matrix_mult( *solution->_Sol[i], * _fe_proj_matrices.GetQitoQjProjection( index, _ml_sol->GetSolutionType( i ), * mesh ) );
241 }
242 else if( name == _index_bdc ) {
243 numVector->matrix_mult( *solution->_Bdc[i], * _fe_proj_matrices.GetQitoQjProjection( index, _ml_sol->GetSolutionType( i ), * mesh ) );
244 }
245 else if( name == _index_res ) {
246 numVector->matrix_mult( *solution->_Res[i], * _fe_proj_matrices.GetQitoQjProjection( index, _ml_sol->GetSolutionType( i ), * mesh ) );
247 }
248 else if( name == _index_eps ) {
249 numVector->matrix_mult( *solution->_Eps[i], * _fe_proj_matrices.GetQitoQjProjection( index, _ml_sol->GetSolutionType( i ), * mesh ) );
250 }
251 numVector->localize_to_one( vector1, 0 );
252 fout.write( ( char* ) &vector1[0], nvt * sizeof( double ) );
253 }
254 //END LAGRANGIAN Fem SOLUTION
255
256 //BEGIN DISCONTINUOUS Fem SOLUTION
257 else if( _ml_sol->GetSolutionType( i ) < NFE_FAMS ) { // ********** on the elements **********
258 fout.write( ( char* ) printName.c_str(), sizeof( char ) * 8 );
259 fout.write( ( char* ) &zero, sizeof( unsigned ) );
260
261 if( name == _index_sol ) {
262 solution->_Sol[i]->localize_to_one( vector2, 0 );
263 }
264 else if( name == _index_bdc ) {
265 solution->_Bdc[i]->localize_to_one( vector2, 0 );
266 }
267 else if( name == _index_res ) {
268 solution->_Res[i]->localize_to_one( vector2, 0 );
269 }
270 else if( name == _index_eps ) {
271 solution->_Eps[i]->localize_to_one( vector2, 0 );
272 }
273 vector1.resize( nel );
274 for( unsigned ii = 0; ii < nel; ii++ ) {
275 vector1[ii] = vector2[ mesh->GetSolutionDof( 0, ii, _ml_sol->GetSolutionType( i ) ) ];
276 }
277
278 fout.write( ( char* ) &vector1[0], nel * sizeof( double ) );
279 }
280 //END DISCONTINUOUS Fem SOLUTION
281
282 }
283
284 }
285 }
286 }
287 //END SOLUTION
288
289 sprintf( buffer, "%s", "endvars" );
290 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
291 //END VARIABLES
292
293 sprintf( buffer, "%s", "endgmv" );
294 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
295 fout.close();
296 //END GMV FILE PRINT
297
298 delete numVector;
299 delete [] buffer;
300
301 return;
302 }

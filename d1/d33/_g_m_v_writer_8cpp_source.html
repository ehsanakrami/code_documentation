46 void GMVWriter::Write( const std::string output_path, const char order[], const std::vector<std::string>& vars, const unsigned time_step ) {
47
48 // ********** linear -> index==0 *** quadratic -> index==1 **********
49 unsigned index = ( strcmp( order, "linear" ) ) ? 1 : 0;
50
51 std::string filename_prefix;
52 if( _ml_sol != NULL )
53 filename_prefix = "sol";
54 else
55 filename_prefix = "mesh";
56
57 std::ostringstream filename;
58 filename << output_path << "/" << filename_prefix << ".level" << _gridn << "." << time_step << "." << order << ".gmv";
59
60 std::ofstream fout;
61
62 if( _iproc != 0 ) {
63 fout.rdbuf(); //redirect to dev_null
64 }
65 else {
66 fout.open( filename.str().c_str() );
67 if( fout.is_open() ) {
68 std::cout << std::endl << " The output is printed to file " << filename.str() << " in GMV format" << std::endl;
69 }
70 else {
71 std::cout << std::endl << " The output file " << filename.str() << " cannot be opened.\n";
72 abort();
73 }
74 }
75
76 Mesh* mesh = _ml_mesh->GetLevel( _gridn - 1 );
77 Solution* solution = _ml_sol->GetSolutionLevel( _gridn - 1 );
78 unsigned nvt = mesh->GetTotalNumberOfDofs( index );
79 unsigned nel = mesh->GetNumberOfElements();
80 unsigned dim = mesh->GetDimension();
81
82 std::vector < double > vector1;
83 vector1.reserve( nvt );
84
85 std::vector < double > vector2;
86 if( nvt > ( dim + 1 ) *nel )
87 vector2.reserve( nvt );
88 else
89 vector2.reserve( ( dim + 1 ) *nel );
90
91 NumericVector* numVector;
92 numVector = NumericVector::build().release();
93 numVector->init( mesh->dofmap_get_dof_offset(index, _nprocs), mesh->dofmap_get_own_size(index, _iproc), true, AUTOMATIC );
94
95
96 //BEGIN GMV FILE PRINT
97 char* buffer = new char[10];
98 sprintf( buffer, "%s", "gmvinput" );
99 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
100 sprintf( buffer, "%s", "ieeei4r8" );
101 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
102
103
104 //BEGIN COORDINATES
105 sprintf( buffer, "%s", "nodes" );
106 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
107 fout.write( ( char* ) &nvt, sizeof( unsigned ) );
108
109 for( int i = 0; i < 3; i++ ) {
110 if( !_surface ) {
111 numVector->matrix_mult( *mesh->_topology->_Sol[i],
112 *mesh->GetQitoQjProjection( index, 2 ) );
113 if( _graph && i == 2 ) {
114 unsigned indGraphVar = _ml_sol->GetIndex( _graphVariable.c_str() );
115 numVector->matrix_mult( *solution->_Sol[indGraphVar],
116 *mesh->GetQitoQjProjection( index, _ml_sol->GetSolutionType( indGraphVar ) ) );
117 }
118 }
119 else {
120 unsigned indSurfVar = _ml_sol->GetIndex( _surfaceVariables[i].c_str() );
121 numVector->matrix_mult( *solution->_Sol[indSurfVar],
122 *mesh->GetQitoQjProjection( index, _ml_sol->GetSolutionType( indSurfVar ) ) );
123 }
124
125 numVector->localize_to_one( vector1, 0 );
126 if( _ml_sol != NULL && _moving_mesh && dim > i ) {
127 unsigned indDXDYDZ = _ml_sol->GetIndex( _moving_vars[i].c_str() );
128 numVector->matrix_mult( *solution->_Sol[indDXDYDZ],
129 *mesh->GetQitoQjProjection( index, _ml_sol->GetSolutionType( indDXDYDZ ) ) );
130 numVector->localize_to_one( vector2, 0 );
131 if( _iproc == 0 ) {
132 for( unsigned i = 0; i < nvt; i++ )
133 vector1[i] += vector2[i];
134 }
135 }
136 fout.write( ( char* ) &vector1[0], nvt * sizeof( double ) );
137
138 }
139 //END COORDINATES
140
141 //BEGIN CONNETTIVITY
142 const int eltp[2][6] = {{8, 4, 6, 4, 3, 2}, {20, 10, 15, 8, 6, 3}};
143 sprintf( buffer, "%s", "cells" );
144 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
145
146 fout.write( ( char* ) &nel, sizeof( unsigned ) );
147
148 unsigned topology[27];
149
150 //mesh->_topology->_Sol[mesh->GetTypeIndex()]->localize_to_one( vector1, 0 );
151
152 for( unsigned isdom = 0; isdom < _nprocs; isdom++ ) {
153 mesh->el->LocalizeElementDof( isdom );
154 mesh->el->LocalizeElementQuantities(isdom);
155 if( _iproc == 0 ) {
156 for( unsigned ii = mesh->_elementOffset[isdom]; ii < mesh->_elementOffset[isdom + 1]; ii++ ) {
157 //short unsigned ielt = static_cast < short unsigned >( vector1[ii] + 0.25 );
158 short unsigned ielt = mesh->GetElementType(ii);
159 if( ielt == 0 )
160 sprintf( buffer, "phex%d", eltp[index][0] );
161 else if( ielt == 1 )
162 sprintf( buffer, "ptet%d", eltp[index][1] );
163 else if( ielt == 2 )
164 sprintf( buffer, "pprism%d", eltp[index][2] );
165 else if( ielt == 3 ) {
166 if( eltp[index][3] == 8 )
167 sprintf( buffer, "%dquad", eltp[index][3] );
168 else
169 sprintf( buffer, "quad" );
170 }
171 else if( ielt == 4 ) {
172 if( eltp[index][4] == 6 )
173 sprintf( buffer, "%dtri", eltp[index][4] );
174 else
175 sprintf( buffer, "tri" );
176 }
177 else if( ielt == 5 ) {
178 if( eltp[index][5] == 3 )
179 sprintf( buffer, "%dline", eltp[index][5] );
180 else
181 sprintf( buffer, "line" );
182 }
183 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
184 fout.write( ( char* ) &NVE[ielt][index], sizeof( unsigned ) );
185 for( unsigned j = 0; j < NVE[ielt][index]; j++ ) {
186 unsigned jnode_Metis = mesh->GetSolutionDof( j, ii, index );
187 topology[j] = jnode_Metis + 1;
188 }
189 fout.write( ( char* ) topology, sizeof( unsigned ) *NVE[ielt][index] );
190 }
191 }
192 mesh->el->FreeLocalizedElementDof();
193 mesh->el->FreeLocalizedElementQuantities();
194 }
195
196 //END CONNETTIVITY
197
198 //BEGIN VARIABLES
199 const unsigned zero = 0u;
200 const unsigned one = 1u;
201 sprintf( buffer, "%s", "variable" );
202 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
203
204 //BEGIN METIS PARTITIONING
205 strcpy( buffer, "METIS_DD" );
206 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
207 fout.write( ( char* ) &zero, sizeof( unsigned ) );
208
209 vector1.resize( nel );
210 unsigned icount = 0;
211 for( int isdom = 0; isdom < _nprocs; isdom++ ) {
212 for( unsigned ii = mesh->_elementOffset[isdom]; ii < mesh->_elementOffset[isdom + 1]; ii++ ) {
213 vector1[icount] = isdom;
214 icount++;
215 }
216 }
217 fout.write( ( char* ) &vector1[0], nel * sizeof( double ) );
218 //END METIS PARTITIONING
219
220 //BEGIN SOLUTION
221 if( _ml_sol != NULL ) {
222 bool printAll = 0;
223 for( unsigned ivar = 0; ivar < vars.size(); ivar++ ) {
224 printAll += !( vars[ivar].compare( "All" ) ) + !( vars[ivar].compare( "all" ) ) + !( vars[ivar].compare( "ALL" ) );
225 }
226 for( unsigned ivar = 0; ivar < !printAll * vars.size() + printAll * _ml_sol->GetSolutionSize(); ivar++ ) {
227 unsigned i = ( printAll == 0 ) ? _ml_sol->GetIndex( vars[ivar].c_str() ) : ivar;
228
229 for( int name = 0; name < 4; name++ ) {
230 //BEGIN LAGRANGIAN Fem SOLUTION
231 if( name == 0 ) {
232 sprintf( buffer, "%s", _ml_sol->GetSolutionName( i ) );
233 }
234 else if( name == 1 ) {
235 sprintf( buffer, "%s %s", "Bdc", _ml_sol->GetSolutionName( i ) );
236 }
237 else if( name == 2 ) {
238 sprintf( buffer, "%s %s", "Res", _ml_sol->GetSolutionName( i ) );
239 }
240 else {
241 sprintf( buffer, "%s %s", "Eps", _ml_sol->GetSolutionName( i ) );
242 }
243 if( name == 0 || ( _debugOutput && solution->_ResEpsBdcFlag[i] ) ) {
244 if( _ml_sol->GetSolutionType( i ) < 3 ) { // ********** on the nodes **********
245 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
246 fout.write( ( char* ) &one, sizeof( unsigned ) );
247 if( name == 0 ) {
248 numVector->matrix_mult( *solution->_Sol[i], *mesh->GetQitoQjProjection( index, _ml_sol->GetSolutionType( i ) ) );
249 }
250 else if( name == 1 ) {
251 numVector->matrix_mult( *solution->_Bdc[i], *mesh->GetQitoQjProjection( index, _ml_sol->GetSolutionType( i ) ) );
252 }
253 else if( name == 2 ) {
254 numVector->matrix_mult( *solution->_Res[i], *mesh->GetQitoQjProjection( index, _ml_sol->GetSolutionType( i ) ) );
255 }
256 else {
257 numVector->matrix_mult( *solution->_Eps[i], *mesh->GetQitoQjProjection( index, _ml_sol->GetSolutionType( i ) ) );
258 }
259 numVector->localize_to_one( vector1, 0 );
260 fout.write( ( char* ) &vector1[0], nvt * sizeof( double ) );
261 //END LAGRANGIAN Fem SOLUTION
262 }
263 else {
264 //BEGIN DISCONTINUOUS Fem SOLUTION
265 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
266 fout.write( ( char* ) &zero, sizeof( unsigned ) );
267
268 if( name == 0 ) {
269 solution->_Sol[i]->localize_to_one( vector2, 0 );
270 }
271 else if( name == 1 ) {
272 solution->_Bdc[i]->localize_to_one( vector2, 0 );
273 }
274 else if( name == 2 ) {
275 solution->_Res[i]->localize_to_one( vector2, 0 );
276 }
277 else {
278 solution->_Eps[i]->localize_to_one( vector2, 0 );
279 }
280 vector1.resize( nel );
281 for( unsigned ii = 0; ii < nel; ii++ ) {
282 vector1[ii] = vector2[ mesh->GetSolutionDof( 0, ii, _ml_sol->GetSolutionType( i ) ) ];
283 }
284
285 fout.write( ( char* ) &vector1[0], nel * sizeof( double ) );
286 //END DISCONTINUOUS Fem SOLUTION
287 }
288 }
289 }
290 }
291 }
292 //END SOLUTION
293
294 sprintf( buffer, "%s", "endvars" );
295 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
296 //END VARIABLES
297
298 sprintf( buffer, "%s", "endgmv" );
299 fout.write( ( char* ) buffer, sizeof( char ) * 8 );
300 fout.close();
301 //END GMV FILE PRINT
302
303 delete numVector;
304 delete [] buffer;
305
306 return;
307 }

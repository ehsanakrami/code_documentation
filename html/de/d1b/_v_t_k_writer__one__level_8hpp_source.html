245 template < class ARRAY_TYPE >
246 void VTKWriter::print_Mesh_related_Element_based_fields(const std::string field_string, const std::string field_datatype,
247 std::ofstream & fout, std::ofstream & Pfout,
248 void* buffer_void,
249 const unsigned elemetOffset, const unsigned elemetOffsetp1,
250 const unsigned * dim_array_elvar,
251 const Mesh * mesh,
252 const unsigned fe_index,
253 std::vector <char> & enc) const {
254
255
256 // point pointer to common memory area buffer of void type;
257 ARRAY_TYPE * var_el = static_cast< ARRAY_TYPE * >( buffer_void );
258
259 int icount = 0;
260
261 if (field_string == "Material") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetElementMaterial(iel); icount++; } }
262 else if (field_string == "Group") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetElementGroup(iel); icount++; } }
263 else if (field_string == "TYPE") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetElementType(iel); icount++; } }
264 else if (field_string == "Level") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetMeshElements()->GetElementLevel(iel); icount++; } }
265 else if (field_string == "Metis partition") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = _iproc; icount++; } }
266
267 else if (field_string == "types") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { short unsigned ielt = mesh->GetElementType( iel );
268 var_el[icount] = femusToVtkCellType[fe_index][ielt];
269 icount++;
270 }
271 }
272 else if (field_string == "offsets") { int offset_el = 0;
273 for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { offset_el += mesh->GetElementDofNumber( iel, fe_index ); var_el[icount] = offset_el; icount++; } }
274 else { abort(); }
275
276
277 print_data_array< ARRAY_TYPE >(field_string, field_datatype, fout, Pfout, dim_array_elvar, var_el, enc);
278
279 }

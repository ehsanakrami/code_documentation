576 for (unsigned i = 0; i < dim; i++) {
577 for (unsigned j = 0; j < dim; j++) {

590 - solx_uv[2][0] * solx_uv[1][1]) / sqrt (detg);
591 normal[1] = (solx_uv[2][0] * solx_uv[0][1]
592 - solx_uv[0][0] * solx_uv[2][1]) / sqrt (detg);
593 normal[2] = (solx_uv[0][0] * solx_uv[1][1]
594 - solx_uv[1][0] * solx_uv[0][1]) / sqrt (detg);

610 V[1] = X_uv[1][1] - normal[2] * X_uv[0][0] + normal[0] * X_uv[2][0];

624 M[1][1] = V[1] + normal[0] * W[2] - normal[2] * W[0];
625 M[2][1] = V[2] + normal[1] * W[0] - normal[0] * W[1];

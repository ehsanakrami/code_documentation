92void NewmarkTransientSystem<Base>::UpdateAcceleration(const std::vector<std::string>& vel_vars, const std::vector<std::string>& acc_vars)
93{
94// const double gamma = 0.5;
95// const double a5 = -1.*(1. - gamma)/gamma;
96// const double a1 = 1./(gamma*this->_dt);
97// const double a2 = -1./(gamma*this->_dt);
98
99// unsigned dim = this->_msh[0]->GetDimension();
100 // vel and acc must have the same number of variables
101 assert(vel_vars.size() == acc_vars.size());
102
103 const unsigned dim = vel_vars.size();
104
105 unsigned axyz[3];
106 unsigned vxyz[3];
107// const char accname[3][3] = {"AX","AY","AZ"};
108// const char velname[3][2] = {"U","V","W"};
109
110 for(unsigned i=0; i<dim; i++) {
111 axyz[i] = this->_equation_systems.GetIndex(acc_vars.at(i));
112 vxyz[i] = this->_equation_systems.GetIndex(vel_vars.at(i));
113 }
114
115 for (int ig=0;ig< this->_gridn;ig++) {
116 for(unsigned i=0; i<dim; i++) {
117 this->_solution[ig]->_Sol[axyz[i]]->scale(_a5);
118 this->_solution[ig]->_Sol[axyz[i]]->add(_a1,*(this->_solution[ig]->_Sol[vxyz[i]]));
119 this->_solution[ig]->_Sol[axyz[i]]->add(_a2,*(this->_solution[ig]->_SolOld[vxyz[i]]));
120 }
121 }
122}

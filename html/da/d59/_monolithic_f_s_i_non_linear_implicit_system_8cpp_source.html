48 void MonolithicFSINonLinearImplicitSystem::Build_RestrictionTranspose_OneElement_OneFEFamily_With_Pair_In_System(const LinearEquation& lspdef,
49 const LinearEquation& lspdec,
50 const int& ielc,
51 SparseMatrix* Projmat,
52 const unsigned& index_sol,
53 const unsigned& kkindex_sol,
54 const unsigned& index_pair_sol,
55 const unsigned& kkindex_pair_sol,
56 const elem_type * elem_type_in) const
57 {
58
59 const unsigned soltype_in = elem_type_in->GetSolType();
60 const unsigned ndofs = elem_type_in->GetNDofs();
61 const unsigned ndofs_fine = elem_type_in->GetNDofsFine();
62
63
64 if(lspdec.GetMeshFromLinEq()->GetRefinedElementIndex(ielc)) { // coarse2fine prolongation
65
66 //BEGIN project nodeSolidMark
67 std::vector < double > fineNodeSolidMark(ndofs_fine, 0);
68 std::vector < bool > coarseNodeSolidMark(ndofs, 0);
69
70 if( soltype_in == 2 ) {
71 for(unsigned j = 0; j < ndofs; j++) {
72 int jadd = lspdec.GetMeshFromLinEq()->GetSolutionDof(j, ielc, soltype_in);
73 coarseNodeSolidMark[j] = lspdec.GetMeshFromLinEq()->GetSolidMark(jadd);
74 }
75
76 for(unsigned i = 0; i < ndofs_fine; i++) {
77 int ncols = elem_type_in->Get_Prolongator_Num_Columns(i);
78
79 for(int k = 0; k < ncols; k++) {
80 int j = elem_type_in->Get_Prolongator_Index(i, k);
81 fineNodeSolidMark[i] += elem_type_in->Get_Prolongator_Value(i, k) * coarseNodeSolidMark[j];
82 }
83 }
84 }
85
86 //END project nodeSolidMark
87
88 std::vector <int> cols( ndofs );
89 std::vector <double> copy_prol_val;
90 copy_prol_val.reserve( ndofs );
91
92 for(int i = 0; i < ndofs_fine; i++) {
93
94 const std::pair<int, int> id_0_1 = elem_type_in->GetKVERT_IND(i);
95
96 int irow = lspdef.GetSystemDof(index_sol, kkindex_sol, ielc, id_0_1.first, id_0_1.second, lspdec.GetMeshFromLinEq());
97
98 const bool isolidmark = (fineNodeSolidMark[i] > 0.99 && fineNodeSolidMark[i] < 1.01) ? true : false;
99
100 int ncols = elem_type_in->Get_Prolongator_Num_Columns(i);
101 cols.assign(ncols, 0);
102 copy_prol_val.assign(ncols, 0);
103
104 for(int k = 0; k < ncols; k++) {
105 int j = elem_type_in->Get_Prolongator_Index(i, k);
106 bool jsolidmark = coarseNodeSolidMark[j];
107
108 if(isolidmark == jsolidmark) {
109 int jcolumn = lspdec.GetSystemDof(index_sol, kkindex_sol, j, ielc);
110 cols[k] = jcolumn;
111 copy_prol_val[k] = elem_type_in->Get_Prolongator_Value(i, k);
112 }
113 else {
114 int jcolumn = lspdec.GetSystemDof(index_pair_sol, kkindex_pair_sol, j, ielc);
115 cols[k] = jcolumn;
116 copy_prol_val[k] = (index_sol != index_pair_sol) ? elem_type_in->Get_Prolongator_Value(i, k) : 0.;
117 }
118 }
119
120 Projmat->insert_row(irow, ncols, cols, &copy_prol_val[0]);
121 }
122
123 }
124 else {
125
126 std::vector <int> jcol(1);
127 double one = 1.;
128
129 for(int i = 0; i < ndofs; i++) {
130
131 int irow = lspdef.GetSystemDof(index_sol, kkindex_sol, ielc, 0, i, lspdec.GetMeshFromLinEq());
132 jcol[0] = lspdec.GetSystemDof(index_sol, kkindex_sol, i, ielc);
133 Projmat->insert_row(irow, 1, jcol, &one);
134
135 }
136 }
137

139 }

147 void MonolithicFSINonLinearImplicitSystem::BuildProlongatorMatrix(unsigned gridf) {
148
149 // ------------------- Sparsity pattern size - BEGIN
150
151 if(gridf < 1) {
152 std::cout << "Error! In function \"BuildProlongatorMatrix\" argument less then 1" << std::endl;
153 exit(0);
154 }
155
156 int iproc;
157 MPI_Comm_rank(MPI_COMM_WORLD, &iproc);
158
159 LinearEquationSolver* LinSolf = _LinSolver[gridf];
160 LinearEquationSolver* LinSolc = _LinSolver[gridf - 1];
161 Mesh* mshc = _msh[gridf - 1];
162 int nf = LinSolf->KKIndex[LinSolf->KKIndex.size() - 1u];
163 int nc = LinSolc->KKIndex[LinSolc->KKIndex.size() - 1u];
164 int nf_loc = LinSolf->KKoffset[LinSolf->KKIndex.size() - 1][iproc] - LinSolf->KKoffset[0][iproc];
165 int nc_loc = LinSolc->KKoffset[LinSolc->KKIndex.size() - 1][iproc] - LinSolc->KKoffset[0][iproc];
166
167 NumericVector *NNZ_d = NumericVector::build().release();
168 NNZ_d->init(*LinSolf->_EPS);
169 NNZ_d->zero();
170
171 NumericVector *NNZ_o = NumericVector::build().release();
172 NNZ_o->init(*LinSolf->_EPS);
173 NNZ_o->zero();
174
175 for(unsigned k = 0; k < _SolSystemPdeIndex.size(); k++) {
176 unsigned SolIndex = _SolSystemPdeIndex[k];
177 unsigned SolType = _ml_sol->GetSolutionType(SolIndex);
178 // loop on the coarse grid
179 for(int isdom = iproc; isdom < iproc + 1; isdom++) {
180 for(int iel = mshc->GetElementOffset(isdom); iel < mshc->GetElementOffset(isdom + 1); iel++) {
181 short unsigned ielt = mshc->GetElementType(iel);
182 Get_Prolongation_SparsityPatternSize_OneElement_OneFEFamily_In_System(*LinSolf, *LinSolc, iel, NNZ_d, NNZ_o, SolIndex, k,
183 mshc->GetFiniteElement(ielt, SolType) );
184 }
185 }
186 }
187
188 NNZ_d->close();
189 NNZ_o->close();
190
191 unsigned offset = LinSolf->KKoffset[0][iproc];
192 std::vector <int> nnz_d(nf_loc);
193 std::vector <int> nnz_o(nf_loc);
194 for(int i = 0; i < nf_loc; i++) {
195 nnz_d[i] = static_cast <int>((*NNZ_d)(offset + i));
196 nnz_o[i] = static_cast <int>((*NNZ_o)(offset + i));
197 }
198
199 delete NNZ_d;
200 delete NNZ_o;
201 // ------------------- Sparsity pattern size - END
202
203 // ------------------- Prolongator - BEGIN
204
205
206 _PP[gridf] = SparseMatrix::build().release();
207 _PP[gridf]->init(nf, nc, nf_loc, nc_loc, nnz_d, nnz_o);
208
209 SparseMatrix *RRt;
210 RRt = SparseMatrix::build().release();
211 RRt->init(nf, nc, nf_loc, nc_loc, nnz_d, nnz_o);
212
213 for(unsigned k = 0; k < _SolSystemPdeIndex.size(); k++) {
214 unsigned SolIndex = _SolSystemPdeIndex[k];
215 unsigned solPairIndex = _ml_sol->GetSolutionPairIndex(k);
216 unsigned SolType = _ml_sol->GetSolutionType(SolIndex);
217 unsigned solPairPdeIndex = GetSolPdeIndex(_ml_sol->GetSolName_from_index(solPairIndex));
218
219 bool testIfPressure = 0;
220 if(_ml_sol->TestIfSolutionIsPressure(SolIndex)) testIfPressure = 1;
221
222 // loop on the coarse grid
223 for(int isdom = iproc; isdom < iproc + 1; isdom++) {
224 for(int iel = mshc->GetElementOffset(isdom); iel < mshc->GetElementOffset(isdom + 1); iel++) {
225 short unsigned ielt = mshc->GetElementType(iel);
226
227 if(!testIfPressure) {
228 Build_RestrictionTranspose_OneElement_OneFEFamily_With_Pair_In_System(*LinSolf, *LinSolc, iel, RRt, SolIndex, k, solPairIndex, solPairPdeIndex, mshc->GetFiniteElement(ielt, SolType));
229 }
230 else {
231 Build_Prolongation_OneElement_OneFEFamily_In_System(*LinSolf, *LinSolc, iel, RRt, SolIndex, k,
232 mshc->GetFiniteElement(ielt, SolType) );
233 }
234
235 Build_Prolongation_OneElement_OneFEFamily_In_System(*LinSolf, *LinSolc, iel, _PP[gridf], SolIndex, k,
236 mshc->GetFiniteElement(ielt, SolType) );
237
238 }
239 }
240 }
241
242 _PP[gridf]->close();
243 RRt->close();
244
245 _RR[gridf] = SparseMatrix::build().release();
246 RRt->get_transpose(*_RR[gridf]);
247 delete RRt;
248
249 // ------------------- Prolongator - END
250

254 void MonolithicFSINonLinearImplicitSystem::BuildAmrProlongatorMatrix(unsigned level) {
255
256 // ------------------- Sparsity pattern size - BEGIN
257
258 int iproc;
259 MPI_Comm_rank(MPI_COMM_WORLD, &iproc);
260
261 LinearEquationSolver* LinSol = _LinSolver[level];
262
263 const Mesh* mesh = _msh[level];
264 const std::vector < std::map < unsigned, std::map < unsigned, double > > > & amrRestriction = mesh->GetAmrRestrictionMap();
265
266 const std::vector < std::map < unsigned, bool > > & amrSolidMark = mesh->GetAmrSolidMark();
267
268 int n = LinSol->KKIndex[LinSol->KKIndex.size() - 1u];
269 int n_loc = LinSol->KKoffset[LinSol->KKIndex.size() - 1][iproc] - LinSol->KKoffset[0][iproc];
270
271 NumericVector* NNZ_d = NumericVector::build().release();
272 NNZ_d->init(*LinSol->_EPS);
273 NNZ_d->zero();
274
275 NumericVector* NNZ_o = NumericVector::build().release();
276 NNZ_o->init(*LinSol->_EPS);
277 NNZ_o->zero();
278
279 for(unsigned k = 0; k < _SolSystemPdeIndex.size(); k++) {
280
281 unsigned solIndex = _SolSystemPdeIndex[k];
282 unsigned solType = _ml_sol->GetSolutionType(solIndex);
283 unsigned kOffset = LinSol->KKoffset[k][iproc];
284
285 unsigned solOffset = mesh->_dofOffset[solType][iproc];
286 unsigned solOffsetp1 = mesh->_dofOffset[solType][iproc + 1];
287 for(unsigned i = solOffset; i < solOffsetp1; i++) {
288 if(solType > 2 || amrRestriction[solType].find(i) == amrRestriction[solType].end()) {
289 NNZ_d->set(kOffset + (i - solOffset), 1);
290 }
291 else {
292 double cnt_d = 0;
293 double cnt_o = 0;
294 for(std::map<unsigned, double> ::const_iterator it = amrRestriction[solType].at(i).begin(); it != amrRestriction[solType].at(i).end(); it++) {
295 if(it->first >= solOffset && it->first < solOffsetp1) {
296 cnt_d++;
297 }
298 else {
299 cnt_o++;
300 }
301 }
302 NNZ_d->set(kOffset + (i - solOffset), cnt_d);
303 NNZ_o->set(kOffset + (i - solOffset), cnt_o);
304 }
305 }
306 }
307
308 NNZ_d->close();
309 NNZ_o->close();
310
311 unsigned offset = LinSol->KKoffset[0][iproc];
312 std::vector <int> nnz_d(n_loc);
313 std::vector <int> nnz_o(n_loc);
314
315 for(int i = 0; i < n_loc; i++) {
316 nnz_d[i] = static_cast <int>((*NNZ_d)(offset + i));
317 nnz_o[i] = static_cast <int>((*NNZ_o)(offset + i));
318 }
319
320 delete NNZ_d;
321 delete NNZ_o;
322 // ------------------- Sparsity pattern size - END
323
324
325 // ------------------- Prolongator - BEGIN
326
327 _PPamr[level] = SparseMatrix::build().release();
328 _PPamr[level]->init(n, n, n_loc, n_loc, nnz_d, nnz_o);
329
330 _RRamr[level] = SparseMatrix::build().release();
331 _RRamr[level]->init(n, n, n_loc, n_loc, nnz_d, nnz_o);
332
333 for(unsigned k = 0; k < _SolSystemPdeIndex.size(); k++) {
334
335 unsigned solIndex = _SolSystemPdeIndex[k];
336 unsigned solPairIndex = _ml_sol->GetSolutionPairInverseIndex(k);
337
338 unsigned kPair = GetSolPdeIndex(_ml_sol->GetSolName_from_index(solPairIndex));
339
340 unsigned kOffset = LinSol->KKoffset[k][iproc];
341 unsigned kPairOffset = LinSol->KKoffset[kPair][iproc];
342
343 unsigned solType = _ml_sol->GetSolutionType(solIndex);
344
345 unsigned solOffset = mesh->_dofOffset[solType][iproc];
346 unsigned solOffsetp1 = mesh->_dofOffset[solType][iproc + 1];
347
348 for(unsigned i = solOffset; i < solOffsetp1; i++) {
349 unsigned irow = kOffset + (i - solOffset);
350 if(solType > 2 || amrRestriction[solType].find(i) == amrRestriction[solType].end()) {
351 std::vector <int> col(1, irow);
352 double value = 1.;
353 _PPamr[level]->insert_row(irow, 1, col, &value);
354 _RRamr[level]->insert_row(irow, 1, col, &value);
355 }
356 else {
357 bool solidMarki = amrSolidMark[solType].at(i);
358 const int ncols = amrRestriction[solType].at(i).size();
359 std::vector <int> colPP(ncols);
360 std::vector <int> colRR(ncols);
361 std::vector <double> valuePP(ncols);
362 std::vector <double> valueRR(ncols);
363 unsigned j = 0;
364 for(std::map<unsigned, double> ::const_iterator it = amrRestriction[solType].at(i).begin(); it != amrRestriction[solType].at(i).end(); it++) {
365 bool solidMarkj = amrSolidMark[solType].at(it->first);
366 if(it->first >= solOffset && it->first < solOffsetp1) {
367 colPP[j] = kOffset + (it->first - solOffset);
368 if(solidMarki == true && solidMarkj == false && k != kPair){
369 colRR[j] = kPairOffset + (it->first - solOffset);
370 }
371 else{
372 colRR[j] = kOffset + (it->first - solOffset);
373 }
374 }
375 else {
376 unsigned jproc = _msh[level]->BisectionSearch_find_processor_of_dof(it->first, solType);
377 colPP[j] = LinSol->KKoffset[k][jproc] + (it->first - mesh->_dofOffset[solType][jproc]);
378 if(solidMarki == true && solidMarkj == false && k != kPair){
379 colRR[j] = LinSol->KKoffset[kPair][jproc] + (it->first - mesh->_dofOffset[solType][jproc]);
380 }
381 else{
382 colRR[j] = LinSol->KKoffset[k][jproc] + (it->first - mesh->_dofOffset[solType][jproc]);
383 }
384 }
385 valuePP[j] = it->second;
386 if( solidMarki == true && solidMarkj == false && k == kPair){
387 valueRR[j] = 0.;
388 }
389 else{
390 valueRR[j] = it->second;
391 }
392 j++;
393 }
394 _PPamr[level]->insert_row(irow, ncols, colPP, &valuePP[0]);
395 _RRamr[level]->insert_row(irow, ncols, colRR, &valueRR[0]);
396 }
397 }
398 }
399 _PPamr[level]->close();
400 _RRamr[level]->close();
401
402 _PPamr[level]->get_transpose(*_PPamr[level]);
403
404 // ------------------- Prolongator - END
405
406
407 }

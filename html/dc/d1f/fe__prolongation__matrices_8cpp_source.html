163 void FE_Prolongation_Matrices::Get_Prolongation_SparsityPatternSize_OneElement_OneFEFamily(const Mesh& meshf,
164 const Mesh& meshc,
165 const int& ielc,
166 NumericVector* NNZ_d,
167 NumericVector* NNZ_o,
168 const char is_fine_or_coarse[],
169 const elem_type * elem_type_in) const
170 {
171
172 const unsigned soltype_in = elem_type_in->GetSolType();
173 const unsigned ndofs = elem_type_in->GetNDofs();
174 const unsigned ndofs_fine = elem_type_in->GetNDofsFine();
175
176 unsigned n_elemdofs = 0;
177 if ( !strcmp(is_fine_or_coarse, "fine") ) n_elemdofs = ndofs_fine;
178 else if ( !strcmp(is_fine_or_coarse, "coarse") ) n_elemdofs = ndofs;
179
180 if(meshc.GetRefinedElementIndex(ielc)) { // coarse2fine prolongation
181
182 for(int i = 0; i < n_elemdofs ; i++) {
183
184 const std::pair<int, int> id_0_1 = elem_type_in->GetKVERT_IND(i);
185
186 const int irow = meshf.GetSolutionDof(ielc, id_0_1.first, id_0_1.second, soltype_in, &meshc);
187
188 const int iproc = meshf.BisectionSearch_find_processor_of_dof(irow, soltype_in);
189
190 const int ncols = elem_type_in->Get_Prolongator_Num_Columns(i);
191
192 unsigned counter_off_diag = 0;
193
194 for(int k = 0; k < ncols; k++) {
195 int j = elem_type_in->Get_Prolongator_Index(i, k);
196 int jcolumn = meshc.GetSolutionDof(j, ielc, soltype_in);
197
198 if(jcolumn < meshc.dofmap_get_dof_offset(soltype_in, iproc) ||
199 jcolumn >= meshc.dofmap_get_dof_offset(soltype_in, iproc + 1) ) counter_off_diag++;
200 }
201
202 NNZ_d->set(irow, ncols - counter_off_diag);
203 NNZ_o->set(irow, counter_off_diag);
204 }
205
206 }
207 else { // coarse2coarse prolongation
208
209 for(int i = 0; i < ndofs; i++) {
210
211 int irow = meshf.GetSolutionDof(ielc, 0, i , soltype_in, &meshc);
212
213 int iproc = meshf.BisectionSearch_find_processor_of_dof(irow, soltype_in);
214 int jcolumn = meshc.GetSolutionDof(i, ielc, soltype_in);
215
216 if( jcolumn < meshc.dofmap_get_dof_offset(soltype_in, iproc) ||
217 jcolumn >= meshc.dofmap_get_dof_offset(soltype_in, iproc + 1) ) {
218 NNZ_o->set(irow, 1);
219 }
220 else {
221 NNZ_d->set(irow, 1);
222 }
223 }
224
225 }

232 void FE_Prolongation_Matrices::Build_Prolongation_OneElement_OneFEFamily(const Mesh& meshf,
233 const Mesh& meshc,
234 const int& ielc,
235 SparseMatrix* Projmat,

247
248 if(meshc.GetRefinedElementIndex(ielc)) { // coarse2fine prolongation
249
250 std::vector<int> jcols( ndofs );

252 for(int i = 0; i < n_elemdofs /*_nf*/; i++) {
253
254 const std::pair<int, int> id_0_1 = elem_type_in->GetKVERT_IND(i);
255
256 const int irow = meshf.GetSolutionDof(ielc, id_0_1.first, id_0_1.second, soltype_in, &meshc);
257 const int ncols = elem_type_in->Get_Prolongator_Num_Columns(i);
258
259 jcols.assign(ncols, 0);
260
261 for(int k = 0; k < ncols; k++) {
262 int j = elem_type_in->Get_Prolongator_Index(i, k);
263 int jcolumn = meshc.GetSolutionDof(j, ielc, soltype_in);
264 jcols[k] = jcolumn;
265 }
266
267 Projmat->insert_row(irow, ncols, jcols, elem_type_in->Get_Prolongator_Values_Row(i) );
268 }
269
270 }
271 else { // coarse2coarse prolongation
272
273 std::vector <int> jcol(1);
274 double one = 1.;
275
276 for(int i = 0; i < ndofs; i++) {
277
278 const int irow = meshf.GetSolutionDof(ielc, 0, i , soltype_in, &meshc);
279 jcol[0] = meshc.GetSolutionDof(i, ielc, soltype_in);
280 Projmat->insert_row(irow, 1, jcol, &one);
281
282 }
283
284 }

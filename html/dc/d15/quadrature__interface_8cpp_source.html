36 Gauss::Gauss(const char *geom_elem, const char *order_gauss) : _order(order_gauss)
37 {
38 if (!strcmp(order_gauss, "zero") || !strcmp(order_gauss, "first")) {
39 gauss_order = 0;
40 }
41 else if (!strcmp(order_gauss, "second") || !strcmp(order_gauss, "third") ) {
42 gauss_order = 1;
43 }
44 else if (!strcmp(order_gauss, "fourth") || !strcmp(order_gauss, "fifth") ) {
45 gauss_order = 2;
46 }
47 else if (!strcmp(order_gauss, "sixth") || !strcmp(order_gauss, "seventh") ) {
48 gauss_order = 3;
49 }
50 else if (!strcmp(order_gauss, "eighth") || !strcmp(order_gauss, "ninth") ) {
51 gauss_order = 4;
52 }
53 else if (!strcmp(order_gauss, "tenth") || !strcmp(order_gauss, "eleventh") ) {
54 gauss_order = 5;
55 }
56 else {
57 std::cout << order_gauss << "is not a valid option for the Gauss points of" << geom_elem << std::endl;
58 abort();
59 }

61 if (!strcmp(geom_elem, "hex")) {
62 GaussWeight = hex_gauss::Gauss[gauss_order];

64 }

72 }
73 else if (!strcmp(geom_elem, "quad")) {

75 GaussPoints = quad_gauss::GaussPoints[gauss_order];
76 }
77 else if (!strcmp(geom_elem, "tri")) {
78 GaussWeight = tri_gauss::Gauss[gauss_order];
79 GaussPoints = tri_gauss::GaussPoints[gauss_order];
80 }
81 else if (!strcmp(geom_elem, "line")) {
82 GaussWeight = line_gauss::Gauss[gauss_order];
83 GaussPoints = line_gauss::GaussPoints[gauss_order];
84 }

86 GaussWeight = point_gauss::Gauss[gauss_order];

88 }

695// Internal helper function to create std::vector <something_useable> from
696// std::vector <bool> for compatibility with MPI bitwise operations
697template <typename T>
698inline void pack_vector_bool(const std::vector<bool> &in,
699 std::vector<T> &out) {
700 unsigned int data_bits = 8*sizeof(T);
701 unsigned int in_size = in.size();
702 unsigned int out_size = in_size/data_bits + (in_size%data_bits?1:0);
703 out.clear();
704 out.resize(out_size);
705 for (unsigned int i=0; i != in_size; ++i) {
706 unsigned int index = i/data_bits;
707 unsigned int offset = i%data_bits;
708 out[index] += (in[i]?1:0) << offset;
709 }
710}
711
712// Internal helper function to create std::vector <something_useable> from
713// std::vector <bool> for compatibility with MPI byte operations
714template <typename T>
715inline void unpack_vector_bool(const std::vector<T> &in,
716 std::vector<bool> &out) {
717 unsigned int data_bits = 8*sizeof(T);
718 // We need the output vector to already be properly sized
719 unsigned int out_size = out.size();
720 assert(out_size/data_bits + (out_size%data_bits?1:0) == in.size());
721
722 for (unsigned int i=0; i != out_size; ++i) {
723 unsigned int index = i/data_bits;
724 unsigned int offset = i%data_bits;
725 out[i] = in[index] << (data_bits-1-offset) >> (data_bits-1);
726 }
727}

785template <typename T> inline bool verify(const T &r) {
786 int n_proc; /*int iproc;*/
787 MPI_Comm_size(MPI_COMM_WORLD,&n_proc);
788// MPI_Comm_rank (MPI_COMM_WORLD,&iproc);
789 if (n_proc > 1) {
790 T tempmin = r, tempmax = r;
791 Parallel::min(tempmin);
792 Parallel::max(tempmax);
793 bool verified = (r == tempmin) &&
794 (r == tempmax);
795 Parallel::min(verified);
796 return verified;
797 }
798 return true;
799}

1241template <typename T>
1242inline void send_receive(const unsigned int dest_processor_id,
1243 std::vector<std::vector<T> > &send,
1244 const unsigned int source_processor_id,
1245 std::vector<std::vector<T> > &recv) {
1246 int n_proc;
1247 int iproc;
1248 MPI_Comm_size(MPI_COMM_WORLD,&n_proc);
1249 MPI_Comm_rank (MPI_COMM_WORLD,&iproc);
1250
1251 // START_LOG("send_receive()", "Parallel");
1252
1253 if (dest_processor_id == iproc &&
1254 source_processor_id == iproc) {
1255 recv = send;
1256 // STOP_LOG("send_receive()", "Parallel");
1257 return;
1258 }
1259
1260 // temporary buffers - these will be sized in bytes
1261 // and manipulated with MPI_Pack and friends
1262 std::vector<char> sendbuf, recvbuf;
1263
1264 // figure out how many bytes we need to pack all the data
1265 int packedsize=0, sendsize=0;
1266
1267 // The outer buffer size
1268 MPI_Pack_size(1,
1269 datatype<unsigned int>(),
1270 MPI_COMM_WORLD,
1271 &packedsize);
1272 sendsize += packedsize;
1273
1274 for (unsigned int i=0; i<send.size(); i++) {
1275 // The size of the ith inner buffer
1276 MPI_Pack_size(1,
1277 datatype<unsigned int>(),
1278 MPI_COMM_WORLD,
1279 &packedsize);
1280 sendsize += packedsize;
1281
1282 // The data for each inner buffer
1283 MPI_Pack_size(send[i].size(),
1284 datatype<T>(),
1285 MPI_COMM_WORLD,
1286 &packedsize);
1287 sendsize += packedsize;
1288 }
1289
1290 assert(sendsize /* should at least be 1! */);
1291 sendbuf.resize(sendsize);
1292
1293 // Pack the send buffer
1294 int pos=0;
1295
1296 // ... the size of the outer buffer
1297 sendsize = send.size();
1298 MPI_Pack(&sendsize, 1, datatype<unsigned int>(),
1299 &sendbuf[0], sendbuf.size(), &pos,
1300 MPI_COMM_WORLD);
1301
1302 for (unsigned int i=0; i<send.size(); i++) {
1303 // ... the size of the ith inner buffer
1304 sendsize = send[i].size();
1305 MPI_Pack(&sendsize, 1, datatype<unsigned int>(),
1306 &sendbuf[0], sendbuf.size(), &pos,
1307 MPI_COMM_WORLD);
1308
1309 // ... the contents of the ith inner buffer
1310 if (!send[i].empty())
1311 MPI_Pack(&send[i][0], send[i].size(), datatype<T>(),
1312 &sendbuf[0], sendbuf.size(), &pos,
1313 MPI_COMM_WORLD);
1314 }
1315
1316 assert(static_cast<unsigned int>(pos) == sendbuf.size());
1317
1318 Parallel::request request;
1319
1320 Parallel::nonblocking_send(dest_processor_id,
1321 sendbuf,
1322 MPI_PACKED,
1323 request,
1324 /* tag = */ 123);
1325
1326 Parallel::receive(source_processor_id,
1327 recvbuf,
1328 MPI_PACKED,
1329 /* tag = */ 123);
1330
1331 // Unpack the received buffer
1332 assert(!recvbuf.empty());
1333 pos=0;
1334 MPI_Unpack(&recvbuf[0], recvbuf.size(), &pos,
1335 &sendsize, 1, datatype<unsigned int>(),
1336 MPI_COMM_WORLD);
1337
1338 // ... size the outer buffer
1339 recv.resize(sendsize);
1340
1341 for (unsigned int i=0; i<recv.size(); i++) {
1342 MPI_Unpack(&recvbuf[0], recvbuf.size(), &pos,
1343 &sendsize, 1, datatype<unsigned int>(),
1344 MPI_COMM_WORLD);
1345
1346 // ... size the inner buffer
1347 recv[i].resize(sendsize);
1348
1349 // ... unpack the inner buffer if it is not empty
1350 if (!recv[i].empty())
1351 MPI_Unpack(&recvbuf[0], recvbuf.size(), &pos,
1352 &recv[i][0], recv[i].size(), datatype<T>(),
1353 MPI_COMM_WORLD);
1354 }
1355
1356 Parallel::wait(request);
1357
1358 // STOP_LOG("send_receive()", "Parallel");
1359}

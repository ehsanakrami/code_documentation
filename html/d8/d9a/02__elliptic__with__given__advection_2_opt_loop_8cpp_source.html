81double ComputeIntegral (const uint Level, const MultiLevelMeshTwo* mesh, const SystemTwo* eqn, const std::string output_time) {
82
83 Mesh *mymsh = eqn->GetMLProb()._ml_msh->GetLevel(Level);
84 //====== processor index
85 const uint myproc = mesh->_iproc;
86 const uint space_dim = mesh->get_dim();
87
88 const uint mesh_vb = VV;

91 double integral = 0.;

94//parallel sum

96 const uint nel_b = mesh->_off_el[mesh_vb][mesh->_NoLevels*myproc+Level];
97
98 for (uint iel=0; iel < (nel_e - nel_b); iel++) {
99
100 CurrentElem<double> currelem(iel,myproc,Level,VV,eqn,*mesh, NULL, mymsh);
101 // CurrentGaussPointBase & currgp = // CurrentGaussPointBase::build(currelem,eqn->GetMLProb().GetQuadratureRule(currelem.GetDim()));

103 //==========
104 CurrentQuantity Tempold(currelem);
105 Tempold._SolName = "Qty_Temperature";
106 Tempold._qtyptr = eqn->GetMLProb().GetQtyMap().GetQuantity("Qty_Temperature");
107 Tempold.VectWithQtyFillBasic();
108 Tempold.Allocate();
109
110 //==========
111 CurrentQuantity Tlift(currelem);
112 Tlift._SolName = "Qty_TempLift";
113 Tlift._qtyptr = eqn->GetMLProb().GetQtyMap().GetQuantity("Qty_TempLift");
114 Tlift.VectWithQtyFillBasic();
115 Tlift.Allocate();

117 //===========
118 CurrentQuantity Tdes(currelem);
119 Tdes._SolName = "Qty_TempDes";
120 Tdes._dim = Tempold._dim;
121 Tdes._FEord = Tempold._FEord;
122 Tdes._ndof = Tempold._ndof;
123 Tdes.Allocate();
124
125//========= DOMAIN MAPPING
126 CurrentQuantity xyz(currelem);
127 xyz._dim = space_dim;
128 xyz._FEord = MESH_MAPPING_FE;
129 xyz._ndof = currelem.GetElemType(xyz._FEord)->GetNDofs();
130 xyz.Allocate();
131
132 currelem.SetDofobjConnCoords();
133
134 currelem.ConvertElemCoordsToMappingOrd(xyz);
135// ===============
136 xyz.SetElemAverage();
137 int el_flagdom = ElFlagControl(xyz._el_average,eqn->GetMLProb()._ml_msh);
138//====================
139
140 Tempold.GetElemDofs();
141 Tlift.GetElemDofs();
142
143 TempDesired(Tdes,currelem);

145 const uint el_ngauss = eqn->GetMLProb().GetQuadratureRule(currelem.GetDim()).GetGaussPointsNumber();
146
147 for (uint qp = 0; qp < el_ngauss; qp++) {
148
149 for (uint fe = 0; fe < QL; fe++) {
150// currgp.SetPhiElDofsFEVB_g (fe,qp);
151// currgp.SetDPhiDxezetaElDofsFEVB_g (fe,qp);

153
154 const double Jac_g = 1.;// currgp.JacVectVV_g(xyz);

156
157 Tempold.val_g();
158 Tlift.val_g();
159 Tdes.val_g();
160
161 double deltau_squarenorm_g = 0.;
162 deltau_squarenorm_g += (Tempold._val_g[0] + Tlift._val_g[0] - Tdes._val_g[0])*
163 (Tempold._val_g[0] + Tlift._val_g[0] - Tdes._val_g[0]);

167 }//gauss loop
168
169 }//element loop
170
172 std::cout << "integral on processor 0: " << integral << std::endl;
173
174 double J=0.;
175#ifdef HAVE_MPI
176 MPI_Allreduce( &integral, &J, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD ); //THIS IS THE RIGHT ONE!!
177#else
178 J = integral;
179#endif
180
181 std::cout << "@@@@@@@@@@@@@@@@ functional value: " << J << std::endl;
182//=====functional evaluation =======
183
184
187
188 std::string app_path = "./";
189 std::string intgr_fname = app_path + Files::_application_output_directory + "/" + "alpha";
190
191 std::ofstream intgr_fstream;
192
193 if (Parallel::get_rank() ==0 ) {
194 intgr_fstream.open(intgr_fname.c_str(),std::ios_base::app);
195 intgr_fstream << output_time << " " << eqn->GetMLProb().GetInputParser().get("alphaT") << " " << eqn->GetMLProb().GetInputParser().get("injsuc")<< " " << J << " " << std::endl ;
196 intgr_fstream.close(); //you have to close to disassociate the file from the stream
197 }

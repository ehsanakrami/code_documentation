265 template < class ARRAY_TYPE >
266 void VTKWriter::print_Mesh_related_Element_based_fields(const std::string field_string, const std::string field_datatype,
267 std::ofstream & fout, std::ofstream & Pfout,
268 void* buffer_void,
269 const unsigned elemetOffset, const unsigned elemetOffsetp1,
270 const unsigned * dim_array_elvar,
271 const Mesh * mesh,
272 const unsigned fe_index,
273 std::vector <char> & enc) const {
274
275
276 // point pointer to common memory area buffer of void type;
277 ARRAY_TYPE * var_el = static_cast< ARRAY_TYPE * >( buffer_void );
278
279 int icount = 0;
280
281 if (field_string == "Material") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetElementMaterial(iel); icount++; } }
282 else if (field_string == "Group") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetElementGroup(iel); icount++; } }
283 else if (field_string == "TYPE") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetElementType(iel); icount++; } }
284 else if (field_string == "Level") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetMeshElements()->GetElementLevel(iel); icount++; } }
285 else if (field_string == "Metis partition") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = _writer_one_level.processor_id(); icount++; } }
286
287 else if (field_string == "types") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { short unsigned ielt = mesh->GetElementType( iel );
288 var_el[icount] = femusToVtkCellType[fe_index][ielt];
289 icount++;
290 }
291 }
292 else if (field_string == "offsets") { int offset_el = 0;
293 for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { offset_el += mesh->GetElementDofNumber( iel, fe_index ); var_el[icount] = offset_el; icount++; } }
294 else { abort(); }
295
296
297 print_data_array< ARRAY_TYPE >(field_string, field_datatype, fout, Pfout, dim_array_elvar, var_el, enc);
298
299 }

128
129 // for a quarter-circle in Quadrant 1
130
131 double xx = x[0];
132 double yy = x[1];
133
134 return xx * yy * (1.0 + (xx*xx + yy*yy) ); // forced to be zero on the x and y axis, and the circle edge
135 }
136
137
138 std::vector < type > gradient(const std::vector < type >& x) const {
139
140 std::vector < type > solGrad(x.size(), 0.);
141
142 double xx = x[0];
143 double yy = x[1];
144
145 solGrad[0] = yy * (1.0 + (xx*xx + yy*yy) ) + xx * yy * (2. * xx);
146 solGrad[1] = xx * (1.0 + (xx*xx + yy*yy) ) + yy * xx * (2. * yy);
147
148 return solGrad;
149 }
150
151
152 type laplacian(const std::vector < type >& x) const {
153
154 double xx = x[0];
155 double yy = x[1];

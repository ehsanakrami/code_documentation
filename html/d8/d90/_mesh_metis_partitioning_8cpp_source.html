41 void MeshMetisPartitioning::DoPartition(std::vector <unsigned>& partition, const bool& AMR) {
42
43 int nnodes = _mesh.GetNumberOfNodes();
44 int nelem = _mesh.GetNumberOfElements();
45
46 std::vector <int> epart(nelem);
47
48 if(_nprocs == 1) {
49 //serial computation
50 for(unsigned i = 0; i < nelem; i++) {
51 epart[i] = 0;
52 }
53 }
54 else if(_nprocs > nelem) {
55 std::cout << "Error In MeshMetis::DoPartition, the number of processes " << _nprocs

57 abort();
58 }
59 else if(_nprocs == nelem) {
60 for(unsigned i = 0; i < nelem; i++) {
61 epart[i] = i;
62 }
63 }
64 else {
65
66#ifndef HAVE_METIS
67 std::cerr << "Fatal error: Metis library was not found. Metis partioning algorithm cannot be called!" << std::endl;
68 exit(1);
69#endif
70
71 unsigned eind_size = _mesh.el->GetElementNumber(geom_elems[HEX]) * _mesh.el->GetNVE(0, CONTINUOUS_BIQUADRATIC)
72 + _mesh.el->GetElementNumber(geom_elems[TET]) * _mesh.el->GetNVE(1, CONTINUOUS_BIQUADRATIC)
73 + _mesh.el->GetElementNumber(geom_elems[WEDGE]) * _mesh.el->GetNVE(2, CONTINUOUS_BIQUADRATIC)
74 + _mesh.el->GetElementNumber(geom_elems[QUAD]) * _mesh.el->GetNVE(3, CONTINUOUS_BIQUADRATIC)

78 std::vector < idx_t > eptr(nelem + 1);

113 int err = METIS_PartMeshDual(&nelem, &nnodes, &eptr[0], &eind[0], NULL, NULL, &ncommon, &_nprocs, NULL, options, &objval, &epart[0], &npart[0]);
114
115 if(err == METIS_OK) {
116 std::cout << " METIS PARTITIONING IS OK " << std::endl;
117 }
118 else if(err == METIS_ERROR_INPUT) {

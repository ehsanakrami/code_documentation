143 void FE_Proj_Matrices::Get_QitoQjProjection_SparsityPatternSize_OneElement_OneFEFamily_Lagrange_Continuous(const Mesh& mesh,
144 const int& iel,
145 NumericVector* NNZ_d,
146 NumericVector* NNZ_o,
147 const unsigned& itype,
148 const elem_type * elem_type_in_jtype,
149 const unsigned ndofs_itype_in) const
150 {
151
152 assert(itype < NFE_FAMS_C_ZERO_LAGRANGE);
153
154
155 const unsigned soltype_in = elem_type_in_jtype->GetSolType();
156 const basis * pt_basis_in = elem_type_in_jtype->GetBasis();
157 const unsigned ndofs = elem_type_in_jtype->GetNDofs();
158
159 const unsigned ndofs_Lagrange = ndofs_itype_in;
160
161 const bool identity = ( ndofs_Lagrange <= ndofs ) ? true : false;
162
163 for(int i = 0; i < ndofs_Lagrange; i++) {
164
165 const int irow = mesh.GetSolutionDof(i, iel, itype);
166 const int iproc = mesh.BisectionSearch_find_processor_of_dof(irow, itype);
167 const int ncols = (identity) ? 1 : ndofs;
168
169 unsigned counter_off_diag = 0;
170 unsigned counter_all_diag = 0;
171
172 for(int k = 0; k < ncols; k++) {
173
174 const double phi = (identity) ? 1. : pt_basis_in->eval_phi( pt_basis_in->GetIND(k), pt_basis_in->GetXcoarse(i) );
175
176 if(fabs(phi) > 1.0e-14) {
177 counter_all_diag ++;
178 int kcolumn = (identity) ? mesh.GetSolutionDof(i, iel, soltype_in) : mesh.GetSolutionDof(k, iel, soltype_in);
179 if( kcolumn < mesh.dofmap_get_dof_offset(soltype_in, iproc) ||
180 kcolumn >= mesh.dofmap_get_dof_offset(soltype_in, iproc + 1) ) counter_off_diag++;
181 }
182
183 }
184
185 NNZ_d->set(irow, counter_all_diag - counter_off_diag);
186 NNZ_o->set(irow, counter_off_diag);
187 }
188
189
190 }

58 void MeshRefinement::FlagElementsToRefine(const unsigned& type) {
59
60 //BEGIN temporary parallel vector initialization
61 NumericVector* numberOfRefinedElement;
62 numberOfRefinedElement = NumericVector::build().release();
63
64 if(_nprocs == 1) numberOfRefinedElement->init(_nprocs, 1, false, SERIAL);
65 else numberOfRefinedElement->init(_nprocs, 1, false, PARALLEL);
66
67 numberOfRefinedElement->zero();
68 //END temporary parallel vector initialization
69
70 //BEGIN flag element to be refined
71 if(type == 0) { // Flag all element
72 for(int iel = _mesh.GetElementOffset(_iproc); iel < _mesh.GetElementOffset(_iproc + 1); iel++) {
73 if(_mesh.el->GetIfElementCanBeRefined(iel)) {
74 _mesh.GetTopology()->_Sol[_mesh.GetAmrIndex()]->set(iel, 1.);
75 numberOfRefinedElement->add(_iproc, 1.);
76 }
77 }
78 }
79 else if(type == 1) { // Flag AMR elements
80 for(int iel = _mesh.GetElementOffset(_iproc); iel < _mesh.GetElementOffset(_iproc + 1); iel++) {
81 if(_mesh.el->GetIfElementCanBeRefined(iel)) {
82 if((*_mesh.GetTopology()->_Sol[ _mesh.GetAmrIndex() ])(iel) > 0.5) {
83 numberOfRefinedElement->add(_iproc, 1.);
84 }
85 else if( _mesh.get_is_refinement_function_defined() ) {
86 short unsigned ielt = _mesh.GetElementType(iel);
87 unsigned nve = _mesh.GetElementDofNumber(iel, 0);
88 std::vector < double > x(3, 0.);
89
90 for(unsigned i = 0; i < nve; i++) {
91 unsigned inode_metis = _mesh.GetSolutionDof(i, iel, 2);
92 x[0] += (*_mesh.GetTopology()->_Sol[0])(inode_metis);
93 x[1] += (*_mesh.GetTopology()->_Sol[1])(inode_metis);
94 x[2] += (*_mesh.GetTopology()->_Sol[2])(inode_metis);
95 }
96
97 x[0] /= nve;
98 x[1] /= nve;
99 x[2] /= nve;
100
101 if(_mesh._SetRefinementFlag(x, _mesh.GetElementGroup(iel), _mesh.GetLevel())) {
102 _mesh.GetTopology()->_Sol[ _mesh.GetAmrIndex() ]->set(iel, 1.);
103 numberOfRefinedElement->add(_iproc, 1.);
104 }
105 }
106 }
107 else {
108 _mesh.GetTopology()->_Sol[ _mesh.GetAmrIndex() ]->set(iel, 0.);
109 }
110 }
111 }
112 else if(type == 2) { // Flag only even elements (for debugging purposes)
113 for(int iel = _mesh.GetElementOffset(_iproc); iel < _mesh.GetElementOffset(_iproc + 1); iel++) {
114 if(_mesh.el->GetIfElementCanBeRefined(iel)) {
115 if((*_mesh.GetTopology()->_Sol[_mesh.GetAmrIndex()])(iel) < 0.5 && iel % 2 == 0) {
116 _mesh.GetTopology()->_Sol[_mesh.GetAmrIndex()]->set(iel, 1.);
117 numberOfRefinedElement->add(_iproc, 1.);
118 }
119 }
120 }
121 }
122
123 _mesh.GetTopology()->_Sol[_mesh.GetAmrIndex()]->close();
124 //END flag element to be refined
125
126 //BEGIN update elem
127 numberOfRefinedElement->close();
128 double totalNumber = numberOfRefinedElement->l1_norm();
129 _mesh.el->SetRefinedElementNumber(static_cast < unsigned >(totalNumber + 0.25));
130 delete numberOfRefinedElement;
131
132 //END update elem
133 }

136 bool MeshRefinement::FlagElementsToRefineBasedOnError(const double& treshold, NumericVector& error) {
137
138 unsigned type = 2;
139
140 //BEGIN temporary parallel vector initialization
141 NumericVector* numberOfRefinedElement;
142 numberOfRefinedElement = NumericVector::build().release();
143
144 if(_nprocs == 1) numberOfRefinedElement->init(_nprocs, 1, false, SERIAL);
145 else numberOfRefinedElement->init(_nprocs, 1, false, PARALLEL);
146
147 numberOfRefinedElement->zero();
148 //END temporary parallel vector initialization
149
150
151 //BEGIN flag element to be refined
152 for(int iel = _mesh.GetElementOffset(_iproc); iel < _mesh.GetElementOffset(_iproc + 1); iel++) {
153 if(_mesh.el->GetIfElementCanBeRefined(iel)) {
154 if((*_mesh.GetTopology()->_Sol[_mesh.GetAmrIndex()])(iel) < 0.5 && error(iel) > treshold) {
155 _mesh.GetTopology()->_Sol[_mesh.GetAmrIndex()]->set(iel, 1.);
156 numberOfRefinedElement->add(_iproc, 1.);
157 }
158 }
159 }
160
161 _mesh.GetTopology()->_Sol[_mesh.GetAmrIndex()]->close();
162 //END flag element to be refined
163
164 //BEGIN update elem
165 numberOfRefinedElement->close();
166 double totalNumber = numberOfRefinedElement->l1_norm();
167 _mesh.el->SetRefinedElementNumber(static_cast < unsigned >(totalNumber + 0.25));
168 delete numberOfRefinedElement;
169
170 bool elementsHaveBeenRefined = true;
171 if(totalNumber < _nprocs) {
172 elementsHaveBeenRefined = false;
173 }
174 return elementsHaveBeenRefined;
175 //END update elem
176 }

188void MeshRefinement::RefineMesh(const unsigned& igrid, Mesh* mshc, /*const*/ elem_type* otherFiniteElement[N_GEOM_ELS][NFE_FAMS]) {

191//==== Equivalent of: ReadCoarseMeshBeforePartitioning - BEGIN ========
192
193
194//==== Level - BEGIN ==============================
195 _mesh.SetLevel(igrid);
196//==== Level - END ==============================
197
198
199//==== AMR - BEGIN ==============================
200 _mesh.SetIfHomogeneous(true);
201//==== AMR - END ==============================
202
203
204//==== info from the coarse mesh - BEGIN ==============================
205
206 const unsigned dim = mshc->GetDimension();
207
208 _mesh.SetDimension(dim);
209
210 _mesh.get_prol_matrices().SetCoarseMesh(mshc);
211
212 elem* elc = mshc->GetMeshElements();
213
214 _mesh.SetFiniteElementPtr(otherFiniteElement);
215
216//==== info from the coarse mesh - END ==============================
217
218
219//====== BEGIN ELEMENTS ==============================
220
221 // total number of elements on the fine level
222
223 const unsigned n_elems_per_elem_after_ref = _mesh.GetMeshElements()->GetRefIndex(dim);
224 int nelem = elem::InitializeNumberOfElementsFromCoarseList(elc, n_elems_per_elem_after_ref );
225
226 _mesh.SetNumberOfElements(nelem);
227
228 unsigned elementOffsetCoarse = mshc->GetElementOffset(_iproc);
229 unsigned elementOffsetCoarseP1 = mshc->GetElementOffset(_iproc + 1);
230
231 std::vector < double > coarseLocalizedAmrVector;
232 mshc->GetTopology()->_Sol[mshc->GetAmrIndex()]->localize_to_all(coarseLocalizedAmrVector);
233
234 mshc->GetMeshElements()->AllocateChildrenElement (n_elems_per_elem_after_ref, mshc);
235 mshc->GetMeshElements()->AllocateChildrenElementDof(n_elems_per_elem_after_ref, mshc);
236
237 _mesh.el = new elem(elc, mshc->GetDimension(), n_elems_per_elem_after_ref, coarseLocalizedAmrVector);
238
239
240 unsigned jel = 0;
241 //divide each coarse element in 8(3D), 4(2D) or 2(1D) fine elements and find all the vertices
242
243 _mesh.el->SetElementGroupNumber(elc->GetElementGroupNumber());
244
245 bool AMR = false;
246
247
248
249 for(unsigned isdom = 0; isdom < _nprocs; isdom++) {
250
251 elc->LocalizeElementDof(isdom);
252 elc->LocalizeElementNearFace(isdom);
253 elc->LocalizeElement_Level_Type_Group_Material(isdom);
254
255 for(unsigned iel = mshc->GetElementOffset(isdom); iel < mshc->GetElementOffset(isdom + 1); iel++) {
256
257 if(static_cast < unsigned short >(coarseLocalizedAmrVector[iel] + 0.25) == 1) {
258
259 unsigned elt = elc->GetElementType(iel);
260 // project element type, group, material; child element -----------------
261 for(unsigned j = 0; j < n_elems_per_elem_after_ref; j++) {
262 _mesh.el->SetElementType(jel + j, elc->GetElementType(iel));
263 _mesh.el->SetElementGroup(jel + j, elc->GetElementGroup(iel));
264
265 unsigned gr_mat = elc->GetElementMaterial(iel);
266 _mesh.el->SetElementMaterial(jel + j, gr_mat);
267
268
269 _mesh.el->SetElementLevel(jel + j, elc->GetElementLevel(iel) + 1);
270 if(iel >= elementOffsetCoarse && iel < elementOffsetCoarseP1) {
271 elc->SetChildElement(iel, j, jel + j);
272 }
273 }
274
275 // project vertex indices -----------------
276 for(unsigned j = 0; j < n_elems_per_elem_after_ref; j++)
277 for(unsigned inode = 0; inode < elc->GetNVE(elt, CONTINUOUS_LINEAR); inode++) {
278 unsigned jDof = otherFiniteElement[elt][ CONTINUOUS_LINEAR ]->GetBasis()->GetFine2CoarseVertexMapping(j, inode);
279 _mesh.el->SetElementDofIndex(jel + j, inode, elc->GetElementDofIndex(iel, jDof));
280 }
281
282 // project face indices -----------------
283 for(unsigned iface = 0; iface < elc->GetNFC(elt, 1); iface++) {
284 int value = elc->GetFaceElementIndex(iel, iface);
285
286 if(value < -1)
287 for(unsigned jface = 0; jface < _mesh.GetMeshElements()->GetRefFaceIndex(dim); jface++)
288 _mesh.el->SetFaceElementIndex(jel + coarse2FineFaceMapping[elt][iface][jface][0], coarse2FineFaceMapping[elt][iface][jface][1], value);
289 }
290
291 // update element numbers -----------------
292 jel += n_elems_per_elem_after_ref;
293 _mesh.el->AddToElementNumber(n_elems_per_elem_after_ref, elt);
294 }
295 else {
296
297 _mesh.SetIfHomogeneous(false);
298 AMR = true;
299
300 unsigned elt = elc->GetElementType(iel);
301
302 // project element type, group, material; child element -----------------
303 _mesh.el->SetElementType(jel, elc->GetElementType(iel));
304 _mesh.el->SetElementGroup(jel , elc->GetElementGroup(iel));
305 _mesh.el->SetElementMaterial(jel, elc->GetElementMaterial(iel));
306
307 unsigned gr_mat = elc->GetElementMaterial(iel);
308 _mesh.el->SetElementMaterial(jel, gr_mat);
309
310 _mesh.el->SetElementLevel(jel, elc->GetElementLevel(iel));
311 if(iel >= elementOffsetCoarse && iel < elementOffsetCoarseP1) {
312 elc->SetChildElement(iel, 0, jel);
313 }
314
315 // project nodes indices -----------------
316 for(unsigned inode = 0; inode < elc->GetNVE(elt, CONTINUOUS_BIQUADRATIC); inode++)
317 _mesh.el->SetElementDofIndex(jel, inode, elc->GetElementDofIndex(iel, inode));
318
319 // project face indices -----------------
320 for(unsigned iface = 0; iface < elc->GetNFC(elt, 1); iface++) {
321 int value = elc->GetFaceElementIndex(iel, iface);
322
323 if(value < -1) {
324 _mesh.el->SetFaceElementIndex(jel, iface, value);
325 }
326 }
327
328 // update element numbers -----------------
329 jel++;
330 _mesh.el->AddToElementNumber(1, elt);
331 }
332
333 }
334
335 elc->FreeLocalizedElementDof();
336 elc->FreeLocalizedElementNearFace();
337 elc->FreeLocalizedElement_Level_Type_Group_Material();
338
339 }
340
341
342 std::vector< double > ().swap(coarseLocalizedAmrVector);
343
344//====== END ELEMENTS ==============================
345
346
347//====== BEGIN NODES ==============================
348
349 int nnodes = elc->GetNodeNumber();
350 _mesh.SetNumberOfNodes(nnodes);
351 _mesh.el->SetNodeNumber(nnodes);
352
353 //find all the elements near each vertex
354 _mesh.el->BuildElementNearVertex();
355
356 //initialize to UINT_MAX all the middle edge points
357 for(unsigned iel = 0; iel < _mesh.GetNumberOfElements(); iel++) {
358 if(_mesh.el->GetIfFatherHasBeenRefined(iel)) {
359 for(unsigned inode = _mesh.el->GetElementDofNumber(iel, 0); inode < _mesh.el->GetElementDofNumber(iel, 1); inode++) {
360 _mesh.el->SetElementDofIndex(iel, inode, UINT_MAX);
361 }
362 }
363 }
364
365 //find all the middle edge points
366 for(unsigned iel = 0; iel < _mesh.GetNumberOfElements(); iel++) {
367 if(_mesh.el->GetIfFatherHasBeenRefined(iel)) {
368 unsigned ielt = _mesh.el->GetElementType(iel);
369 unsigned istart = _mesh.el->GetElementDofNumber(iel, 0);
370 unsigned iend = _mesh.el->GetElementDofNumber(iel, 1);
371
372 for(unsigned inode = istart; inode < iend; inode++) {
373 if(UINT_MAX == _mesh.el->GetElementDofIndex(iel, inode)) {
374 nnodes++;
375 _mesh.el->SetElementDofIndex(iel, inode, nnodes - 1u);
376 unsigned im = _mesh.el->GetElementDofIndex(iel, edge2VerticesMapping[ielt][inode - istart][0]);
377 unsigned ip = _mesh.el->GetElementDofIndex(iel, edge2VerticesMapping[ielt][inode - istart][1]);
378
379 //find all the near elements which share the same middle edge point
380 for(unsigned j = 0; j < _mesh.el->GetElementNearVertexNumber(im); j++) {
381 unsigned jel = _mesh.el->GetElementNearVertex(im, j);
382
383 if(_mesh.el->GetIfFatherHasBeenRefined(jel) && jel > iel) { // to skip coarse elements
384 unsigned jm = 0, jp = 0;
385 unsigned jelt = _mesh.el->GetElementType(jel);
386
387 for(unsigned jnode = 0; jnode < _mesh.el->GetElementDofNumber(jel, 0); jnode++) {
388 if(_mesh.el->GetElementDofIndex(jel, jnode) == im) {
389 jm = jnode + 1u;
390 break;
391 }
392 }
393
394 if(jm != 0) { //TODO this can be changed and put inside (by Sara)
395 for(unsigned jnode = 0; jnode < _mesh.el->GetElementDofNumber(jel, 0); jnode++) {
396 if(_mesh.el->GetElementDofIndex(jel, jnode) == ip) {
397 jp = jnode + 1u;
398 break;
399 }
400 }
401
402 if(jp != 0) {
403 if(jp < jm) {
404 unsigned tp = jp;
405 jp = jm;
406 jm = tp;
407 }
408
409 _mesh.el->SetElementDofIndex(jel, vertices2EdgeMapping[jelt][--jm][--jp], nnodes - 1u);
410 }
411 }
412 }
413 }
414 }
415 }
416 }
417 }
418
419 _mesh.SetNumberOfNodes(nnodes);
420 _mesh.el->SetNodeNumber(nnodes);
421
422
423 AddFaceDofAndElementDof();
424//====== END NODES ==============================
425
426
427
428//==== Equivalent of: ReadCoarseMeshBeforePartitioning - END ========
429
430
431
432
433
434//==== Partition: BEGIN ========
435
436 std::vector < unsigned > partition = _mesh.PartitionForElements_refinement(AMR, mshc);
437
438 _mesh.FillISvectorDofMapAllFEFamilies(partition);
439
440 std::vector<unsigned> ().swap(partition);
441
442//==== Partition: END ========
443
444
445
446//====================================
447//==== SetChildElementDof ========
448//====================================
449 elc->SetChildElementDof(_mesh.el);
450
451
452
453//==== BuildElem_NearFace_NearElem_using_NearVertex - BEGIN ========
454 _mesh.GetMeshElements()->BuildElem_NearFace_NearElem_using_NearVertex();
455
456 _mesh.GetMeshElements()->ScatterElement_Level_Type_Group_Material___NearFace();
457
458 _mesh.GetMeshElements()->ScatterElementDof();
459//==== BuildElem_NearFace_NearElem_using_NearVertex - END ========
460
461
462//====================================
463

465 _mesh.Topology_InitializeCoordinates();
466

468 unsigned solType = CONTINUOUS_BIQUADRATIC;
469
470 _mesh.GetTopology()->_Sol[0]->matrix_mult(*mshc->GetTopology()->_Sol[0], *_mesh.get_prol_matrices().GetCoarseToFineProjection(solType, _mesh));
471 _mesh.GetTopology()->_Sol[1]->matrix_mult(*mshc->GetTopology()->_Sol[1], *_mesh.get_prol_matrices().GetCoarseToFineProjection(solType, _mesh));
472 _mesh.GetTopology()->_Sol[2]->matrix_mult(*mshc->GetTopology()->_Sol[2], *_mesh.get_prol_matrices().GetCoarseToFineProjection(solType, _mesh));
473 _mesh.GetTopology()->_Sol[0]->close();
474 _mesh.GetTopology()->_Sol[1]->close();
475 _mesh.GetTopology()->_Sol[2]->close();
476//==== Topology, Coordinates - END ========
477
478
479
480//==== Topology, AMR - BEGIN ========
481 _mesh.Topology_InitializeAMR();
482//==== Topology, AMR - END ========
483
484
485//==== Topology, Solidnodeflag - BEGIN ========
486 _mesh.Topology_InitializeSolidNodeFlag();
487 _mesh.Topology_FillSolidNodeFlag();
488//==== Topology, Solidnodeflag - END ========
489
490
491

493 _mesh.InitializeAndPossiblyFillAmrRestriction(AMR);
494//==== AMR (only uses Topology in one point, does not modify it) - END ========
495
496
497//====================================
498//==== CharacteristicLength ========
499//====================================
500 _mesh.SetCharacteristicLength( mshc->GetCharacteristicLength() );
501
502
503//==== Print Info ========
504 _mesh.PrintInfo();
505

513 void MeshRefinement::AddFaceDofAndElementDof() {
514
515 unsigned int nnodes = _mesh.GetNumberOfNodes();
516
517 //intialize to UINT_MAX
518 for(unsigned iel = 0; iel < _mesh.el->GetElementNumber(); iel++) {
519 if(_mesh.el->GetIfFatherHasBeenRefined(iel)) {
520 for(unsigned inode = _mesh.el->GetElementDofNumber(iel, 1); inode < _mesh.el->GetElementDofNumber(iel, 2); inode++) {
521 _mesh.el->SetElementDofIndex(iel, inode, UINT_MAX);
522 }
523 }
524 }
525
526 // generate face dofs on quad faces for hex and wedge elements
527 for(unsigned iel = 0; iel < _mesh.el->GetElementNumber(); iel++) {
528 if(_mesh.el->GetIfFatherHasBeenRefined(iel)) {
529 for(unsigned iface = 0; iface < _mesh.el->GetElementFaceNumber(iel, 0); iface++) { // I think is on all the faces that are quads
530 unsigned inode = _mesh.el->GetElementDofNumber(iel, 1) + iface;
531
532 if(UINT_MAX == _mesh.el->GetElementDofIndex(iel, inode)) {
533 _mesh.el->SetElementDofIndex(iel, inode, ++nnodes - 1u);
534 unsigned i1 = _mesh.el->GetFaceVertexIndex(iel, iface, 0);
535 unsigned i2 = _mesh.el->GetFaceVertexIndex(iel, iface, 1);
536 unsigned i3 = _mesh.el->GetFaceVertexIndex(iel, iface, 2);
537
538 for(unsigned j = 0; j < _mesh.el->GetElementNearVertexNumber(i1); j++) {
539 unsigned jel = _mesh.el->GetElementNearVertex(i1, j);
540
541 if(_mesh.el->GetIfFatherHasBeenRefined(jel) && jel > iel) {
542 for(unsigned jface = 0; jface < _mesh.el->GetElementFaceNumber(jel, 0); jface++) {
543 unsigned jnode = _mesh.el->GetElementDofNumber(jel, 1) + jface;
544
545 if(UINT_MAX == _mesh.el->GetElementDofIndex(jel, jnode)) {
546 unsigned j1 = _mesh.el->GetFaceVertexIndex(jel, jface, 0);
547 unsigned j2 = _mesh.el->GetFaceVertexIndex(jel, jface, 1);
548 unsigned j3 = _mesh.el->GetFaceVertexIndex(jel, jface, 2);
549 unsigned j4 = _mesh.el->GetFaceVertexIndex(jel, jface, 3);
550
551 if((i1 == j1 || i1 == j2 || i1 == j3 || i1 == j4) &&
552 (i2 == j1 || i2 == j2 || i2 == j3 || i2 == j4) &&
553 (i3 == j1 || i3 == j2 || i3 == j3 || i3 == j4)) {
554 _mesh.el->SetElementDofIndex(jel, jnode, nnodes - 1u);
555 }
556 }
557 }
558 }
559 }
560 }
561 }
562 // generate face dofs on tri faces for tet and wedge elements
563 unsigned elementType = _mesh.el->GetElementType(iel);
564 if(elementType == 1 || elementType == 2) {
565 for(unsigned iface = _mesh.el->GetElementFaceNumber(iel, 0); iface < _mesh.el->GetElementFaceNumber(iel, 1); iface++) { //on all the faces that are triangles
566 unsigned inode = _mesh.el->GetElementDofNumber(iel, 1) + iface;
567 if(UINT_MAX == _mesh.el->GetElementDofIndex(iel, inode)) {
568 _mesh.el->SetElementDofIndex(iel, inode, ++nnodes - 1u);
569 unsigned i1 = _mesh.el->GetFaceVertexIndex(iel, iface, 0);
570 unsigned i2 = _mesh.el->GetFaceVertexIndex(iel, iface, 1);
571 unsigned i3 = _mesh.el->GetFaceVertexIndex(iel, iface, 2);
572 for(unsigned j = 0; j < _mesh.el->GetElementNearVertexNumber(i1); j++) {
573 unsigned jel = _mesh.el->GetElementNearVertex(i1, j);
574 if(_mesh.el->GetIfFatherHasBeenRefined(jel) && jel > iel) {
575 for(unsigned jface = _mesh.el->GetElementFaceNumber(jel, 0); jface < _mesh.el->GetElementFaceNumber(jel, 1); jface++) {
576 unsigned jnode = _mesh.el->GetElementDofNumber(jel, 1) + jface;
577 if(UINT_MAX == _mesh.el->GetElementDofIndex(jel, jnode)) {
578 unsigned j1 = _mesh.el->GetFaceVertexIndex(jel, jface, 0);
579 unsigned j2 = _mesh.el->GetFaceVertexIndex(jel, jface, 1);
580 unsigned j3 = _mesh.el->GetFaceVertexIndex(jel, jface, 2);
581 if((i1 == j1 || i1 == j2 || i1 == j3) &&
582 (i2 == j1 || i2 == j2 || i2 == j3) &&
583 (i3 == j1 || i3 == j2 || i3 == j3)) {
584 _mesh.el->SetElementDofIndex(jel, jnode, nnodes - 1u);
585 }
586 }
587 }
588 }
589 }
590 }
591 }
592 }
593
594 }
595 }
596
597 // generates element dofs for hex, tet, wedge, quad and triangle elements
598 for(unsigned iel = 0; iel < _mesh.el->GetElementNumber(); iel++) {
599 if(_mesh.el->GetIfFatherHasBeenRefined(iel)) {
600 if(0 == _mesh.el->GetElementType(iel)) { //hex
601 _mesh.el->SetElementDofIndex(iel, 26, ++nnodes - 1u);
602 }
603 if(1 == _mesh.el->GetElementType(iel)) { //tet
604 _mesh.el->SetElementDofIndex(iel, 14, ++nnodes - 1u);
605 }
606 if(2 == _mesh.el->GetElementType(iel)) { //wedge
607 _mesh.el->SetElementDofIndex(iel, 20, ++nnodes - 1u);
608 }
609 if(3 == _mesh.el->GetElementType(iel)) { //quad
610 _mesh.el->SetElementDofIndex(iel, 8, ++nnodes - 1u);
611 }
612 if(4 == _mesh.el->GetElementType(iel)) { //triangle
613 _mesh.el->SetElementDofIndex(iel, 6, ++nnodes - 1u);
614 }
615 }
616 }
617
618 _mesh.el->SetNodeNumber(nnodes);
619 _mesh.SetNumberOfNodes(nnodes);
620
621 }

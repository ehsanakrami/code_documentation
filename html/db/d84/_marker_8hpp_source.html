85 double GetCoordinates (const Solution *sol, const unsigned &k, const unsigned &i , const double &s) {
86 if (!sol->GetIfFSI()) {
87 return (*sol->GetMesh()->GetTopology()->_Sol[k]) (i);
88 }
89 else {
90 const char varname[3][3] = {"DX", "DY", "DZ"};
91 unsigned solIndex = sol->GetIndex (&varname[k][0]);
92 return (*sol->GetMesh()->GetTopology()->_Sol[k]) (i)
93 + (1. - s) * (*sol->_SolOld[solIndex]) (i)
94 + s * (*sol->_Sol[solIndex]) (i);

433 unsigned step = (_step == UINT_MAX) ? n * order : _step;

436void AssembleBilaplaceProblem_AD(MultiLevelProblem& ml_prob) {
437 // ml_prob is the global object from/to where get/set all the data
438 // level is the level of the PDE system to be assembled
439
440 // call the adept stack object
441 adept::Stack& s = FemusInit::_adeptStack;
442
443 // extract pointers to the several objects that we are going to use
444
445 NonLinearImplicitSystem* mlPdeSys = &ml_prob.get_system<NonLinearImplicitSystem> ("Biharmonic"); // pointer to the linear implicit system named "Biharmonic"
446
447 const unsigned level = mlPdeSys->GetLevelToAssemble();
448
449 Mesh* msh = ml_prob._ml_msh->GetLevel(level); // pointer to the mesh (level) object
450 elem* el = msh->GetMeshElements(); // pointer to the elem object in msh (level)
451
452 MultiLevelSolution* mlSol = ml_prob._ml_sol; // pointer to the multilevel solution object
453 Solution* sol = ml_prob._ml_sol->GetSolutionLevel(level); // pointer to the solution (level) object
454
455 LinearEquationSolver* pdeSys = mlPdeSys->_LinSolver[level]; // pointer to the equation (level) object
456 SparseMatrix* KK = pdeSys->_KK; // pointer to the global stifness matrix object in pdeSys (level)
457 NumericVector* RES = pdeSys->_RES; // pointer to the global residual vector object in pdeSys (level)
458
459 const unsigned dim = msh->GetDimension(); // get the domain dimension of the problem
460 unsigned iproc = msh->processor_id(); // get the process_id (for parallel computation)
461
462 //solution variable
463 unsigned soluIndex = mlSol->GetIndex("u"); // get the position of "u" in the ml_sol object
464 unsigned soluType = mlSol->GetSolutionType(soluIndex); // get the finite element type for "u"
465 unsigned soluPdeIndex = mlPdeSys->GetSolPdeIndex("u"); // get the position of "u" in the pdeSys object
466
467 //adouble is a type inside adept library correpsonding to the double type. Need to be defined inside the library.
468 std::vector < adept::adouble > solu; // local solution
469
470 unsigned solvIndex = mlSol->GetIndex("v"); // get the position of "v" in the ml_sol object
471 unsigned solvType = mlSol->GetSolutionType(solvIndex); // get the finite element type for "v"
472 unsigned solvPdeIndex = mlPdeSys->GetSolPdeIndex("v"); // get the position of "v" in the pdeSys object
473
474 std::vector < adept::adouble > solv; // local solution
475
476
477 std::vector < std::vector < double > > x(dim); // local coordinates
478 unsigned xType = 2; // get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC)
479
480 std::vector < int > sysDof; // local to global pdeSys dofs
481 std::vector <double> phi; // local test function
482 std::vector <double> phi_x; // local test function first order partial derivatives
483 std::vector <double> phi_xx; // local test function second order partial derivatives
484 double weight; // gauss point weight
485
486 std::vector< double > Res; // local redidual vector
487 std::vector< adept::adouble > aResu; // local redidual vector
488 std::vector< adept::adouble > aResv; // local redidual vector
489
490
491 // reserve memory for the local standar vectors
492 const unsigned maxSize = static_cast< unsigned >(ceil(pow(3, dim))); // conservative: based on line3, quad9, hex27
493 solu.reserve(maxSize);
494 solv.reserve(maxSize);
495
496 for (unsigned i = 0; i < dim; i++)
497 x[i].reserve(maxSize);
498
499 sysDof.reserve(2 * maxSize);
500 phi.reserve(maxSize);
501 phi_x.reserve(maxSize * dim);
502 unsigned dim2 = (3 * (dim - 1) + !(dim - 1)); // dim2 is the number of second order partial derivatives (1,3,6 depending on the dimension)
503 phi_xx.reserve(maxSize * dim2);
504
505 Res.reserve(2 * maxSize);
506 aResu.reserve(maxSize);
507 aResv.reserve(maxSize);
508
509 std::vector < double > Jac; // local Jacobian matrix (ordered by column, adept)
510 Jac.reserve(4 * maxSize * maxSize);
511
512
513 KK->zero(); // Set to zero all the entries of the Global Matrix
514
515
516 for (int iel = msh->GetElementOffset(iproc); iel < msh->GetElementOffset(iproc + 1); iel++) { //Loop for each of the element and computing the Jacobian matrix using "adept"
517
518 short unsigned ielGeom = msh->GetElementType(iel);
519 unsigned nDofs = msh->GetElementDofNumber(iel, soluType); // number of solution element dofs
520 unsigned nDofs2 = msh->GetElementDofNumber(iel, xType); // number of coordinate element dofs
521
522 // resize local arrays
523 sysDof.resize(2 * nDofs);
524 solu.resize(nDofs);
525 solv.resize(nDofs);
526
527 for (int i = 0; i < dim; i++) {
528 x[i].resize(nDofs2);
529 }
530
531 aResu.assign(nDofs, 0.); //resize
532 aResv.assign(nDofs, 0.); //resize
533
534 // local storage of global mapping and solution
535 for (unsigned i = 0; i < nDofs; i++) {
536 unsigned solDof = msh->GetSolutionDof(i, iel, soluType); // global to global mapping between solution node and solution dof
537 solu[i] = (*sol->_Sol[soluIndex])(solDof); // global extraction and local storage for the solution
538 solv[i] = (*sol->_Sol[solvIndex])(solDof); // global extraction and local storage for the solution
539 sysDof[i] = pdeSys->GetSystemDof(soluIndex, soluPdeIndex, i, iel); // global to global mapping between solution node and pdeSys dof
540 sysDof[nDofs + i] = pdeSys->GetSystemDof(solvIndex, solvPdeIndex, i, iel); // global to global mapping between solution node and pdeSys dof
541 }
542
543 // local storage of coordinates
544 for (unsigned i = 0; i < nDofs2; i++) {
545 unsigned xDof = msh->GetSolutionDof(i, iel, xType); // global to global mapping between coordinates node and coordinate dof
546
547 for (unsigned jdim = 0; jdim < dim; jdim++) {
548 x[jdim][i] = (*msh->GetTopology()->_Sol[jdim])(xDof); // global extraction and local storage for the element coordinates
549 }
550 }
551
552 // start a new recording of all the operations involving adept::adouble variables
553 s.new_recording();
554
555 // *** Gauss point loop ***
556 for (unsigned ig = 0; ig < msh->_finiteElement[ielGeom][soluType]->GetGaussPointNumber(); ig++) {
557 // *** get gauss point weight, test function and test function partial derivatives ***
558 msh->_finiteElement[ielGeom][soluType]->Jacobian(x, ig, weight, phi, phi_x, phi_xx);
559
560 // evaluate the solution, the solution derivatives and the coordinates in the gauss point
561 adept::adouble soluGauss = 0;
562 std::vector < adept::adouble > soluGauss_x(dim, 0.);
563
564 adept::adouble solvGauss = 0;
565 std::vector < adept::adouble > solvGauss_x(dim, 0.);
566
567 std::vector < double > xGauss(dim, 0.);
568
569 for (unsigned i = 0; i < nDofs; i++) {
570 soluGauss += phi[i] * solu[i];
571 solvGauss += phi[i] * solv[i];
572
573 for (unsigned jdim = 0; jdim < dim; jdim++) {
574 soluGauss_x[jdim] += phi_x[i * dim + jdim] * solu[i];
575 solvGauss_x[jdim] += phi_x[i * dim + jdim] * solv[i];
576 xGauss[jdim] += x[jdim][i] * phi[i];
577 }
578 }
579
580 // *** phi_i loop ***
581 for (unsigned i = 0; i < nDofs; i++) {
582
583 adept::adouble Laplace_u = 0.;
584 adept::adouble Laplace_v = 0.;
585
586 for (unsigned jdim = 0; jdim < dim; jdim++) {
587 Laplace_u += - phi_x[i * dim + jdim] * soluGauss_x[jdim];
588 Laplace_v += - phi_x[i * dim + jdim] * solvGauss_x[jdim];
589 }
590
591 double exactSolValue = GetExactSolutionValue(xGauss);
592
593 //RHS part of the Biharmonic Equation
594 double pi = acos(-1.);
595 aResv[i] += (solvGauss * phi[i] - Laplace_u) * weight;
596 //aResu[i] += ( (36*x[0] + 36*x[1] - 24) * phi[i] - Laplace_v) * weight;
597 aResu[i] += (LaplaceGetExactSolutionLaplacian(xGauss) * phi[i] - Laplace_v) * weight; //Orginal
598 } // end phi_i loop
599 } // end gauss point loop
600
601
602 // Add the local Matrix/Vector into the global Matrix/Vector
603
604 //copy the value of the adept::adoube aRes in double Res and store
605 Res.resize(2 * nDofs);
606
607 for (int i = 0; i < nDofs; i++) {
608 Res[i] = -aResu[i].value();
609 Res[nDofs + i] = -aResv[i].value();
610 }
611
612 RES->add_vector_blocked(Res, sysDof);
613
614 Jac.resize( 4 * nDofs * nDofs );
615
616 // define the dependent variables
617 s.dependent(&aResu[0], nDofs);
618 s.dependent(&aResv[0], nDofs);
619
620 // define the independent variables
621 s.independent(&solu[0], nDofs);
622 s.independent(&solv[0], nDofs);
623
624 // get the jacobian matrix (ordered by column)
625 s.jacobian(&Jac[0], true);
626
627 KK->add_matrix_blocked(Jac, sysDof, sysDof);
628
629 s.clear_independents();
630 s.clear_dependents();
631
632 } //end element loop for each process
633
634 RES->close();
635 KK->close();
636
637
638}

75 value = kmax;

359 soluOld.reserve (maxSize);

363 unsigned xType = 2; // get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC)

366 x[k].reserve (maxSize); // dim x maxsize is reserved for x.
367 }

369 std::vector <double> phi; // local test function
370 std::vector <double> phi_x; // local test function first order partial derivatives
371
372 double weight; // gauss point weight
373 phi.reserve (maxSize);
374 phi_x.reserve (maxSize * dim); // This is probably gradient but he is doing the life difficult for me!
375

377 aRes.reserve (maxSize);
378

380 l2GMap.reserve (maxSize);
381 std::vector < double > Res; // local redidual vector

383 std::vector < double > Jac;

386 KK->zero(); // Set to zero all the entries of the Global Matrix

485
486 double soluOld_gss = 0;
487 std::vector < double > gradSoluOld_gss (dim, 0.);
488
489
490 for (unsigned i = 0; i < nDofu; i++) {
491 solu_gss += phi[i] * solu[i];
492 soluOld_gss += phi[i] * soluOld[i];
493
494 for (unsigned k = 0; k < dim; k++) {
495 gradSolu_gss[k] += phi_x[i * dim + k] * solu[i];
496 gradSoluOld_gss[k] += phi_x[i * dim + k] * soluOld[i];
497 }
498 }
499
500 // *** phi_i loop ***
501 for (unsigned i = 0; i < nDofu; i++) {
502
503 adept::adouble graduGradphi = 0.;

505
506 for (unsigned k = 0; k < dim; k++) {
507 graduGradphi += phi_x[i * dim + k] * gradSolu_gss[k];
508 graduOldGradphi += phi_x[i * dim + k] * gradSoluOld_gss[k];
509 }
510
511 aRes[i] += ( (solu_gss - soluOld_gss) * phi[i] / dt + (1. * graduGradphi + 0. * graduOldGradphi)) * weight;
512
513 } // end phi_i loop
514 } // end gauss point loop
515
516 //--------------------------------------------------------------------------------------------------------
517 // Add the local Matrix/Vector into the global Matrix/Vector
518
519 //copy the value of the adept::adoube aRes in double Res and store
520 Res.resize (nDofu); //resize
521
522 for (int i = 0; i < nDofu; i++) {
523 Res[i] = - aRes[i].value();
524 }
525
526 RES->add_vector_blocked (Res, l2GMap);
527
528 // define the dependent variables
529 s.dependent (&aRes[0], nDofu);
530
531 // define the independent variables
532 s.independent (&solu[0], nDofu);
533
534 // get the jacobian matrix (ordered by row major )
535 Jac.resize (nDofu * nDofu); //resize
536 s.jacobian (&Jac[0], true);
537
538 //store K in the global matrix KK
539 KK->add_matrix_blocked (Jac, l2GMap, l2GMap);
540
541 s.clear_independents();
542 s.clear_dependents();
543
544 } //end element loop for each process
545
546 RES->close();
547
548 KK->close();
549
550 // ***************** END ASSEMBLY *******************
551}

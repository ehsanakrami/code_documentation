120void ChangeTriangleConfiguration2 (const std::vector<unsigned> & ENVN, std::vector <double> &angle) {
121 unsigned type = 3; // there are 2 or 3 leading angles
122 if (ENVN[0] < ENVN[1]) { // 0 leads on 1
123 if (ENVN[0] < ENVN[2]) type = 0; // 0 is leading angle
124 else if (ENVN[0] > ENVN[2]) type = 2; // 2 is leading angle
125 }
126 else if (ENVN[0] > ENVN[1]) { // 1 leads on 0
127 if (ENVN[1] < ENVN[2]) type = 1; // 1 is leading angle
128 else if (ENVN[1] > ENVN[2]) type = 2; // 2 is leading angle
129 }
130 else { // 0 equals 1
131 if (ENVN[0] > ENVN[2]) type = 2; // 2 is leading angle
132 }
133
134 double scale;
135 if (type == 0) {
136 scale = (M_PI - angle[0]) / (angle[1] + angle [2]);
137 angle[1] *= scale;
138 angle[2] *= scale;
139 }
140 else if (type == 1) {
141 scale = (M_PI - angle[1]) / (angle[0] + angle [2]);
142 angle[0] *= scale;
143 angle[2] *= scale;
144 }
145 else if (type == 2) {
146 scale = (M_PI - angle[2]) / (angle[1] + angle [0]);
147 angle[1] *= scale;
148 angle[0] *= scale;
149 }
150 else {
151 scale = M_PI / (angle[0] + angle[1] + angle[2]);
152 angle[0] *= scale;
153 angle[1] *= scale;
154 angle[2] *= scale;

158double GetPWillmoreEnergy (MultiLevelSolution &mlSol) {
159
160 unsigned level = mlSol.GetMLMesh()->GetNumberOfLevels() - 1;
161
162 Solution* sol = mlSol.GetSolutionLevel(level);
163 Mesh* msh = mlSol.GetMLMesh()->GetLevel(level);
164 elem* el = msh->el;
165
166 // Convenience variables to encode the dimension.
167 const unsigned dim = 2;
168 const unsigned DIM = 3;
169
170 // Get the process_id (for parallel computation).
171 unsigned iproc = msh->processor_id();
172
173 // Extract the solution vector; get solDx positions in the ml_sol object.
174 unsigned solDxIndex[DIM];
175 solDxIndex[0] = mlSol.GetIndex ("Dx1");
176 solDxIndex[1] = mlSol.GetIndex ("Dx2");
177 solDxIndex[2] = mlSol.GetIndex ("Dx3");
178
179 // Extract the finite element type for solx.
180 unsigned solxType;
181 solxType = mlSol.GetSolutionType (solDxIndex[0]);
182
183 // Define solx and solxOld.
184 std::vector < double > solx[DIM];
185 std::vector < double > solxOld[DIM];
186
187 // Get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC).
188 unsigned xType = 2;
189
190 // Get positions of Y in the ml_sol object.
191 unsigned solYIndex[DIM];
192 solYIndex[0] = mlSol.GetIndex ("Y1");
193 solYIndex[1] = mlSol.GetIndex ("Y2");
194 solYIndex[2] = mlSol.GetIndex ("Y3");
195
196 // Extract the finite element type for Y.
197 unsigned solYType;
198 solYType = mlSol.GetSolutionType (solYIndex[0]);
199
200
201 // Define solY and solYOld.
202 std::vector < double > solY[DIM];
203 std::vector < double > solYOld[DIM];
204
205 // Initialize area, volume, P-Willmore energy.
206
207 double energy = 0.;
208
209 // ELEMENT LOOP: each process loops only on the elements that it owns.
210 for (int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
211
212 // Number of solution element dofs.
213 short unsigned ielGeom = msh->GetElementType (iel);
214 unsigned nxDofs = msh->GetElementDofNumber (iel, solxType);
215 unsigned nYDofs = msh->GetElementDofNumber (iel, solYType);
216
217 // Resize solution vectors.
218 for (unsigned K = 0; K < DIM; K++) {
219 solx[K].resize (nxDofs);
220 solxOld[K].resize (nxDofs);
221 solY[K].resize (nYDofs);
222 solYOld[K].resize (nYDofs);
223 }
224
225
226 // Loop which handles local storage of global mapping and solution X.
227 for (unsigned i = 0; i < nxDofs; i++) {
228
229 // Global-to-local mapping between solution node and solution dof.
230 unsigned iDDof = msh->GetSolutionDof (i, iel, solxType);
231 unsigned iXDof = msh->GetSolutionDof (i, iel, xType);
232
233 for (unsigned K = 0; K < DIM; K++) {
234 solxOld[K][i] = (*msh->GetTopology()->_Sol[K]) (iXDof) + (*sol->_SolOld[solDxIndex[K]]) (iDDof);
235 solx[K][i] = (*msh->GetTopology()->_Sol[K]) (iXDof) + (*sol->_Sol[solDxIndex[K]]) (iDDof);
236 }
237 }
238
239 // Loop which handles local storage of global mapping and solution Y.
240 for (unsigned i = 0; i < nYDofs; i++) {
241
242 // Global-to-local mapping between solution node and solution dof.
243 unsigned iYDof = msh->GetSolutionDof (i, iel, solYType);
244 for (unsigned K = 0; K < DIM; K++) {
245
246 // Global-to-local solutions.
247 solYOld[K][i] = (*sol->_SolOld[solYIndex[K]]) (iYDof);
248 solY[K][i] = (*sol->_Sol[solYIndex[K]]) (iYDof);
249
250 }
251 }
252
253 // begin GAUSS POINT LOOP
254 for (unsigned ig = 0; ig < msh->_finiteElement[ielGeom][solxType]->GetGaussPointNumber(); ig++) {
255
256 const double *phix; // local test function
257 const double *phix_uv[dim]; // first order derivatives in (u,v)
258
259 const double *phiY; // local test function
260
261
262 double weight; // gauss point weight
263
264 //Extract Gauss point weight, test functions, and their partial derivatives.
265 // "0" is derivative in u, "1" is derivative in v.
266 weight = msh->_finiteElement[ielGeom][solxType]->GetGaussWeight (ig);
267
268 phix = msh->_finiteElement[ielGeom][solxType]->GetPhi (ig);
269 phix_uv[0] = msh->_finiteElement[ielGeom][solxType]->GetDPhiDXi (ig);
270 phix_uv[1] = msh->_finiteElement[ielGeom][solxType]->GetDPhiDEta (ig);
271
272 phiY = msh->_finiteElement[ielGeom][solYType]->GetPhi (ig);
273
274
275 double solYg[3] = {0., 0., 0.};
276 double solx_uv[3][2] = {{0., 0.}, {0., 0.}, {0., 0.}};
277
278 for (unsigned K = 0; K < DIM; K++) {
279 for (unsigned i = 0; i < nYDofs; i++) {
280 solYg[K] += phiY[i] * 0.5 * (solYOld[K][i] + solY[K][i]);
281 }
282 for (int j = 0; j < dim; j++) {
283 for (unsigned i = 0; i < nxDofs; i++) {
284 solx_uv[K][j] += phix_uv[j][i] * 0.5 * (solx[K][i] + solxOld[K][i]);
285 }
286 }
287 }
288
289 // Computing the metric, metric determinant, and area element.
290 double g[dim][dim] = {{0., 0.}, {0., 0.}};
291 for (unsigned i = 0; i < dim; i++) {
292 for (unsigned j = 0; j < dim; j++) {
293 for (unsigned K = 0; K < DIM; K++) {
294 g[i][j] += solx_uv[K][i] * solx_uv[K][j];
295 }
296 }
297 }
298 double detg = g[0][0] * g[1][1] - g[0][1] * g[1][0];
299 double Area = weight * sqrt (detg);
300
301 // Computing the unit normal vector N.
302 double normal[DIM];
303 normal[0] = normalSign * (solx_uv[1][0] * solx_uv[2][1]
304 - solx_uv[2][0] * solx_uv[1][1]) / sqrt (detg);
305 normal[1] = normalSign * (solx_uv[2][0] * solx_uv[0][1]
306 - solx_uv[0][0] * solx_uv[2][1]) / sqrt (detg);
307 normal[2] = normalSign * (solx_uv[0][0] * solx_uv[1][1]
308 - solx_uv[1][0] * solx_uv[0][1]) / sqrt (detg);
309
310 // Computing Y.N and |Y|^2, which are essentially 2H and 4H^2.
311 double YdotN = 0.;
312 double YdotY = 0.;
313 for (unsigned K = 0; K < DIM; K++) {
314 YdotN += solYg[K] * normal[K];
315 YdotY += solYg[K] * solYg[K];
316 }
317 // double signYdotN = (YdotN.value() >= 0.) ? 1. : -1.;
318 double signYdotN = 1.;
319
320 // Some necessary quantities when working with polynomials.
321 double sumP3 = 0.;
322 for (unsigned p = 0; p < 3; p++) {
323 double signP = (P[p] % 2u == 0) ? 1. : signYdotN;
324 sumP3 += signP * ap[p] * pow (YdotY, P[p] / 2.);
325 }
326
327 energy += sumP3 * Area;
328
329 } // end GAUSS POINT LOOP.
330
331
332 } // End ELEMENT LOOP for each process.
333
334
335 double energyAll;
336 MPI_Reduce (&energy, &energyAll, 1, MPI_DOUBLE, MPI_SUM, 0, MPI_COMM_WORLD);
337
338 return energyAll;

90 MyVector <unsigned> rowSizeElDof(_nel);
91 MyVector <unsigned> rowSizeElNearFace(_nel);
92 unsigned jel = 0;
93 for (unsigned isdom = 0; isdom < elc->_nprocs; isdom++) {
94 elc->_elementType.broadcast(isdom);
95 for (unsigned iel = elc->_elementType.begin(); iel < elc->_elementType.end(); iel++) {
96 short unsigned elType = elc->_elementType[iel];
97 int increment = 1;
98 if (static_cast < short unsigned >(coarseAmrVector[iel] + 0.25) == 1) {
99 increment = NRE[elType];
100 }
101 for (unsigned j = 0; j < increment; j++) {
102 rowSizeElDof[jel + j] += NVE[elType][2];
103 rowSizeElNearFace[jel + j] += NFC[elType][1];
104 }
105 jel += increment;
106 }
107 elc->_elementType.clearBroadcast();
108 }
109
110 _elementDof = MyMatrix <unsigned> (rowSizeElDof);
111
112 _elementNearFace = MyMatrix <int> (rowSizeElNearFace, -1);

439 void elem::BuildElementNearFace() {
440
441 for(unsigned iel = 0; iel < GetElementNumber(); iel++) {
442 for(unsigned iface = 0; iface < GetElementFaceNumber(iel); iface++) {
443 if(GetFaceElementIndex(iel, iface) <= 0) {
444 unsigned i1 = GetFaceVertexIndex(iel, iface, 0);
445 unsigned i2 = GetFaceVertexIndex(iel, iface, 1);
446 unsigned i3 = GetFaceVertexIndex(iel, iface, 2);
447
448 for(unsigned j = 0; j < GetElementNearVertexNumber(i1); j++) {
449 unsigned jel = GetElementNearVertex(i1, j);
450
451 if(jel > iel) {
452 for(unsigned jface = 0; jface < GetElementFaceNumber(jel); jface++) {
453 if(GetFaceElementIndex(jel, jface) <= 0) {
454 unsigned j1 = GetFaceVertexIndex(jel, jface, 0);
455 unsigned j2 = GetFaceVertexIndex(jel, jface, 1);
456 unsigned j3 = GetFaceVertexIndex(jel, jface, 2);
457 unsigned j4 = GetFaceVertexIndex(jel, jface, 3);
458
459 const bool faces_coincide_three_dim = ( GetDimension() == 3 &&
460 (i1 == j1 || i1 == j2 || i1 == j3 || i1 == j4) &&
461 (i2 == j1 || i2 == j2 || i2 == j3 || i2 == j4) &&
462 (i3 == j1 || i3 == j2 || i3 == j3 || i3 == j4));
463
464 const bool faces_coincide_two_dim = ( GetDimension() == 2 &&
465 (i1 == j1 || i1 == j2) &&
466 (i2 == j1 || i2 == j2));
467
468 const bool faces_coincide_one_dim = ( GetDimension() == 1 && (i1 == j1));
469
470 if(faces_coincide_three_dim
471 ||
472 faces_coincide_two_dim
473 ||
474 faces_coincide_one_dim
475 ) {
476 SetFaceElementIndex(iel, iface, jel + 1u);
477 SetFaceElementIndex(jel, jface, iel + 1u);
478 }
479 }
480 }
481 }
482 }
483 }

495 void elem::BuildElementNearElement()
496 {
497 MyVector < unsigned > rowSize(_elementOffset, 1);
498 for (unsigned iel = rowSize.begin(); iel < rowSize.end(); iel++) {
499 std::map< unsigned, bool> elements;
500 for (unsigned i = 0; i < GetElementDofNumber(iel, 0); i++) {
501 unsigned inode = GetElementDofIndex(iel, i);
502 for (unsigned j = _elementNearVertex.begin(inode); j < _elementNearVertex.end(inode); j++) {
503 elements[_elementNearVertex[inode][j]] = true;
504 }
505 }
506 rowSize[iel] = elements.size();
507 }
508
509 _elementNearElement = MyMatrix <unsigned> (rowSize, UINT_MAX);
510
511 for (unsigned iel = _elementNearElement.begin(); iel < _elementNearElement.end(); iel++) {
512 std::map< unsigned, bool> elements;
513 _elementNearElement[iel][0] = iel;
514 for (unsigned i = 0; i < GetElementDofNumber(iel, 0); i++) {
515 unsigned inode = GetElementDofIndex(iel, i);
516 for (unsigned j = _elementNearVertex.begin(inode); j < _elementNearVertex.end(inode); j++) {
517 if (_elementNearVertex[inode][j] != iel) {
518 elements[_elementNearVertex[inode][j]] = true;
519 }
520 }
521 }
522 unsigned j = 1;
523 for (std::map<unsigned, bool>::iterator it = elements.begin(); it != elements.end(); it++, j++) {
524 _elementNearElement[iel][j] = it->first;
525 }
526 }
527
528 }

531 void elem::BuildElementNearVertex()
532 {
533 MyVector <unsigned> rowSize(_nvt, 0);
534 for (unsigned iel = 0; iel < _nel; iel++) {
535 for (unsigned inode = 0; inode < GetElementDofNumber(iel, 0); inode++) {
536 rowSize[GetElementDofIndex(iel, inode)]++;
537 }
538 }
539 _elementNearVertex = MyMatrix <unsigned> (rowSize, _nel);
540 for (unsigned iel = 0; iel < _nel; iel++) {
541 for (unsigned inode = 0; inode < GetElementDofNumber(iel, 0); inode++) {
542 unsigned irow = GetElementDofIndex(iel, inode);
543 unsigned j = 0;
544 while (_nel != _elementNearVertex[irow][j]) j++;
545 _elementNearVertex[irow][j] = iel;
546 }
547 }

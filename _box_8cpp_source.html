80void Box::TransformPointToRef(const double* x_in,double* x_out) const {
81
82 double thetaz = _domain_rtmap.get("thetaz");
83 double transfl = _domain_rtmap.get("transflag");
84
85 double xpr[3] = {0.,0.,0.};
86 double Dx[3] = {0.,0.,0.};
87
88 xpr[0] = x_in[0]*cos(thetaz) + x_in[1]*sin(thetaz);
89 xpr[1] = x_in[1]*cos(thetaz) - x_in[0]*sin(thetaz);
90
91 Dx[0]=0.5*(_le[0]-_lb[0])*transfl;
92 Dx[1]=0.5*(_le[1]-_lb[1])*transfl;
93 x_out[0]=xpr[0] + Dx[0];
94 x_out[1]=xpr[1] + Dx[1];
95
96 if (_spacedim == 3) {
97 Dx[2]=0.5*(_le[2]-_lb[2])*transfl;
98 xpr[2]=x_in[2]; //no rotation wrt planes parallel to xy occurs
99 x_out[2]=xpr[2] + Dx[2];
100 }
101
102 return;
103}

693 void Mesh::dofmap_Node_based_dof_offsets_Complete_linear_quadratic() {
694
695 //BEGIN completing k = 0, 1
696
697 for(unsigned k = 0; k < 2; k++) {
698
699 std::vector < unsigned > ownedGhostCounter(_nprocs , 0);
700 unsigned counter = 0;
701
702 _originalOwnSize[k].resize(_nprocs);
703
704 for(int isdom = 0; isdom < _nprocs; isdom++) {
705
706 //owned nodes
707 for(unsigned inode = _dofOffset[2][isdom]; inode < _ownSize[k][isdom] + _dofOffset[2][isdom]; inode++) {
708 counter++;
709 }
710
711 for(unsigned inode = 0; inode < _ghostDofs[k][isdom].size(); inode++) {
712 unsigned ghostNode = _ghostDofs[k][isdom][inode];
713
714 unsigned ksdom = IsdomBisectionSearch(ghostNode, 2);
715
716 int upperBound = _dofOffset[2][ksdom] + _ownSize[k][ksdom];
717
718 if(ghostNode < upperBound) {
719 _ghostDofs[k][isdom][inode] = ghostNode - _dofOffset[2][ksdom] + _dofOffset[k][ksdom];
720 }
721 else if(_ownedGhostMap[k].find(ghostNode) != _ownedGhostMap[k].end()) {
722 _ghostDofs[k][isdom][inode] = _ownedGhostMap[k][ghostNode];
723 }
724 else { // owned ghost nodes
725 _ownedGhostMap[k][ ghostNode ] = counter;
726 counter++;
727 ownedGhostCounter[isdom]++;
728
729 for(unsigned jnode = inode; jnode < _ghostDofs[k][isdom].size() - 1; jnode++) {
730 _ghostDofs[k][isdom][jnode] = _ghostDofs[k][isdom][jnode + 1];
731 }
732
733 _ghostDofs[k][isdom].resize(_ghostDofs[k][isdom].size() - 1);
734 inode--;
735 }
736 }
737
738 _originalOwnSize[k][isdom] = _ownSize[k][isdom];
739 _ownSize[k][isdom] += ownedGhostCounter[isdom];
740 _dofOffset[k][isdom + 1] = _dofOffset[k][isdom] + _ownSize[k][isdom];
741 }
742 }
743
744 //END completing for k = 0, 1

903

906 switch(solType) {
907
908 case 0: { // linear Lagrange
909 unsigned iNode = el->GetElementDofIndex(iel, i); //GetMeshDof(iel, i, solType);
910 unsigned isdom = IsdomBisectionSearch(iNode, 2);
911
912 if(iNode < _dofOffset[2][isdom] + _originalOwnSize[0][isdom]) {
913 dof = (iNode - _dofOffset[2][isdom]) + _dofOffset[0][isdom];
914 }
915 else {
916 dof = _ownedGhostMap[0].find(iNode)->second;
917 }
918 }
919 break;
920
921 case 1: { // quadratic Lagrange
922 unsigned iNode = el->GetElementDofIndex(iel, i); //GetMeshDof(iel, i, solType);
923 unsigned isdom = IsdomBisectionSearch(iNode, 2);
924

926 dof = (iNode - _dofOffset[2][isdom]) + _dofOffset[1][isdom];
927 }
928 else {
929 dof = _ownedGhostMap[1].find(iNode)->second;
930 }
931 }
932 break;
933
934 case 2: // bi-quadratic Lagrange
935 dof = el->GetElementDofIndex(iel, i); //GetMeshDof(iel, i, solType);
936 break;
937
938 case 3: // piecewise constant
939 // in this case use i=0
940 dof = iel;
941 break;
942
943 case 4: // piecewise linear discontinuous
944 unsigned isdom = IsdomBisectionSearch(iel, 3);

947 unsigned ownSize = offsetp1 - offset;
948 unsigned offsetPWLD = offset * (_dimension + 1);
949 unsigned locIel = iel - offset;

952 }

389 void CurrentElem<real_num_mov>::set_elem_center_3d(const unsigned int iel, const unsigned int solType_coords) {
390
391 unsigned nDofx = _mesh_new->GetElementDofNumber(iel, solType_coords);
392
393 std::fill(_elem_center_3d.begin(), _elem_center_3d.end(), 0.);
394
395 for (unsigned j = 0; j < 3; j++) {
396 for (unsigned i = 0; i < nDofx; i++) {
397 _elem_center_3d[j] += _coords_at_dofs_3d[j][i];
398 }
399 }
400
401 for (unsigned j = 0; j < 3; j++) { _elem_center_3d[j] = _elem_center_3d[j]/nDofx; }
402
403 }

405
406// since the Mesh is Bi-TriQuadratic order, here I need the formula for the center of an Edge, a Triangle, a Quadrilateral
407// On the other hand, instead of COMPUTING this quantity, I'd rather RETRIEVE IT from the LAST NODE!
408template < typename real_num_mov >
409 void CurrentElem<real_num_mov>::set_elem_center_bdry_3d() {
410
411
412 std::fill(_elem_center_bdry_3d.begin(), _elem_center_bdry_3d.end(), 0.);
413
414 for (unsigned j = 0; j < _space_dim; j++) {
415 for (unsigned i = 0; i < _coords_at_dofs_bdry_3d[j].size(); i++) {
416 _elem_center_bdry_3d[j] += _coords_at_dofs_bdry_3d[j][i];
417 }
418 }
419
420 for (unsigned j = 0; j < _space_dim; j++) {
421 _elem_center_bdry_3d[j] = _elem_center_bdry_3d[j]/_coords_at_dofs_bdry_3d[j].size();
422 }

210template < typename real_num_mov >
211 void CurrentElem<real_num_mov>::SetDofobjConnCoords() {
212
213 const uint mydim = _mesh.get_dim();
214 const uint el_nnodes = _el_conn.size();
215
216 for (uint n=0; n<el_nnodes; n++) {
217
218 _el_conn[n] = _mesh._el_map[_mesh_vb][( _iel + _mesh._off_el[_mesh_vb][_mesh._NoLevels*_proc + _Level] )*el_nnodes+n];
219
220 for (uint idim=0; idim < mydim; idim++) {
221 const uint indxn = n+idim*el_nnodes;
222 _xx_nds[indxn] = _mesh._xyz[_el_conn[n]+idim*_mesh._NoNodesXLev[_mesh._NoLevels-1]];
223 }
224 }
225
226
227 int sum_elems_prev_sd_at_lev = 0;
228 for (uint pr = 0; pr< _proc; pr++) { sum_elems_prev_sd_at_lev += _mesh._off_el[_mesh_vb][_mesh._NoLevels*pr + _Level + 1] - _mesh._off_el[_mesh_vb][ _mesh._NoLevels*pr + _Level]; }
229 uint iel_DofObj = _iel + sum_elems_prev_sd_at_lev;
230 if (_mesh_vb == VV) { _vol_iel_DofObj = iel_DofObj; }
231 else if (_mesh_vb == BB) { _vol_iel_DofObj = _mesh._el_bdry_to_vol[_Level][iel_DofObj]; }
232
233
234
235 return;
236 }

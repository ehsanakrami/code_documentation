51 void LinearEquationSolverPetsc::BuildBdcIndex (const vector <unsigned>& variable_to_be_solved) {
52
53 _bdcIndexIsInitialized = 1;
54
55 unsigned BDCIndexSize = KKoffset[KKIndex.size() - 1][processor_id()] - KKoffset[0][processor_id()];
56 _bdcIndex.resize (BDCIndexSize);
57
58 vector <bool> ThisSolutionIsIncluded (_SolPdeIndex.size(), false);
59
60 for (unsigned iind = 0; iind < variable_to_be_solved.size(); iind++) {
61 unsigned PdeIndexSol = variable_to_be_solved[iind];
62 ThisSolutionIsIncluded[PdeIndexSol] = true;
63 }
64
65 unsigned count0 = 0;
66
67 for (int k = 0; k < _SolPdeIndex.size(); k++) {
68 unsigned indexSol = _SolPdeIndex[k];
69 unsigned soltype = _SolType[indexSol];
70
71 for (unsigned inode_mts = _msh->_dofOffset[soltype][processor_id()];
72 inode_mts < _msh->_dofOffset[soltype][processor_id() + 1]; inode_mts++) {
73 int local_mts = inode_mts - _msh->_dofOffset[soltype][processor_id()];
74 int idof_kk = KKoffset[k][processor_id()] + local_mts;
75
76 if (!ThisSolutionIsIncluded[k] || (* (*_Bdc) [indexSol]) (inode_mts) < 1.5) {
77 _bdcIndex[count0] = idof_kk;
78 count0++;
79 }
80 }
81 }
82
83 _bdcIndex.resize (count0);
84 std::vector < PetscInt > (_bdcIndex).swap (_bdcIndex);
85 std::sort (_bdcIndex.begin(), _bdcIndex.end());
86
87 return;
88 }

92 void LinearEquationSolverPetsc::Solve (const vector <unsigned>& variable_to_be_solved, const bool& ksp_clean) {
93
94 PetscLogDouble t1;
95 PetscTime (&t1);
96
97 if (_bdcIndexIsInitialized == 0) BuildBdcIndex (variable_to_be_solved);
98
99 //BEGIN ASSEMBLE matrix with Dirichlet penalty BCs by penalty
100 Mat KK = (static_cast<PetscMatrix*> (_KK))->mat();
101 if (ksp_clean) {
102 this->Clear();
103 SetPenalty();
104 RemoveNullSpace();
105 this->Init (KK, KK);
106 }
107 //END ASSEMBLE
108
109// PetscViewer viewer;
110// PetscViewerDrawOpen(PETSC_COMM_WORLD,NULL,NULL,0,0,900,900,&viewer);
111// PetscObjectSetName((PetscObject)viewer,"FSI matrix");
112// PetscViewerPushFormat(viewer,PETSC_VIEWER_DRAW_LG);
113// MatView(KK,viewer);
114// double a;
115// std::cin>>a;
116
117
118 //BEGIN SOLVE and UPDATE
119 ZerosBoundaryResiduals();
120 KSPSolve (_ksp, (static_cast< PetscVector* > (_RES))->vec(), (static_cast< PetscVector* > (_EPSC))->vec());
121 *_EPS += *_EPSC;
122 _RESC->matrix_mult (*_EPSC, *_KK);
123 *_RES -= *_RESC;
124 //END SOLVE and UPDATE
125
126 //BEGIN PRINT Computational info
127 if (_printSolverInfo) {
128 int its;
129 KSPGetIterationNumber (_ksp, &its);
130
131 KSPConvergedReason reason;
132 KSPGetConvergedReason (_ksp, &reason);
133
134 PetscReal rnorm;
135 KSPGetResidualNorm (_ksp, &rnorm);
136
137 PetscLogDouble t2;
138 PetscTime (&t2);
139
140 PetscPrintf (PETSC_COMM_WORLD, " *************** ML linear solver time: %e \n", t2 - t1);
141 PetscPrintf (PETSC_COMM_WORLD, " *************** Number of outer ksp solver iterations = %i \n", its);
142 PetscPrintf (PETSC_COMM_WORLD, " *************** Convergence reason = %i \n", reason);
143 PetscPrintf (PETSC_COMM_WORLD, " *************** Residual norm = %10.8g \n", rnorm);
144 }
145
146 //END PRINT
147 }

151 void LinearEquationSolverPetsc::Init (Mat& Amat, Mat& Pmat) {
152
153 if (!this->initialized()) {
154 this->_is_initialized = true;
155
156 KSPCreate (MPI_COMM_WORLD, &_ksp);
157 KSPGetPC (_ksp, &_pc);
158
159 this->SetSolver (_ksp, _levelSolverType);
160
161 KSPSetOperators (_ksp, Amat, Pmat);
162 KSPSetTolerances (_ksp, _rtol, _abstol, _dtol, _maxits);
163
164 if (_levelSolverType != PREONLY) {
165 KSPSetInitialGuessKnoll (_ksp, PETSC_TRUE);
166 KSPSetNormType (_ksp, KSP_NORM_PRECONDITIONED);
167 }
168 else {
169 KSPSetInitialGuessKnoll (_ksp, PETSC_FALSE);
170 KSPSetNormType (_ksp, KSP_NORM_NONE);
171 }
172
173 KSPSetFromOptions (_ksp);
174 KSPGMRESSetRestart (_ksp, _restart);
175
176 SetPreconditioner (_ksp, _pc);
177
178 }

183 void LinearEquationSolverPetsc::MGInit (const MgSmootherType& mg_smoother_type, const unsigned& levelMax, const SolverType & mgSolverType) {
184
185 KSPCreate (PETSC_COMM_WORLD, &_ksp);
186
187 _mgSolverType = mgSolverType;
188 double otherRCF = _richardsonScaleFactor;
189 _richardsonScaleFactor = .99999;
190 SetSolver (_ksp, _mgSolverType);
191 _richardsonScaleFactor = otherRCF;
192
193 KSPGetPC (_ksp, &_pc);
194 PCSetType (_pc, PCMG);
195 PCMGSetLevels (_pc, levelMax, NULL);
196
197 if (mg_smoother_type == FULL) {
198 PCMGSetType (_pc, PC_MG_FULL);
199 }
200 else if (mg_smoother_type == MULTIPLICATIVE) {
201 PCMGSetType (_pc, PC_MG_MULTIPLICATIVE);
202 }
203 else if (mg_smoother_type == ADDITIVE) {
204 PCMGSetType (_pc, PC_MG_ADDITIVE);
205 }
206 else if (mg_smoother_type == KASKADE) {
207 PCMGSetType (_pc, PC_MG_KASKADE);
208 }
209 else {
210 std::cout << "Wrong mg_type for PETSCsolve()" << std::endl;
211 abort();
212 }
213 };

217 void LinearEquationSolverPetsc::MGSetLevel (
218 LinearEquationSolver* LinSolver, const unsigned& levelMax,
219 const vector <unsigned>& variable_to_be_solved, SparseMatrix* PP, SparseMatrix* RR,
220 const unsigned& npre, const unsigned& npost) {
221
222 unsigned level = _msh->GetLevel();
223
224 // ***************** NODE/ELEMENT SEARCH *******************
225 if (_bdcIndexIsInitialized == 0) BuildBdcIndex (variable_to_be_solved);
226 // ***************** END NODE/ELEMENT SEARCH *******************
227
228 KSP* kspMG = LinSolver->GetKSP();
229 PC pcMG;
230 KSPGetPC (*kspMG, &pcMG);
231
232 KSP subksp;
233
234 if (level == 0) {
235 PCMGGetCoarseSolve (pcMG, &subksp);
236 KSPSetTolerances (subksp, PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT, npre);
237 }
238 else {
239 PCMGGetSmoother (pcMG, level , &subksp);
240 KSPSetTolerances (subksp, PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT, npre);
241 }
242
243 if (level != 0 && _levelSolverType == PREONLY) {
244 _levelSolverType = RICHARDSON;
245 _richardsonScaleFactor = 1.;
246 }
247
248 this->SetSolver (subksp, _levelSolverType);
249 std::ostringstream levelName;
250 levelName << "level-" << level;
251 KSPSetOptionsPrefix (subksp, levelName.str().c_str());
252 KSPSetFromOptions (subksp);
253
254 //ZerosBoundaryResiduals();
255
256 SetPenalty();
257 RemoveNullSpace();
258
259 Mat KK = (static_cast< PetscMatrix* > (_KK))->mat();
260
261 KSPSetOperators (subksp, KK, KK);
262
263 PC subpc;
264 KSPGetPC (subksp, &subpc);
265 SetPreconditioner (subksp, subpc);
266
267 if (level < levelMax) {
268 PCMGSetX (pcMG, level, (static_cast< PetscVector* > (_EPS))->vec());
269 PCMGSetRhs (pcMG, level, (static_cast< PetscVector* > (_RES))->vec());
270 }
271
272 if (level > 0) {
273 PCMGSetR (pcMG, level, (static_cast<PetscVector*> (_RESC))->vec());
274 PCMGSetInterpolation (pcMG, level, (static_cast< PetscMatrix* > (PP))->mat());
275 PCMGSetRestriction (pcMG, level, (static_cast< PetscMatrix* > (RR))->mat());
276
277 if (npre != npost) {
278 KSP subkspUp;
279 PCMGGetSmootherUp (pcMG, level , &subkspUp);
280 KSPSetTolerances (subkspUp, PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT, npost);
281 this->SetSolver (subkspUp, _levelSolverType);
282 KSPSetPC (subkspUp, subpc);
283 PC subpcUp;
284 KSPGetPC (subkspUp, &subpcUp);
285 KSPSetUp (subkspUp);
286 }
287 }

355 void LinearEquationSolverPetsc::RemoveNullSpace() {
356
357 if (_msh->GetLevel() != 0) {
358 std::vector < Vec > nullspBase;
359 GetNullSpaceBase (nullspBase);
360 if (nullspBase.size() != 0) {
361 MatNullSpace nullsp;
362 MatNullSpaceCreate (PETSC_COMM_WORLD, PETSC_FALSE, nullspBase.size(), &nullspBase[0], &nullsp);
363
364 PetscBool isNull;
365 MatNullSpaceTest (nullsp, (static_cast< PetscMatrix* > (_KK))->mat(), &isNull);
366 if (!isNull) std::cout << "The null space created for KK is not correct!" << std::endl;
367
368 MatSetNullSpace ( (static_cast< PetscMatrix* > (_KK))->mat(), nullsp);
369 MatSetTransposeNullSpace ( (static_cast< PetscMatrix* > (_KK))->mat(), nullsp);
370 MatNullSpaceDestroy (&nullsp);
371
372 for (unsigned i = 0; i < nullspBase.size(); i++) {
373 VecDestroy (&nullspBase[i]);
374 }
375 }
376 }
377
378 }

382 void LinearEquationSolverPetsc::GetNullSpaceBase (std::vector < Vec >& nullspBase) {
383
384 for (int k = 0; k < _SolPdeIndex.size(); k++) {
385 unsigned indexSol = _SolPdeIndex[k];
386
387 if (_solution->GetIfRemoveNullSpace (indexSol)) {
388 Vec EPS = (static_cast< PetscVector* > (_EPS))->vec();
389 unsigned nullspSize = nullspBase.size();
390 nullspBase.resize (nullspSize + 1);
391 VecDuplicate (EPS, &nullspBase[nullspSize]);
392
393 unsigned soltype = _SolType[indexSol];
394 unsigned owndofs = _msh->_dofOffset[soltype][processor_id() + 1] - _msh->_dofOffset[soltype][processor_id()];
395 if (soltype == 4) owndofs /= (_msh->GetDimension() + 1);
396
397 for (unsigned i = 0; i < owndofs; i++) {
398 int idof_kk = KKoffset[k][processor_id()] + i;
399 unsigned inode_mts = _msh->_dofOffset[soltype][processor_id()] + i;
400 if ( (* (*_Bdc) [indexSol]) (inode_mts) > 1.9) {
401 VecSetValue (nullspBase[nullspSize], idof_kk, 1., INSERT_VALUES);
402 }
403 }
404
405 VecAssemblyBegin (nullspBase[nullspSize]);
406 VecAssemblyEnd (nullspBase[nullspSize]);
407 VecNormalize (nullspBase[nullspSize], NULL);
408
409 nullspSize++;
410 }
411 }
412 }

463

472 return;
473

475 ierr = KSPSetType (ksp, (char*) KSPTCQMR);

488

492 return;

541 std::pair<unsigned int, double> LinearEquationSolverPetsc::solve (SparseMatrix& matrix_in,
542 SparseMatrix& precond_in, NumericVector& solution_in, NumericVector& rhs_in,
543 const double tol, const unsigned int m_its) {
544
545// START_LOG("solve()", "PetscLinearSolverM");
546 // Make sure the data passed in are really of Petsc types
547 PetscMatrix* matrix = libmeshM_cast_ptr<PetscMatrix*> (&matrix_in);
548 PetscMatrix* precond = libmeshM_cast_ptr<PetscMatrix*> (&precond_in);
549 PetscVector* solution = libmeshM_cast_ptr<PetscVector*> (&solution_in);
550 PetscVector* rhs = libmeshM_cast_ptr<PetscVector*> (&rhs_in);
551 this->init (matrix);
552
553 int ierr = 0;
554 int its = 0, max_its = static_cast<int> (m_its);
555 PetscReal final_resid = 0.;
556 // Close the matrices and vectors in case this wasn't already done.
557 matrix->close();
558 precond->close();
559 solution->close();
560 rhs->close();
561
562// // If matrix != precond, then this means we have specified a
563// // special preconditioner, so reset preconditioner type to PCMAT.
564// if (matrix != precond)
565// {
566// this->_preconditioner_type = USER_PRECOND;
567// this->set_petsc_preconditioner_type ();
568// }
569 if (this->_preconditioner) this->_preconditioner->set_matrix (matrix_in);
570
571 // 2.2.1 & newer style
572 // Set operators. The input matrix works as the preconditioning matrix
573 if (!this->same_preconditioner) {
574 //ierr = KSPSetOperators(_ksp, matrix->mat(), precond->mat(),SAME_NONZERO_PATTERN);
575 ierr = KSPSetOperators (_ksp, matrix->mat(), precond->mat()); //PETSC3p5
576 CHKERRABORT (MPI_COMM_WORLD, ierr);
577 }
578 else {
579 //ierr = KSPSetOperators(_ksp, matrix->mat(), precond->mat(),SAME_PRECONDITIONER);
580 ierr = KSPSetOperators (_ksp, matrix->mat(), precond->mat()); //PETSC3p5
581 CHKERRABORT (MPI_COMM_WORLD, ierr);
582 }
583
584 // Set the tolerances for the iterative solver. Use the user-supplied
585 // tolerance for the relative residual & leave the others at default values.
586 ierr = KSPSetTolerances (_ksp, tol, PETSC_DEFAULT, PETSC_DEFAULT, max_its);
587 CHKERRABORT (MPI_COMM_WORLD, ierr);
588 // Solve the linear system
589
590// PetscLogEvent USER_EVENT;
591// PetscLogDouble user_event_flops;
592// PetscLogEventRegister("User event",0,&USER_EVENT);
593// PetscLogEventBegin(USER_EVENT,0,0,0,0);
594
595
596 ierr = KSPSolve (_ksp, rhs->vec(), solution->vec());
597 CHKERRABORT (MPI_COMM_WORLD, ierr);
598// PetscLogFlops(user_event_flops);
599// PetscLogEventEnd(USER_EVENT,0,0,0,0);
600
601 // Get the number of iterations required for convergence
602 ierr = KSPGetIterationNumber (_ksp, &its);
603 CHKERRABORT (MPI_COMM_WORLD, ierr);
604 // Get the norm of the final residual to return to the user.
605 ierr = KSPGetResidualNorm (_ksp, &final_resid);
606 CHKERRABORT (MPI_COMM_WORLD, ierr);
607
608// STOP_LOG("solve()", "PetscLinearSolverM");
609 return std::make_pair (its, final_resid);
610 }

390 void Mesh::SetCharacteristicLengthOfCoarsestLevel() {
391
392 //compute max and min coords -----------
393 std::vector < double > xMax(3, 0.);
394 std::vector < double > xMin(3, 0.);
395 for(unsigned i = 0; i < _coords[0].size(); i++) {
396 for(unsigned k = 0; k < 3; k++) {
397 if(xMax[k] < _coords[k][i]) xMax[k] = _coords[k][i];
398 if(xMin[k] > _coords[k][i]) xMin[k] = _coords[k][i];
399 }
400 }
401 //compute max and min coords - end -----------
402
403 _cLength = sqrt(pow(xMax[0] - xMin[0], 2) + pow(xMax[1] - xMin[1], 2) + pow(xMax[2] - xMin[2], 2));
404

1009
1010 unsigned dof;
1011
1012 switch(solType) {
1013
1014 case CONTINUOUS_LINEAR: { // linear Lagrange

287void MultiLevelMesh::RefineMesh( const unsigned short &igridn,
288 const unsigned short &igridr,
289 bool (* SetRefinementFlag)(const std::vector < double >& x, const int &ElemGroupNumber, const int &level) )
290{
291
292 _gridn0 = igridn;
293
294 _level0.resize(_gridn0);
295
296 //totally refined meshes **************
297 RefineMeshesTotally(igridr);
298
299
300 //partially refined meshes **************
301 RefineMeshesPartially(igridr, SetRefinementFlag);
302
303
304 // copy _level0 into the new levels _level **************
305 CopyLevelsZeroIntoNewLevels();
306
307}

311 void MultiLevelMesh::RefineMeshesTotally(const unsigned short &igridr) {
312
313 for (unsigned i = 1; i < igridr; i++) {
314 MeshRefinement meshcoarser(*_level0[i-1u]);
315 meshcoarser.FlagAllElementsToBeRefined();
316
317 _level0[i] = new Mesh();
318 MeshRefinement meshfiner(*_level0[i]);
319 meshfiner.RefineMesh(i, _level0[i-1], _finiteElement);
320 }
321
322
323 }

326 void MultiLevelMesh::RefineMeshesPartially(const unsigned short &igridr,
327 bool (* SetRefinementFlag)(const std::vector < double > &x, const int &ElemGroupNumber,const int &level) ) {
328
329
330 if(SetRefinementFlag == NULL) {
331
332 }
333 else{
334 Mesh::_SetRefinementFlag = SetRefinementFlag;
335 Mesh::set_is_refinement_function_defined(true);
336 }
337
338 //Here there are two alternatives:
339 //When igridn is STRICTLY LARGER THAN igridr, one must provide a refinement flag function, ...
340 //When igridn is SMALLER THAN OR EQUAL igridr, this loop is ignored
341 for (unsigned i = igridr; i < _gridn0; i++) {
342
343 if(Mesh::get_is_refinement_function_defined() == false) {
344 std::cout << "Set Refinement Region flag is not defined! " << std::endl;
345 exit(1);
346 }
347 else {
348 MeshRefinement meshcoarser(*_level0[i-1u]);
349 meshcoarser.FlagElementsToBeRefined();
350 //meshcoarser.FlagOnlyEvenElementsToBeRefined();
351 }
352
353 _level0[i] = new Mesh();
354 MeshRefinement meshfiner(*_level0[i]);
355 meshfiner.RefineMesh(i,_level0[i-1],_finiteElement);
356
357 }

422void MultiLevelMesh::AddAMRMeshLevel()
423{
424
425 //AMR refine mesh
426 _level0.resize(_gridn0+1u);
427
428 MeshRefinement meshcoarser(*_level0[_gridn0-1u]);
429 meshcoarser.FlagElementsToBeRefined();
430
431 _level0[_gridn0] = new Mesh();
432 MeshRefinement meshfiner(*_level0[_gridn0]);
433 meshfiner.RefineMesh(_gridn0,_level0[_gridn0-1u],_finiteElement);
434
435 _level.resize(_gridn+1u);
436 _level[_gridn]=_level0[_gridn0];
437
438 _gridn0++;
439 _gridn++;
440}

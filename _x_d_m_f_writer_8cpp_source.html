67 void XDMFWriter::Write( const std::string output_path, const char order[], const std::vector<std::string>& vars, const unsigned time_step ) {
68
69#ifdef HAVE_HDF5
70
71 bool print_all = 0;
72 for( unsigned ivar = 0; ivar < vars.size(); ivar++ ) {
73 print_all += !( vars[ivar].compare( "All" ) ) + !( vars[ivar].compare( "all" ) ) + !( vars[ivar].compare( "ALL" ) );
74 }
75
76 unsigned index_nd = 0;
77 if( !strcmp( order, "linear" ) ) { //linear
78 index_nd = 0;
79 }
80 else if( !strcmp( order, "quadratic" ) ) { //quadratic
81 index_nd = 1;
82 }
83 else if( !strcmp( order, "biquadratic" ) ) { //tensor-product quadratic (real and fake)
84 index_nd = 2;
85 }
86
87 Mesh* mesh = _ml_mesh->GetLevel( _gridn - 1 );
88 Solution* solution = _ml_sol->GetSolutionLevel( _gridn - 1 );
89
91 std::string type_elem;
92 unsigned iel0 = mesh->_elementOffset[_iproc];
93 unsigned elemtype = mesh->GetElementType( iel0 );
94
95 type_elem = XDMFWriter::type_el[index_nd][elemtype];
96
97 if( type_elem.compare( "Not_implemented" ) == 0 )
98 {
99 std::cerr << "XDMF-Writer error: element type not supported!" << std::endl;
100 abort();
101 }
102
103 unsigned nvt = mesh->dofmap_get_dof_offset(index_nd, _nprocs);
104 unsigned nel = mesh->GetNumberOfElements();
105 unsigned dim = mesh->GetDimension();
106 unsigned maxDim = ( nvt > ( dim + 1 ) * nel ) ? nvt : ( dim + 1 ) * nel;
107 unsigned ndofs = mesh->el->GetNVE( elemtype, index_nd );
108
109 std::vector < int > var_conn( nel * ndofs );
110
111 std::vector < double > vector1;
112 vector1.reserve( nvt );
113
114 std::vector < double > vector2;
115 vector2.reserve( maxDim );
116
117 NumericVector* numVector = NumericVector::build().release();
118 numVector->init( nvt, mesh->dofmap_get_own_size(index_nd, _iproc), true, AUTOMATIC );
119
120 //BEGIN XMF FILE PRINT
121
122 std::string filename_prefix;
123 if( _ml_sol != NULL ) filename_prefix = "sol";
124 else filename_prefix = "mesh";
125
126 std::ostringstream xdmf_filename;
127 xdmf_filename << output_path << "/" << filename_prefix << ".level" << _gridn << "." << time_step << "." << order << ".xmf";
128
129 std::ofstream fout;
130
131 if( _iproc != 0 ) {
132 fout.rdbuf(); //redirect to dev_null
133 }
134 else {
135 fout.open( xdmf_filename.str().c_str() );
136 if( fout.is_open() ) {
137 std::cout << std::endl << " The output is printed to file " << xdmf_filename.str() << " in XDMF-HDF5 format" << std::endl;
138 }
139 else {
140 std::cout << std::endl << " The output file " << xdmf_filename.str() << " cannot be opened.\n";
141 abort();
142 }
143 }
144
145 // The HDF5 file name
146 std::ostringstream hdf5_filename;
147 std::ostringstream hdf5_filename2;
148 hdf5_filename2 << filename_prefix << ".level" << _gridn << "." << time_step << "." << order << ".h5";
149 hdf5_filename << output_path << "/" << hdf5_filename2.str();
150
151 // head ************************************************
152 fout << "<?xml version=\"1.0\" ?>" << std::endl;
153 fout << "<!DOCTYPE Xdmf SYSTEM \"Xdmf.dtd []\">" << std::endl;
154 fout << "<Xdmf>" << std::endl;
155 fout << "<Domain>" << std::endl;
156 fout << "<Grid Name=\"Mesh\">" << std::endl;
157 fout << "<Time Value =\"" << time_step << "\" />" << std::endl;
158 fout << "<Topology Type=\"" << type_elem << "\" Dimensions=\"" << nel << "\">" << std::endl;
159 //Connectivity
160 fout << "<DataStructure DataType=\"Int\" Dimensions=\"" << nel << " " << ndofs << "\"" << " Format=\"HDF\">" << std::endl;
161 fout << hdf5_filename2.str() << ":/CONNECTIVITY" << std::endl;
162 fout << "</DataStructure>" << std::endl;
163 fout << "</Topology>" << std::endl;
164 fout << "<Geometry Type=\"X_Y_Z\">" << std::endl;
165 //Node_X
166 fout << "<DataStructure DataType=\"Double\" Precision=\"8\" Dimensions=\"" << nvt << " 1\"" << " Format=\"HDF\">" << std::endl;
167 fout << hdf5_filename2.str() << ":/NODES_X1" << std::endl;
168 fout << "</DataStructure>" << std::endl;
169 //Node_Y
170 fout << "<DataStructure DataType=\"Double\" Precision=\"8\" Dimensions=\"" << nvt << " 1\"" << " Format=\"HDF\">" << std::endl;
171 fout << hdf5_filename2.str() << ":/NODES_X2" << std::endl;
172 fout << "</DataStructure>" << std::endl;
173 //Node_Z
174 fout << "<DataStructure DataType=\"Double\" Precision=\"8\" Dimensions=\"" << nvt << " 1\"" << " Format=\"HDF\">" << std::endl;
175 fout << hdf5_filename2.str() << ":/NODES_X3" << std::endl;
176 fout << "</DataStructure>" << std::endl;
177 fout << "</Geometry>" << std::endl;
178 //Metis partitions
179 fout << "<Attribute Name=\"" << "Domain_partitions" << "\" AttributeType=\"Scalar\" Center=\"Cell\">" << std::endl;
180 fout << "<DataItem DataType=\"Double\" Dimensions=\"" << nel << " 1\"" << " Format=\"HDF\">" << std::endl;
181 fout << hdf5_filename2.str() << ":/DOMAIN_PARTITIONS" << std::endl;
182 fout << "</DataItem>" << std::endl;
183 fout << "</Attribute>" << std::endl;
184
185 if( _ml_sol != NULL ) {
186 // Solution Variables
187 for( unsigned i = 0; i < ( 1 - print_all ) * vars.size() + print_all * _ml_sol->GetSolutionSize(); i++ ) {
188 unsigned indx = ( print_all == 0 ) ? _ml_sol->GetIndex( vars[i].c_str() ) : i;
189 //Printing biquadratic solution on the nodes
190 if( _ml_sol->GetSolutionType( indx ) < 3 ) {
191 std::string solName = _ml_sol->GetSolutionName( indx );
192 for( int name = 0; name < 1 + 3 * _debugOutput * solution->_ResEpsBdcFlag[i]; name++ ) {
193 std::string printName;
194 if( name == 0 ) printName = solName;
195 else if( name == 1 ) printName = "Bdc" + solName;
196 else if( name == 2 ) printName = "Res" + solName;
197 else printName = "Eps" + solName;
198 fout << "<Attribute Name=\"" << printName << "\" AttributeType=\"Scalar\" Center=\"Node\">" << std::endl;
199 fout << "<DataItem DataType=\"Double\" Precision=\"8\" Dimensions=\"" << nvt << " 1\"" << " Format=\"HDF\">" << std::endl;
200 fout << hdf5_filename2.str() << ":/" << printName << std::endl;
201 fout << "</DataItem>" << std::endl;
202 fout << "</Attribute>" << std::endl;
203 }
204 }
205 else if( _ml_sol->GetSolutionType( indx ) >= 3 ) { //Printing picewise constant solution on the element
206 std::string solName = _ml_sol->GetSolutionName( indx );
207 for( int name = 0; name < 1 + 3 * _debugOutput * solution->_ResEpsBdcFlag[i]; name++ ) {
208 std::string printName;
209 if( name == 0 ) printName = solName;
210 else if( name == 1 ) printName = "Bdc" + solName;
211 else if( name == 2 ) printName = "Res" + solName;
212 else printName = "Eps" + solName;
213 fout << "<Attribute Name=\"" << printName << "\" AttributeType=\"Scalar\" Center=\"Cell\">" << std::endl;
214 fout << "<DataItem DataType=\"Double\" Precision=\"8\" Dimensions=\"" << nel << " 1\"" << " Format=\"HDF\">" << std::endl;
215 fout << hdf5_filename2.str() << ":/" << printName << std::endl;
216 fout << "</DataItem>" << std::endl;
217 fout << "</Attribute>" << std::endl;
218 }
219 }
220 }
221 }
222 //end ml_sol
223
224 fout << "</Grid>" << std::endl;
225 fout << "</Domain>" << std::endl;
226 fout << "</Xdmf>" << std::endl;
227 fout.close();
228 //END XMF FILE PRINT
229
230 //BEGIN HD5 FILE PRINT
231 hid_t file_id;
232 if( _iproc == 0 ) file_id = H5Fcreate( hdf5_filename.str().c_str(), H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT );
233 hsize_t dimsf[2];
234 herr_t status;
235 hid_t dataspace;
236 hid_t dataset;
237
238 //BEGIN COORDINATES
239 for( int i = 0; i < 3; i++ ) {
240 numVector->matrix_mult( *mesh->_topology->_Sol[i], *mesh->GetQitoQjProjection( index_nd, 2 ) );
241 numVector->localize_to_one( vector1, 0 );
242
243 if( _ml_sol != NULL && _moving_mesh && dim > i ) {
244 unsigned varind_DXDYDZ = _ml_sol->GetIndex( _moving_vars[i].c_str() );
245 numVector->matrix_mult( *solution->_Sol[varind_DXDYDZ],
246 *mesh->GetQitoQjProjection( index_nd, _ml_sol->GetSolutionType( varind_DXDYDZ ) ) );
247 numVector->localize_to_one( vector2, 0 );
248 if( _iproc == 0 ) {
249 for( unsigned ii = 0; ii < nvt; ii++ ) vector1[ii] += vector2[ii];
250 }
251 }
252
253 if( _iproc == 0 ) {
254 dimsf[0] = nvt ;
255 dimsf[1] = 1;
256 std::ostringstream Name;
257 Name << "/NODES_X" << i + 1;
258 dataspace = H5Screate_simple( 2, dimsf, NULL );
259 dataset = H5Dcreate( file_id, Name.str().c_str(), H5T_NATIVE_DOUBLE,
260 dataspace, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT );
261 status = H5Dwrite( dataset, H5T_NATIVE_DOUBLE, H5S_ALL, H5S_ALL, H5P_DEFAULT, &vector1[0] );
262 H5Sclose( dataspace );
263 H5Dclose( dataset );
264 }
265 } //end 3d loop
266 //END COORDINATES
267
268 //BEGIN CONNETTIVITY
269 unsigned icount = 0;
270 for( unsigned isdom = 0; isdom < _nprocs; isdom++ ) {
271 mesh->el->LocalizeElementDof( isdom );
272 if( _iproc == 0 ) {
273 for( unsigned iel = mesh->_elementOffset[isdom]; iel < mesh->_elementOffset[isdom + 1]; iel++ ) {
274 for( unsigned j = 0; j < ndofs; j++ ) {
275 unsigned vtk_loc_conn = FemusToVTKorToXDMFConn[j];
276 var_conn[icount] = mesh->GetSolutionDof( vtk_loc_conn, iel, index_nd );
277 icount++;
278 }
279 }
280 }
281 mesh->el->FreeLocalizedElementDof();
282 }
283
284 if(_iproc == 0) {
285 dimsf[0] = nel * ndofs ;
286 dimsf[1] = 1;
287 dataspace = H5Screate_simple( 2, dimsf, NULL );
288 dataset = H5Dcreate( file_id, "/CONNECTIVITY", H5T_NATIVE_INT,
289 dataspace, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT );
290 status = H5Dwrite( dataset, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT, &var_conn[0] );
291 H5Sclose( dataspace );
292 H5Dclose( dataset );
293 }
294
295 //END CONNETTIVITY
296
297 //BEGIN METIS PARTITIONING
298 if( _iproc == 0 ) {
299 unsigned icount = 0;
300 for( int isdom = 0; isdom < _nprocs; isdom++ ) {
301 for( unsigned ii = mesh->_elementOffset[isdom]; ii < mesh->_elementOffset[isdom + 1]; ii++ ) {
302 vector1[icount] = isdom;
303 icount++;
304 }
305 }
306 dimsf[0] = nel;
307 dimsf[1] = 1;
308 dataspace = H5Screate_simple( 2, dimsf, NULL );
309 dataset = H5Dcreate( file_id, "/DOMAIN_PARTITIONS", H5T_NATIVE_DOUBLE,
310 dataspace, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT );
311 status = H5Dwrite( dataset, H5T_NATIVE_DOUBLE, H5S_ALL, H5S_ALL, H5P_DEFAULT, &vector1[0] );
312 H5Sclose( dataspace );
313 H5Dclose( dataset );
314 }
315 //END METIS PARTITIONING
316
317 //BEGIN SOLUTION
318 if( _ml_sol != NULL ) {
319 //BEGIN DISCONTINUOUS Fem SOLUTION
320 for( unsigned i = 0; i < ( 1 - print_all ) *vars.size() + print_all * _ml_sol->GetSolutionSize(); i++ ) {
321 unsigned indx = ( print_all == 0 ) ? _ml_sol->GetIndex( vars[i].c_str() ) : i;
322 if( _ml_sol->GetSolutionType( indx ) >= 3 ) {
323 for( int name = 0; name < 1 + 3 * _debugOutput * solution->_ResEpsBdcFlag[i]; name++ ) {
324
325 std::string solName = _ml_sol->GetSolutionName( indx );
326 std::string printName;
327 if( name == 0 ) {
328 solution->_Sol[indx]->localize_to_one( vector2, 0 );
329 printName = solName;
330 }
331 else if( name == 1 ) {
332 solution->_Bdc[indx]->localize_to_one( vector2, 0 );
333 printName = "Bdc" + solName;
334 }
335 else if( name == 2 ) {
336 solution->_Res[indx]->localize_to_one( vector2, 0 );
337 printName = "Res" + solName;
338 }
339 else {
340 solution->_Eps[indx]->localize_to_one( vector2, 0 );
341 printName = "Eps" + solName;
342 }
343
344 vector1.resize( nel );
345 for( unsigned ii = 0; ii < nel; ii++ ) {
346 vector1[ii] = vector2[ mesh->GetSolutionDof( 0, ii, _ml_sol->GetSolutionType( indx ) ) ];
347 }
348
349 if( _iproc == 0 ) {
350 dimsf[0] = nel;
351 dimsf[1] = 1;
352 dataspace = H5Screate_simple( 2, dimsf, NULL );
353 dataset = H5Dcreate( file_id, printName.c_str(), H5T_NATIVE_DOUBLE,
354 dataspace, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT );
355 status = H5Dwrite( dataset, H5T_NATIVE_DOUBLE, H5S_ALL, H5S_ALL, H5P_DEFAULT, &vector1[0] );
356 H5Sclose( dataspace );
357 H5Dclose( dataset );
358 }
359 }
360 }
361 }
362 //END DISCONTINUOUS Fem SOLUTION
363
364 //BEGIN LAGRANGIAN Fem SOLUTION
365 for( unsigned i = 0; i < ( 1 - print_all ) *vars.size() + print_all * _ml_sol->GetSolutionSize(); i++ ) {
366 unsigned indx = ( print_all == 0 ) ? _ml_sol->GetIndex( vars[i].c_str() ) : i;
367 if( _ml_sol->GetSolutionType( indx ) < 3 ) {
368 for( int name = 0; name < 1 + 3 * _debugOutput * solution->_ResEpsBdcFlag[i]; name++ ) {
369
370 std::string solName = _ml_sol->GetSolutionName( indx );
371 std::string printName;
372 if( name == 0 ) {
373 numVector->matrix_mult( *solution->_Sol[indx],
374 *mesh->GetQitoQjProjection( index_nd, _ml_sol->GetSolutionType( indx ) ) );
375 printName = solName;
376 }
377 else if( name == 1 ) {
378 numVector->matrix_mult( *solution->_Bdc[indx],
379 *mesh->GetQitoQjProjection( index_nd, _ml_sol->GetSolutionType( indx ) ) );
380 printName = "Bdc" + solName;
381 }
382 else if( name == 2 ) {
383 numVector->matrix_mult( *solution->_Res[indx],
384 *mesh->GetQitoQjProjection( index_nd, _ml_sol->GetSolutionType( indx ) ) );
385 printName = "Res" + solName;
386 }
387 else {
388 numVector->matrix_mult( *solution->_Eps[indx],
389 *mesh->GetQitoQjProjection( index_nd, _ml_sol->GetSolutionType( indx ) ) );
390 printName = "Eps" + solName;
391 }
392
393 numVector->localize_to_one( vector1, 0 );
394
395 if( _iproc == 0 ) {
396 dimsf[0] = nvt;
397 dimsf[1] = 1;
398 dataspace = H5Screate_simple( 2, dimsf, NULL );
399 dataset = H5Dcreate( file_id, printName.c_str(), H5T_NATIVE_DOUBLE,
400 dataspace, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT );
401 status = H5Dwrite( dataset, H5T_NATIVE_DOUBLE, H5S_ALL, H5S_ALL, H5P_DEFAULT, &vector1[0] );
402 H5Sclose( dataspace );
403 H5Dclose( dataset );
404 }
405 }
406 }
407 }
408 //END LAGRANGIAN Fem SOLUTION
409 }
410 //END SOLUTION
411
412 if( _iproc == 0 ) H5Fclose( file_id );
413
414 //END HD5 FILE PRINT
415
416 delete numVector;
417
418#endif
419
420 return;
421 }

478 void XDMFWriter::transient_print_xmf( const std::string output_path, const uint t_idx_in, const uint t_idx_final, const int print_step, const uint nolevels_in ) {
479
480 //multigrid
481 uint NoLevels = nolevels_in;
482
483 // time parameters
484 const uint ndigits = DEFAULT_NDIGITS;
485
486 // dir names
487 std::string basetime = DEFAULT_BASETIME;
488 std::string ext_xdmf = DEFAULT_EXT_XDMF;
489 std::string basesol = DEFAULT_BASESOL;
490 std::string aux_xdmf = DEFAULT_AUX_XDMF;
491
492// =================================
493// ============= LEVELS ============
494// =================================
495
496 for( uint l = 0; l < NoLevels; l++ ) {
497

500 Name << output_path << "/" << basetime << "."
501 << std::setw( ndigits ) << std::setfill( '0' ) << t_idx_in << "-"

503 << ext_xdmf;
504 std::ofstream out( Name.str().c_str() );

506 std::cout << "transient_print_xmf: cannot print timeN.xmf" << std::endl;
507 abort();
508 }
509

511 std::string gname[3];
512 gname[0] = basesol;
513
514 // Mesh -----------------------------------
515 out << "<?xml version=\"1.0\" ?> \n";
516 out << "<!DOCTYPE Xdmf SYSTEM "
517 << "\"" << output_path << "/" << aux_xdmf << "\"" << "[]>\n";
518 out << "<Xdmf xmlns:xi=\"http://www.w3.org/2001/XInclude\" Version=\"2.2\"> \n";
519 out << "<Domain> \n";
520 for( uint kp = 0; kp < nprt; kp++ ) {
521 out << "<Grid Name=\"" << gname[kp].c_str() << "\" GridType=\"Collection\" CollectionType=\"Temporal\"> \n";
522 // time loop for grid sequence
523 for( uint it = t_idx_in; it <= t_idx_final; it++ ) if( it % print_step == 0 ) {
524 out << "<xi:include href=\""
525 << basesol << "." << std::setw( ndigits ) << std::setfill( '0' ) << it << "_l" << l << ext_xdmf
526 << "\"" << " xpointer=\"xpointer(//Xdmf/Domain/Grid[" << kp + 1 << "])\" >\n";
527 out << "<xi:fallback />\n";
528 out << " </xi:include>\n";
529 }
530 out << "</Grid> \n";
531 }
532 // Grid Collection end
533 out << "</Domain> \n";
534 out << "</Xdmf> \n";
535 out.close();
536
537 } //end levels
538
539 return;
540 }

554 hid_t XDMFWriter::read_Ihdf5( hid_t file, const std::string& name, int* data ) {

556#ifdef HAVE_HDF5
557

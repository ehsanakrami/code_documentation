295 void uq::ComputeTensorProductSet (std::vector < std::vector <unsigned> > & Tp,
296 const unsigned & numberOfQuadraturePoints,
297 const unsigned & numberOfEigPairs) { //p is max poly degree
298
299 unsigned tensorProductDim = pow (numberOfQuadraturePoints, numberOfEigPairs);
300
301 if (_output) {
302 std::cout << "tensorProductDim = " << tensorProductDim << std::endl;
303 }
304
305 Tp.resize (tensorProductDim);
306 for (unsigned i = 0; i < tensorProductDim; i++) {
307 Tp[i].resize (numberOfEigPairs);
308 }
309
310 unsigned index = 0;
311 unsigned counters[numberOfEigPairs + 1];
312 memset (counters, 0, sizeof (counters));
313
314 while (!counters[numberOfEigPairs]) {
315
316 for (unsigned j = 0; j < numberOfEigPairs; j++) {
317 Tp[index][j] = counters[numberOfEigPairs - 1 - j];
318 if (_output) {
319 std::cout << " Tp[" << index << "][" << j << "]= " << Tp[index][j] ;
320 }
321 }
322 if (_output) {
323 std::cout << std::endl;
324 }
325 index++;
326
327 unsigned i;
328 for (i = 0; counters[i] == numberOfQuadraturePoints - 1; i++) { // inner loops that are at maxval restart at zero
329 counters[i] = 0;
330 }
331 ++counters[i]; // the innermost loop that isn't yet at maxval, advances by 1
332 }
333 }

336 const std::vector < std::vector <unsigned> > & uq::GetTensorProductSet (const unsigned & numberOfQuadraturePoints,
337 const unsigned & numberOfEigPairs) {
338
339 std::pair<unsigned, unsigned> TpIndex = std::make_pair (numberOfQuadraturePoints, numberOfEigPairs);
340
341 std::map<std::pair<unsigned, unsigned>, std::vector < std::vector <unsigned> > >::iterator it;
342
343 it = _Tp.find (TpIndex);
344 if (it == _Tp.end()) {
345 ComputeTensorProductSet (_Tp[TpIndex], numberOfQuadraturePoints, numberOfEigPairs);
346 }
347 return _Tp[TpIndex];
348 };

363 void uq::ComputeHermitePolynomial (std::vector < std::vector < double > > & hermitePoly,
364 const unsigned &numberOfQuadraturePoints, const unsigned & maxPolyOrder) {
365
366 if (numberOfQuadraturePoints < 1 || numberOfQuadraturePoints > 16) {
367 std::cout << "The selected order of integration has not been implemented yet, choose an integer in [1,16]" << std::endl;
368 abort();
369 }
370
371 else {
372 hermitePoly.resize (maxPolyOrder + 1);
373 for (unsigned i = 0; i < maxPolyOrder + 1; i++) {
374 hermitePoly[i].resize (numberOfQuadraturePoints);
375 }
376
377 const double* hermiteQuadraturePoints = GetHermiteQuadraturePoints (numberOfQuadraturePoints);
378

17double hypergeometric(double a, double b, double c, double x)
18{
19 const double TOLERANCE = 1.0e-10;
20 double term = a * b * x / c;
21 double value = 1.0 + term;
22 int n = 1;
23
24 while(abs(term) > TOLERANCE) {
25 a++, b++, c++, n++;
26 term *= a * b * x / c / n;
27 value += term;
28 }

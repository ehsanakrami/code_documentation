501void GaussianEleminationWithPivoting (std::vector<std::vector < double > > & A, std::vector < double> &x, const bool &output) {
502 // A is nx(n+1) augmented matrix
503 unsigned n = A.size();
504
505 std::vector < unsigned > idx (n);
506 for (unsigned i = 0; i < n; i++) {
507 idx[i] = i;
508 }
509
510 if (output) {
511 std::cout << "Before pivoting, A = " << std::endl;
512 for (unsigned i = 0; i < n; i++) {
513 for (unsigned j = 0; j <= n; j++) {
514 std::cout << A[idx[i]][j] << " " ;
515 }
516 std::cout << std::endl;
517 }
518 std::cout << std::endl;
519 }
520
521 const double tol = 1e-12;
522 double max_row = 0.0, abs;
523 for (unsigned i = 0; i < n; i++) {
524 for (unsigned k = i + 1; k < n; k++) {
525 max_row = fabs (A[idx[i]][i]);
526 if ( (abs = fabs (A[idx[k]][i])) > max_row) {
527 max_row = abs;
528 unsigned idxi = idx[i];
529 idx[i] = idx[k];
530 idx[k] = idxi;
531 }
532 }
533 if (max_row < tol) {
534 std::cout << "Degenerate matrix" << std::endl;
535 //exit (0);
536 }
537
538 for (unsigned j = i + 1; j < n; j++) {
539 double mji = A[idx[j]][i] / A[idx[i]][i];
540 for (unsigned k = i ; k < n + 1; k++) {
541 A[idx[j]][k] -= mji * A[idx[i]][k];
542 }
543 }
544 }
545 if (fabs (A[idx[n - 1]][n - 1] / A[idx[n - 1]][n]) < tol) {
546 std::cout << "Zero row! Matrix is singular" << std::endl;
547 for (unsigned i = 0; i < n; i++) {
548 for (unsigned j = 0; j <= n; j++) {
549 std::cout << A[idx[i]][j] << " " ;
550 }
551 std::cout << std::endl;
552 }
553 std::cout << std::endl << std::endl;
554 //exit (0);
555 }
556
557 x[n - 1] = A[idx[n - 1]][n] / A[idx[n - 1]][n - 1];
558 for (unsigned ip1 = n - 1; ip1 > 0; ip1--) {
559 unsigned i = ip1 - 1u;
560 x[i] = A[idx[i]][n];
561 for (int j = ip1; j < n; j++) {
562 x[i] -= A[idx[i]][j] * x[j];
563 }
564 x[i] /= A[idx[i]][i];
565 }
566
567 if (output) {
568 std::cout << "After pivoting, A = " << std::endl;
569 for (unsigned i = 0; i < n; i++) {
570 for (unsigned j = 0; j <= n; j++) {
571 std::cout << A[idx[i]][j] << " ";
572 }
573 std::cout << std::endl;
574 }
575 std::cout << std::endl << std::endl;
576 std::cout << "Solution is:" << std::endl;
577 for (unsigned i = 0; i < n; i++) {
578 std::cout << x[i] << std::endl;
579 }
580 }

596 std::cout << std::endl;
597 }

620 A[idx[j]][i] = mji;
621 for (unsigned k = i + 1; k < n; k++) {
622 A[idx[j]][k] -= mji * A[idx[i]][k];
623 }
624 }

635 //exit (0);

640 for (unsigned i = 0; i < n; i++) {

652 std::vector < double > &b, std::vector < double > &x, const bool &output) {
653
654 unsigned n = A.size();
655
656 if (output) {

662 }
663 for (unsigned i = 0; i < n; i++) {
664 for (unsigned j = i + 1; j < n; j++) {
665 b[idx[j]] -= A[idx[j]][i] * b[idx[i]];

669 if (output) {
670 std::cout << "RHS after pivoting:" << std::endl;

673 }
674 std::cout << std::endl;
675 }
676
677 x[n - 1] = b[idx[n - 1]] / A[idx[n - 1]][n - 1];
678 for (unsigned ip1 = n - 1; ip1 > 0; ip1--) {
679 unsigned i = ip1 - 1u;
680 x[i] = b[idx[i]];
681 for (int j = ip1; j < n; j++) {
682 x[i] -= A[idx[i]][j] * x[j];
683 }
684 x[i] /= A[idx[i]][i];
685 }
686
687 if (output) {
688 std::cout << "Solution is:" << std::endl;
689 for (unsigned i = 0; i < n; i++) {

738 counters[i] = 0;

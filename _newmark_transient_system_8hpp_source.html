91void NewmarkTransientSystem<Base>::UpdateAcceleration(const std::vector<std::string>& vel_vars, const std::vector<std::string>& acc_vars)
92{
93// const double gamma = 0.5;
94// const double a5 = -1.*(1. - gamma)/gamma;
95// const double a1 = 1./(gamma*this->_dt);
96// const double a2 = -1./(gamma*this->_dt);
97
98// unsigned dim = this->_msh[0]->GetDimension();
99 // vel and acc must have the same number of variables
100 assert(vel_vars.size() == acc_vars.size());
101
102 const unsigned dim = vel_vars.size();
103
104 unsigned axyz[3];
105 unsigned vxyz[3];
106// const char accname[3][3] = {"AX","AY","AZ"};
107// const char velname[3][2] = {"U","V","W"};
108
109 for(unsigned i=0; i<dim; i++) {
110 axyz[i] = this->_equation_systems.GetIndex(acc_vars.at(i));
111 vxyz[i] = this->_equation_systems.GetIndex(vel_vars.at(i));
112 }
113
114 for (int ig=0;ig< this->_gridn;ig++) {
115 for(unsigned i=0; i<dim; i++) {
116 this->_solution[ig]->_Sol[axyz[i]]->scale(_a5);
117 this->_solution[ig]->_Sol[axyz[i]]->add(_a1,*(this->_solution[ig]->_Sol[vxyz[i]]));
118 this->_solution[ig]->_Sol[axyz[i]]->add(_a2,*(this->_solution[ig]->_SolOld[vxyz[i]]));
119 }
120 }
121}

334void Files::ComposeOutdirName(const bool use_output_time_folder) {
335
336
337 struct tm* sys_time = NULL;
338 std::string new_out;
339 char* out_char;
340 int outchar_size;
341 std::ostringstream outname;
342
343//****** Proc0 generates the string
344 if (paral::get_rank() == 0) {
345
346 time_t curtime = 0;
347
348 curtime = time(NULL); //C function
349// time_t curtime_mpi = (time_t) MPI_Wtime(); //MPI Function for time
350
351 sys_time = localtime(&curtime); //converts time into broken-down representation
352
353 const uint year = 1900 + sys_time->tm_year;
354 const uint month = 1 +sys_time->tm_mon;
355 const uint day = sys_time->tm_mday;
356 const uint hour = sys_time->tm_hour;
357 const uint minute = sys_time->tm_min;
358 const uint second = sys_time->tm_sec;
359
360
361 outname << year
362 << "-" << std::setw(2) << std::setfill('0') << month
363 << "-" << std::setw(2) << std::setfill('0') << day << "_"
364 << std::setw(2) << std::setfill('0') << hour
365 << "-" << std::setw(2) << std::setfill('0') << minute
366 << "-" << std::setw(2) << std::setfill('0') << second << "/";
367
368 new_out = outname.str(); // declare it outside the block as all the processes will see it
369
370 outchar_size = outname.str().size();
371
372 }
373
374//****** Proc0 broadcasts the string size and the string content to every proc
375#ifdef HAVE_MPI
376 MPI_Bcast(&outchar_size, 1, MPI_INT, 0, MPI_COMM_WORLD);
377#endif
378
379out_char = new char[outchar_size +1]; //for the null
380
381 if (paral::get_rank() == 0) std::strcpy(out_char, new_out.c_str());
382
383
384#ifdef HAVE_MPI
385MPI_Bcast(out_char, outchar_size, MPI_CHAR, 0, MPI_COMM_WORLD);
386#endif
387
388 out_char[outchar_size] = NULL; //HEY MAYBE NOT 100% ORTHODOX BUT OTHERWISE IT DOESN'T WORK!!!

390// printf("^^^^^^^^^^^^^^^^^^^ %s\n",out_char);

394 // new_out.resize(outchar_size);

397 delete [] out_char;
398
399
400 if (use_output_time_folder) _output_time = new_out;
401 else _output_time = "";
402
403 //************************
404 //set the input and output_path variables

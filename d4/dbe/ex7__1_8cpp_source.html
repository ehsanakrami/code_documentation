564void AssemblePoissonMatrixandRhs(MultiLevelProblem& ml_prob) {
565
566 //pointers and references
567 LinearImplicitSystem& mylin_impl_sys = ml_prob.get_system<LinearImplicitSystem>("Poisson");
568 const unsigned level = mylin_impl_sys.GetLevelToAssemble();
569
570 Solution* mysolution = ml_prob._ml_sol->GetSolutionLevel(level);
571 LinearEquationSolver* mylsyspde = mylin_impl_sys._LinSolver[level];
572 Mesh* mymsh = ml_prob._ml_msh->GetLevel(level);
573 elem* myel = mymsh->el;
574 SparseMatrix* myKK = mylsyspde->_KK;
575 NumericVector* myRES = mylsyspde->_RES;
576 MultiLevelSolution* ml_sol = ml_prob._ml_sol;
577
578 //data
579 const unsigned dim = mymsh->GetDimension();
580 unsigned nel = mymsh->GetNumberOfElements();
581 unsigned igrid = mymsh->GetLevel();
582 unsigned iproc = mymsh->processor_id();
583
584 //solution variable
585 unsigned SolIndex;
586 unsigned SolPdeIndex;
587 SolIndex = ml_sol->GetIndex("Sol");
588 SolPdeIndex = mylin_impl_sys.GetSolPdeIndex("Sol");
589 //solution order
590 unsigned order_ind = ml_sol->GetSolutionType(SolIndex);
591
592
593 //coordinates
594 std::vector < std::vector < double> > coordinates(dim);
595
596 // declare
597 std::vector < int > metis_node;
598 std::vector < int > KK_dof;
599 std::vector <double> phi;
600 std::vector <double> gradphi;
601 std::vector <double> nablaphi;
602 double weight;
603 std::vector < double > F;
604 std::vector < double > B;
605 std::vector <double> normal(3.0);
606 double src_term = 0.;
607 std::vector <double> xyzt(4, 0.);
608 ParsedFunction* bdcfunc = NULL;
609
610 // reserve
611 const unsigned max_size = static_cast< unsigned >(ceil(pow(3, dim)));
612 metis_node.reserve(max_size);
613 KK_dof.reserve(max_size);
614
615 for(int i = 0; i < dim; i++)
616 coordinates[i].reserve(max_size);
617
618 phi.reserve(max_size);
619 gradphi.reserve(max_size * dim);
620 nablaphi.reserve(max_size * (3 * (dim - 1) + !(dim - 1)));
621 F.reserve(max_size);
622 B.reserve(max_size * max_size);
623
624 // Set to zeto all the entries of the Global Matrix
625 myKK->zero();
626
627 // *** element loop ***
628 for(int iel = mymsh->_elementOffset[iproc]; iel < mymsh->_elementOffset[iproc + 1]; iel++) {
629
630 short unsigned ielt = mymsh->GetElementType(iel);
631 unsigned nve = mymsh->GetElementDofNumber(iel, order_ind);
632
633 // resize
634 metis_node.resize(nve);
635 KK_dof.resize(nve);
636 phi.resize(nve);
637 gradphi.resize(nve * dim);
638 nablaphi.resize(nve * (3 * (dim - 1) + !(dim - 1)));
639
640 for(int i = 0; i < dim; i++) {
641 coordinates[i].resize(nve);
642 }
643
644 // set to zero all the entries of the FE matrices
645 F.resize(nve);
646 memset(&F[0], 0, nve * sizeof(double));
647 B.resize(nve * nve);
648 memset(&B[0], 0, nve * nve * sizeof(double));
649
650
651 // get local to global mappings
652 for(unsigned i = 0; i < nve; i++) {
653 unsigned inode_coord_metis = mymsh->GetSolutionDof(i, iel, 2);
654 metis_node[i] = mymsh->GetSolutionDof(i, iel, order_ind);

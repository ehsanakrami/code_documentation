200void MultiLevelMeshTwo::FillNodeSto() {
201
202// ================================================
203// NODES (in nd_sto and nod_val)
204// ================================================
205 _nd_sto = new NodeSto*[_n_nodes];
206 for (int i=0;i<_n_nodes;i++) {
207 _nd_sto[i]= new NodeSto(100000,_NoLevels); //initialization values are IMPORTANT!!!
208 }
209
210 // Storing the node information with element loop-------------- ==> we SUPERIMPOSE stuff, we fill more than once
211 // looping over elements is essential if you want to separate the LINEAR from the QUADRATIC nodes
212 for (int i=0;i<_n_elements_sum_levs[VV];i++) {
213 for (uint k=0;k<_elnodes[VV][QQ];k++) {
214
215 int knode=_el_sto[i]->_elnds[k];
216 _nd_sto[knode]->_id = knode; //libmesh node numbering
217 if (_el_sto[i]->_subd > _nd_sto[knode]->_subd) _nd_sto[knode]->_subd = _el_sto[i]->_subd; // HIGHEST subdomain number among the elements the node belongs to //libmesh does the MINIMUM: by definition, a Node's processor ID is the minimum processor ID for all of the elements which share the node.
218 if (_el_sto[i]->_lev < _nd_sto[knode]->_lev) _nd_sto[knode]->_lev = _el_sto[i]->_lev; // COARSEST level among the elements the node belongs to
219 if ( k < _elnodes[VV][LL] &&
220 _el_sto[i]->_lev < _nd_sto[knode]->_levp) _nd_sto[knode]->_levp = _el_sto[i]->_lev; //COARSEST level also for linear
221 _nd_sto[knode]->_var = 1;
222 }
223 }
224
225 return;
226
227}

231void MultiLevelMeshTwo::ComputeElemOffsetsBySubdLevel() {
232 _off_el = new int*[VB];
233 int sum_el_vb[VB];
234 for (int vb=0;vb < VB; vb++) {
235 _off_el[vb]=new int[_NoSubdom*_NoLevels+1];
236 _off_el[vb][0]=0;
237 sum_el_vb[vb]=0;
238 }
239
240 for (int vb=0;vb < VB; vb++) {
241 for (int isub=0;isub<_NoSubdom*_NoLevels; isub++) {
242 sum_el_vb[vb] += _n_elements_sl_vb[vb][isub];
243 _off_el[vb][isub+1] = sum_el_vb[vb];
244 }
245 }
246
247 return;

282void MultiLevelMeshTwo::ComputeNodeMapExtLevels() {
283
284 int NegativeOneFlag = -1;
285
286 _Qnode_fine_Qnode_lev=new int*[_NoLevels+1];
287
288// quadratic levels
289 for (int ilev=0;ilev< _NoLevels;ilev++) {
290 _Qnode_fine_Qnode_lev[ilev] = new int[_n_nodes];
291 int count=0;
292 for (int inode=0;inode<_n_nodes;inode++) _Qnode_fine_Qnode_lev[ilev][inode] = NegativeOneFlag;
293 for (int iproc=0; iproc < _NoSubdom; iproc++) {
294 for (int inode = _off_nd[QQ][iproc*_NoLevels];
295 inode < _off_nd[QQ][iproc*_NoLevels+ilev+1];inode++)
296 _Qnode_fine_Qnode_lev[ilev][inode]=count++;//post increment
297 }
298 }
299
300// linear coarse level
301 _Qnode_fine_Qnode_lev[_NoLevels]=new int[_n_nodes];
302 int count=0;
303 for (int inode=0;inode<_n_nodes;inode++) _Qnode_fine_Qnode_lev[_NoLevels][inode] = NegativeOneFlag;
304 for (int iproc=0; iproc < _NoSubdom; iproc++) {
305 for (int inode=0;inode < _off_nd[LL][iproc*_NoLevels+1]
306 - _off_nd[LL][iproc*_NoLevels];inode++)
307 _Qnode_fine_Qnode_lev[_NoLevels][_off_nd[QQ][iproc*_NoLevels]+inode] = count++;
308 }
309
310 //After _Qnode_fine_Qnode_lev is filled, I can fill the inverse map... can I do it on the fly?! //Let me do it here now
311 _Qnode_lev_Qnode_fine = new uint *[_NoLevels+1];
312 uint n_nodes_top=_NoNodesXLev[_NoLevels-1];
313
314 for (uint ilev=0;ilev<=_NoLevels;ilev++) {
315 _Qnode_lev_Qnode_fine[ilev] = new uint [_NoNodesXLev[ilev]];
316 for (uint inode=0;inode<n_nodes_top;inode++) {
317 uint val_lev = _Qnode_fine_Qnode_lev[ilev][inode];
318 if ( val_lev != NegativeOneFlag ) _Qnode_lev_Qnode_fine[ilev][ val_lev ] = inode;
319 }
320 }
321
322
323 return;
324}

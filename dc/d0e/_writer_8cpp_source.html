148 std::string Writer::print_sol_bdc_res_eps_name(const std::string solName, const unsigned name) const {
149
150 std::string printName;
151
152 if( name == _index_sol ) printName = solName;
153 else if( name == _index_bdc ) printName = solName + "_" + _name_bdc;
154 else if( name == _index_res ) printName = solName + "_" + _name_res;
155 else if( name == _index_eps ) printName = solName + "_" + _name_eps;
156 else { abort(); }
157
158 return printName;
159 }

176
177 unsigned index = 0;
178
179 if( !strcmp( order_str.c_str(), fe_fams_for_files[ FILES_CONTINUOUS_LINEAR ].c_str() ) ) { index = FILES_CONTINUOUS_LINEAR; }

181 else if( !strcmp( order_str.c_str(), fe_fams_for_files[ FILES_CONTINUOUS_BIQUADRATIC ].c_str() ) ) { index = FILES_CONTINUOUS_BIQUADRATIC; }

246#ifdef HAVE_MPI

258 return _request;

264 MPI_Wait(&_request, &status);
265#endif

267 }

270#ifdef HAVE_MPI
271 int val=0;
272 MPI_Test(&_request, &val, MPI_STATUS_IGNORE);
273 if (val) {

276 }
277
278 return val;
279#else
280 return true;
281#endif

659// Internal helper function to create vector<something_useable> from
660// vector<bool> for compatibility with MPI bitwise operations
661template <typename T>
662inline void pack_vector_bool(const std::vector<bool> &in,
663 std::vector<T> &out) {
664 unsigned int data_bits = 8*sizeof(T);
665 unsigned int in_size = in.size();
666 unsigned int out_size = in_size/data_bits + (in_size%data_bits?1:0);
667 out.clear();
668 out.resize(out_size);
669 for (unsigned int i=0; i != in_size; ++i) {
670 unsigned int index = i/data_bits;
671 unsigned int offset = i%data_bits;
672 out[index] += (in[i]?1:0) << offset;
673 }
674}
675
676// Internal helper function to create vector<something_useable> from
677// vector<bool> for compatibility with MPI byte operations
678template <typename T>
679inline void unpack_vector_bool(const std::vector<T> &in,
680 std::vector<bool> &out) {
681 unsigned int data_bits = 8*sizeof(T);
682 // We need the output vector to already be properly sized
683 unsigned int out_size = out.size();
684 assert(out_size/data_bits + (out_size%data_bits?1:0) == in.size());
685
686 for (unsigned int i=0; i != out_size; ++i) {
687 unsigned int index = i/data_bits;
688 unsigned int offset = i%data_bits;
689 out[i] = in[index] << (data_bits-1-offset) >> (data_bits-1);
690 }
691}

749template <typename T> inline bool verify(const T &r) {
750 int n_proc; /*int iproc;*/
751 MPI_Comm_size(MPI_COMM_WORLD,&n_proc);
752// MPI_Comm_rank (MPI_COMM_WORLD,&iproc);
753 if (n_proc > 1) {
754 T tempmin = r, tempmax = r;
755 Parallel::min(tempmin);
756 Parallel::max(tempmax);
757 bool verified = (r == tempmin) &&
758 (r == tempmax);
759 Parallel::min(verified);
760 return verified;
761 }
762 return true;
763}

1205template <typename T>
1206inline void send_receive(const unsigned int dest_processor_id,
1207 std::vector<std::vector<T> > &send,
1208 const unsigned int source_processor_id,
1209 std::vector<std::vector<T> > &recv) {
1210 int n_proc;
1211 int iproc;
1212 MPI_Comm_size(MPI_COMM_WORLD,&n_proc);
1213 MPI_Comm_rank (MPI_COMM_WORLD,&iproc);
1214
1215 // START_LOG("send_receive()", "Parallel");
1216
1217 if (dest_processor_id == iproc &&
1218 source_processor_id == iproc) {
1219 recv = send;
1220 // STOP_LOG("send_receive()", "Parallel");
1221 return;
1222 }
1223
1224 // temporary buffers - these will be sized in bytes
1225 // and manipulated with MPI_Pack and friends
1226 std::vector<char> sendbuf, recvbuf;
1227
1228 // figure out how many bytes we need to pack all the data
1229 int packedsize=0, sendsize=0;
1230
1231 // The outer buffer size
1232 MPI_Pack_size(1,
1233 datatype<unsigned int>(),
1234 MPI_COMM_WORLD,
1235 &packedsize);
1236 sendsize += packedsize;
1237
1238 for (unsigned int i=0; i<send.size(); i++) {
1239 // The size of the ith inner buffer
1240 MPI_Pack_size(1,
1241 datatype<unsigned int>(),
1242 MPI_COMM_WORLD,
1243 &packedsize);
1244 sendsize += packedsize;
1245
1246 // The data for each inner buffer
1247 MPI_Pack_size(send[i].size(),
1248 datatype<T>(),
1249 MPI_COMM_WORLD,
1250 &packedsize);
1251 sendsize += packedsize;
1252 }
1253
1254 assert(sendsize /* should at least be 1! */);
1255 sendbuf.resize(sendsize);
1256
1257 // Pack the send buffer
1258 int pos=0;
1259
1260 // ... the size of the outer buffer
1261 sendsize = send.size();
1262 MPI_Pack(&sendsize, 1, datatype<unsigned int>(),
1263 &sendbuf[0], sendbuf.size(), &pos,
1264 MPI_COMM_WORLD);
1265
1266 for (unsigned int i=0; i<send.size(); i++) {
1267 // ... the size of the ith inner buffer
1268 sendsize = send[i].size();
1269 MPI_Pack(&sendsize, 1, datatype<unsigned int>(),
1270 &sendbuf[0], sendbuf.size(), &pos,
1271 MPI_COMM_WORLD);
1272
1273 // ... the contents of the ith inner buffer
1274 if (!send[i].empty())
1275 MPI_Pack(&send[i][0], send[i].size(), datatype<T>(),
1276 &sendbuf[0], sendbuf.size(), &pos,
1277 MPI_COMM_WORLD);
1278 }
1279
1280 assert(static_cast<unsigned int>(pos) == sendbuf.size());
1281
1282 Parallel::request request;
1283
1284 Parallel::nonblocking_send(dest_processor_id,
1285 sendbuf,
1286 MPI_PACKED,
1287 request,
1288 /* tag = */ 123);
1289
1290 Parallel::receive(source_processor_id,
1291 recvbuf,
1292 MPI_PACKED,
1293 /* tag = */ 123);
1294
1295 // Unpack the received buffer
1296 assert(!recvbuf.empty());
1297 pos=0;
1298 MPI_Unpack(&recvbuf[0], recvbuf.size(), &pos,
1299 &sendsize, 1, datatype<unsigned int>(),
1300 MPI_COMM_WORLD);
1301
1302 // ... size the outer buffer
1303 recv.resize(sendsize);
1304
1305 for (unsigned int i=0; i<recv.size(); i++) {
1306 MPI_Unpack(&recvbuf[0], recvbuf.size(), &pos,
1307 &sendsize, 1, datatype<unsigned int>(),
1308 MPI_COMM_WORLD);
1309
1310 // ... size the inner buffer
1311 recv[i].resize(sendsize);
1312
1313 // ... unpack the inner buffer if it is not empty
1314 if (!recv[i].empty())
1315 MPI_Unpack(&recvbuf[0], recvbuf.size(), &pos,
1316 &recv[i][0], recv[i].size(), datatype<T>(),
1317 MPI_COMM_WORLD);
1318 }
1319
1320 Parallel::wait(request);
1321
1322 // STOP_LOG("send_receive()", "Parallel");
1323}

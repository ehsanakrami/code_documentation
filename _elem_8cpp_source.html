69 elem::elem(elem* elc, const unsigned dim_in, const unsigned refindex, const std::vector < double >& coarseAmrVector)
70 {
71
72 _dim = dim_in;
73
74 _coarseElem = elc;
75
76 _level = elc->_level + 1;
77
78 _nelt[0] = _nelt[1] = _nelt[2] = _nelt[3] = _nelt[4] = _nelt[5] = 0;
79 _nel = elc->GetRefinedElementNumber() * refindex; //refined
80 _nel += elc->GetElementNumber() - elc->GetRefinedElementNumber(); // + non-refined;
81
82 _elementType.resize(_nel);
83 _elementGroup.resize(_nel);
84 _elementMaterial.resize(_nel);
85 _elementLevel.resize(_nel, _level);
86
87 //**************************
88 MyVector <unsigned> rowSizeElDof(_nel);
89 MyVector <unsigned> rowSizeElNearFace(_nel);
90 unsigned jel = 0;
91 for (unsigned isdom = 0; isdom < elc->_nprocs; isdom++) {
92 elc->_elementType.broadcast(isdom);
93 for (unsigned iel = elc->_elementType.begin(); iel < elc->_elementType.end(); iel++) {
94 short unsigned elType = elc->_elementType[iel];
95 int increment = 1;
96 if (static_cast < short unsigned >(coarseAmrVector[iel] + 0.25) == 1) {
97 increment = NRE[elType];
98 }
99 for (unsigned j = 0; j < increment; j++) {
100 rowSizeElDof[jel + j] += NVE[elType][2];
101 rowSizeElNearFace[jel + j] += NFC[elType][1];
102 }
103 jel += increment;
104 }
105 elc->_elementType.clearBroadcast();
106 }
107 _elementDof = MyMatrix <unsigned> (rowSizeElDof);
108 _elementNearFace = MyMatrix <int> (rowSizeElNearFace, -1);
109
110 rowSizeElDof.clear();
111 rowSizeElNearFace.clear();
112
113 }

138 void elem::ReorderMeshElements(const std::vector < unsigned >& elementMapping)
139 {
140
141 //BEGIN reordering _elementType
142 MyVector <short unsigned> tempElementType;
143 tempElementType = _elementType;
144 for (unsigned iel = 0; iel < _nel; iel++) {
145 _elementType[ elementMapping [iel] ] = tempElementType[iel] ;
146 }
147 tempElementType.clear();
148 //END reordering _elementType
149
150 //BEGIN reordering _elementLevel
151 if (_level != 0) {
152 MyVector <short unsigned> tempElementLevel;
153 tempElementLevel = _elementLevel;
154 for (unsigned iel = 0; iel < _nel; iel++) {
155 _elementLevel[elementMapping [iel]] = tempElementLevel[iel];
156 }
157 tempElementLevel.clear();
158 }
159 //END reordering _elementCanBeRefined
160
161 //BEGIN reordering _elementGroup and _elementMaterial
162
163 MyVector <short unsigned> tempElementGroup;
164 MyVector <short unsigned> tempElementMaterial;
165 tempElementGroup = _elementGroup;
166 tempElementMaterial = _elementMaterial;
167
168 for (unsigned iel = 0; iel < _nel; iel++) {
169 _elementGroup[elementMapping[iel]] = tempElementGroup[iel];
170 _elementMaterial[elementMapping[iel]] = tempElementMaterial[iel] ;
171 }
172 tempElementGroup.clear();
173 tempElementMaterial.clear();
174
175 //END reordering _elementGroup and _elementMaterial
176
177
178 //BEGIN reordering _elementDof (rows)
179 MyVector <unsigned> rowSize(_nel, 0);
180 for (unsigned i = rowSize.begin(); i < rowSize.end(); i++) {
181 rowSize[elementMapping[i]] = _elementDof.size(i);
182 }
183
184 MyMatrix <unsigned> tmpElDof = _elementDof;
185 _elementDof = MyMatrix <unsigned>(rowSize, 0);
186 for (unsigned i = tmpElDof.begin(); i < tmpElDof.end(); i++) {
187 for (unsigned j = tmpElDof.begin(i); j < tmpElDof.end(i); j++) {
188 _elementDof[elementMapping [i]][j] = tmpElDof[i][j];
189 }
190 }
191 tmpElDof.clear();
192 //END reordering OF _elementDof
193
194 //BEGIN reordering _elementNearFace (rows)
195 for (unsigned i = rowSize.begin(); i < rowSize.end(); i++) {
196 rowSize[elementMapping[i]] = _elementNearFace.size(i);
197 }
198 MyMatrix <int> tmpElNearFace = _elementNearFace;
199 _elementNearFace = MyMatrix <int>(rowSize, 0);
200 for (unsigned i = tmpElNearFace.begin(); i < tmpElNearFace.end(); i++) {
201 for (unsigned j = tmpElNearFace.begin(i); j < tmpElNearFace.end(i); j++) {
202 _elementNearFace[elementMapping [i]][j] = tmpElNearFace[i][j];
203 }
204 }
205 tmpElNearFace.clear();
206 //END reordering _elementNearFace
207
208 //BEGIN reordering _childElementDof (columns) on coarse level
209 if (_level != 0) {
210 for (unsigned i = _coarseElem->_childElem.begin(); i < _coarseElem->_childElem.end(); i++) {
211 for (unsigned j = _coarseElem->_childElem.begin(i); j < _coarseElem->_childElem.end(i); j++) {
212 _coarseElem->_childElem[i][j] = elementMapping[ _coarseElem->_childElem[i][j]];
213 }
214 }
215 }
216 //END reordering _childElementDof
217
218 }

392 void elem::BuildElementNearFace() {
393
394 for(unsigned iel = 0; iel < GetElementNumber(); iel++) {
395 for(unsigned iface = 0; iface < GetElementFaceNumber(iel); iface++) {
396 if(GetFaceElementIndex(iel, iface) <= 0) {
397 unsigned i1 = GetFaceVertexIndex(iel, iface, 0);
398 unsigned i2 = GetFaceVertexIndex(iel, iface, 1);
399 unsigned i3 = GetFaceVertexIndex(iel, iface, 2);
400
401 for(unsigned j = 0; j < GetElementNearVertexNumber(i1); j++) {
402 unsigned jel = GetElementNearVertex(i1, j);
403
404 if(jel > iel) {
405 for(unsigned jface = 0; jface < GetElementFaceNumber(jel); jface++) {
406 if(GetFaceElementIndex(jel, jface) <= 0) {
407 unsigned j1 = GetFaceVertexIndex(jel, jface, 0);
408 unsigned j2 = GetFaceVertexIndex(jel, jface, 1);
409 unsigned j3 = GetFaceVertexIndex(jel, jface, 2);
410 unsigned j4 = GetFaceVertexIndex(jel, jface, 3);
411
412 const bool faces_coincide_three_dim = (GetDimension()/*Mesh::_dimension*/ == 3 &&
413 (i1 == j1 || i1 == j2 || i1 == j3 || i1 == j4) &&
414 (i2 == j1 || i2 == j2 || i2 == j3 || i2 == j4) &&
415 (i3 == j1 || i3 == j2 || i3 == j3 || i3 == j4));
416
417 const bool faces_coincide_two_dim = (GetDimension()/*Mesh::_dimension*/ == 2 &&
418 (i1 == j1 || i1 == j2) &&
419 (i2 == j1 || i2 == j2));
420
421 const bool faces_coincide_one_dim = (GetDimension()/*Mesh::_dimension*/ == 1 && (i1 == j1));
422
423 if(faces_coincide_three_dim
424 ||
425 faces_coincide_two_dim
426 ||
427 faces_coincide_one_dim
428 ) {
429 SetFaceElementIndex(iel, iface, jel + 1u);
430 SetFaceElementIndex(jel, jface, iel + 1u);
431 }
432 }
433 }
434 }
435 }
436 }

448 void elem::BuildElementNearElement()
449 {
450 MyVector < unsigned > rowSize(_elementOffset, 1);
451 for (unsigned iel = rowSize.begin(); iel < rowSize.end(); iel++) {
452 std::map< unsigned, bool> elements;
453 for (unsigned i = 0; i < GetElementDofNumber(iel, 0); i++) {
454 unsigned inode = GetElementDofIndex(iel, i);
455 for (unsigned j = _elementNearVertex.begin(inode); j < _elementNearVertex.end(inode); j++) {
456 elements[_elementNearVertex[inode][j]] = true;
457 }
458 }
459 rowSize[iel] = elements.size();
460 }
461 _elementNearElement = MyMatrix <unsigned> (rowSize, UINT_MAX);
462 for (unsigned iel = _elementNearElement.begin(); iel < _elementNearElement.end(); iel++) {
463 std::map< unsigned, bool> elements;
464 _elementNearElement[iel][0] = iel;
465 for (unsigned i = 0; i < GetElementDofNumber(iel, 0); i++) {
466 unsigned inode = GetElementDofIndex(iel, i);
467 for (unsigned j = _elementNearVertex.begin(inode); j < _elementNearVertex.end(inode); j++) {
468 if (_elementNearVertex[inode][j] != iel) {
469 elements[_elementNearVertex[inode][j]] = true;
470 }
471 }
472 }
473 unsigned j = 1;
474 for (std::map<unsigned, bool>::iterator it = elements.begin(); it != elements.end(); it++, j++) {
475 _elementNearElement[iel][j] = it->first;
476 }
477 }
478
479 }

482 void elem::BuildElementNearVertex()
483 {
484 MyVector <unsigned> rowSize(_nvt, 0);
485 for (unsigned iel = 0; iel < _nel; iel++) {
486 for (unsigned inode = 0; inode < GetElementDofNumber(iel, 0); inode++) {
487 rowSize[GetElementDofIndex(iel, inode)]++;
488 }
489 }
490 _elementNearVertex = MyMatrix <unsigned> (rowSize, _nel);
491 for (unsigned iel = 0; iel < _nel; iel++) {
492 for (unsigned inode = 0; inode < GetElementDofNumber(iel, 0); inode++) {
493 unsigned irow = GetElementDofIndex(iel, inode);
494 unsigned j = 0;
495 while (_nel != _elementNearVertex[irow][j]) j++;
496 _elementNearVertex[irow][j] = iel;
497 }
498 }

665 void elem::GetAMRRestriction(Mesh *msh)
666 {
667
668 std::vector < std::map < unsigned, std::map < unsigned, double > > > & restriction = msh->GetAmrRestrictionMap();
669 restriction.resize(3);
670
671 std::vector < std::map < unsigned, bool > > &interfaceSolidMark = msh->GetAmrSolidMark();
672 interfaceSolidMark.resize(3);
673
674 std::vector < MyVector<unsigned> > interfaceElement;
675 std::vector < MyMatrix<unsigned> > interfaceLocalDof;
676 std::vector < std::vector < MyMatrix<unsigned> > > interfaceDof;
677 std::vector < std::vector < MyMatrix<unsigned> > > levelInterfaceSolidMark;
678 std::vector < std::vector < MyMatrix< double > > > interfaceNodeCoordinates;
679
680 interfaceElement.resize(_level + 1);
681 interfaceLocalDof.resize(_level + 1);
682 interfaceDof.resize(3);
683 levelInterfaceSolidMark.resize(3);
684 for (unsigned i = 0; i < 3; i++) {
685 interfaceDof[i].resize(_level + 1);
686 levelInterfaceSolidMark[i].resize(_level + 1);
687 }
688 interfaceNodeCoordinates.resize(_level + 1);
689 unsigned dim = msh->GetDimension();
690
691 for (unsigned ilevel = 0; ilevel <= _level; ilevel++) {
692 //BEGIN interface element search
693 interfaceElement[ilevel] = MyVector <unsigned> (_elementOwned);
694 unsigned counter = 0;
695 for (unsigned i = _elementLevel.begin(); i < _elementLevel.end(); i++) {
696 if (ilevel == _elementLevel[i]) {
697 for (unsigned j = _elementNearFace.begin(i); j < _elementNearFace.end(i); j++) {
698 if (-1 == _elementNearFace[i][j]) {
699 interfaceElement[ilevel][counter] = i;
700 counter++;
701 break;
702 }
703 }
704 }
705 }
706 interfaceElement[ilevel].resize(counter);
707 interfaceElement[ilevel].stack();
708 //END interface element search
709
710 //BEGIN interface node search
711 std::vector< unsigned > offset = interfaceElement[ilevel].getOffset();
712 interfaceLocalDof[ilevel] = MyMatrix <unsigned>(offset, NVE[0][2], UINT_MAX);
713 for (unsigned i = interfaceElement[ilevel].begin(); i < interfaceElement[ilevel].end(); i++) {
714 unsigned iel = interfaceElement[ilevel][i];
715 std::map <unsigned, bool> ldofs;
716 for (unsigned jface = _elementNearFace.begin(iel); jface < _elementNearFace.end(iel); jface++) {
717 if (-1 == _elementNearFace[iel][jface]) {
718 for (unsigned k = 0; k < GetNFACENODES(GetElementType(iel), jface, 2); k++) {
719 unsigned index = GetIG(GetElementType(iel), jface, k);
720 ldofs[index] = true;
721 }
722 }
723 }
724 unsigned j = 0;
725 for (std::map<unsigned, bool>::iterator it = ldofs.begin(); it != ldofs.end(); it++) {
726 interfaceLocalDof[ilevel][i][j] = it->first;
727 j++;
728 }
729 }
730 interfaceLocalDof[ilevel].shrinkToFit(UINT_MAX);
731 //END interface node search
732
733 //BEGIN interface node dof global search, one for each soltype
734 MyVector <unsigned> rowSize = interfaceLocalDof[ilevel].getRowSize();
735 for (unsigned soltype = 0; soltype < 3; soltype++) {
736 interfaceDof[soltype][ilevel] = MyMatrix< unsigned > (rowSize, UINT_MAX);
737 levelInterfaceSolidMark[soltype][ilevel] = MyMatrix< unsigned > (rowSize, UINT_MAX);
738 for (unsigned i = interfaceLocalDof[ilevel].begin(); i < interfaceLocalDof[ilevel].end(); i++) {
739 unsigned iel = interfaceElement[ilevel][i];
740 unsigned counter = 0;
741 for (unsigned j = interfaceLocalDof[ilevel].begin(i); j < interfaceLocalDof[ilevel].end(i); j++) {
742 unsigned jloc = interfaceLocalDof[ilevel][i][j];
743 if (jloc < GetElementDofNumber(iel, soltype)) {
744 unsigned jdof = msh->GetSolutionDof(jloc, iel, soltype);
745 interfaceDof[soltype][ilevel][i][counter] = jdof;
746 unsigned jdof2 = msh->GetSolutionDof(jloc, iel, 2);
747 levelInterfaceSolidMark[soltype][ilevel][i][counter] = msh->GetSolidMark(jdof2);
748 counter++;
749 }
750 else {
751 break;
752 }
753 }
754 }
755 interfaceDof[soltype][ilevel].shrinkToFit(UINT_MAX);
756 levelInterfaceSolidMark[soltype][ilevel].shrinkToFit(UINT_MAX);
757 }
758 //END interface node dof global search, one for each soltype
759
760 //BEGIN interface node coordinates
761 interfaceNodeCoordinates[ilevel].resize(dim);
762 for (unsigned k = 0; k < dim; k++) {
763 interfaceNodeCoordinates[ilevel][k] = MyMatrix< double > (rowSize, 0.);
764 }
765 for (unsigned i = interfaceLocalDof[ilevel].begin(); i < interfaceLocalDof[ilevel].end(); i++) {
766 unsigned iel = interfaceElement[ilevel][i];
767 for (unsigned j = interfaceLocalDof[ilevel].begin(i); j < interfaceLocalDof[ilevel].end(i); j++) {
768 unsigned jnode = interfaceLocalDof[ilevel][i][j];
769 unsigned xDof = msh->GetSolutionDof(jnode, iel, 2);
770 for (unsigned k = 0; k < dim; k++) {
771 interfaceNodeCoordinates[ilevel][k][i][j] = (*msh->_topology->_Sol[k])(xDof);
772 }
773 }
774 }
775 //END interface node coordinates search
776 }
777
778 for (unsigned soltype = 0; soltype < 3; soltype++) {
779 for (int ilevel = 0; ilevel < _level; ilevel++) {
780 for (int jlevel = ilevel + 1; jlevel <= _level; jlevel++) {
781 for (unsigned lproc = 0; lproc < _nprocs; lproc++) {
782 interfaceDof[soltype][jlevel].broadcast(lproc);
783 levelInterfaceSolidMark[soltype][jlevel].broadcast(lproc);
784 for (unsigned d = 0; d < dim; d++) {
785 interfaceNodeCoordinates[jlevel][d].broadcast(lproc);
786 }
787 std::map< unsigned, bool> candidateNodes;
788 std::map< unsigned, bool> elementNodes;
789

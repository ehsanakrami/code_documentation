324 inline void PetscMatrix::add (const double a_in, // double constant
325 SparseMatrix &X_in // sparse matrix
326 ) {
327 assert (this->initialized());
328
329 // crash due to incompatible sparsity structure...
330 assert (this->m() == X_in.m());
331 assert (this->n() == X_in.n());
332
333 PetscScalar a = static_cast<PetscScalar> (a_in);
334 PetscMatrix* X = dynamic_cast<PetscMatrix*> (&X_in);
335
336 // the matrix from which we copy the values has to be assembled/closed
337 // X->close ();
338 assert (X != NULL);
339 assert (X->closed());
340 semiparallel_only();
341 int ierr = 0;
342
343#if PETSC_VERSION_LESS_THAN(2,3,0) // 2.2.x & earlier style
344 ierr = MatAXPY (&a, X->_mat, _mat, SAME_NONZERO_PATTERN);
345 CHKERRABORT (MPI_COMM_WORLD, ierr);
346#else // 2.3.x & newer
347 ierr = MatAXPY (_mat, a, X->_mat, DIFFERENT_NONZERO_PATTERN);
348 CHKERRABORT (MPI_COMM_WORLD, ierr);
349#endif
350 }

352// ========================================================
353 inline double PetscMatrix::operator() (const int i, // index i
354 const int j // index j
355 ) const {
356
357 // do no use this function !!!!!!!!
358 assert (this->initialized());
359#if PETSC_VERSION_LESS_THAN(2,2,1) // PETSc 2.2.0 & older
360 PetscScalar *petsc_row;
361 int* petsc_cols;
362#else // PETSc 2.2.1 & newer
363 const PetscScalar *petsc_row;
364 const PetscInt *petsc_cols;
365#endif
366
367 // If the entry is not in the sparse matrix, it is 0.
368 double value = 0.;
369 int ierr = 0, ncols = 0, i_val = static_cast<int> (i), j_val = static_cast<int> (j);
370
371 assert (this->closed()); // the matrix needs to be closed for this to work
372 ierr = MatGetRow (_mat, i_val, &ncols, &petsc_cols, &petsc_row);
373 CHKERRABORT (MPI_COMM_WORLD, ierr);
374 // Perform a binary search to find the contiguous index in
375 // petsc_cols (resp. petsc_row) corresponding to global index j_val
376 std::pair<const int*, const int*> p =
377 std::equal_range (&petsc_cols[0], &petsc_cols[0] + ncols, j_val);
378 // Found an entry for j_val
379 if (p.first != p.second) {
380 // The entry in the contiguous row corresponding to the j_val column of interest
381 const int j = std::distance (const_cast<int*> (&petsc_cols[0]), const_cast<int*> (p.first));
382
383 assert (j < ncols);
384 assert (petsc_cols[j] == j_val);
385 value = static_cast<double> (petsc_row[j]);
386 }
387 ierr = MatRestoreRow (_mat, i_val,
388 &ncols, &petsc_cols, &petsc_row);
389 CHKERRABORT (MPI_COMM_WORLD, ierr);
390 return value;
391 }

395 inline int PetscMatrix::MatGetRowM (const int i_val, // index i
396 int cols[],
397 double vals[]
398 ) {
399 // Set up
400 assert (this->initialized());
401 const PetscScalar *petsc_row;
402 const PetscInt *petsc_cols;
403 int ierr = 0, ncols = 0;
404 // Get row
405 assert (this->closed()); // the matrix needs to be closed for this to work
406 ierr = MatGetRow (_mat, i_val, &ncols, &petsc_cols, &petsc_row);
407 CHKERRABORT (MPI_COMM_WORLD, ierr);
408 // close row
409 ierr = MatRestoreRow (_mat, i_val, &ncols, &petsc_cols, &petsc_row);
410 CHKERRABORT (MPI_COMM_WORLD, ierr);
411 // print row
412 if (&cols[0] != PETSC_NULL) for (int j = 0; j < ncols; j++) {
413 cols[j] = petsc_cols[j];
414 vals[j] = petsc_row[j];
415 }
416 // return # of columns
417 return ncols;
418 }

441 inline void PetscMatrix::matrix_add (const double a_in, SparseMatrix &X_in, const char pattern_type []) {
442 assert (this->initialized());
443
444 // sanity check. but this cannot avoid
445 // crash due to incompatible sparsity structure...
446 assert (this->m() == X_in.m());
447 assert (this->n() == X_in.n());
448
449 PetscScalar a = static_cast<PetscScalar> (a_in);
450 PetscMatrix* X = dynamic_cast<PetscMatrix*> (&X_in);
451
452 assert (X != NULL);
453
454 int ierr = 0;
455
456 // the matrix from which we copy the values has to be assembled/closed
457 // X->close ();
458 assert (X->closed());
459
460 semiparallel_only();
461
462 if (!strcmp (pattern_type, "different_nonzero_pattern")) {
463 ierr = MatAXPY (_mat, a, X->_mat, DIFFERENT_NONZERO_PATTERN);
464 }
465 else if (!strcmp (pattern_type, "subset_nonzero_pattern")) {
466 ierr = MatAXPY (_mat, a, X->_mat, SUBSET_NONZERO_PATTERN);
467 }
468 else if (!strcmp (pattern_type, "same_nonzero_pattern")) {
469 ierr = MatAXPY (_mat, a, X->_mat, SAME_NONZERO_PATTERN);
470 }
471 else {
472 std::cout << "Error in pattern_type in function PetscMatrix::matrix_add " << std::endl;
473 exit (0);
474 }
475 CHKERRABORT (MPI_COMM_WORLD, ierr);
476 }

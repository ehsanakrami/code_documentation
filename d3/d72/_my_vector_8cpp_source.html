167 template <class Type> void MyVector<Type>::stack() {
168
169 if(!_serial) {
170 std::cout << "Error in MyVector.stack(), vector is in " << status() << " status" << std::endl;
171 abort();
172 }
173
174 _offset.resize(_nprocs+1);
175 _offset[0]=0;
176
177 for(unsigned jproc = 0; jproc<_nprocs; jproc++ ){
178 if(jproc != _iproc){
179 MPI_Send(&_size, 1, MPI_UNSIGNED, jproc, 1, MPI_COMM_WORLD);
180 MPI_Recv(&_offset[jproc+1], 1, MPI_UNSIGNED, jproc, 1, MPI_COMM_WORLD, NULL);
181 }
182 else{
183 _offset[_iproc+1]=_size;
184 }
185 }
186
187 for(unsigned i=0;i<_nprocs;i++){
188 _offset[i+1] += _offset[i];
189 }
190
191 _serial = false;
192
193 _begin = _offset[_iproc];
194 _end = _offset[_iproc + 1];
195 _size = _end - _begin;
196
197 }

207 template <class Type> void MyVector<Type>::buildOffset() {
208 unsigned locsize = _size / _nprocs;
209 unsigned reminder = _size % _nprocs;
210 _offset.resize(_nprocs + 1);
211 _offset[0] = 0;
212 for(unsigned i = 1; i < _nprocs; i++) {
213 _offset[i] = _offset[i - 1] + locsize;
214 if(i <= reminder) _offset[i] += 1;
215 }
216 _offset[_nprocs] = _size;
217 }

138 void FE_convergence< type >::convergence_study(MultiLevelProblem & ml_prob,
139 MultiLevelMesh & ml_mesh,
140 MultiLevelMesh & ml_mesh_all_levels,
141 const unsigned max_number_of_meshes,

144 const unsigned volume_or_boundary,

150 const MultiLevelSolution::BoundaryFuncMLProb SetBoundaryCondition
151 ) {
152
153 // Controls - BEGIN
154 if ( equation_solve == true && SetBoundaryCondition == NULL) { std::cout << "Must provide a BC function" << std::endl; abort(); }
155
156 if (conv_rate_computation_method == 1 && exact_sol.size() == 0) { std::cout << "Must provide exact sol" << std::endl; abort(); }
157 // Controls - END
158
159
160
161
162 std::vector < std::vector < std::vector < std::vector < double > > > > norms = FE_convergence::initialize_vector_of_norms ( volume_or_boundary,
163 unknowns.size(),
164 max_number_of_meshes, norm_flag);

167 unknowns,
168 exact_sol,
169 ml_mesh_all_levels,
170 max_number_of_meshes,
171 SetInitialCondition,
172 SetBoundaryCondition,
173 equation_solve);
174
175 //prepare Abstract quantities for all fe fams for all geom elems: all quadrature evaluations are performed beforehand in the main function
176 std::vector < std::vector < /*const*/ elem_type_templ_base<type, double> * > > elem_all;
177 ml_prob.get_all_abstract_fe(elem_all);
178
179 for (int lev = 0; lev < max_number_of_meshes; lev++) {
180
181 const MultiLevelSolution ml_sol_single_level = main_in.run_on_single_level(ml_prob,
182 ml_mesh,
183 lev,
184 unknowns,
185 exact_sol,
186 SetInitialCondition,
187 SetBoundaryCondition,
188 equation_solve
189 );

191
192 FE_convergence::compute_error_norms_per_unknown_per_level ( elem_all,
193 ml_prob.GetQuadratureRuleAllGeomElems(),
194 & ml_sol_single_level,
195 & ml_sol_all_levels,
196 unknowns,
197 lev,
198 norm_flag,
199 norms,

201 volume_or_boundary,

212template < class type>
213/*static*/ std::vector < std::vector < std::vector < std::vector < type > > > > FE_convergence< type >::initialize_vector_of_norms(const unsigned volume_or_boundary,
214 const unsigned unknowns_size,
215 const unsigned max_number_of_meshes, const unsigned norm_flag) {
216
217 // VB, how many Unknowns, how many mesh levels, how many norms
218
219 std::vector < std::vector < std::vector < std::vector < type > > > > norms( volume_or_boundary + 1 );
220
221 for (unsigned int vb = 0; vb < norms.size(); vb++) { //0: volume, 1: boundary
222 norms[vb].resize( unknowns_size );
223 for (unsigned int u = 0; u < norms[vb].size(); u++) {
224 norms[vb][u].resize( max_number_of_meshes );
225 for (int i = 0; i < norms[vb][u].size(); i++) { // loop on the mesh level
226 norms[vb][u][i].resize(norm_flag + 1);
227 std::fill(norms[vb][u][i].begin(), norms[vb][u][i].end(), 0.);
228 }
229 }
230 }
231
232 return norms;
233
234}

238template < class type>
239/*static*/ const MultiLevelSolution FE_convergence< type >::initialize_convergence_study(MultiLevelProblem & ml_prob,
240 const std::vector< Unknown > & unknowns,
241 const std::vector< Math::Function< double > * > & exact_sol,
242 MultiLevelMesh & ml_mesh_all_levels,
243 const unsigned max_number_of_meshes,
244 const MultiLevelSolution::InitFuncMLProb SetInitialCondition_in,
245 const MultiLevelSolution::BoundaryFuncMLProb SetBoundaryCondition_in,
246 const bool equation_solve
247 ) {
248
249 //Mesh: construct all levels - BEGIN ==================
250 unsigned numberOfUniformLevels_finest = max_number_of_meshes;
251 ml_mesh_all_levels.RefineMesh(numberOfUniformLevels_finest, numberOfUniformLevels_finest, NULL);
252// ml_mesh_all_levels.EraseCoarseLevels(numberOfUniformLevels - 2); // need to keep at least two levels to send u_(i-1) projected(prolongated) into next refinement
253 //Mesh: construct all levels - END ==================
254
255
256 //Solution - BEGIN ==================
257// std::vector < MultiLevelSolution * > ml_sol_all_levels(unknowns.size());
258// ml_sol_all_levels[u] = new MultiLevelSolution (& ml_mesh_all_levels); //with the declaration outside and a "new" inside it persists outside the loop scopes
259 MultiLevelSolution ml_sol_all_levels(& ml_mesh_all_levels);
260 //Solution - END ==================
261
262 //Problem - BEGIN ==================
263 MultiLevelProblem ml_prob_aux(ml_prob);
264 ml_prob_aux.SetMultiLevelMeshAndSolution(& ml_sol_all_levels);
265 //Problem - END ==================
266
267
268 for (unsigned int u = 0; u < unknowns.size(); u++) {
269 ml_sol_all_levels.AddSolution(unknowns[u]._name.c_str(), unknowns[u]._fe_family, unknowns[u]._fe_order); //We have to do so to avoid buildup of AddSolution with different FE families
270 ml_sol_all_levels.set_analytical_function(unknowns[u]._name.c_str(), exact_sol[u]);
271 ml_sol_all_levels.Initialize(unknowns[u]._name.c_str(), SetInitialCondition_in, & ml_prob_aux);
272 }
273
274
275 // Only for an EQUATION - BEGIN ==================
276 if (equation_solve) {
277
278 ml_sol_all_levels.AttachSetBoundaryConditionFunction(SetBoundaryCondition_in);
279 for (unsigned int u = 0; u < unknowns.size(); u++) {
280 ml_sol_all_levels.GenerateBdc(unknowns[u]._name.c_str(), "Steady", & ml_prob_aux);
281 }
282
283 }
284 // Only for an EQUATION - END ==================
285
286 return ml_sol_all_levels;
287}

384 if (conv_rate_computation_method == 1 && ex_sol_in == NULL) { std::cout << "Please provide analytical solution" << std::endl; abort(); }

398 const Solution* sol = ml_sol->GetSolutionLevel(level);
399
400 const unsigned dim = msh->GetDimension();
401 unsigned dim2 = (3 * (dim - 1) + !(dim - 1)); // dim2 is the number of second order partial derivatives (1,3,6 depending on the dimension)
402
403 unsigned iproc = msh->processor_id();
404
405
406 //***************************************************
407 constexpr unsigned int space_dim = 3;
408 const unsigned int dim_offset_grad = 3;

410 // ======================================
411 // reserve memory for the local standar vectors

414
415 //solution variable
416 unsigned sol_uIndex = ml_sol->GetIndex(unknown.c_str()); // ml_sol->GetIndex("u"); // get the position of "u" in the ml_sol object
417 unsigned sol_uType = ml_sol->GetSolutionType(sol_uIndex); // get the finite element type for "u"
418
419
420
421
422
423 CurrentElem < double > geom_element_iel(dim, msh);
424
425
426 //----------------- V initialization - BEGIN
427 //***************************************************
428 std::vector < std::vector < double > > JacI_qp(space_dim);
429 std::vector < std::vector < double > > Jac_qp(dim);
430 for (unsigned d = 0; d < dim; d++) { Jac_qp[d].resize(space_dim); }
431 for (unsigned d = 0; d < space_dim; d++) { JacI_qp[d].resize(dim); }

434 type weight = 0.; // gauss point weight
435 //***************************************************

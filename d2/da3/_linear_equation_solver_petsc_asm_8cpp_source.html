86 void LinearEquationSolverPetscAsm::BuildAMSIndex(const vector <unsigned>& variable_to_be_solved) {
87
88 unsigned nel = _msh->GetNumberOfElements();
89
90 bool FastVankaBlock = true;
91
92 if(_NSchurVar != 0) {
93 FastVankaBlock = (_SolType[_SolPdeIndex[variable_to_be_solved[variable_to_be_solved.size() - _NSchurVar]]] < 3) ? false : true;
94 }
95
96 unsigned iproc = processor_id();
97
98 unsigned DofOffset = KKoffset[0][iproc];
99 unsigned DofOffsetSize = KKoffset[KKIndex.size() - 1][iproc] - KKoffset[0][iproc];
100 vector < unsigned > indexa(DofOffsetSize, DofOffsetSize);
101 vector < unsigned > indexb(DofOffsetSize, DofOffsetSize);
102
103 vector <bool> owned(DofOffsetSize, false);
104
105 map<int, bool> mymap;
106
107 unsigned ElemOffset = _msh->_dofOffset[3][iproc];
108 unsigned ElemOffsetp1 = _msh->_dofOffset[3][iproc + 1];
109 unsigned ElemOffsetSize = ElemOffsetp1 - ElemOffset;
110 vector <PetscInt> indexci(ElemOffsetSize);
111 vector < unsigned > indexc(ElemOffsetSize, ElemOffsetSize);
112
113 vector < vector < unsigned > > block_elements;
114
115 MeshASMPartitioning meshasmpartitioning(*_msh);
116
117 meshasmpartitioning.DoPartition(_elementBlockNumber, block_elements, _blockTypeRange);
118
119 vector <bool> ThisVaribaleIsNonSchur(_SolPdeIndex.size(), true);
120
121 for(unsigned iind = variable_to_be_solved.size() - _NSchurVar; iind < variable_to_be_solved.size(); iind++) {
122 unsigned PdeIndexSol = variable_to_be_solved[iind];
123 ThisVaribaleIsNonSchur[PdeIndexSol] = false;
124 }
125
126 // *** Start Vanka Block ***
127
128 _localIsIndex.resize(block_elements.size());
129 _overlappingIsIndex.resize(block_elements.size());
130
131 for(int vb_index = 0; vb_index < block_elements.size(); vb_index++) { //loop on the vanka-blocks
132 _localIsIndex[vb_index].resize(DofOffsetSize);
133 _overlappingIsIndex[vb_index].resize(DofOffsetSize);
134
135 PetscInt PAsize = 0;
136 PetscInt PBsize = 0;
137
138 PetscInt Csize = 0;
139
140 // ***************** NODE/ELEMENT SERCH *******************
141 for(int kel = 0; kel < block_elements[vb_index].size(); kel++) { //loop on the vanka-block elements
142 unsigned iel = block_elements[vb_index][kel];
143 for(unsigned j = 0; j < _msh->el->GetElementNearElementSize(iel,!FastVankaBlock);j++){
144 unsigned jel = _msh->el->GetElementNearElement(iel,j);
145 if( jel >= ElemOffset && jel<ElemOffsetp1 ){
146
147 //add elements for velocity to be solved
148 if(indexc[jel - ElemOffset] == ElemOffsetSize) {
149 indexci[Csize] = jel - ElemOffset;
150 indexc[jel - ElemOffset] = Csize++;
151
152 //add non-schur variables to be solved
153 for(int indexSol = 0; indexSol < _SolPdeIndex.size(); indexSol++) {
154 if(ThisVaribaleIsNonSchur[indexSol]) {
155 unsigned SolPdeIndex = _SolPdeIndex[indexSol];
156 unsigned SolType = _SolType[SolPdeIndex];
157 unsigned nvej = _msh->GetElementDofNumber(jel, SolType);
158
159 for(unsigned jj = 0; jj < nvej; jj++) {
160 unsigned jdof = _msh->GetSolutionDof(jj, jel, SolType);
161 unsigned kkdof = GetSystemDof(SolPdeIndex, indexSol, jj, jel);
162
163 if(jdof >= _msh->_dofOffset[SolType][iproc] &&
164 jdof < _msh->_dofOffset[SolType][iproc + 1]) {
165 if(indexa[kkdof - DofOffset] == DofOffsetSize && owned[kkdof - DofOffset] == false) {
166 owned[kkdof - DofOffset] = true;
167 _localIsIndex[vb_index][PAsize] = kkdof;
168 indexa[kkdof - DofOffset] = PAsize++;
169 }
170
171 if(indexb[kkdof - DofOffset] == DofOffsetSize) {
172 _overlappingIsIndex[vb_index][PBsize] = kkdof;
173 indexb[kkdof - DofOffset] = PBsize++;
174 }
175 }
176 else {
177 mymap[kkdof] = true;
178 }
179 }
180 }
181 }
182 }
183 }
184 }
185
186 //-----------------------------------------------------------------------------------------
187 //Add Schur nodes (generally pressure type variables) to be solved
188 {
189 for(int indexSol = 0; indexSol < _SolPdeIndex.size(); indexSol++) {
190 if(!ThisVaribaleIsNonSchur[indexSol]) {
191 unsigned SolPdeIndex = _SolPdeIndex[indexSol];
192 unsigned SolType = _SolType[SolPdeIndex];
193 unsigned nvei = _msh->GetElementDofNumber(iel, SolType);
194
195 for(unsigned ii = 0; ii < nvei; ii++) {
196 unsigned inode_Metis = _msh->GetSolutionDof(ii, iel, SolType);
197 unsigned kkdof = GetSystemDof(SolPdeIndex, indexSol, ii, iel);
198
199 if(inode_Metis >= _msh->_dofOffset[SolType][iproc] &&
200 inode_Metis < _msh->_dofOffset[SolType][iproc + 1]) {
201 if(indexa[kkdof - DofOffset] == DofOffsetSize && owned[kkdof - DofOffset] == false) {
202 owned[kkdof - DofOffset] = true;
203 _localIsIndex[vb_index][PAsize] = kkdof;
204 indexa[kkdof - DofOffset] = PAsize++;
205 }
206
207 if(indexb[kkdof - DofOffset] == DofOffsetSize) {
208 _overlappingIsIndex[vb_index][PBsize] = kkdof;
209 indexb[kkdof - DofOffset] = PBsize++;
210 }
211 }
212 else {
213 mymap[kkdof] = true;
214 }
215 }
216 }
217 }
218 }
219 //-----------------------------------------------------------------------------------------
220 }
221
222 // *** re-initialize indeces(a,c,d)
223 for(PetscInt i = 0; i < PAsize; i++) {
224 indexa[_localIsIndex[vb_index][i] - DofOffset] = DofOffsetSize;
225 }
226
227 for(PetscInt i = 0; i < PBsize; i++) {
228 indexb[_overlappingIsIndex[vb_index][i] - DofOffset] = DofOffsetSize;
229 }
230
231 for(PetscInt i = 0; i < Csize; i++) {
232 indexc[indexci[i]] = ElemOffsetSize;
233 }
234
235 _localIsIndex[vb_index].resize(PAsize);
236 std::vector < PetscInt >(_localIsIndex[vb_index]).swap(_localIsIndex[vb_index]);
237
238 _overlappingIsIndex[vb_index].resize(PBsize + mymap.size());
239 int i = 0;
240 for(std::map<int, bool>::iterator it = mymap.begin(); it != mymap.end(); ++it, ++i) {
241 _overlappingIsIndex[vb_index][PBsize + i] = it->first;
242 }
243 std::vector < PetscInt >(_overlappingIsIndex[vb_index]).swap(_overlappingIsIndex[vb_index]);
244
245
246 mymap.clear();
247
248 std::sort(_localIsIndex[vb_index].begin(), _localIsIndex[vb_index].end());
249 std::sort(_overlappingIsIndex[vb_index].begin(), _overlappingIsIndex[vb_index].end());
250
251
252 }
253
254 //BEGIN Generate std::vector<IS> for ASM PC ***********
255 _localIs.resize(_localIsIndex.size());
256 _overlappingIs.resize(_overlappingIsIndex.size());
257
258 for(unsigned vb_index = 0; vb_index < _localIsIndex.size(); vb_index++) {
259 ISCreateGeneral(MPI_COMM_SELF, _localIsIndex[vb_index].size(), &_localIsIndex[vb_index][0], PETSC_USE_POINTER , &_localIs[vb_index]);
260 ISCreateGeneral(MPI_COMM_SELF, _overlappingIsIndex[vb_index].size(), &_overlappingIsIndex[vb_index][0], PETSC_USE_POINTER , &_overlappingIs[vb_index]);
261 }
262
263 //END Generate std::vector<IS> for ASM PC ***********
264
265 return;
266 }

270 void LinearEquationSolverPetscAsm::SetPreconditioner(KSP& subksp, PC& subpc) {
271
272 PetscPreconditioner::set_petsc_preconditioner_type(ASM_MULTIPLICATIVE_PRECOND, subpc);
273
274 if(!_standardASM) {
275 PCASMSetLocalSubdomains(subpc, _localIsIndex.size(), &_overlappingIs[0], &_localIs[0]);
276 }
277 PCASMSetOverlap(subpc, _overlap);
278
279 KSPSetUp(subksp);
280
281 KSP* subksps;
282 PCASMGetSubKSP(subpc, &_nlocal, PETSC_NULL, &subksps);
283 PetscReal epsilon = 1.e-16;
284
285 if(!_standardASM) {
286 for(int i = 0; i < _blockTypeRange[1]; i++) {
287 PC subpcs;
288 KSPGetPC(subksps[i], &subpcs);
289 KSPSetTolerances(subksps[i], PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT, 1);
290 KSPSetFromOptions(subksps[i]);
291 PetscPreconditioner::set_petsc_preconditioner_type(MLU_PRECOND, subpcs);
292 PCFactorSetZeroPivot(subpcs, epsilon);
293 PCFactorSetShiftType(subpcs, MAT_SHIFT_NONZERO);
294 }
295
296 for(int i = _blockTypeRange[1]; i < _blockTypeRange[2]; i++) {
297 PC subpcs;
298 KSPGetPC(subksps[i], &subpcs);
299 KSPSetTolerances(subksps[i], PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT, 1);
300 KSPSetFromOptions(subksps[i]);
301
302 if(this->_preconditioner_type == ILU_PRECOND)
303 PCSetType(subpcs, (char*) PCILU);
304 else
305 PetscPreconditioner::set_petsc_preconditioner_type(this->_preconditioner_type, subpcs);
306
307 PCFactorSetZeroPivot(subpcs, epsilon);
308 PCFactorSetShiftType(subpcs, MAT_SHIFT_NONZERO);
309 }
310 }
311 else {
312 for(int i = 0; i < _nlocal; i++) {
313 PC subpcs;
314 KSPGetPC(subksps[i], &subpcs);
315 KSPSetTolerances(subksps[i], PETSC_DEFAULT, PETSC_DEFAULT, PETSC_DEFAULT, 1);
316 KSPSetFromOptions(subksps[i]);
317
318 if(this->_preconditioner_type == ILU_PRECOND)
319 PCSetType(subpcs, (char*) PCILU);
320 else
321 PetscPreconditioner::set_petsc_preconditioner_type(this->_preconditioner_type, subpcs);
322
323 PCFactorSetZeroPivot(subpcs, epsilon);
324 PCFactorSetShiftType(subpcs, MAT_SHIFT_NONZERO);
325 }
326 }
327 }

219void SystemTwo::Initialize() {
220
221
222 const uint coords_fine_offset = GetMLProb().GetMeshTwo()._NoNodesXLev[GetGridn()-1];
223 std::vector<double> xp(GetMLProb().GetMeshTwo().get_dim());
224 std::vector<double> u_value(_dofmap._n_vars);
225
226 std::cout << "\n====================== Initialize: Now we are setting them for all levels! ========================" << "\n \n";
227
228 for (uint Level = 0; Level< GetGridn(); Level++) {
229
230 Mesh *mymsh = GetMLProb()._ml_msh->GetLevel(Level);
231 const unsigned myproc = mymsh->processor_id();
232
233 uint iel_b = GetMLProb().GetMeshTwo()._off_el[VV][ GetMLProb().GetMeshTwo()._iproc*GetGridn() + Level ];
234 uint iel_e = GetMLProb().GetMeshTwo()._off_el[VV][ GetMLProb().GetMeshTwo()._iproc*GetGridn() + Level + 1];
235
236 for (uint iel=0; iel < (iel_e - iel_b); iel++) {
237
238 CurrentElem<double> currelem(iel,myproc,Level,VV,this,GetMLProb().GetMeshTwo(),GetMLProb().GetElemType(),mymsh);
239
240 currelem.SetDofobjConnCoords();
241
242 const uint el_dof_objs = NVE[ GetMLProb().GetMeshTwo()._geomelem_flag[currelem.GetDim()-1] ][BIQUADR_FE];
243
244 for (uint q=0; q < _UnknownQuantitiesVector.size() ; q++) {
245
246 std::vector<double> value(_UnknownQuantitiesVector[q]->_dim,0.);
247 //the fact is that THERE ARE DIFFERENT DOF OBJECTS for DIFFERENT FE families
248 //for each family we should only pick the dof objects that are needed
249 //what changes between the FE families is the DOF OBJECT YOU PROVIDE: it could be a NODE or a CELL
250 // Notice that for some elements you don't have the midpoint of the element!
251
252 for (uint ivar=0; ivar < _UnknownQuantitiesVector[q]->_dim; ivar++) {
253
254 for (uint k=0; k < currelem.GetElemType(_UnknownQuantitiesVector[q]->_FEord)->GetNDofs() ; k++) {
255
256 const int fine_node = GetMLProb().GetMeshTwo()._el_map[VV][ k + ( iel + iel_b )*el_dof_objs ];
257 for (uint idim = 0; idim < GetMLProb().GetMeshTwo().get_dim(); idim++) xp[idim] = GetMLProb().GetMeshTwo()._xyz[ fine_node + idim*coords_fine_offset ];
258
259 _UnknownQuantitiesVector[q]->initialize_xyz(&xp[0],value);
260 const int dof_pos_lev = _dofmap.GetDofQuantityComponent(Level,_UnknownQuantitiesVector[q],ivar,fine_node);
261 _LinSolver[Level]->_EPS->set( dof_pos_lev, value[ivar] );
262
263 } //dof objects
264 }
265
266 } //qty
267
268
269 } // end of element loop
270
271 _LinSolver[Level]->_EPS->localize(*_LinSolver[Level]->_EPSC);
272 _LinSolver[Level]->_EPSC->close();
273
274 } //end Level
275
276
277#ifdef DEFAULT_PRINT_INFO
278 std::cout << "\n Initialize(Base): Initial solution defined by ic_read" << "\n \n";
279#endif
280
281 return;
282}

94 void MultiLevelSolution::AddSolutionLevel() {
95 // add level solution
96 _solution.resize(_gridn + 1);
97 _solution[_gridn] = new Solution(_mlMesh->GetLevel(_gridn));
98
99 // add all current solutions and initialize to zero
100 for(unsigned i = 0; i < _solName.size(); i++) {
101 _solution[_gridn]->AddSolution(_solName[i], _family[i], _order[i], _solTimeOrder[i], _pdeType[i]);
102 }
103
104 for(unsigned k = 0; k < _solName.size(); k++) {
105 _solution[_gridn]->ResizeSolutionVector(_solName[k]);
106 _solution[_gridn]->_Sol[k]->matrix_mult(*_solution[_gridn - 1]->_Sol[k],
107 *_mlMesh->GetLevel(_gridn)->GetCoarseToFineProjection(_solType[k]));
108 _solution[_gridn]->_Sol[k]->close();
109 if(_solTimeOrder[k] == 2) {
110 _solution[_gridn]->_SolOld[k]->matrix_mult(*_solution[_gridn - 1]->_SolOld[k],
111 *_mlMesh->GetLevel(_gridn)->GetCoarseToFineProjection(_solType[k]));
112 _solution[_gridn]->_SolOld[k]->close();
113 }
114 }
115
116 _gridn++;
117
118 for(int k = 0; k < _solName.size(); k++) {
119 GenerateBdc(k, _gridn - 1, 0.);
120
121 }
122
123 }

127 unsigned tmorder, const bool& PdeType) {
128
129 unsigned n = _solType.size();

131 _family.resize(n + 1u);

133 _solName.resize(n + 1u);
134 _bdcType.resize(n + 1u);

140 _solPairIndex.resize(n + 1u);
141 _solPairInverseIndex.resize(n + 1u);

161 << std::setw(12) << order << " and time discretization order " << tmorder << endl;

164 _solution[ig]->AddSolution(_solName[n], _family[n], _order[n], _solTimeOrder[n], _pdeType[n]);

171 void MultiLevelSolution::AddSolution(const char name[], const FEFamily fefamily, const FEOrder order_v, const FEOrder order_b,
172 unsigned tmorder, const bool& PdeType) {
173
174
175 unsigned n = _solType.size();
176
177// ---------------------
178//------ first resize -----------
179// ---------------------
180
181
182// ID related---
183 _solName.resize(n + 1u);
184
185// is_an_unknown_of_a_pde---
186 _pdeType.resize(n + 1u);
187
188// For pressure variables---
189 _testIfPressure.resize(n + 1u);
190 _addAMRPressureStability.resize(n + 1u);
191// For pressure variables, fix it at one point for PCFieldSplit, in Boundary Conditions ---
192 _fixSolutionAtOnePoint.resize(n + 1u);
193
194// For FSI, pairing velocities and displacements ---
195 _solPairIndex.resize(n + 1u);
196 _solPairInverseIndex.resize(n + 1u);
197
198// Time discretization---
199 _solTimeOrder.resize(n + 1u);
200
201// These are the things that should change for different FE... - BEGIN ------
202// FE related---
203 _family.resize(n + 1u);
204 _order.resize(n + 1u);
205 _solType.resize(n + 1u);
206
207// FE related and Time related: Boundary Conditions---
208 _bdcType.resize(n + 1u);
209// These are the things that should change for different FE... - END ------
210
211
212// ---------------------
213//------ then fill -----------
214// ---------------------
215
216// ID related---
217 _solName[n] = new char [DEFAULT_SOL_NCHARS];
218 strcpy(_solName[n], name);
219
220// is_an_unknown_of_a_pde---
221 _pdeType[n] = PdeType;
222
223// For pressure variables---
224 _testIfPressure[n] = 0;
225 _addAMRPressureStability[n] = false;
226// For pressure variables, if solution is fixed at one point (then null space must be removed) ---
227 _fixSolutionAtOnePoint[n] = false;
228
229// For FSI, pairing velocities and displacements ---
230 _solPairIndex[n] = n;
231 _solPairInverseIndex[n] = n;
232
233// Time discretization---
234 _solTimeOrder[n] = tmorder;
235
236// These are the things that should change for different FE...------
237// FE related---
238 _family[n] = fefamily;
239 _order[n] = order_v;
240// _order_b[n] = order_b;
241 _solType[n] = Solution::compute_fe_sol_type(fefamily, order_v, order_b);
242
243// FE related and Time related: Boundary Conditions
244 _bdcType[n] = new char [20];
245 sprintf(_bdcType[n], "undefined");
246// These are the things that should change for different FE... - end------
247
248
249// ---------------------
250 cout << " Add variable " << std::setw(3) << _solName[n] << " discretized with FE type "
251 << std::setw(12) << order_v << " and time discretization order " << tmorder << endl;
252
253 for(unsigned ig = 0; ig < _gridn; ig++) {
254 _solution[ig]->AddSolution(_solName[n], _family[n], _order[n], order_b, _solTimeOrder[n], _pdeType[n]);
255 }
256 }

329 void MultiLevelSolution::Initialize(const char name[], InitFunc func, InitFuncMLProb funcMLProb, const MultiLevelProblem* ml_prob) {
330
331
332
333 std::vector< unsigned > sol_start_end = solution_start_and_end(std::string (name));
334
335
336 for(unsigned i = sol_start_end[0]; i < sol_start_end[1]; i++) {
337 unsigned sol_type = _solType[i];
338
339 for(unsigned ig = 0; ig < _gridn; ig++) {
340
341 _solution[ig]->ResizeSolutionVector(_solName[i]);
342 _solution[ig]->_Sol[i]->zero();
343
344 if(func || funcMLProb) {
345 double value;
346
347 if(sol_type < 3) {
348 for(int isdom = _iproc; isdom < _iproc + 1; isdom++) {
349 for(int iel = _mlMesh->GetLevel(ig)->_elementOffset[isdom];
350 iel < _mlMesh->GetLevel(ig)->_elementOffset[isdom + 1]; iel++) {
351 unsigned nloc_dof = _mlMesh->GetLevel(ig)->GetElementDofNumber(iel, sol_type);
352
353 for(int j = 0; j < nloc_dof; j++) {
354 unsigned inode_Metis = _mlMesh->GetLevel(ig)->GetSolutionDof(j, iel, sol_type);
355 unsigned icoord_Metis = _mlMesh->GetLevel(ig)->GetSolutionDof(j, iel, 2);
356 std::vector < double > xx(3);
357 xx[0] = (*_mlMesh->GetLevel(ig)->_topology->_Sol[0])(icoord_Metis);
358 xx[1] = (*_mlMesh->GetLevel(ig)->_topology->_Sol[1])(icoord_Metis);
359 xx[2] = (*_mlMesh->GetLevel(ig)->_topology->_Sol[2])(icoord_Metis);
360
361 value = (func) ? func(xx) : funcMLProb(ml_prob, xx, name);
362
363 _solution[ig]->_Sol[i]->set(inode_Metis, value);
364
365 if(_solTimeOrder[i] == 2) {
366 _solution[ig]->_SolOld[i]->set(inode_Metis, value);
367 }
368 }
369 }
370 }
371 }
372 else if(sol_type < 5) {
373 for(int isdom = _iproc; isdom < _iproc + 1; isdom++) {
374 for(int iel = _mlMesh->GetLevel(ig)->_elementOffset[isdom];
375 iel < _mlMesh->GetLevel(ig)->_elementOffset[isdom + 1]; iel++) {
376
377 unsigned nloc_dof = _mlMesh->GetLevel(ig)->GetElementDofNumber(iel, 0);
378
379 // evaluate at average of biquadratic nodes (element center) - BEGIN -----
380 std::vector < double > xx(3, 0.);
381
382 for(int j = 0; j < nloc_dof; j++) {
383 unsigned icoord_Metis = _mlMesh->GetLevel(ig)->GetSolutionDof(j, iel, 2);
384 xx[0] += (*_mlMesh->GetLevel(ig)->_topology->_Sol[0])(icoord_Metis);
385 xx[1] += (*_mlMesh->GetLevel(ig)->_topology->_Sol[1])(icoord_Metis);
386 xx[2] += (*_mlMesh->GetLevel(ig)->_topology->_Sol[2])(icoord_Metis);
387 }
388
389 xx[0] /= nloc_dof;
390 xx[1] /= nloc_dof;
391 xx[2] /= nloc_dof;
392 // evaluate at average of biquadratic nodes (element center) - END -----
393
394 value = (func) ? func(xx) : funcMLProb(ml_prob, xx, name);
395
396 unsigned placeholder_index = 0/*2*/;
397
398 unsigned solDof = _mlMesh->GetLevel(ig)->GetSolutionDof(placeholder_index, iel, sol_type);
399
400 _solution[ig]->_Sol[i]->set(solDof, value);
401
402 if(_solTimeOrder[i] == 2) {
403 _solution[ig]->_SolOld[i]->set(solDof, value);
404 }
405 }
406 }
407 }
408
409 _solution[ig]->_Sol[i]->close();
410
411 if(_solTimeOrder[i] == 2) {
412 _solution[ig]->_SolOld[i]->close();
413 }
414 }
415
416 }
417 }
418
419 return;
420 }

552 void MultiLevelSolution::SetBoundaryCondition_new(const std::string name, const std::string facename,
553 const BDCType bdctype, const bool istimedependent, FunctionBase* func) {
554
555
556
557 unsigned int ivar = GetIndex(name.c_str());
558 unsigned int iface = 0;
559 bool ishomogeneous = true;
560
561 if(func != NULL) {
562 ishomogeneous = false;
563 }
564
565 std::map<unsigned int, std::string>::iterator iter;
566 iter = _mlMesh->GetLevel(0)->_boundaryinfo.begin();
567
568 for(iter = _mlMesh->GetLevel(0)->_boundaryinfo.begin(); iter != _mlMesh->GetLevel(0)->_boundaryinfo.end(); ++iter) {

570 iface = iter->first;
571 break;
572 }
573 }
574
575 if(iter == _mlMesh->GetLevel(0)->_boundaryinfo.end()) {
576 std::cout << " Error: the facename " << facename << " does not exist!" << std::endl;
577 exit(1);
578 }
579
580 _boundaryConditions[ivar][iface] = bdctype;
581 _isHomogeneous[ivar][iface] = ishomogeneous;
582 _nonHomogeneousBCFunction[ivar][iface] = func;
583
584 }

594void MultiLevelSolution::GenerateBdc(const char* name, const char* bdc_type, const MultiLevelProblem* ml_prob) {
595
596 if(_useParsedBCFunction == false && _bdcFuncSet == false && _bdcFuncSetMLProb == false) {
597 cout << "Error: The boundary condition user-function is not set! Please call the AttachSetBoundaryConditionFunction routine"
598 << endl;
599 abort();
600 }
601
602 if(_bdcFuncSetMLProb == true && ml_prob != NULL) {
603 _mlBCProblem = ml_prob;
604 }
605 else if(_bdcFuncSetMLProb == true && ml_prob == NULL) {
606 std::cout << "Warning, no MultilevelProblem pointer has been passed in MultiLevelSolution::GenerateBdc(...}!!!" << std::endl;
607 std::cout << "The MultiLevelProblem pointer in the boundary function will be unavailable, and if used will cause segmentation-fault." << std::endl;
608 std::cout << "It is highly recommended either to pass the MultilevelProblem pointer in MultiLevelSolution::GenerateBdc(...} or to use" << std::endl;
609 std::cout << "MultiLevelSolution::AttachSetBoundaryConditionFunction( (*BoundaryFunc) ) rather than" << std::endl;
610 std::cout << "MultiLevelSolution::AttachSetBoundaryConditionFunction( (*BoundaryFuncMLProb) )" << std::endl;
611 }
612
613 unsigned i_start;
614 unsigned i_end;
615
616 if(!strcmp(name, "All") || !strcmp(name, "all") || !strcmp(name, "ALL")) {
617
618 i_start = 0;
619 i_end = _solType.size();
620
621 for(unsigned k = i_start; k < i_end; k++) {
622 if(_solution[0]->_ResEpsBdcFlag[k]) {
623 sprintf(_bdcType[k], "Steady");
624 cout << " Set " << std::setw(15) << _bdcType[k] << " Boundary_condition"
625 << " for variable " << std::setw(3) << _solName[k] << endl;
626 }
627 else {
628 sprintf(_bdcType[k], "Not-available");
629 }
630
631 }
632 }
633 else {
634 i_start = GetIndex(name);
635 i_end = i_start + 1u;
636
637 if(_solution[0]->_ResEpsBdcFlag[i_start]) {
638 if(!strcmp(bdc_type, "Steady")) {
639 strcpy(_bdcType[i_start], bdc_type);
640 }
641 else if(!strcmp(bdc_type, "Time_dependent")) {
642 strcpy(_bdcType[i_start], bdc_type);
643 }
644 else {
645 cout << "Error! Invalid boundary condition specified for " << _solName[i_start]
646 << " in GenerateBdc function" << endl;
647 exit(1);
648 }
649
650 cout << " Set " << std::setw(14) << _bdcType[i_start] << " Boundary_condition"
651 << " for variable " << std::setw(3) << _solName[i_start] << endl;
652 }
653 else {
654 sprintf(_bdcType[i_start], "Not-available");
655 }
656 }
657
658 for(unsigned i = i_start; i < i_end; i++) {
659 GenerateBdc(i, 0, 0.);
660 }
661 }

674//---------------------------------------------------------------------------------------------------
679 void MultiLevelSolution::GenerateBdc(const unsigned int k, const unsigned int grid0, const double time) {
680
681 for(unsigned igridn = grid0; igridn < _gridn; igridn++) {
682 if(_solution[igridn]->_ResEpsBdcFlag[k]) {
683 Mesh* msh = _mlMesh->GetLevel(igridn);
684
685 std::vector < std::map < unsigned, std::map < unsigned, double > > > & amrRestriction = msh->GetAmrRestrictionMap();
686
687 // default Neumann
688 for(unsigned j = msh->dofmap_get_dof_offset(_solType[k], _iproc); j < msh->dofmap_get_dof_offset(_solType[k], _iproc + 1); j++) {
689 _solution[igridn]->_Bdc[k]->set(j, 2.);
690 }
691
692 if(_solType[k] < 3) { // boundary condition for lagrangian elements
693 //AMR - related
694 for(int iel = msh->_elementOffset[_iproc]; iel < msh->_elementOffset[_iproc + 1]; iel++) {
695 for(unsigned jface = 0; jface < msh->GetElementFaceNumber(iel); jface++) {
696 if(msh->el->GetBoundaryIndex(iel, jface) == 0) { // interior boundary (AMR) u = 0
697 short unsigned ielt = msh->GetElementType(iel);
698 unsigned nv1 = msh->GetElementFaceDofNumber(iel, jface, _solType[k]); // only the face dofs
699 for(unsigned iv = 0; iv < nv1; iv++) {
700 unsigned i = msh->GetLocalFaceVertexIndex(iel, jface, iv);
701 unsigned idof = msh->GetSolutionDof(i, iel, _solType[k]);
702 if(amrRestriction[_solType[k]].find(idof) != amrRestriction[_solType[k]].end() &&
703 amrRestriction[_solType[k]][idof][idof] == 0) {
704 _solution[igridn]->_Bdc[k]->set(idof, 1.);
705 }
706 }
707 }
708 }
709 }
710 //AMR - related - end
711 for(int iel = msh->_elementOffset[_iproc]; iel < msh->_elementOffset[_iproc + 1]; iel++) {
712
713 for(unsigned jface = 0; jface < msh->GetElementFaceNumber(iel); jface++) {
714

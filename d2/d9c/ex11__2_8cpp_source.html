698  }
699 
700  }
701  }
702

704 
705  //BEGIN phiV_i loop: Momentum balance
706  for(unsigned i = 0; i < nDofsV; i++) {
707 
708  for(unsigned k = 0; k < dim; k++) {
709  unsigned irow = nDofsT + k * nDofsV + i;
710 
711  for(unsigned l = 0; l < dim; l++) {
712  Res[irow] += -sqrt(Pr / Ra) * phiV_x[i * dim + l] * (gradSolV_gss[k][l] + gradSolV_gss[l][k]) * weight;
713  Res[irow] += -phiV[i] * solV_gss[l] * gradSolV_gss[k][l] * weight;
714  }
715 
716  Res[irow] += solP_gss * phiV_x[i * dim + k] * weight;
717 
718  if(k == 1) {
719  Res[irow] += beta * solT_gss * phiV[i] * weight;
720  }
721 
722  if(assembleMatrix) {
723  unsigned irowMat = nDofsTVP * irow;
724 
725  for(unsigned l = 0; l < dim; l++) {
726  for(unsigned j = 0; j < nDofsV; j++) {
727  unsigned jcol1 = (nDofsT + k * nDofsV + j);
728  unsigned jcol2 = (nDofsT + l * nDofsV + j);
729  Jac[ irowMat + jcol1] += sqrt(Pr / Ra) * phiV_x[i * dim + l] * phiV_x[j * dim + l] * weight;
730  Jac[ irowMat + jcol2] += sqrt(Pr / Ra) * phiV_x[i * dim + l] * phiV_x[j * dim + k] * weight;
731  Jac[ irowMat + jcol1] += phiV[i] * solV_gss[l] * phiV_x[j * dim + l] * weight;
732  Jac[ irowMat + jcol2] += phiV[i] * phiV[j] * gradSolV_gss[k][l] * weight;
733  }
734  }
735 
736  for(unsigned j = 0; j < nDofsP; j++) {
737  unsigned jcol = (nDofsT + dim * nDofsV) + j;
738  Jac[ irowMat + jcol] += - phiV_x[i * dim + k] * phiP[j] * weight;
739  }
740 
741  if(k == 1) {
742  for(unsigned j = 0; j < nDofsT; j++) {
743  Jac[ irowMat + j ] += -beta * phiV[i] * phiT[j] * weight;
744  }
745  }
746  }
747 
748  }
749  }
750 
751  //END phiV_i loop
752 
753  //BEGIN phiP_i loop: mass balance
754  for(unsigned i = 0; i < nDofsP; i++) {
755  unsigned irow = nDofsT + dim * nDofsV + i;
756 
757  for(int k = 0; k < dim; k++) {
758  Res[irow] += (gradSolV_gss[k][k]) * phiP[i] * weight;
759 
760  if(assembleMatrix) {
761  unsigned irowMat = nDofsTVP * irow;
762 
763  for(unsigned j = 0; j < nDofsV; j++) {
764  unsigned jcol = (nDofsT + k * nDofsV + j);
765  Jac[ irowMat + jcol ] += - phiP[i] * phiV_x[j * dim + k] * weight;
766  }
767  }
768 
769  }
770  }
771 
772  //END phiP_i loop
773 
774  }
775 
776  //END Gauss point loop
777 
778  //BEGIN local to global Matrix/Vector assembly
779  RES->add_vector_blocked(Res, sysDof);
780 
781  if(assembleMatrix) {
782  KK->add_matrix_blocked(Jac, sysDof, sysDof);
783  }
784 
785  //END local to global Matrix/Vector assembly
786 
787  }
788 
789  //END element loop
790 
791  RES->close();
792 
793  if(assembleMatrix) {
794  KK->close();
795  }
796

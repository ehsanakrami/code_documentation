44 char* buffer = new char[size + 1000];
45 buffer[size] = 0;

466void AssemblePoissonMatrixandRhs(MultiLevelProblem& ml_prob) {
467
468 //pointers and references
469 LinearImplicitSystem& mylin_impl_sys = ml_prob.get_system<LinearImplicitSystem>("Poisson");
470 const unsigned level = mylin_impl_sys.GetLevelToAssemble();
471
472 Solution* mysolution = ml_prob._ml_sol->GetSolutionLevel(level);
473 LinearEquationSolver* mylsyspde = mylin_impl_sys._LinSolver[level];
474 Mesh* mymsh = ml_prob._ml_msh->GetLevel(level);
475 elem* myel = mymsh->el;
476 SparseMatrix* myKK = mylsyspde->_KK;
477 NumericVector* myRES = mylsyspde->_RES;
478 MultiLevelSolution* ml_sol = ml_prob._ml_sol;
479
480 //data
481 const unsigned dim = mymsh->GetDimension();
482 unsigned nel = mymsh->GetNumberOfElements();
483 unsigned igrid = mymsh->GetLevel();
484 unsigned iproc = mymsh->processor_id();
485
486 //solution variable
487 unsigned SolIndex;
488 unsigned SolPdeIndex;
489 SolIndex = ml_sol->GetIndex("Sol");
490 SolPdeIndex = mylin_impl_sys.GetSolPdeIndex("Sol");
491 //solution order
492 unsigned order_ind = ml_sol->GetSolutionType(SolIndex);
493
494
495 //coordinates
496 vector< vector < double> > coordinates(dim);
497
498 // declare
499 vector< int > metis_node;
500 vector< int > KK_dof;
501 vector <double> phi;
502 vector <double> gradphi;
503 vector <double> nablaphi;
504 double weight;
505 vector< double > F;
506 vector< double > B;
507 vector<double> normal(3.0);
508 double src_term = 0.;
509 vector<double> xyzt(4, 0.);
510 ParsedFunction* bdcfunc = NULL;
511
512 // reserve
513 const unsigned max_size = static_cast< unsigned >(ceil(pow(3, dim)));
514 metis_node.reserve(max_size);
515 KK_dof.reserve(max_size);
516
517 for (int i = 0; i < dim; i++)
518 coordinates[i].reserve(max_size);
519
520 phi.reserve(max_size);
521 gradphi.reserve(max_size * dim);
522 nablaphi.reserve(max_size * (3 * (dim - 1) + !(dim - 1)));
523 F.reserve(max_size);
524 B.reserve(max_size * max_size);
525
526 // Set to zeto all the entries of the Global Matrix
527 myKK->zero();
528
529 // *** element loop ***
530 for (int iel = mymsh->_elementOffset[iproc]; iel < mymsh->_elementOffset[iproc + 1]; iel++) {
531
532 short unsigned ielt = mymsh->GetElementType(iel);
533 unsigned nve = mymsh->GetElementDofNumber(iel, order_ind);
534
535 // resize
536 metis_node.resize(nve);
537 KK_dof.resize(nve);
538 phi.resize(nve);
539 gradphi.resize(nve * dim);
540 nablaphi.resize(nve * (3 * (dim - 1) + !(dim - 1)));
541
542 for (int i = 0; i < dim; i++) {
543 coordinates[i].resize(nve);
544 }
545
546 // set to zero all the entries of the FE matrices
547 F.resize(nve);
548 memset(&F[0], 0, nve * sizeof(double));
549 B.resize(nve * nve);
550 memset(&B[0], 0, nve * nve * sizeof(double));
551
552
553 // get local to global mappings
554 for (unsigned i = 0; i < nve; i++) {
555 unsigned inode_coord_metis = mymsh->GetSolutionDof(i, iel, 2);
556 metis_node[i] = mymsh->GetSolutionDof(i, iel, order_ind);
557
558 for (unsigned ivar = 0; ivar < dim; ivar++) {
559 coordinates[ivar][i] = (*mymsh->_topology->_Sol[ivar])(inode_coord_metis);
560 }
561
562 KK_dof[i] = mylsyspde->GetSystemDof(SolIndex, SolPdeIndex, i, iel);
563 }
564
565
566 // *** Gauss poit loop ***
567 for (unsigned ig = 0; ig < ml_prob._ml_msh->_finiteElement[ielt][order_ind]->GetGaussPointNumber(); ig++) {
568 // *** get Jacobian and test function and test function derivatives ***
569 ml_prob._ml_msh->_finiteElement[ielt][order_ind]->Jacobian(coordinates, ig, weight, phi, gradphi, nablaphi);
570 //current solution
571 double SolT = 0;
572 vector < double > gradSolT(dim, 0.);
573
574 for (unsigned ivar = 0; ivar < dim; ivar++) {
575 gradSolT[ivar] = 0;
576 }
577
578 xyzt.assign(4, 0.);
579 unsigned SolType = ml_sol->GetSolutionType("Sol");
580
581 for (unsigned i = 0; i < nve; i++) {
582 double soli = (*mysolution->_Sol[SolIndex])(metis_node[i]);
583
584 for (unsigned ivar = 0; ivar < dim; ivar++) {
585 xyzt[ivar] += coordinates[ivar][i] * phi[i];
586 }
587
588 SolT += phi[i] * soli;
589
590 for (unsigned ivar2 = 0; ivar2 < dim; ivar2++) gradSolT[ivar2] += gradphi[i * dim + ivar2] * soli;
591 }
592
593 // *** phi_i loop ***
594 for (unsigned i = 0; i < nve; i++) {
595 //BEGIN RESIDUALS A block ===========================
596 double Adv_rhs = 0;
597 double Lap_rhs = 0;
598
599 for (unsigned ivar = 0; ivar < dim; ivar++) {
600 Lap_rhs += gradphi[i * dim + ivar] * gradSolT[ivar];
601 }
602
603
604 //src_term = Source(xyzt);
605#ifdef HAVE_FPARSER
606 //src_term = fpsource.Eval(xyzt);
607 src_term = fpsource(&xyzt[0]);
608#endif
609
610 F[i] += (-Lap_rhs + src_term * phi[i]) * weight;
611
612 //END RESIDUALS A block ===========================
613 // *** phi_j loop ***
614 for (unsigned j = 0; j < nve; j++) {
615 double Lap = 0;
616 double Adv1 = 0;
617
618 for (unsigned ivar = 0; ivar < dim; ivar++) {
619 // Laplacian
620 Lap += gradphi[i * dim + ivar] * gradphi[j * dim + ivar] * weight;
621 }
622
623 B[i * nve + j] += Lap;
624 } // end phij loop
625
626 } // endif assemble_matrix
627 } // end gauss point loop
628
629 //number of faces for each type of element
630 //number of faces for each type of element
631 unsigned nfaces = mymsh->GetElementFaceNumber(iel);
632
633 // loop on faces
634 for (unsigned jface = 0; jface < nfaces; jface++) {
635
636 // look for boundary faces
637 if (myel->GetBoundaryIndex(iel, jface) > 0) {
638
639 unsigned int faceIndex = myel->GetBoundaryIndex(iel, jface) - 1u;
640
641 if (ml_sol->GetBoundaryCondition("Sol", faceIndex) == NEUMANN && !ml_sol->Ishomogeneous("Sol", faceIndex)) {
642
643 bdcfunc = (ParsedFunction*)(ml_sol->GetBdcFunction("Sol", faceIndex));
644 unsigned nve = mymsh->GetElementFaceDofNumber(iel, jface, order_ind);
645 const unsigned felt = mymsh->GetElementFaceType(iel, jface);
646
647 for (unsigned i = 0; i < nve; i++) {
648 unsigned ilocal = mymsh->GetLocalFaceVertexIndex(iel, jface, i);
649 unsigned inode_coord_metis = mymsh->GetSolutionDof(ilocal, iel, 2);
650
651 for (unsigned ivar = 0; ivar < dim; ivar++) {
652 coordinates[ivar][i] = (*mymsh->_topology->_Sol[ivar])(inode_coord_metis);
653 }
654 }
655
656 if (felt != 6)
657 {
658 for (unsigned igs = 0; igs < ml_prob._ml_msh->_finiteElement[felt][order_ind]->GetGaussPointNumber(); igs++) {
659 ml_prob._ml_msh->_finiteElement[felt][order_ind]->JacobianSur(coordinates, igs, weight, phi, gradphi, normal);
660
661 xyzt.assign(4, 0.);
662
663 for (unsigned i = 0; i < nve; i++) {
664 for (unsigned ivar = 0; ivar < dim; ivar++) {
665 xyzt[ivar] += coordinates[ivar][i] * phi[i];
666 }
667 }
668
669 // *** phi_i loop ***
670 for (unsigned i = 0; i < nve; i++) {
671 double surfterm_g = (*bdcfunc)(&xyzt[0]);
672 double bdintegral = phi[i] * surfterm_g * weight;
673 unsigned int ilocalnode = mymsh->GetLocalFaceVertexIndex(iel, jface, i);
674 F[ilocalnode] += bdintegral;
675 }
676 }
677 }
678 else // 1D : the side elems are points and does not still exist the point elem
679 {
680 // in 1D it is only one point
681 xyzt[0] = coordinates[0][0];
682 xyzt[1] = 0.;
683 xyzt[2] = 0.;
684 xyzt[3] = 0.;
685
686 double bdintegral = (*bdcfunc)(&xyzt[0]);
687 unsigned int ilocalnode = mymsh->GetLocalFaceVertexIndex(iel, jface, 0);
688 F[ilocalnode] += bdintegral;
689 }
690 }
691 }
692 }
693
694
695 //--------------------------------------------------------------------------------------------------------
696 //Sum the local matrices/vectors into the global Matrix/vector
697
698 myRES->add_vector_blocked(F, KK_dof);
699 myKK->add_matrix_blocked(B, KK_dof, KK_dof);
700 } //end list of elements loop for each subdomain
701
702 myRES->close();
703 myKK->close();
704
705 // ***************** END ASSEMBLY *******************
706
707}

711double GetRelativeError(MultiLevelSolution& ml_sol, const bool& H1) {
712
713 int iproc, nprocs;
714 MPI_Comm_rank(MPI_COMM_WORLD, &iproc);
715 MPI_Comm_size(MPI_COMM_WORLD, &nprocs);
716
717
718 NumericVector* error_vec;
719 NumericVector* solution_vec;
720 error_vec = NumericVector::build().release();
721 solution_vec = NumericVector::build().release();
722
723 if (nprocs == 1) {
724 error_vec->init(nprocs, 1, false, SERIAL);
725 solution_vec->init(nprocs, 1, false, SERIAL);
726 }
727 else {
728 error_vec->init(nprocs, 1, false, PARALLEL);
729 solution_vec->init(nprocs, 1, false, PARALLEL);
730 }
731
732 error_vec->zero();
733 solution_vec->zero();
734
735 unsigned gridn = ml_sol._mlMesh->GetNumberOfLevels();
736
737 for (int ilevel = gridn - 1; ilevel < gridn; ilevel++) {
738 Solution* solution = ml_sol.GetSolutionLevel(ilevel);
739 Mesh* msh = ml_sol._mlMesh->GetLevel(ilevel);
740 unsigned iproc = msh->processor_id();
741
742
743 const unsigned dim = msh->GetDimension();
744 vector< vector < double> > coordinates(dim);
745
746 vector< int > metis_node;
747
748 vector <double> phi;
749 vector <double> gradphi;
750 vector <double> nablaphi;
751
752 double weight;
753
754 // reserve
755 const unsigned max_size = static_cast< unsigned >(ceil(pow(3, dim)));
756 metis_node.reserve(max_size);
757
758 for (int i = 0; i < dim; i++)
759 coordinates[i].reserve(max_size);
760
761 phi.reserve(max_size);
762 gradphi.reserve(max_size * dim);
763 nablaphi.reserve(max_size * (3 * (dim - 1) + !(dim - 1)));
764
765
766 unsigned SolIndex;
767 SolIndex = ml_sol.GetIndex("Sol");
768 unsigned SolOrder = ml_sol.GetSolutionType(SolIndex);
769
770
771 for (int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
772
773 short unsigned ielt = msh->GetElementType(iel);
774 unsigned nve = msh->GetElementDofNumber(iel, SolOrder);
775 // resize
776 metis_node.resize(nve);
777 phi.resize(nve);
778 gradphi.resize(nve * dim);
779 nablaphi.resize(nve * (3 * (dim - 1) + !(dim - 1)));
780
781 for (int i = 0; i < dim; i++) {
782 coordinates[i].resize(nve);
783 }
784
785 // get local to global mappings
786 for (unsigned i = 0; i < nve; i++) {
787 unsigned inode_coord_metis = msh->GetSolutionDof(i, iel, 2);
788 metis_node[i] = msh->GetSolutionDof(i, iel, SolOrder);
789
790 for (unsigned idim = 0; idim < dim; idim++) {
791 coordinates[idim][i] = (*msh->_topology->_Sol[idim])(inode_coord_metis);
792 }
793 }
794
795 for (unsigned ig = 0; ig < ml_sol._mlMesh->_finiteElement[ielt][SolOrder]->GetGaussPointNumber(); ig++) {
796 // *** get Jacobian and test function and test function derivatives ***
797 ml_sol._mlMesh->_finiteElement[ielt][SolOrder]->Jacobian(coordinates, ig, weight, phi, gradphi, nablaphi);
798 //current solution
799 double SolT = 0;
800 vector < double > gradSolT(dim, 0.);
801
802 for (unsigned ivar = 0; ivar < dim; ivar++) {
803 gradSolT[ivar] = 0;
804 }
805
806 double pi = acos(-1.);
807 double x[4] = {0., 0., 0., 0.};
808
809 unsigned SolType = ml_sol.GetSolutionType("Sol");
810
811 for (unsigned i = 0; i < nve; i++) {
812 double soli = (*solution->_Sol[SolIndex])(metis_node[i]);
813
814 for (unsigned ivar = 0; ivar < dim; ivar++) {
815 x[ivar] += coordinates[ivar][i] * phi[i];
816 }
817
818 SolT += phi[i] * soli;
819
820 for (unsigned ivar2 = 0; ivar2 < dim; ivar2++) {
821 gradSolT[ivar2] += gradphi[i * dim + ivar2] * soli;
822 }
823 }
824
825 double SolExact, dSolExactdx, dSolExactdy;
826#ifdef HAVE_FPARSER
827 SolExact = fp_sol(x);
828 dSolExactdx = fp_dsoldx(x);
829 dSolExactdy = fp_dsoldy(x);
830#endif
831
832 error_vec->add(iproc, ((SolT - SolExact) * (SolT - SolExact) +
833 H1 * ((gradSolT[0] - dSolExactdx) * (gradSolT[0] - dSolExactdx) +
834 (gradSolT[1] - dSolExactdy) * (gradSolT[1] - dSolExactdy))
835 )*weight);
836
837 solution_vec->add(iproc, (SolExact * SolExact +
838 H1 * (dSolExactdx * dSolExactdx + dSolExactdy * dSolExactdy))*weight);
839 }
840 }
841
842 }
843
844 error_vec->close();
845 solution_vec->close();
846
847 double l2_error = error_vec->l1_norm();
848 double l2_solution = solution_vec->l1_norm();
849
850 delete error_vec;
851 delete solution_vec;
852
853 return sqrt(l2_error / l2_solution);
854
855}

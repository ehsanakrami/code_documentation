40 void PetscMatrix::init (const int m,
41 const int n,
42 const int m_l,
43 const int n_l,
44 const int nnz,
45 const int noz) {
46 // Set matrix dimension
47 _m = m;
48 _n = n;
49 _m_l = m_l;
50 _n_l = n_l;
51
52// _len_row.resize(_m_l);
53// _len_diag_row.resize(_m_l);
54 // We allow 0x0 matrices now
55 //if ((m==0) || (n==0))
56 // return;
57
58 // Clear initialized matrices
59 if (this->initialized()) this->clear();
60 this->_is_initialized = true;
61
62 // processor info
63 int proc_id;
64 MPI_Comm_rank (MPI_COMM_WORLD, &proc_id);
65 int numprocs;
66 MPI_Comm_size (MPI_COMM_WORLD, &numprocs);
67 int ierr = 0;
68 int m_global = static_cast<int> (m);
69 int n_global = static_cast<int> (n);
70 int m_local = static_cast<int> (m_l);
71 int n_local = static_cast<int> (n_l);
72 int n_nz = static_cast<int> (nnz);
73 int n_oz = static_cast<int> (noz);
74
75// create a sequential matrix on one processor
76 if (numprocs == 1) {
77 assert ( (m_l == m) && (n_l == n));
78
79 // Create matrix. Revisit later to do preallocation and make more efficient
80 ierr = MatCreateSeqAIJ (MPI_COMM_WORLD, m_global, n_global,
81 n_nz, PETSC_NULL, &_mat);
82 CHKERRABORT (MPI_COMM_WORLD, ierr);
83 ierr = MatSetFromOptions (_mat);
84 CHKERRABORT (MPI_COMM_WORLD, ierr);
85 }
86
87 else {
88 parallel_only();
89
90 ierr = MatCreate (MPI_COMM_WORLD, &_mat);
91 CHKERRABORT (MPI_COMM_WORLD, ierr);
92
93 ierr = MatSetSizes (_mat, m_l, n_l, m, n);
94 CHKERRABORT (MPI_COMM_WORLD, ierr);
95
96 ierr = MatSetType (_mat, MATMPIAIJ); // Automatically chooses seqaij or mpiaij
97 CHKERRABORT (MPI_COMM_WORLD, ierr);
98
99 ierr = MatMPIAIJSetPreallocation (_mat, nnz, PETSC_NULL, noz, PETSC_NULL);
100 CHKERRABORT (MPI_COMM_WORLD, ierr);
101
102 }
103
104 this->zero();
105 }

107 void PetscMatrix::init (const int nr, const int nc, const std::vector < SparseMatrix*> &P) {
108
109 unsigned dim = P.size();
110
111 if (nr * nc != dim) {
112 std::cout << "error in function PetscMatrix::init (const int nr, const int nc, const std::vector < SparseMatrix*> &P)\n";
113 std::cout << "the product of nr * nc should be equal to the dimension of P\n";
114 abort();
115 }
116
117 int numprocs;
118 MPI_Comm_size (MPI_COMM_WORLD, &numprocs);
119
120 std::cout << "matrix block structure:";
121 std::vector < Mat > KK (dim);
122 for (unsigned k = 0; k < dim; k++) {
123 if (k % (nc) == 0) std::cout << std::endl;
124 if (P[k] != NULL) {
125 KK[k] = (static_cast<PetscMatrix*> (P[k]))->mat();
126 std::cout << "A_" << k / nc << k % nc << " ";
127 }
128 else {
129 KK[k] = PETSC_NULL;
130 std::cout << " 0 ";
131 }
132 }
133 std::cout << std::endl << std::endl;
134
135 if (dim != 1) {
136 Mat nMat;
137 MatCreateNest (MPI_COMM_WORLD, nr, NULL, nc, NULL, &KK[0], &nMat);
138
139 if (numprocs == 1) {
140 MatConvert (nMat, MATSEQAIJ, MAT_INITIAL_MATRIX, &_mat);
141 }
142 else {
143 MatConvert (nMat, MATMPIAIJ, MAT_INITIAL_MATRIX, &_mat);
144 }
145
146 MatDestroy (&nMat);
147 }
148 else {
149 if (numprocs == 1) {
150 MatConvert (KK[0], MATSEQAIJ, MAT_INITIAL_MATRIX, &_mat);
151 }
152 else {
153 MatConvert (KK[0], MATMPIAIJ, MAT_INITIAL_MATRIX, &_mat);
154 }
155 }
156
157 this->_is_initialized = true;
158 MatGetSize (_mat, &_m, &_n);
159 MatGetLocalSize (_mat, &_m_l, &_n_l);
160 _destroy_mat_on_exit = true;
161
162 }

164 void PetscMatrix::init (const int m, const int n, const int m_l, const int n_l,
165 const std::vector< int > & n_nz, const std::vector< int > & n_oz) {
166 // Set matrix dimension
167 _m = m;
168 _n = n;
169 _m_l = m_l;
170 _n_l = n_l;
171
172 // Clear initialized matrices
173 if (this->initialized())
174 this->clear();
175
176 this->_is_initialized = true;
177
178 // processor info
179 int n_procs;
180 MPI_Comm_size (MPI_COMM_WORLD, &n_procs);
181
182 int ierr = 0;
183
184// create a sequential matrix on one processor
185 if (n_procs == 1) {
186 assert (n_nz.size() == _m_l);
187 ierr = MatCreateSeqAIJ (MPI_COMM_WORLD, _m, _n, 0, &n_nz[0], &_mat);
188 CHKERRABORT (MPI_COMM_WORLD, ierr);
189 ierr = MatSetFromOptions (_mat);
190 CHKERRABORT (MPI_COMM_WORLD, ierr);
191 }
192 else {
193 parallel_only();
194 assert ( (n_nz.size() == _m_l) && (n_oz.size() == _m_l));
195 ierr = MatCreate (MPI_COMM_WORLD, &_mat);
196 CHKERRABORT (MPI_COMM_WORLD, ierr);
197 ierr = MatSetSizes (_mat, _m_l, _n_l, _m, _n);
198 CHKERRABORT (MPI_COMM_WORLD, ierr);
199 ierr = MatSetType (_mat, MATMPIAIJ);
200 CHKERRABORT (MPI_COMM_WORLD, ierr);
201 ierr = MatMPIAIJSetPreallocation (_mat, 1, &n_nz[0], 100, &n_oz[0]);
202 CHKERRABORT (MPI_COMM_WORLD, ierr);
203 }
204 this->zero();
205 }

303 void PetscMatrix::update_sparsity_pattern (const Graph &sparsity_pattern) {
304
305 // dimension
306 const int m = sparsity_pattern._m; // global rows;
307 const int n = sparsity_pattern._n; // global columns
308 const int n_l = sparsity_pattern._nl; // local rows
309 const int m_l = sparsity_pattern._ml; // local columns
310 const int ml_start = sparsity_pattern._ml_start;
311
312 // vectors n_nz (diagonal) and n_oz (offset) -------------------------------
313 std::vector< int> n_nz (m_l);
314 std::vector< int> n_oz (m_l);
315 for (int i = 0; i < m_l; i++) {
316 int len = sparsity_pattern[ml_start + i].size() - 1;
317 n_oz[i] = sparsity_pattern[ml_start + i][len];
318 n_nz[i] = len - n_oz[i];
319 }
320 update_sparsity_pattern (m, n, m_l, n_l, n_oz, n_nz);
321 return;
322 }

507 void PetscMatrix::print_matlab (const std::string& name, const std::string& format) const {
508
509 this->close();
510
511 PetscErrorCode ierr = 0;
512 PetscViewer petsc_viewer;
513 ierr = PetscViewerCreate (MPI_COMM_WORLD,
514 &petsc_viewer);
515 CHKERRABORT (MPI_COMM_WORLD, ierr);
516
521 if (name != "") {
522
523 if (format == "binary") {
524
525 ierr = PetscViewerBinaryOpen (MPI_COMM_WORLD,
526 name.c_str(),
527 FILE_MODE_WRITE,
528 &petsc_viewer);
529 CHKERRABORT (MPI_COMM_WORLD, ierr);
530
531 ierr = MatView (_mat, petsc_viewer);
532 CHKERRABORT (MPI_COMM_WORLD, ierr);
533 }
534 else if (format == "ascii") {
535 ierr = PetscViewerASCIIOpen (MPI_COMM_WORLD,
536 name.c_str(),
537 &petsc_viewer);
538 CHKERRABORT (MPI_COMM_WORLD, ierr);
539
540 ierr = PetscViewerPushFormat (petsc_viewer,
541 PETSC_VIEWER_ASCII_MATLAB);
542 CHKERRABORT (MPI_COMM_WORLD, ierr);
543
544 ierr = MatView (_mat, petsc_viewer);
545 CHKERRABORT (MPI_COMM_WORLD, ierr);
546 }
547 else {
548 std::cout << "Provide either \"ascii\" or \"binary\" for the second argument" << std::endl;
549 abort();
550 }
551
552 }
553
557 else {
558 ierr = PetscViewerPushFormat (PETSC_VIEWER_STDOUT_WORLD,
559 PETSC_VIEWER_ASCII_MATLAB);
560 CHKERRABORT (MPI_COMM_WORLD, ierr);
561 ierr = MatView (_mat, PETSC_VIEWER_STDOUT_WORLD);
562 CHKERRABORT (MPI_COMM_WORLD, ierr);
563 }
564
568 ierr = PetscViewerDestroy (&petsc_viewer);
569 CHKERRABORT (MPI_COMM_WORLD, ierr);
570
571 }

757 void PetscMatrix::RemoveZeroEntries (double & tolerance) {
758
759 int rowStart, rowEnd;
760 MatGetOwnershipRange (_mat, &rowStart, &rowEnd);
761 int colStart, colEnd;
762 MatGetOwnershipRangeColumn (_mat, &colStart, &colEnd);
763
764 std::vector < int > sizeDiag (rowEnd - rowStart, 0);
765 std::vector < int > sizeOff (rowEnd - rowStart, 0);
766
767 std::vector < std::vector < int > > nCols (rowEnd - rowStart);
768 std::vector < std::vector < double > > nVals (rowEnd - rowStart);
769
770 int n;
771 const int *cols;
772 const double *vals;
773
774 for (int i = 0; i < rowEnd - rowStart; i++) {
775
776 int row = rowStart + i;
777
778 MatGetRow (_mat, row, &n, &cols, &vals);
779
780 nCols[i].resize (n);
781 nVals[i].resize (n);
782
783 int k = 0;
784 for (int j = 0; j < n; j++) {
785 if (fabs (vals[j]) > tolerance) {
786 if (colStart <= cols[j] && cols[j] < colEnd) {
787 sizeDiag[i]++;
788 }
789 else {
790 sizeOff[i]++;
791 }
792 nCols[i][k] = cols[j];
793 nVals[i][k] = vals[j];
794 k++;
795 }
796 }
797
798 MatRestoreRow (_mat, i, &n, &cols, &vals);
799 nCols[i].resize (sizeDiag[i] + sizeOff[i]);
800 nVals[i].resize (sizeDiag[i] + sizeOff[i]);
801 }
802
803 MatDestroy (&_mat);
804
805 MatCreate (MPI_COMM_WORLD, &_mat);
806 MatSetSizes (_mat, _m_l, _n_l, _m, _n);
807
808 int n_procs;
809 MPI_Comm_size (MPI_COMM_WORLD, &n_procs);
810
811 // create a sequential matrix on one processor
812 if (n_procs == 1) {
813 MatCreateSeqAIJ (MPI_COMM_WORLD, _m, _n, 0, &sizeDiag[0], &_mat);
814 MatSetFromOptions (_mat);
815 }
816 else {
817 MatSetType (_mat, MATMPIAIJ);
818 MatMPIAIJSetPreallocation (_mat, 0, &sizeDiag[0], 0, &sizeOff[0]);
819 MatSetFromOptions (_mat);
820 }
821
822 for (int i = 0; i < rowEnd - rowStart; i++) {
823 int row = rowStart + i;
824 int n = nCols[i].size();
825 MatSetValues (_mat, 1, &row, n, &nCols[i][0], &nVals[i][0], INSERT_VALUES);
826 }
827
828 MatAssemblyBegin (_mat, MAT_FINAL_ASSEMBLY);
829 MatAssemblyEnd (_mat, MAT_FINAL_ASSEMBLY);
830
831 }

959 void PetscMatrix::_get_submatrix (SparseMatrix& submatrix,
960 const std::vector< int> &rows,
961 const std::vector< int> &cols,
962 const bool reuse_submatrix) const {
963
964 // Can only extract submatrices from closed matrices
965 this->close();
966
967 // Make sure the SparseMatrix passed in is really a PetscMatrix
968 PetscMatrix* petsc_submatrix = libmeshM_cast_ptr<PetscMatrix*> (&submatrix);
969
970 // If we're not reusing submatrix and submatrix is already initialized
971 // then we need to clear it, otherwise we get a memory leak.
972 if (!reuse_submatrix && submatrix.initialized()) submatrix.clear();
973 // Construct row and column index sets.
974 int ierr = 0;
975 IS isrow, iscol;
976
977 ierr = ISCreateGeneral (MPI_COMM_WORLD, rows.size(), (int*) &rows[0], PETSC_COPY_VALUES, &isrow); // PETSC_COPY_VALUES is my first choice; see also PETSC_OWN_POINTER, PETSC_USE_POINTER
978 CHKERRABORT (MPI_COMM_WORLD, ierr);
979
980 ierr = ISCreateGeneral (MPI_COMM_WORLD, cols.size(), (int*) &cols[0], PETSC_COPY_VALUES, &iscol);
981 CHKERRABORT (MPI_COMM_WORLD, ierr);
982
983//---
984
985#if !PETSC_VERSION_LESS_THAN(3,0,1) || !PETSC_VERSION_RELEASE
986// ierr = MatGetSubMatrix(_mat,
987// isrow,
988// iscol,
989// (reuse_submatrix ? MAT_REUSE_MATRIX : MAT_INITIAL_MATRIX),
990// &(petsc_submatrix->_mat));
991 ierr = MatCreateSubMatrix (_mat,
992 isrow,
993 iscol,
994 (reuse_submatrix ? MAT_REUSE_MATRIX : MAT_INITIAL_MATRIX),
995 & (petsc_submatrix->_mat));
996 CHKERRABORT (MPI_COMM_WORLD, ierr);
997#else
998 ierr = MatGetSubMatrix (_mat,
999 isrow,
1000 iscol,
1001 PETSC_DECIDE,
1002 (reuse_submatrix ? MAT_REUSE_MATRIX : MAT_INITIAL_MATRIX),
1003 & (petsc_submatrix->_mat));
1004 CHKERRABORT (MPI_COMM_WORLD, ierr);
1005#endif
1006
1007 // Specify that the new submatrix is initialized and close it.
1008 petsc_submatrix->_is_initialized = true;
1009 petsc_submatrix->close();
1010
1011 // Clean up PETSc data structures
1012 ierr = ISDestroy (&isrow);
1013 CHKERRABORT (MPI_COMM_WORLD, ierr);
1014 ierr = ISDestroy (&iscol);
1015 CHKERRABORT (MPI_COMM_WORLD, ierr);

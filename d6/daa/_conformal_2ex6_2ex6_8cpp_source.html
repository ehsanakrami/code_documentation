340 for (unsigned i = 0; i < nxDofs; i++) {
341 unsigned iXDof = msh->GetSolutionDof (i, iel, xType);
342 for (unsigned K = 0; K < DIM; K++) {
343 solxHat[K][i] = (*msh->_topology->_Sol[K]) (iXDof);
344 solx[K][i] = (*msh->_topology->_Sol[K]) (iXDof) + solDx[K][i];
345 }
346 }

349 for (unsigned ig = 0; ig < msh->_finiteElement[ielGeom][solType]->GetGaussPointNumber(); ig++) {
350
351 const double *phix; // local test function
352 const double *phi1; // local test function
353 const double *phix_uv[dim]; // local test function first order partial derivatives
354
355 double weight; // gauss point weight
356
357 // Get Gauss point weight, test function, and first order derivatives.
358 if (ielGeom == QUAD) {
359 phix = msh->_finiteElement[ielGeom][solType]->GetPhi (ig);
360
361 phix_uv[0] = msh->_finiteElement[ielGeom][solType]->GetDPhiDXi (ig);
362 phix_uv[1] = msh->_finiteElement[ielGeom][solType]->GetDPhiDEta (ig);
363
364 weight = msh->_finiteElement[ielGeom][solType]->GetGaussWeight (ig);
365 }
366
367 // Special adjustments for triangles.
368 else {
369 msh->_finiteElement[ielGeom][solType]->Jacobian (xT, ig, weight, stdVectorPhi, stdVectorPhi_uv);
370 phix = &stdVectorPhi[0];
371 phi_uv0.resize (nxDofs);
372 phi_uv1.resize (nxDofs);
373 for (unsigned i = 0; i < nxDofs; i++) {
374 phi_uv0[i] = stdVectorPhi_uv[i * dim];
375 phi_uv1[i] = stdVectorPhi_uv[i * dim + 1];
376 }
377 phix_uv[0] = &phi_uv0[0];
378 phix_uv[1] = &phi_uv1[0];
379 }
380
381 phi1 = msh->_finiteElement[ielGeom][solType1]->GetPhi (ig);
382
383 // Initialize and compute values of x, Dx, NDx, x_uv at the Gauss points.
384 double solxHatg[DIM] = {0., 0.};
385 adept::adouble solx_uv[2][2] = {{0., 0.}, {0., 0.}};
386 for (unsigned K = 0; K < DIM; K++) {
387 for (unsigned i = 0; i < nxDofs; i++) {
388 solxHatg[K] += phix[i] * solxHat[K][i];
389 }
390 for (int j = 0; j < dim; j++) {
391 for (unsigned i = 0; i < nxDofs; i++) {
392 solx_uv[K][j] += phix_uv[j][i] * solx[K][i];

402 std::vector < std::vector < adept::adouble > > g (dim);
403 for (unsigned i = 0; i < dim; i++) g[i].assign (dim, 0.);
404
405 for (unsigned i = 0; i < dim; i++) {
406 for (unsigned j = 0; j < dim; j++) {
407 for (unsigned K = 0; K < DIM; K++) {
408 g[i][j] += solx_uv[K][i] * solx_uv[K][j];
409 }
410 }
411 }
412
413 adept::adouble detg = g[0][0] * g[1][1] - g[0][1] * g[1][0];
414 adept::adouble Area = weight * sqrt (detg);
415 adept::adouble Area2 = weight;// Trick to give equal weight to each element.
416
417// adept::adouble norm2Xz = (1. / 4.) * (pow ( (solx_uv[0][0] + solx_uv[1][1]), 2) + pow ( (solx_uv[1][0] - solx_uv[0][1]), 2));
418//
419// // Discretize the equation \delta CD = 0 on the basis d/du, d/dv.
420// adept::adouble XzBarXz_Bar[DIM];
421//
422// // Comment out for working code
423//
424//
425// XzBarXz_Bar[0] = (1. / 4.) * (pow (solx_uv[0][0], 2) + pow (solx_uv[1][0], 2) - pow (solx_uv[0][1], 2) - pow (solx_uv[1][1], 2));
426// XzBarXz_Bar[1] = (1. / 2.) * (solx_uv[0][0] * solx_uv[0][1] + solx_uv[1][0] * solx_uv[1][1]);
427//
428// // Comment out for working code
429
430 double mu[2] = {0., 0.};
431
432 for (unsigned i = 0; i < nDofs1; i++) {
433 for (unsigned K = 0; K < DIM; K++) {
434 mu[K] += phi1[i] * solMu[K][i];
435 }
436 }

438
439
440// if (counter == 0) mu[0] = 0.8;

467 //M[1][1]= -(1 + mu1) * V[0] - mu2 * V[1];
468
469

471 for (unsigned k = 0; k < dim; k++) {

488 for (int i = 0; i < nxDofs; i++) {
489 Res[ k * nxDofs + i] = -aResDx[k][i].value();
490 }
491 }
492
493 RES->add_vector_blocked (Res, SYSDOF);
494
495 // Resize Jacobian.
496 Jac.resize ( (dim * nxDofs) * (dim * nxDofs));
497
498 // Define the dependent variables.
499 for (int k = 0; k < dim; k++) {
500 s.dependent (&aResDx[k][0], nxDofs);
501 }
502
503 // Define the independent variables.
504 for (int k = 0; k < dim; k++) {
505 s.independent (&solDx[k][0], nxDofs);
506 }
507
508 // Get the jacobian matrix (ordered by row).
509 s.jacobian (&Jac[0], true);
510
511 KK->add_matrix_blocked (Jac, SYSDOF, SYSDOF);
512
513 s.clear_independents();
514 s.clear_dependents();
515
516 } //end ELEMENT LOOP for each process.
517
518 RES->close();
519 KK->close();
520

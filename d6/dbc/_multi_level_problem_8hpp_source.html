441template <typename T_sys>
442inline
443const T_sys & MultiLevelProblem::get_system (const unsigned int num) const
444{
445 assert(num < this->n_systems());
446
447 const_system_iterator pos = _systems.begin();
448 const const_system_iterator end = _systems.end();
449
450 for (; pos != end; ++pos)
451 if (pos->second->number() == num)
452 break;
453
454 // Check for errors
455 if (pos == end)
456 {
457 std::cerr << "ERROR: no system number " << num << " found!" << std::endl;
458 }
459
460 // Attempt dynamic cast
461 return *static_cast<T_sys*>(pos->second);
462}

489template <typename T_sys>
490inline
491T_sys & MultiLevelProblem::add_system (const std::string& name,const LinearEquationSolverType & smoother_type )
492{
493 T_sys* ptr = NULL;
494
495 if (!_systems.count(name))
496 {
497 ptr = new T_sys(*this, name, this->n_systems(),smoother_type);
498 _systems.insert (std::make_pair(name, ptr));
499 }
500 else
501 {
502 // We now allow redundant add_system calls, to make it
503 // easier to load data from files for user-derived system
504 // subclasses
505 std::cerr << "ERROR: There was already a system"
506 << " named " << name
507 << std::endl;
508
509 ptr = &(this->get_system<T_sys>(name));
510 }
511
512 // Return a dynamically casted reference to the newly added System.
513 return *ptr;
514}

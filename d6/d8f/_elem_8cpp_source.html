71 elem::elem(elem* elc, const unsigned dim_in, const unsigned refindex, const std::vector < double >& coarseAmrVector)
72 {
73
74 _dim = dim_in;
75
76 _coarseElem = elc;
77
78 _level = elc->_level + 1;
79
80
81 InitializeNumberOfElementsPerGeomType();
82
83 _nel = InitializeNumberOfElementsFromCoarseList(elc, refindex);
84
85
86 ResizeElement_Level_Type_Group_Material(_nel, _level);
87
88
89 //**************************
90 MyVector <unsigned> rowSizeElDof(_nel);
91 MyVector <unsigned> rowSizeElNearFace(_nel);
92 unsigned jel = 0;
93 for (unsigned isdom = 0; isdom < elc->_nprocs; isdom++) {
94 elc->_elementType.broadcast(isdom);
95 for (unsigned iel = elc->_elementType.begin(); iel < elc->_elementType.end(); iel++) {
96 short unsigned elType = elc->_elementType[iel];
97 int increment = 1;
98 if (static_cast < short unsigned >(coarseAmrVector[iel] + 0.25) == 1) {
99 increment = NRE[elType];
100 }
101 for (unsigned j = 0; j < increment; j++) {
102 rowSizeElDof[jel + j] += NVE[elType][2];
103 rowSizeElNearFace[jel + j] += NFC[elType][1];
104 }
105 jel += increment;
106 }
107 elc->_elementType.clearBroadcast();
108 }
109
110 _elementDof = MyMatrix <unsigned> (rowSizeElDof);
111
112 _elementNearFace = MyMatrix <int> (rowSizeElNearFace, -1);
113
114 rowSizeElDof.clear();
115 rowSizeElNearFace.clear();
116
117 }

443 void elem::BuildElementNearFace() {
444
445 for(unsigned iel = 0; iel < GetElementNumber(); iel++) {
446 for(unsigned iface = 0; iface < GetElementFaceNumber(iel); iface++) {
447 if(GetFaceElementIndex(iel, iface) <= 0) {
448 unsigned i1 = GetFaceVertexIndex(iel, iface, 0);
449 unsigned i2 = GetFaceVertexIndex(iel, iface, 1);
450 unsigned i3 = GetFaceVertexIndex(iel, iface, 2);
451
452 for(unsigned j = 0; j < GetElementNearVertexNumber(i1); j++) {
453 unsigned jel = GetElementNearVertex(i1, j);
454
455 if(jel > iel) {
456 for(unsigned jface = 0; jface < GetElementFaceNumber(jel); jface++) {
457 if(GetFaceElementIndex(jel, jface) <= 0) {
458 unsigned j1 = GetFaceVertexIndex(jel, jface, 0);
459 unsigned j2 = GetFaceVertexIndex(jel, jface, 1);
460 unsigned j3 = GetFaceVertexIndex(jel, jface, 2);
461 unsigned j4 = GetFaceVertexIndex(jel, jface, 3);
462
463 const bool faces_coincide_three_dim = (GetDimension()/*Mesh::_dimension*/ == 3 &&
464 (i1 == j1 || i1 == j2 || i1 == j3 || i1 == j4) &&
465 (i2 == j1 || i2 == j2 || i2 == j3 || i2 == j4) &&
466 (i3 == j1 || i3 == j2 || i3 == j3 || i3 == j4));
467
468 const bool faces_coincide_two_dim = (GetDimension()/*Mesh::_dimension*/ == 2 &&
469 (i1 == j1 || i1 == j2) &&
470 (i2 == j1 || i2 == j2));
471
472 const bool faces_coincide_one_dim = (GetDimension()/*Mesh::_dimension*/ == 1 && (i1 == j1));
473
474 if(faces_coincide_three_dim
475 ||
476 faces_coincide_two_dim
477 ||
478 faces_coincide_one_dim
479 ) {
480 SetFaceElementIndex(iel, iface, jel + 1u);
481 SetFaceElementIndex(jel, jface, iel + 1u);
482 }
483 }
484 }
485 }
486 }
487 }
488 }

499 void elem::BuildElementNearElement()
500 {
501 MyVector < unsigned > rowSize(_elementOffset, 1);
502 for (unsigned iel = rowSize.begin(); iel < rowSize.end(); iel++) {
503 std::map< unsigned, bool> elements;
504 for (unsigned i = 0; i < GetElementDofNumber(iel, 0); i++) {
505 unsigned inode = GetElementDofIndex(iel, i);
506 for (unsigned j = _elementNearVertex.begin(inode); j < _elementNearVertex.end(inode); j++) {
507 elements[_elementNearVertex[inode][j]] = true;
508 }
509 }
510 rowSize[iel] = elements.size();
511 }
512
513 _elementNearElement = MyMatrix <unsigned> (rowSize, UINT_MAX);
514
515 for (unsigned iel = _elementNearElement.begin(); iel < _elementNearElement.end(); iel++) {
516 std::map< unsigned, bool> elements;
517 _elementNearElement[iel][0] = iel;
518 for (unsigned i = 0; i < GetElementDofNumber(iel, 0); i++) {
519 unsigned inode = GetElementDofIndex(iel, i);
520 for (unsigned j = _elementNearVertex.begin(inode); j < _elementNearVertex.end(inode); j++) {
521 if (_elementNearVertex[inode][j] != iel) {
522 elements[_elementNearVertex[inode][j]] = true;
523 }
524 }
525 }
526 unsigned j = 1;
527 for (std::map<unsigned, bool>::iterator it = elements.begin(); it != elements.end(); it++, j++) {
528 _elementNearElement[iel][j] = it->first;
529 }
530 }
531
532 }

535 void elem::BuildElementNearVertex()
536 {
537 MyVector <unsigned> rowSize(_nvt, 0);
538 for (unsigned iel = 0; iel < _nel; iel++) {
539 for (unsigned inode = 0; inode < GetElementDofNumber(iel, 0); inode++) {
540 rowSize[GetElementDofIndex(iel, inode)]++;
541 }
542 }
543 _elementNearVertex = MyMatrix <unsigned> (rowSize, _nel);
544 for (unsigned iel = 0; iel < _nel; iel++) {
545 for (unsigned inode = 0; inode < GetElementDofNumber(iel, 0); inode++) {
546 unsigned irow = GetElementDofIndex(iel, inode);
547 unsigned j = 0;
548 while (_nel != _elementNearVertex[irow][j]) j++;
549 _elementNearVertex[irow][j] = iel;
550 }
551 }

723 void elem::GetAMRRestriction(Mesh *msh) const
724 {
725
726 std::vector < std::map < unsigned, std::map < unsigned, double > > > & restriction = msh->GetAmrRestrictionMap();
727 restriction.resize(NFE_FAMS_C_ZERO_LAGRANGE);
728
729 std::vector < std::map < unsigned, bool > > &interfaceSolidMark = msh->GetAmrSolidMark();
730 interfaceSolidMark.resize(NFE_FAMS_C_ZERO_LAGRANGE);
731
732 std::vector < MyVector<unsigned> > interfaceElement;
733 std::vector < MyMatrix<unsigned> > interfaceLocalDof;
734 std::vector < std::vector < MyMatrix<unsigned> > > interfaceDof;
735 std::vector < std::vector < MyMatrix<unsigned> > > levelInterfaceSolidMark;
736 std::vector < std::vector < MyMatrix< double > > > interfaceNodeCoordinates;
737
738 interfaceElement.resize(_level + 1);
739 interfaceLocalDof.resize(_level + 1);
740 interfaceDof.resize(NFE_FAMS_C_ZERO_LAGRANGE);
741 levelInterfaceSolidMark.resize(NFE_FAMS_C_ZERO_LAGRANGE);
742 for (unsigned i = 0; i < NFE_FAMS_C_ZERO_LAGRANGE; i++) {
743 interfaceDof[i].resize(_level + 1);
744 levelInterfaceSolidMark[i].resize(_level + 1);
745 }
746 interfaceNodeCoordinates.resize(_level + 1);
747 unsigned dim = msh->GetDimension();
748
749 for (unsigned ilevel = 0; ilevel <= _level; ilevel++) {
750 //BEGIN interface element search
751 interfaceElement[ilevel] = MyVector <unsigned> (_elementOwned);
752 unsigned counter = 0;
753 for (unsigned i = _elementLevel.begin(); i < _elementLevel.end(); i++) {
754 if (ilevel == _elementLevel[i]) {
755 for (unsigned j = _elementNearFace.begin(i); j < _elementNearFace.end(i); j++) {
756 if (-1 == _elementNearFace[i][j]) {
757 interfaceElement[ilevel][counter] = i;
758 counter++;
759 break;
760 }
761 }
762 }
763 }
764 interfaceElement[ilevel].resize(counter);
765 interfaceElement[ilevel].stack();
766 //END interface element search
767
768 //BEGIN interface node search
769 std::vector< unsigned > offset = interfaceElement[ilevel].getOffset();
770 interfaceLocalDof[ilevel] = MyMatrix <unsigned>(offset, NVE[0][2], UINT_MAX);
771 for (unsigned i = interfaceElement[ilevel].begin(); i < interfaceElement[ilevel].end(); i++) {
772 unsigned iel = interfaceElement[ilevel][i];
773 std::map <unsigned, bool> ldofs;
774 for (unsigned jface = _elementNearFace.begin(iel); jface < _elementNearFace.end(iel); jface++) {
775 if (-1 == _elementNearFace[iel][jface]) {
776 for (unsigned k = 0; k < GetNFACENODES(GetElementType(iel), jface, 2); k++) {
777 unsigned index = GetIG(GetElementType(iel), jface, k);
778 ldofs[index] = true;
779 }
780 }
781 }
782 unsigned j = 0;
783 for (std::map<unsigned, bool>::iterator it = ldofs.begin(); it != ldofs.end(); it++) {
784 interfaceLocalDof[ilevel][i][j] = it->first;
785 j++;
786 }
787 }
788 interfaceLocalDof[ilevel].shrinkToFit(UINT_MAX);
789 //END interface node search
790
791 //BEGIN interface node dof global search, one for each soltype
792 MyVector <unsigned> rowSize = interfaceLocalDof[ilevel].getRowSize();
793 for (unsigned soltype = 0; soltype < NFE_FAMS_C_ZERO_LAGRANGE; soltype++) {
794 interfaceDof[soltype][ilevel] = MyMatrix< unsigned > (rowSize, UINT_MAX);
795 levelInterfaceSolidMark[soltype][ilevel] = MyMatrix< unsigned > (rowSize, UINT_MAX);
796 for (unsigned i = interfaceLocalDof[ilevel].begin(); i < interfaceLocalDof[ilevel].end(); i++) {
797 unsigned iel = interfaceElement[ilevel][i];
798 unsigned counter = 0;
799 for (unsigned j = interfaceLocalDof[ilevel].begin(i); j < interfaceLocalDof[ilevel].end(i); j++) {
800 unsigned jloc = interfaceLocalDof[ilevel][i][j];
801 if (jloc < GetElementDofNumber(iel, soltype)) {
802 unsigned jdof = msh->GetSolutionDof(jloc, iel, soltype);
803 interfaceDof[soltype][ilevel][i][counter] = jdof;
804 unsigned jdof2 = msh->GetSolutionDof(jloc, iel, 2);
805 levelInterfaceSolidMark[soltype][ilevel][i][counter] = msh->GetSolidMark(jdof2);
806 counter++;
807 }
808 else {
809 break;
810 }
811 }
812 }
813 interfaceDof[soltype][ilevel].shrinkToFit(UINT_MAX);
814 levelInterfaceSolidMark[soltype][ilevel].shrinkToFit(UINT_MAX);
815 }
816 //END interface node dof global search, one for each soltype
817
818 //BEGIN interface node coordinates
819 interfaceNodeCoordinates[ilevel].resize(dim);
820 for (unsigned k = 0; k < dim; k++) {

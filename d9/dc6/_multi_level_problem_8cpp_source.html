71System & MultiLevelProblem::add_system (const std::string& sys_type,
72 const std::string& name)
73{
74 // If the user already built a system with this name, we'll
75 // trust them and we'll use it. That way they can pre-add
76 // non-standard derived system classes, and if their restart file
77 // has some non-standard sys_type we won't throw an error.
78 if (_systems.count(name))
79 {
80 return this->get_system(name);
81 }
82 // Build a basic System
83 else if (sys_type == "Basic")
84 this->add_system<System> (name);
85
86 // Build a Newmark system
87// else if (sys_type == "Newmark")
88// this->add_system<NewmarkSystem> (name);
89
90 // build a transient implicit linear system
91 else if ((sys_type == "Transient") ||
92 (sys_type == "TransientImplicit") ||
93 (sys_type == "TransientLinearImplicit"))
94 this->add_system<TransientLinearImplicitSystem> (name);
95
96 // build a transient implicit nonlinear system
97 else if (sys_type == "TransientNonlinearImplicit")
98 this->add_system<TransientNonlinearImplicitSystem> (name);
99
100 // build a linear implicit system
101 else if (sys_type == "LinearImplicit")
102 this->add_system<LinearImplicitSystem> (name);
103
104 // build a nonlinear implicit system
105 else if (sys_type == "NonlinearImplicit")
106 this->add_system<NonLinearImplicitSystem> (name);
107
108 else
109 {
110 std::cerr << "ERROR: Unknown system type: " << sys_type << std::endl;
111 }
112
113 // Return a reference to the new system
114 return this->get_system(name);
115}

129void MultiLevelProblem::clear_systems() {
130
131 while (!_systems.empty())
132 {
133 system_iterator pos = _systems.begin();
134
135 System *sys = pos->second;
136 delete sys;
137 sys = NULL;
138
139 _systems.erase (pos);
140 }

198
199 _qrule.resize(quadr_order_in_vec.size());
200
201 for (int q = 0; q < _qrule.size(); q++) {
202 _qrule[q].reserve(N_GEOM_ELS);

204 for (int iel = 0; iel < femus::geom_elems.size(); iel++) {

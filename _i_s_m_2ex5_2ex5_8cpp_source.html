70double InitalValueU(const std::vector < double >& x) {
71 double time = (x.size() == 4) ? x[3] : 0.;
72 return
73 2.*(sin(pi * (x[0] + 0.5)) * sin(pi * (x[0] + 0.5)) *
74 ( sin(pi * (x[1] + 0.5)) * cos(pi * (x[1] + 0.5)) - sin(pi * (x[2] + 0.5)) * cos(pi * (x[2] + 0.5)) )
75 )* cos(time);
76}

78double InitalValueV(const std::vector < double >& x) {
79 double time = (x.size() == 4) ? x[3] : 0.;
80 return
81 2.*(sin(pi * (x[1] + 0.5)) * sin(pi * (x[1] + 0.5)) *
82 ( sin(pi * (x[2] + 0.5)) * cos(pi * (x[2] + 0.5)) - sin(pi * (x[0] + 0.5)) * cos(pi * (x[0] + 0.5)) )
83 )* cos(time);
84}
85
86double InitalValueW(const std::vector < double >& x) {
87 double time = (x.size() == 4) ? x[3] : 0.;
88 return
89 2.*( sin(pi * (x[2] + 0.5)) * sin(pi * (x[2] + 0.5)) *
90 ( sin(pi * (x[0] + 0.5)) * cos(pi * (x[0] + 0.5)) - sin(pi * (x[1] + 0.5)) * cos(pi * (x[1] + 0.5)) )
91 )* cos(time);
92
93 return 0.;
94}

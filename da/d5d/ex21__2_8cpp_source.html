334void AssembleBoussinesqAppoximation_AD(MultiLevelProblem& ml_prob) {
335 // ml_prob is the global object from/to where get/set all the data
336 // level is the level of the PDE system to be assembled
337 // levelMax is the Maximum level of the MultiLevelProblem
338 // assembleMatrix is a flag that tells if only the residual or also the matrix should be assembled

340 // extract pointers to the several objects that we are going to use
341 TransientNonlinearImplicitSystem* mlPdeSys = &ml_prob.get_system<TransientNonlinearImplicitSystem> ("NS"); // pointer to the linear implicit system named "Poisson"
342 const unsigned level = mlPdeSys->GetLevelToAssemble();

344 Mesh* msh = ml_prob._ml_msh->GetLevel(level); // pointer to the mesh (level) object

352
353 bool assembleMatrix = mlPdeSys->GetAssembleMatrix();
354 // call the adept stack object

356 if(assembleMatrix) s.continue_recording();
357 else s.pause_recording();
358
359 SparseMatrix* KK = pdeSys->_KK; // pointer to the global stifness matrix object in pdeSys (level)
360 NumericVector* RES = pdeSys->_RES; // pointer to the global residual vector object in pdeSys (level)

363 unsigned dim2 = (3 * (dim - 1) + !(dim - 1)); // dim2 is the number of second order partial derivatives (1,3,6 depending on the dimension)

376 solVIndex[1] = mlSol->GetIndex("V"); // get the position of "V" in the ml_sol object

382 unsigned solPIndex;
383 solPIndex = mlSol->GetIndex("P"); // get the position of "P" in the ml_sol object
384 unsigned solPType = mlSol->GetSolutionType(solPIndex); // get the finite element type for "u"

386 unsigned solTPdeIndex;
387 solTPdeIndex = mlPdeSys->GetSolPdeIndex("T"); // get the position of "T" in the pdeSys object

393 solVPdeIndex[0] = mlPdeSys->GetSolPdeIndex("U"); // get the position of "U" in the pdeSys object
394 solVPdeIndex[1] = mlPdeSys->GetSolPdeIndex("V"); // get the position of "V" in the pdeSys object
395
396 if(dim == 3) solVPdeIndex[2] = mlPdeSys->GetSolPdeIndex("W");

398 unsigned solPPdeIndex;
399 solPPdeIndex = mlPdeSys->GetSolPdeIndex("P"); // get the position of "P" in the pdeSys object

517 }
518 }

523 solPold[i] = (*sol->_SolOld[solPIndex])(solPDof); // global extraction and local storage for the solution

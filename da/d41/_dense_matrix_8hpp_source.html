387inline double DenseMatrix::min () const {

391

393 for (int j=0; j!=this->_n; j++) {
394 double current = (double)((*this)(i,j));
395 my_min = (my_min < current? my_min : current);

416inline double DenseMatrix::l1_norm () const {
417 assert (this->_m);
418 assert (this->_n);
419 double columnsum = 0.;
420 for (int i=0; i!=this->_m; i++) columnsum += fabs((*this)(i,0));
421 double my_max = columnsum;
422 for (int j=1; j!=this->_n; j++) {
423 columnsum = 0.;
424 for (int i=0; i!=this->_m; i++) columnsum += fabs((*this)(i,j));
425 my_max = (my_max > columnsum? my_max : columnsum);
426 }
427 return my_max;
428}

431inline double DenseMatrix::linfty_norm () const {
432 assert (this->_m);
433 assert (this->_n);
434 double rowsum = 0.;
435 for (int j=0; j!=this->_n; j++) rowsum += std::abs((*this)(0,j));
436 double my_max = rowsum;

438 rowsum = 0.;
439 for (int j=0; j!=this->_n; j++) rowsum += std::abs((*this)(i,j));
440 my_max = (my_max > rowsum? my_max : rowsum);

56Fluid::Fluid(Parameter& par, const double viscosity, const double density, const char model[],
57 const double k, const double cp, const double alpha): Material(par,density,k,cp, alpha) {
58// cout << "calling 2" << endl;
59
60 const double lref = par.Get_reference_length();
61 const double uref = par.Get_reference_velocity();
62 const double DeltaTref = par.Get_reference_temperature();
63 _viscosity = viscosity;
64 _density = density;
65 _Reynolds = (_density*uref*lref)/_viscosity;
66 _IRe = 1./_Reynolds; // rho*U*Lref/mu
67
68 if (!strcmp(model,"Newtonian")) {
69 _model = 0;
70 } else {
71 cout<<"Error! This fluid model is not implemented "<<endl;
72 exit(1);
73 }
74
75 _Prandtl = (cp*viscosity)/k;
76 _Froude = (uref*uref)/(9.80665*lref); // U^2/Lref*g
77 _Grashof = (density*density*9.80665*alpha*DeltaTref*lref*lref*lref)/(viscosity*viscosity); //*L^3 * delta_T
78
79 _Rayleigh = _Grashof*_Prandtl;
80 _Peclet = _Prandtl*_Reynolds;
81

83// cout << "Density [kg/m3]: " << density << endl;

85// cout << "Heat capacity [J/Kg*K]: " << cp << endl;
86// cout << "Thermal conductivity [W/m*K]: " << k << endl;
87// cout << "Thermal expansion coefficient [1/K]: " << alpha << endl;
88// cout << "Reynolds Number: " << _Reynolds << endl;

94}

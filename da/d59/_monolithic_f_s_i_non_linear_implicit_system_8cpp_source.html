49 void MonolithicFSINonLinearImplicitSystem::BuildProlongatorMatrix(unsigned gridf) {
50
51 if(gridf < 1) {
52 std::cout << "Error! In function \"BuildProlongatorMatrix\" argument less then 1" << std::endl;
53 exit(0);
54 }
55
56 int iproc;
57 MPI_Comm_rank(MPI_COMM_WORLD, &iproc);
58
59 LinearEquationSolver* LinSolf = _LinSolver[gridf];
60 LinearEquationSolver* LinSolc = _LinSolver[gridf - 1];
61 Mesh* mshc = _msh[gridf - 1];
62 int nf = LinSolf->KKIndex[LinSolf->KKIndex.size() - 1u];
63 int nc = LinSolc->KKIndex[LinSolc->KKIndex.size() - 1u];
64 int nf_loc = LinSolf->KKoffset[LinSolf->KKIndex.size() - 1][iproc] - LinSolf->KKoffset[0][iproc];
65 int nc_loc = LinSolc->KKoffset[LinSolc->KKIndex.size() - 1][iproc] - LinSolc->KKoffset[0][iproc];
66
67 NumericVector *NNZ_d = NumericVector::build().release();
68 NNZ_d->init(*LinSolf->_EPS);
69 NNZ_d->zero();
70
71 NumericVector *NNZ_o = NumericVector::build().release();
72 NNZ_o->init(*LinSolf->_EPS);
73 NNZ_o->zero();
74
75 for(unsigned k = 0; k < _SolSystemPdeIndex.size(); k++) {
76 unsigned SolIndex = _SolSystemPdeIndex[k];
77 unsigned SolType = _ml_sol->GetSolutionType(SolIndex);
78 // loop on the coarse grid
79 for(int isdom = iproc; isdom < iproc + 1; isdom++) {
80 for(int iel = mshc->_elementOffset[isdom]; iel < mshc->_elementOffset[isdom + 1]; iel++) {
81 short unsigned ielt = mshc->GetElementType(iel);
82 mshc->_finiteElement[ielt][SolType]->GetSparsityPatternSize(*LinSolf, *LinSolc, iel, NNZ_d, NNZ_o, SolIndex, k);
83 }
84 }
85 }
86
87 NNZ_d->close();
88 NNZ_o->close();
89
90 unsigned offset = LinSolf->KKoffset[0][iproc];
91 vector <int> nnz_d(nf_loc);
92 vector <int> nnz_o(nf_loc);
93 for(int i = 0; i < nf_loc; i++) {
94 nnz_d[i] = static_cast <int>((*NNZ_d)(offset + i));
95 nnz_o[i] = static_cast <int>((*NNZ_o)(offset + i));
96 }
97
98 delete NNZ_d;
99 delete NNZ_o;
100
101 _PP[gridf] = SparseMatrix::build().release();
102 _PP[gridf]->init(nf, nc, nf_loc, nc_loc, nnz_d, nnz_o);
103
104 SparseMatrix *RRt;
105 RRt = SparseMatrix::build().release();
106 RRt->init(nf, nc, nf_loc, nc_loc, nnz_d, nnz_o);
107
108 for(unsigned k = 0; k < _SolSystemPdeIndex.size(); k++) {
109 unsigned SolIndex = _SolSystemPdeIndex[k];
110 unsigned solPairIndex = _ml_sol->GetSolutionPairIndex(k);
111 unsigned SolType = _ml_sol->GetSolutionType(SolIndex);
112 unsigned solPairPdeIndex = GetSolPdeIndex(_ml_sol->GetSolutionName(solPairIndex));
113
114 bool testIfPressure = 0;
115 if(_ml_sol->TestIfSolutionIsPressure(SolIndex)) testIfPressure = 1;
116
117 // loop on the coarse grid
118 for(int isdom = iproc; isdom < iproc + 1; isdom++) {
119 for(int iel = mshc->_elementOffset[isdom]; iel < mshc->_elementOffset[isdom + 1]; iel++) {
120 short unsigned ielt = mshc->GetElementType(iel);
121 if(!testIfPressure) {
122 mshc->_finiteElement[ielt][SolType]->BuildRestrictionTranspose(*LinSolf, *LinSolc, iel, RRt, SolIndex, k, solPairIndex, solPairPdeIndex);
123 }
124 else {
125 mshc->_finiteElement[ielt][SolType]->BuildProlongation(*LinSolf, *LinSolc, iel, RRt, SolIndex, k);
126 }
127 mshc->_finiteElement[ielt][SolType]->BuildProlongation(*LinSolf, *LinSolc, iel, _PP[gridf], SolIndex, k);
128 }
129 }
130 }
131
132 _PP[gridf]->close();
133 RRt->close();
134
135 _RR[gridf] = SparseMatrix::build().release();
136 RRt->get_transpose(*_RR[gridf]);
137 delete RRt;
138
139 }

141 void MonolithicFSINonLinearImplicitSystem::BuildAmrProlongatorMatrix(unsigned level) {
142
143 int iproc;
144 MPI_Comm_rank(MPI_COMM_WORLD, &iproc);
145
146 LinearEquationSolver* LinSol = _LinSolver[level];
147
148 Mesh* mesh = _msh[level];
149 std::vector < std::map < unsigned, std::map < unsigned, double > > > & amrRestriction = mesh->GetAmrRestrictionMap();
150
151 std::vector < std::map < unsigned, bool > > & amrSolidMark = mesh->GetAmrSolidMark();
152
153 int n = LinSol->KKIndex[LinSol->KKIndex.size() - 1u];
154 int n_loc = LinSol->KKoffset[LinSol->KKIndex.size() - 1][iproc] - LinSol->KKoffset[0][iproc];
155
156 NumericVector* NNZ_d = NumericVector::build().release();
157 NNZ_d->init(*LinSol->_EPS);
158 NNZ_d->zero();
159
160 NumericVector* NNZ_o = NumericVector::build().release();
161 NNZ_o->init(*LinSol->_EPS);
162 NNZ_o->zero();
163
164 for(unsigned k = 0; k < _SolSystemPdeIndex.size(); k++) {
165
166 unsigned solIndex = _SolSystemPdeIndex[k];
167 unsigned solType = _ml_sol->GetSolutionType(solIndex);
168 unsigned kOffset = LinSol->KKoffset[k][iproc];
169
170 unsigned solOffset = mesh->_dofOffset[solType][iproc];
171 unsigned solOffsetp1 = mesh->_dofOffset[solType][iproc + 1];
172 for(unsigned i = solOffset; i < solOffsetp1; i++) {
173 if(solType > 2 || amrRestriction[solType].find(i) == amrRestriction[solType].end()) {
174 NNZ_d->set(kOffset + (i - solOffset), 1);
175 }
176 else {
177 double cnt_d = 0;
178 double cnt_o = 0;
179 for(std::map<unsigned, double> ::iterator it = amrRestriction[solType][i].begin(); it != amrRestriction[solType][i].end(); it++) {
180 if(it->first >= solOffset && it->first < solOffsetp1) {
181 cnt_d++;
182 }
183 else {
184 cnt_o++;
185 }
186 }
187 NNZ_d->set(kOffset + (i - solOffset), cnt_d);
188 NNZ_o->set(kOffset + (i - solOffset), cnt_o);
189 }
190 }
191 }
192
193 NNZ_d->close();
194 NNZ_o->close();
195
196 unsigned offset = LinSol->KKoffset[0][iproc];
197 vector <int> nnz_d(n_loc);
198 vector <int> nnz_o(n_loc);
199
200 for(int i = 0; i < n_loc; i++) {
201 nnz_d[i] = static_cast <int>((*NNZ_d)(offset + i));
202 nnz_o[i] = static_cast <int>((*NNZ_o)(offset + i));
203 }
204
205 delete NNZ_d;
206 delete NNZ_o;
207
208 _PPamr[level] = SparseMatrix::build().release();
209 _PPamr[level]->init(n, n, n_loc, n_loc, nnz_d, nnz_o);
210
211 _RRamr[level] = SparseMatrix::build().release();
212 _RRamr[level]->init(n, n, n_loc, n_loc, nnz_d, nnz_o);
213
214 for(unsigned k = 0; k < _SolSystemPdeIndex.size(); k++) {
215
216 unsigned solIndex = _SolSystemPdeIndex[k];
217 unsigned solPairIndex = _ml_sol->GetSolutionPairInverseIndex(k);
218
219 unsigned kPair = GetSolPdeIndex(_ml_sol->GetSolutionName(solPairIndex));
220
221 unsigned kOffset = LinSol->KKoffset[k][iproc];
222 unsigned kPairOffset = LinSol->KKoffset[kPair][iproc];
223
224 unsigned solType = _ml_sol->GetSolutionType(solIndex);
225
226 unsigned solOffset = mesh->_dofOffset[solType][iproc];
227 unsigned solOffsetp1 = mesh->_dofOffset[solType][iproc + 1];
228
229 for(unsigned i = solOffset; i < solOffsetp1; i++) {
230 unsigned irow = kOffset + (i - solOffset);
231 if(solType > 2 || amrRestriction[solType].find(i) == amrRestriction[solType].end()) {
232 std::vector <int> col(1, irow);
233 double value = 1.;
234 _PPamr[level]->insert_row(irow, 1, col, &value);
235 _RRamr[level]->insert_row(irow, 1, col, &value);
236 }
237 else {
238 bool solidMarki = amrSolidMark[solType][i];
239 int ncols = amrRestriction[solType][i].size();
240 std::vector <int> colPP(ncols);
241 std::vector <int> colRR(ncols);
242 std::vector <double> valuePP(ncols);
243 std::vector <double> valueRR(ncols);
244 unsigned j = 0;
245 for(std::map<unsigned, double> ::iterator it = amrRestriction[solType][i].begin(); it != amrRestriction[solType][i].end(); it++) {
246 bool solidMarkj = amrSolidMark[solType][it->first];
247 if(it->first >= solOffset && it->first < solOffsetp1) {
248 colPP[j] = kOffset + (it->first - solOffset);
249 if(solidMarki == true && solidMarkj == false && k != kPair){
250 colRR[j] = kPairOffset + (it->first - solOffset);
251 }
252 else{
253 colRR[j] = kOffset + (it->first - solOffset);
254 }
255 }
256 else {
257 unsigned jproc = _msh[level]->IsdomBisectionSearch(it->first, solType);
258 colPP[j] = LinSol->KKoffset[k][jproc] + (it->first - mesh->_dofOffset[solType][jproc]);
259 if(solidMarki == true && solidMarkj == false && k != kPair){
260 colRR[j] = LinSol->KKoffset[kPair][jproc] + (it->first - mesh->_dofOffset[solType][jproc]);
261 }
262 else{
263 colRR[j] = LinSol->KKoffset[k][jproc] + (it->first - mesh->_dofOffset[solType][jproc]);
264 }
265 }
266 valuePP[j] = it->second;
267 if( solidMarki == true && solidMarkj == false && k == kPair){
268 valueRR[j] = 0.;
269 }
270 else{
271 valueRR[j] = it->second;
272 }
273 j++;
274 }
275 _PPamr[level]->insert_row(irow, ncols, colPP, &valuePP[0]);
276 _RRamr[level]->insert_row(irow, ncols, colRR, &valueRR[0]);
277 }
278 }
279 }
280 _PPamr[level]->close();
281 _RRamr[level]->close();
282
283 _PPamr[level]->get_transpose(*_PPamr[level]);
284 }

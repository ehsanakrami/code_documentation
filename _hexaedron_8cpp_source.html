204 double HexQuadratic::eval_phi(const int *I,const double* x) const {
205 const double ix=I[0]-1.,jx=I[1]-1.,kx=I[2]-1.;
206 return (fabs(ix*jx*kx)==0)?
207 lagQuadratic(x[0],I[0])*lagQuadratic(x[1],I[1])*lagQuadratic(x[2],I[2])
208 :
209 (-2.+ix*x[0]+jx*x[1]+kx*x[2])*lagQuadratic(x[0],I[0])*lagQuadratic(x[1],I[1])*lagQuadratic(x[2],I[2]);
210 }

439 void elem_type::set_prolongation_OneElement_All_FE(const basis* linearElement) {
440
441 const unsigned n_dofs = GetNDofs();
442
443 const double threshold_derivative_nonzero = 1.0e-14;
444
445 int counter = 0;
446
447 std::vector<unsigned int> n_geom_elems_after_refinement = {2, 4, 8}; //this is valid for all six shapes we have: Line (1d); Quad, Tri (2d); Hex, Tet, Wedge (3d)
448
449 for(int i = 0; i < _nf; i++) {
450
451 std::vector<double> jac(_dim + 1, 0.);
452
453 if(_SolType == 4 && i / n_geom_elems_after_refinement[_dim-1] >= 1) { //if piece_wise_linear derivatives
454 for(int k = 0; k < linearElement->n_dofs(); k++) {
455
456 //coordinates of the coarse vertices with respect to the fine elements
457 std::vector<double> xv(_dim);
458 for(int d = 0; d < _dim; d++) xv[d] = * (linearElement->GetXcoarse(linearElement->GetFine2CoarseVertexMapping(i % n_geom_elems_after_refinement[_dim-1], k)) + d);
459
460 for(unsigned int ideriv = 0; ideriv < _dim; ideriv++) {
461 if( i / n_geom_elems_after_refinement[_dim-1] == (ideriv + 1) ) {
462 for(int d = 0; d < _dim; d++) jac[d+1] += linearElement->eval_dphidxyz(ideriv, linearElement->GetIND(k), _pt_basis->GetX(i) ) * xv[d];
463 }
464 } //ideriv
465
466
467 } //k
468 }
469
470 for(int j = 0; j < n_dofs ; j++) {
471 double phi = _pt_basis->eval_phi(_IND[j], _pt_basis->GetX(i) );
472 if(_SolType == 4 && i / n_geom_elems_after_refinement[_dim-1] >= 1) { //if piece_wise_linear
473 phi = jac[j];
474 }
475 if(fabs(phi) >= threshold_derivative_nonzero) {
476 counter++;
477 }
478 }
479
480 } //_nf
481
482 double* pt_d;
483 int* pt_i;
484
485 _prol_val = new double * [_nf + 1];
486 _prol_ind = new int * [_nf + 1];
487 _mem_prol_val = new double [counter];
488 _mem_prol_ind = new int [counter];
489
490 pt_d = _mem_prol_val;
491 pt_i = _mem_prol_ind;
492
493 for(int i = 0; i < _nf; i++) {
494
495 std::vector<double> jac(_dim + 1, 0.);
496
497 if(_SolType == 4 && i / n_geom_elems_after_refinement[_dim-1] >= 1) { //if piece_wise_linear derivatives
498 for(int k = 0; k < linearElement->n_dofs(); k++) {
499
500 //coordinates of the coarse vertices with respect to the fine elements
501 std::vector<double> xv(_dim);
502 for(int d = 0; d < _dim; d++) xv[d] = * (linearElement->GetXcoarse(linearElement->GetFine2CoarseVertexMapping(i % n_geom_elems_after_refinement[_dim-1], k)) + d);
503
504 for(unsigned int ideriv = 0; ideriv < _dim; ideriv++) {
505 if( i / n_geom_elems_after_refinement[_dim-1] == (ideriv + 1) ) {
506 for(int d = 0; d < _dim; d++) jac[d+1] += linearElement->eval_dphidxyz(ideriv, linearElement->GetIND(k), _pt_basis->GetX(i) ) * xv[d];
507 }
508 } //ideriv
509
510 } //k
511
512 }
513
514 _prol_val[i] = pt_d;
515 _prol_ind[i] = pt_i;
516 for(int j = 0; j < n_dofs ; j++) {
517 double phi = _pt_basis->eval_phi(_IND[j], _pt_basis->GetX(i) );
518 if(_SolType == 4 && i / n_geom_elems_after_refinement[_dim-1] >= 1) { //if piece_wise_linear derivatives
519 phi = jac[j];
520 }
521 if(fabs(phi) >= threshold_derivative_nonzero) {
522 *(pt_d++) = phi;
523 *(pt_i++) = j;
524 }
525 }
526 }
527
528 _prol_val[_nf] = pt_d;
529 _prol_ind[_nf] = pt_i;
530
531
532 }

536
537 // shape function and its derivatives evaluated at Gauss'points
538 const int n_gauss = _gauss->GetGaussPointsNumber();
539 const unsigned n_dofs = GetNDofs();
540
541 _phi = new double*[n_gauss];
542 _dphidxi = new double*[n_gauss];
543 _d2phidxi2 = new double*[n_gauss];
544
545 _phi_memory = new double [n_gauss * n_dofs ];
546 _dphidxi_memory = new double [n_gauss * n_dofs ];
547 _d2phidxi2_memory = new double [n_gauss * n_dofs ];
548
549 for(unsigned i = 0; i < n_gauss; i++) {
550 _phi[i] = &_phi_memory[i * n_dofs ];

553 }
554
555 const double* ptx[1] = {_gauss->GetGaussWeightsPointer() + n_gauss}; // you sum an integer to a pointer, which offsets the pointer as a result

557 for(unsigned i = 0; i < n_gauss; i++) {
558 double x[1];
559
560 for(unsigned j = 0; j < 1; j++) {
561 x[j] = *ptx[j];
562 ptx[j]++;
563 }
564

566 _phi[i][j] = _pt_basis->eval_phi(_IND[j], x);

113template <class Base>
114void TransientSystem<Base>::NewmarkAccUpdate() {
115
116 const double gamma = 0.5;
117 const double a5 = -1.*(1. - gamma)/gamma;
118 const double a1 = 1./(gamma*_dt);
119 const double a2 = -1./(gamma*_dt);
120
121 unsigned dim = this->_msh[0]->GetDimension();
122
123 unsigned axyz[3];
124 unsigned vxyz[3];
125 const char accname[3][3] = {"AX","AY","AZ"};
126 const char velname[3][2] = {"U","V","W"};
127
128 for(unsigned i=0; i<dim; i++) {
129 axyz[i] = this->_ml_sol->GetIndex(&accname[i][0]);
130 vxyz[i] = this->_ml_sol->GetIndex(&velname[i][0]);
131 }
132
133 for (int ig=0;ig< this->_gridn;ig++) {
134 for(unsigned i=0; i<dim; i++) {
135 this->_solution[ig]->_Sol[axyz[i]]->scale(a5);
136 this->_solution[ig]->_Sol[axyz[i]]->add(a1,*(this->_solution[ig]->_Sol[vxyz[i]]));
137 this->_solution[ig]->_Sol[axyz[i]]->add(a2,*(this->_solution[ig]->_SolOld[vxyz[i]]));
138 }
139 }
140}

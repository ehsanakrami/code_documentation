239void AssembleBoussinesqAppoximation(MultiLevelProblem& ml_prob) {
240 // ml_prob is the global object from/to where get/set all the data
241 // level is the level of the PDE system to be assembled
242 // levelMax is the Maximum level of the MultiLevelProblem
243 // assembleMatrix is a flag that tells if only the residual or also the matrix should be assembled
244
245 // extract pointers to the several objects that we are going to use
246 NonLinearImplicitSystem* mlPdeSys = &ml_prob.get_system<NonLinearImplicitSystem> ("NS"); // pointer to the linear implicit system named "Poisson"
247 const unsigned level = mlPdeSys->GetLevelToAssemble();
248
249 Mesh* msh = ml_prob._ml_msh->GetLevel(level); // pointer to the mesh (level) object
250 elem* el = msh->el; // pointer to the elem object in msh (level)
251
252 MultiLevelSolution* mlSol = ml_prob._ml_sol; // pointer to the multilevel solution object
253 Solution* sol = ml_prob._ml_sol->GetSolutionLevel(level); // pointer to the solution (level) object
254
255
256 LinearEquationSolver* pdeSys = mlPdeSys->_LinSolver[level]; // pointer to the equation (level) object
257
258 bool assembleMatrix = mlPdeSys->GetAssembleMatrix();
259 // call the adept stack object
260
261 SparseMatrix* KK = pdeSys->_KK; // pointer to the global stifness matrix object in pdeSys (level)
262 NumericVector* RES = pdeSys->_RES; // pointer to the global residual vector object in pdeSys (level)
263
264 const unsigned dim = msh->GetDimension(); // get the domain dimension of the problem
265 unsigned dim2 = (3 * (dim - 1) + !(dim - 1)); // dim2 is the number of second order partial derivatives (1,3,6 depending on the dimension)
266 unsigned iproc = msh->processor_id(); // get the process_id (for parallel computation)
267
268 // reserve memory for the local standar vectors
269 const unsigned maxSize = static_cast< unsigned >(ceil(pow(3, dim))); // conservative: based on line3, quad9, hex27
270
271 //solution variable
272 unsigned solTIndex;
273 solTIndex = mlSol->GetIndex("T"); // get the position of "T" in the ml_sol object
274 unsigned solTType = mlSol->GetSolutionType(solTIndex); // get the finite element type for "T"
275
276 vector < unsigned > solVIndex(dim);
277 solVIndex[0] = mlSol->GetIndex("U"); // get the position of "U" in the ml_sol object
278 solVIndex[1] = mlSol->GetIndex("V"); // get the position of "V" in the ml_sol object
279
280 if(dim == 3) solVIndex[2] = mlSol->GetIndex("W"); // get the position of "V" in the ml_sol object
281
282 unsigned solVType = mlSol->GetSolutionType(solVIndex[0]); // get the finite element type for "u"
283
284 unsigned solPIndex;
285 solPIndex = mlSol->GetIndex("P"); // get the position of "P" in the ml_sol object
286 unsigned solPType = mlSol->GetSolutionType(solPIndex); // get the finite element type for "u"
287
288 unsigned solTPdeIndex;
289 solTPdeIndex = mlPdeSys->GetSolPdeIndex("T"); // get the position of "T" in the pdeSys object
290
291 // std::cout << solTIndex <<" "<<solTPdeIndex<<std::endl;
292
293
294 vector < unsigned > solVPdeIndex(dim);
295 solVPdeIndex[0] = mlPdeSys->GetSolPdeIndex("U"); // get the position of "U" in the pdeSys object
296 solVPdeIndex[1] = mlPdeSys->GetSolPdeIndex("V"); // get the position of "V" in the pdeSys object
297
298 if(dim == 3) solVPdeIndex[2] = mlPdeSys->GetSolPdeIndex("W");
299
300 unsigned solPPdeIndex;
301 solPPdeIndex = mlPdeSys->GetSolPdeIndex("P"); // get the position of "P" in the pdeSys object
302
303 vector < double > solT; // local solution
304 vector < vector < double > > solV(dim); // local solution
305 vector < double > solP; // local solution
306
307 vector < vector < double > > coordX(dim); // local coordinates
308 unsigned coordXType = 2; // get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC)
309
310 solT.reserve(maxSize);
311
312 for(unsigned k = 0; k < dim; k++) {
313 solV[k].reserve(maxSize);
314 coordX[k].reserve(maxSize);
315 }
316
317 solP.reserve(maxSize);
318
319 vector <double> phiV; // local test function
320 vector <double> phiV_x; // local test function first order partial derivatives
321 vector <double> phiV_xx; // local test function second order partial derivatives
322
323 phiV.reserve(maxSize);
324 phiV_x.reserve(maxSize * dim);
325 phiV_xx.reserve(maxSize * dim2);
326
327 vector <double> phiT; // local test function
328 vector <double> phiT_x; // local test function first order partial derivatives
329 vector <double> phiT_xx; // local test function second order partial derivatives
330
331 phiT.reserve(maxSize);
332 phiT_x.reserve(maxSize * dim);
333 phiT_xx.reserve(maxSize * dim2);
334
335 double* phiP;
336 double weight; // gauss point weight
337
338 vector< int > sysDof; // local to global pdeSys dofs
339 sysDof.reserve((dim + 2) *maxSize);
340
341 vector< double > Res; // local redidual vector
342 Res.reserve((dim + 2) *maxSize);
343
344 vector < double > Jac;
345 Jac.reserve((dim + 2) *maxSize * (dim + 2) *maxSize);
346
347 if(assembleMatrix)
348 KK->zero(); // Set to zero all the entries of the Global Matrix
349
350 //BEGIN element loop
351 for(int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
352
353 //BEGIN local dof number extraction
354 unsigned nDofsT = msh->GetElementDofNumber(iel, solTType); //temperature
355 unsigned nDofsV = msh->GetElementDofNumber(iel, solVType); //velocity
356 unsigned nDofsP = msh->GetElementDofNumber(iel, solPType); //pressure
357 unsigned nDofsX = msh->GetElementDofNumber(iel, coordXType); // coordinates
358 unsigned nDofsTVP = nDofsT + dim * nDofsV + nDofsP; // all solutions
359 //END local dof number extraction
360
361 //BEGIN memory allocation
362 Res.resize(nDofsTVP);
363 std::fill(Res.begin(), Res.end(), 0);
364 Jac.resize(nDofsTVP * nDofsTVP);
365 std::fill(Jac.begin(), Jac.end(), 0);
366 sysDof.resize(nDofsTVP);
367
368 solT.resize(nDofsV);
369
370 for(unsigned k = 0; k < dim; k++) {
371 solV[k].resize(nDofsV);
372 coordX[k].resize(nDofsX);
373 }
374
375 solP.resize(nDofsP);
376 //END memory allocation
377
378 //BEGIN global to local extraction
379 for(unsigned i = 0; i < nDofsT; i++) { //temperature
380 unsigned solTDof = msh->GetSolutionDof(i, iel, solTType); //local to global solution dof
381 solT[i] = (*sol->_Sol[solTIndex])(solTDof); //global to local solution value
382 sysDof[i] = pdeSys->GetSystemDof(solTIndex, solTPdeIndex, i, iel); //local to global system dof
383 }
384
385 for(unsigned i = 0; i < nDofsV; i++) { //velocity
386 unsigned solVDof = msh->GetSolutionDof(i, iel, solVType); //local to global solution dof
387
388 for(unsigned k = 0; k < dim; k++) {
389 solV[k][i] = (*sol->_Sol[solVIndex[k]])(solVDof); //global to local solution value
390 sysDof[i + nDofsT + k * nDofsV] = pdeSys->GetSystemDof(solVIndex[k], solVPdeIndex[k], i, iel); //local to global system dof
391 }
392 }
393
394 for(unsigned i = 0; i < nDofsP; i++) { //pressure
395 unsigned solPDof = msh->GetSolutionDof(i, iel, solPType); //local to global solution dof
396 solP[i] = (*sol->_Sol[solPIndex])(solPDof); //global to local solution value
397 sysDof[i + nDofsT + dim * nDofsV] = pdeSys->GetSystemDof(solPIndex, solPPdeIndex, i, iel); //local to global system dof
398 }
399
400 for(unsigned i = 0; i < nDofsX; i++) { //coordinates
401 unsigned coordXDof = msh->GetSolutionDof(i, iel, coordXType); //local to global coordinate dof
402
403 for(unsigned k = 0; k < dim; k++) {
404 coordX[k][i] = (*msh->_topology->_Sol[k])(coordXDof); //global to local coordinate value
405 }
406 }
407
408 //END global to local extraction
409
410 //BEGIN Gauss point loop
411 short unsigned ielGeom = msh->GetElementType(iel);
412
413 for(unsigned ig = 0; ig < msh->_finiteElement[ielGeom][solVType]->GetGaussPointNumber(); ig++) {
414 // *** get gauss point weight, test function and test function partial derivatives ***
415 msh->_finiteElement[ielGeom][solTType]->Jacobian(coordX, ig, weight, phiT, phiT_x, phiT_xx);
416 msh->_finiteElement[ielGeom][solVType]->Jacobian(coordX, ig, weight, phiV, phiV_x, phiV_xx);
417 phiP = msh->_finiteElement[ielGeom][solPType]->GetPhi(ig);
418
419 // evaluate the solution, the solution derivatives and the coordinates in the gauss point
420 double solT_gss = 0;
421 vector < double > gradSolT_gss(dim, 0.);
422
423 for(unsigned i = 0; i < nDofsT; i++) {
424 solT_gss += phiT[i] * solT[i];
425
426 for(unsigned j = 0; j < dim; j++) {
427 gradSolT_gss[j] += phiT_x[i * dim + j] * solT[i];
428 }
429 }
430
431 vector < double > solV_gss(dim, 0);
432 vector < vector < double > > gradSolV_gss(dim);
433
434 for(unsigned k = 0; k < dim; k++) {
435 gradSolV_gss[k].resize(dim);
436 std::fill(gradSolV_gss[k].begin(), gradSolV_gss[k].end(), 0);
437 }
438
439 for(unsigned i = 0; i < nDofsV; i++) {
440 for(unsigned k = 0; k < dim; k++) {
441 solV_gss[k] += phiV[i] * solV[k][i];
442 }
443
444 for(unsigned j = 0; j < dim; j++) {
445 for(unsigned k = 0; k < dim; k++) {
446 gradSolV_gss[k][j] += phiV_x[i * dim + j] * solV[k][i];
447 }
448 }
449 }
450
451 double solP_gss = 0;
452
453 for(unsigned i = 0; i < nDofsP; i++) {
454 solP_gss += phiP[i] * solP[i];
455 }
456
457
458 double alpha = 1.;
459 double beta = 1.;//40000.;
460
461
462 double Pr = Prandtl;
463 double Ra = Rayleigh;
464
465 //BEGIN phiT_i loop: Energy balance
466 for(unsigned i = 0; i < nDofsT; i++) {
467 unsigned irow = i;
468
469 for(unsigned k = 0; k < dim; k++) {
470 Res[irow] += -alpha / sqrt(Ra * Pr) * phiT_x[i * dim + k] * gradSolT_gss[k] * weight;
471 Res[irow] += -phiT[i] * solV_gss[k] * gradSolT_gss[k] * weight;
472
473 if(assembleMatrix) {
474 unsigned irowMat = irow * nDofsTVP;
475
476 for(unsigned j = 0; j < nDofsT; j++) {
477 Jac[ irowMat + j ] += alpha / sqrt(Ra * Pr) * phiT_x[i * dim + k] * phiT_x[j * dim + k] * weight;
478 Jac[ irowMat + j ] += phiT[i] * solV_gss[k] * phiT_x[j * dim + k] * weight;
479 }
480
481 for(unsigned j = 0; j < nDofsV; j++) {
482 unsigned jcol = nDofsT + k * nDofsV + j;
483 Jac[ irowMat + jcol ] += phiT[i] * phiV[j] * gradSolT_gss[k] * weight;
484 }
485 }
486
487 }
488 }
489
490 //END phiT_i loop
491
492 //BEGIN phiV_i loop: Momentum balance
493 for(unsigned i = 0; i < nDofsV; i++) {
494
495 for(unsigned k = 0; k < dim; k++) {
496 unsigned irow = nDofsT + k * nDofsV + i;
497
498 for(unsigned l = 0; l < dim; l++) {
499 Res[irow] += -sqrt(Pr / Ra) * phiV_x[i * dim + l] * (gradSolV_gss[k][l] + gradSolV_gss[l][k]) * weight;
500 Res[irow] += -phiV[i] * solV_gss[l] * gradSolV_gss[k][l] * weight;
501 }
502
503 Res[irow] += solP_gss * phiV_x[i * dim + k] * weight;
504
505 if(k == 1) {
506 Res[irow] += beta * solT_gss * phiV[i] * weight;
507 }
508
509 if(assembleMatrix) {
510 unsigned irowMat = nDofsTVP * irow;
511
512 for(unsigned l = 0; l < dim; l++) {
513 for(unsigned j = 0; j < nDofsV; j++) {
514 unsigned jcol1 = (nDofsT + k * nDofsV + j);
515 unsigned jcol2 = (nDofsT + l * nDofsV + j);
516 Jac[ irowMat + jcol1] += sqrt(Pr / Ra) * phiV_x[i * dim + l] * phiV_x[j * dim + l] * weight;
517 Jac[ irowMat + jcol2] += sqrt(Pr / Ra) * phiV_x[i * dim + l] * phiV_x[j * dim + k] * weight;
518 Jac[ irowMat + jcol1] += phiV[i] * solV_gss[l] * phiV_x[j * dim + l] * weight;
519 Jac[ irowMat + jcol2] += phiV[i] * phiV[j] * gradSolV_gss[k][l] * weight;
520 }
521 }
522
523 for(unsigned j = 0; j < nDofsP; j++) {
524 unsigned jcol = (nDofsT + dim * nDofsV) + j;
525 Jac[ irowMat + jcol] += - phiV_x[i * dim + k] * phiP[j] * weight;
526 }
527
528 if(k == 1) {
529 for(unsigned j = 0; j < nDofsT; j++) {
530 Jac[ irowMat + j ] += -beta * phiV[i] * phiT[j] * weight;
531 }
532 }
533 }
534
535 }
536 }
537
538 //END phiV_i loop
539
540 //BEGIN phiP_i loop: mass balance
541 for(unsigned i = 0; i < nDofsP; i++) {
542 unsigned irow = nDofsT + dim * nDofsV + i;
543
544 for(int k = 0; k < dim; k++) {
545 Res[irow] += -(gradSolV_gss[k][k]) * phiP[i] * weight;
546
547 if(assembleMatrix) {
548 unsigned irowMat = nDofsTVP * irow;
549
550 for(unsigned j = 0; j < nDofsV; j++) {
551 unsigned jcol = (nDofsT + k * nDofsV + j);
552 Jac[ irowMat + jcol ] += phiP[i] * phiV_x[j * dim + k] * weight;
553 }
554 }
555
556 }
557 }
558
559 //END phiP_i loop
560
561 }
562
563 //END Gauss point loop
564
565 //BEGIN local to global Matrix/Vector assembly
566 RES->add_vector_blocked(Res, sysDof);
567
568 if(assembleMatrix) {
569 KK->add_matrix_blocked(Jac, sysDof, sysDof);
570 }
571
572 //END local to global Matrix/Vector assembly
573
574 }
575
576 //END element loop
577
578 RES->close();
579
580 if(assembleMatrix) {
581 KK->close();
582 }
583
584 // ***************** END ASSEMBLY *******************
585}

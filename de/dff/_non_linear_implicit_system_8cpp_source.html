111 bool NonLinearImplicitSystem::HasNonLinearConverged(const unsigned igridn, double &nonLinearEps) {
112
113 bool conv = true;
114 double L2normEps;
115 double L2normSol;
116 double L2normEpsDividedSol;
117 double L2normRes;
118
119 nonLinearEps = 0.;
120 const double absMinNonlinearEps = 1.e-12;
121 const double absMinNormSol = 1.e-12;
122 const double mindeltaNormSol = 1.e-50;
123
124// we need to store the global vector here
125 if (_debug_nonlinear) { *(_eps_fine[_nonliniteration]) = *(_LinSolver[_gridn-1]->_EPS); }
126
127 for(unsigned k = 0; k < _SolSystemPdeIndex.size(); k++) {
128
129 unsigned indexSol = _SolSystemPdeIndex[k];
130 L2normRes = _solution[igridn]->_Res[indexSol]->l2_norm();
131 L2normEps = _solution[igridn]->_Eps[indexSol]->l2_norm();
132 L2normSol = _solution[igridn]->_Sol[indexSol]->l2_norm();
133 L2normEpsDividedSol = L2normEps / (L2normSol + mindeltaNormSol);
134
135 std::cout << " ********* Level Max " << igridn + 1 << " Nonlinear Eps_l2norm/Sol_l2norm " << \
136 std::scientific << _ml_sol->GetSolutionName(indexSol) << "= " << L2normEpsDividedSol << \
137 " ** Eps_l2norm= " << L2normEps << " ** Sol_l2norm= " << L2normSol << std::endl;
138 nonLinearEps = (nonLinearEps > L2normEpsDividedSol) ? nonLinearEps : L2normEpsDividedSol;
139
140 if((L2normEpsDividedSol < _max_nonlinear_convergence_tolerance || L2normEps < absMinNonlinearEps || L2normSol < absMinNormSol ) && conv == true) {
141 conv = true;
142 }
143 else {
144 conv = false;
145 }
146
147 }
148
149
150 return conv;
151 }

155 void NonLinearImplicitSystem::MGsolve(const MgSmootherType& mgSmootherType) {
156
157 _bitFlipCounter = 0;
158
159 clock_t start_mg_time = clock();
160
161 double totalAssembyTime = 0.;
162
163 unsigned grid0;
164
165 if(_mg_type == F_CYCLE) {
166 std::cout << std::endl << " *** Start Nonlinear Full-Cycle ***" << std::endl;
167 grid0 = 0;
168 }
169 else if(_mg_type == V_CYCLE) {
170 std::cout << std::endl << " *** Start Nonlinear V-Cycle ***" << std::endl;
171 grid0 = _gridn - 1;
172 }
173 else {
174 std::cout << "wrong CYCLE type for this solver " << std::endl;
175 abort();
176 }
177
178 unsigned AMRCounter = 0;
179
180 for(unsigned igridn = grid0; igridn < _gridn; igridn++) { //_igridn
181 std::cout << std::endl << " ****** Start Level Max " << igridn + 1 << " ******" << std::endl;
182 clock_t start_nl_time = clock();
183
184 bool ThisIsAMR = (_mg_type == F_CYCLE && _AMRtest && AMRCounter < _maxAMRlevels && igridn == _gridn - 1u) ? 1 : 0;
185
186restart:
187 if(ThisIsAMR) _solution[igridn]->InitAMREps();
188
189
190 for(unsigned nonLinearIterator = 0; nonLinearIterator < _n_max_nonlinear_iterations; nonLinearIterator++) {
191
192 _nonliniteration = nonLinearIterator;
193
194 if (_debug_nonlinear) {
195 _eps_fine.push_back(NumericVector::build().release());
196 _eps_fine.back()->init(*_LinSolver[_gridn-1]->_EPS); //I'd say init also fills the vector
197 *(_eps_fine.back()) = *(_LinSolver[_gridn-1]->_EPS);
198 }
199
200 std::cout << std::endl << " ********* Nonlinear iteration " << nonLinearIterator + 1 << " *********" << std::endl;
201
202 clock_t start_preparation_time = clock();
203 clock_t start_assembly_time = clock();
204 _levelToAssemble = igridn; //Be carefull!!!! this is needed in the _assemble_function
205 _LinSolver[igridn]->SetResZero();
206 _assembleMatrix = _buildSolver;
207 _assemble_system_function(_equation_systems);
208 std::cout << " ********* Level Max " << igridn + 1 << " ASSEMBLY TIME:\t" << \
209 static_cast<double>((clock() - start_assembly_time)) / CLOCKS_PER_SEC << std::endl;
210
211 if(!_ml_msh->GetLevel(igridn)->GetIfHomogeneous()) {
212 if(!_RRamr[igridn]) {
213 (_LinSolver[igridn]->_RESC)->matrix_mult_transpose(*_LinSolver[igridn]->_RES, *_PPamr[igridn]);
214 }
215 else {
216 (_LinSolver[igridn]->_RESC)->matrix_mult(*_LinSolver[igridn]->_RES, *_RRamr[igridn]);
217 }
218 *(_LinSolver[igridn]->_RES) = *(_LinSolver[igridn]->_RESC);
219 }
220
221 if(_buildSolver) {
222
223 _MGmatrixFineReuse = (0 == nonLinearIterator) ? false : true;
224 _MGmatrixCoarseReuse = (igridn - grid0 > 0) ? true : _MGmatrixFineReuse;
225
226 if(!_ml_msh->GetLevel(igridn)->GetIfHomogeneous()) {
227 _LinSolver[igridn]->SwapMatrices();
228 if(!_RRamr[igridn]) {
229 _LinSolver[igridn]->_KK->matrix_PtAP(*_PPamr[igridn], *_LinSolver[igridn]->_KKamr, _MGmatrixFineReuse);
230 }
231 else {
232 _LinSolver[igridn]->_KK->matrix_ABC(*_RRamr[igridn], *_LinSolver[igridn]->_KKamr, *_PPamr[igridn], _MGmatrixFineReuse);
233 }
234 }
235
236 clock_t mg_proj_mat_time = clock();
237 for(unsigned i = igridn; i > 0; i--) {
238 if(_RR[i]) {
239 if(i == igridn)
240 _LinSolver[i - 1u]->_KK->matrix_ABC(*_RR[i], *_LinSolver[i]->_KK, *_PP[i], _MGmatrixFineReuse);
241 else {
242 _LinSolver[i - 1u]->_KK->matrix_ABC(*_RR[i], *_LinSolver[i]->_KK, *_PP[i], _MGmatrixCoarseReuse);
243 if(_LinSolver[i - 1u]->_KKamr) {
244 delete _LinSolver[i - 1u]->_KKamr;
245 _LinSolver[i - 1u]->_KKamr = NULL;
246 }
247 }
248 }
249 else {
250 if(i == igridn)
251 _LinSolver[i - 1u]->_KK->matrix_PtAP(*_PP[i], *_LinSolver[i]->_KK, _MGmatrixFineReuse);
252 else {
253 _LinSolver[i - 1u]->_KK->matrix_PtAP(*_PP[i], *_LinSolver[i]->_KK, _MGmatrixCoarseReuse);
254 if(_LinSolver[i - 1u]->_KKamr) {
255 delete _LinSolver[i - 1u]->_KKamr;
256 _LinSolver[i - 1u]->_KKamr = NULL;
257 }
258 }
259 }
260 }
261 std::cout << " ********* Level Max " << igridn + 1 << " MG PROJECTION MATRICES TIME:\t" \
262 << static_cast<double>((clock() - mg_proj_mat_time)) / CLOCKS_PER_SEC << std::endl;
263
264 clock_t mg_init_time = clock();
265
266 _LinSolver[igridn]->MGInit(mgSmootherType, igridn + 1, _mgOuterSolver);
267
268 for(unsigned i = 0; i <= igridn; i++) {
269 unsigned npre = (i == 0)? _npre0 : _npre;
270 unsigned npost = (i == 0)? 0 : _npost;
271 if(_RR[i])
272 _LinSolver[i]->MGSetLevel(_LinSolver[igridn], igridn, _VariablesToBeSolvedIndex, _PP[i], _RR[i], npre, npost);
273 else
274 _LinSolver[i]->MGSetLevel(_LinSolver[igridn], igridn, _VariablesToBeSolvedIndex, _PP[i], _PP[i], npre, npost);
275 }
276
277 std::cout << " ********* Level Max " << igridn + 1 << " MGINIT TIME:\t" \
278 << static_cast<double>((clock() - mg_init_time)) / CLOCKS_PER_SEC << std::endl;
279 }
280
281 totalAssembyTime += static_cast<double>((clock() - start_assembly_time)) / CLOCKS_PER_SEC;
282 std::cout << " ********* Level Max " << igridn + 1 << " PREPARATION TIME:\t" << \
283 static_cast<double>((clock() - start_preparation_time)) / CLOCKS_PER_SEC << std::endl;
284 clock_t startUpdateResidualTime = clock();
285
286 for(unsigned updateResidualIterator = 0; updateResidualIterator < _maxNumberOfResidualUpdateIterations; updateResidualIterator++) {
287
288 std::cout << " ********* Linear Cycle + Residual Update iteration " << updateResidualIterator + 1 << std::endl;
289
290 bool thisHasConverged;
291
292 thisHasConverged = Vcycle(igridn, mgSmootherType);
293
294 if(thisHasConverged || updateResidualIterator == _maxNumberOfResidualUpdateIterations - 1) break;
295
296 _LinSolver[igridn]->SetResZero();
297 _assembleMatrix = false;
298 _assemble_system_function(_equation_systems);
299 if(!_ml_msh->GetLevel(igridn)->GetIfHomogeneous()) {
300 if(!_RRamr[igridn]) {
301 (_LinSolver[igridn]->_RESC)->matrix_mult_transpose(*_LinSolver[igridn]->_RES, *_PPamr[igridn]);
302 }
303 else {
304 (_LinSolver[igridn]->_RESC)->matrix_mult(*_LinSolver[igridn]->_RES, *_RRamr[igridn]);
305 }
306 *(_LinSolver[igridn]->_RES) = *(_LinSolver[igridn]->_RESC);
307 }
308 if(_bitFlipOccurred) break;
309 }
310
311 if(_buildSolver) {
312 if(!_ml_msh->GetLevel(igridn)->GetIfHomogeneous()) {
313 _LinSolver[igridn]->SwapMatrices();
314 }
315 _LinSolver[igridn]->MGClear();
316 }
317
318 double nonLinearEps;
319 bool nonLinearIsConverged = HasNonLinearConverged(igridn, nonLinearEps);
320
321 std::cout << " ********* Linear Cycle + Residual Update-Cycle TIME:\t" << std::setw(11) << std::setprecision(6) << std::fixed
322 << static_cast<double>((clock() - startUpdateResidualTime)) / CLOCKS_PER_SEC << std::endl;
323
324
325 // *****************
326 if (_debug_nonlinear) { print_iteration_to_file(nonLinearIterator); }
327 // *****************
328
329
330 if(nonLinearIsConverged || _bitFlipOccurred) break;
331
332 } //end nonlinear iterations
333
334 _last_nonliniteration = _nonliniteration;
335
336
337 if(_bitFlipOccurred && _bitFlipCounter == 1){
338 goto restart;
339 }
340
341 if(igridn + 1 < _gridn) ProlongatorSol(igridn + 1);
342
343 if(ThisIsAMR) AddAMRLevel(AMRCounter);
344
345
346 std::cout << std::endl << " ****** Nonlinear-Cycle TIME: " << std::setw(11) << std::setprecision(6) << std::fixed
347 << static_cast<double>((clock() - start_nl_time)) / CLOCKS_PER_SEC << std::endl;
348
349 std::cout << std::endl << " ****** End Level Max " << igridn + 1 << " ******" << std::endl;
350 }
351
352 double totalSolverTime = static_cast<double>((clock() - start_mg_time)) / CLOCKS_PER_SEC;
353 std::cout << std::endl << " *** Nonlinear Solver TIME: " << std::setw(11) << std::setprecision(6) << std::fixed
354 << totalSolverTime << " = assembly TIME( " << totalAssembyTime << " ) + "
355 << " solver TIME( " << totalSolverTime - totalAssembyTime << " ) " << std::endl;
356
357 _totalAssemblyTime += totalAssembyTime;
358 _totalSolverTime += totalSolverTime - totalAssembyTime;
359 }

388 void NonLinearImplicitSystem::compute_convergence_rate() const {
389
390
391 const unsigned index_upper = _last_nonliniteration;
392
393 for(unsigned nonLinearIterator = 0; nonLinearIterator < index_upper; nonLinearIterator++) {
394
395
396 NumericVector* eps_fine_temp = NumericVector::build().release();
397 eps_fine_temp->init(*_LinSolver[_gridn-1]->_EPS);
398
399 eps_fine_temp->close();
400 eps_fine_temp->zero();
401
402 const unsigned index_lower = nonLinearIterator + 1;
403
404 for(unsigned n = index_upper; n >= index_lower; n--) *(eps_fine_temp) += *(_eps_fine[n]);
405
406 const double numerator = eps_fine_temp->/*linfty_norm*/l2_norm();
407 *(eps_fine_temp) += *(_eps_fine[index_lower - 1]);
408 const double denominator = eps_fine_temp->/*linfty_norm*/l2_norm();
409
410 std::cout << std::setw(16) << std::setprecision(16) << std::scientific << nonLinearIterator << " " << numerator / (denominator * denominator) << std::endl;
411
412 }

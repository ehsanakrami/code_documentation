211 template < class ARRAY_TYPE >
212 void VTKWriter::print_element_based_fields(const std::string field_string, const std::string field_datatype,
213 std::ofstream & fout, std::ofstream & Pfout,
214 void* buffer_void,
215 const unsigned elemetOffset, const unsigned elemetOffsetp1,
216 const unsigned * dim_array_elvar,
217 const Mesh * mesh,
218 const unsigned fe_index,
219 std::vector <char> & enc) const {
220
221
222 //NumericVector& material = mesh->_topology->GetSolutionName( "Material" );
223
224 // point pointer to common memory area buffer of void type;
225 ARRAY_TYPE * var_el = static_cast< ARRAY_TYPE * >( buffer_void );
226
227 int icount = 0;
228
229 if (field_string == "Material") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetElementMaterial(iel); icount++; } }
230 else if (field_string == "Group") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetElementGroup(iel); icount++; } }
231 else if (field_string == "TYPE") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->GetElementType(iel); icount++; } }
232 else if (field_string == "Level") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = mesh->el->GetElementLevel(iel); icount++; } }
233 else if (field_string == "Metis partition") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { var_el[icount] = _iproc; icount++; } }
234 else if (field_string == "types") { for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { short unsigned ielt = mesh->GetElementType( iel );
235 var_el[icount] = femusToVtkCellType[fe_index][ielt];
236 icount++;
237 }
238 }
239 else if (field_string == "offsets") { int offset_el = 0;
240 for( unsigned iel = elemetOffset; iel < elemetOffsetp1; iel++ ) { offset_el += mesh->GetElementDofNumber( iel, fe_index ); var_el[icount] = offset_el; icount++; } }
241 else { abort(); }
242
243
244 print_data_array< ARRAY_TYPE >(field_string, field_datatype, fout, Pfout, dim_array_elvar, var_el, enc);
245
246 }

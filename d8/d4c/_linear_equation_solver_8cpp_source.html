36 std::unique_ptr<LinearEquationSolver> LinearEquationSolver::build(const unsigned& igrid, Solution* other_solution, const LinearEquationSolverType& smoother_type, const SolverPackage solver_package)
37 {
38 // Build the appropriate solver
39
40 switch(solver_package) {
41#ifdef HAVE_PETSC
42 case PETSC_SOLVERS: {
43 switch(smoother_type) {
44 case FEMuS_ASM: {
45 std::unique_ptr<LinearEquationSolver> ap(new LinearEquationSolverPetscAsm(igrid, other_solution));
46 return ap;
47 }
48 case FEMuS_DEFAULT: {
49 std::unique_ptr<LinearEquationSolver> ap(new LinearEquationSolverPetsc(igrid, other_solution));
50 return ap;
51 }
52 case FEMuS_FIELDSPLIT: {
53 std::unique_ptr<LinearEquationSolver> ap(new LinearEquationSolverPetscFieldSplit(igrid, other_solution));
54 return ap;
55 }
56 }
57 }
58#endif
59#ifdef HAVE_TRILINOS
60 case TRILINOS_SOLVERS: {
61 std::unique_ptr<LinearEquationSolver> ap(new AztecLinearEquationSolver);
62 return ap;
63 }
64#endif
65 default:
66 std::cerr << "ERROR: Unrecognized solver package: "
67 << solver_package << std::endl;
68 abort();
69 }
70 }

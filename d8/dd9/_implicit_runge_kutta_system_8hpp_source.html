131 template <class Base>
132 void ImplicitRungeKuttaSystem<Base>::AddSolutionToSystemPDE (const char solname[]) {
133
134 std::ostringstream soliname;
135 soliname << solname;
136 unsigned size = _solName.size();
137 _solName.resize (size + 1u);
138 _solName[size] = soliname.str();
139
140 _solIndex.resize (size + 1u);
141 _solIndex[size] = this->_ml_sol->GetIndex (solname);
142
143 _solKiName.resize (size + 1u);
144 _solKiName[size].resize (_RK);
145
146 _solKiIndex.resize (size + 1u);
147 _solKiIndex[size].resize (_RK);
148
149 _solRKType.resize(size + 1u);
150 _solRKType[size] = true;
151
152 _itime.assign (_RK, 0.);
153 _time0 = 0.;
154
155 for (unsigned i = 0; i < _RK; i++) {
156
157 std::ostringstream solkiname;
158 solkiname << solname << "k" << i + 1;
159 _solKiName[size][i] = solkiname.str();
160
161 this->_ml_sol->AddSolution (_solKiName[size][i].c_str(), this->_ml_sol->GetSolutionFamily (_solIndex[size]), this->_ml_sol->GetSolutionOrder (_solIndex[size]));
162
163 this->_ml_sol->Initialize (_solKiName[size][i].c_str()); // Since this is time depend problem.
164
165 _solKiIndex[size][i] = this->_ml_sol->GetIndex (_solKiName[size][i].c_str());
166
167 this->Base::AddSolutionToSystemPDE (_solKiName[size][i].c_str());
168 }
169
170 this->_ml_sol->GenerateRKBdc (_solIndex[size], _solKiIndex[size], 0, _itime, _time0, 1., _Ai);
171 }

208 template <class Base>
209 void ImplicitRungeKuttaSystem<Base>::UpdateSolution() {
210 unsigned level = this->_solution.size() - 1u;
211
212 for (unsigned i = 0; i < _solName.size(); i++) {
213
214 unsigned solIndex = _solIndex[i];
215 if( _solRKType[i] ) {
216 * (this->_solution[level]->_Sol[solIndex]) = * (this->_solution[level]->_SolOld[solIndex]);
217 for (unsigned j = 0; j < _RK; j++) {
218 unsigned solkiIndex = _solKiIndex[i][j];
219 this->_solution[level]->_Sol[solIndex]-> add (_b[j] * this->_dt, * (this->_solution[level]->_Sol[solkiIndex]));
220 }
221 this->_solution[level]->_Sol[solIndex]->close();
222 }
223 else if( this->_ml_sol->GetSolutionTimeOrder(solIndex) == 2 ){
224 * (this->_solution[level]->_Sol[solIndex]) = * (this->_solution[level]->_SolOld[solIndex]);
225 this->_solution[level]->_Sol[solIndex]->scale (1. - _bAiSum);
226 for (unsigned j = 0; j < _RK; j++) {
227 unsigned solkiIndex = _solKiIndex[i][j];
228 this->_solution[level]->_Sol[solIndex]-> add (_bAi[j], * (this->_solution[level]->_Sol[solkiIndex]));
229 }
230 }
231 else{
232 unsigned solkiIndex = _solKiIndex[i][_RK-1];
233 * (this->_solution[level]->_Sol[solIndex]) = * (this->_solution[level]->_Sol[solkiIndex]);
234 }
235 }

255 template <class Base>
256 void ImplicitRungeKuttaSystem<Base>::GetIntermediateSolutions (
257 const std::vector < double > &soluOld,
258 const std::vector < std::vector < adept::adouble > > & solk,
259 std::vector < std::vector < adept::adouble > > & solu) {
260
261 //local storage of global mapping and solution
262 for (unsigned i = 0; i < soluOld.size(); i++) {
263 for (unsigned j = 0; j < _RK; j++) {
264 solu[j][i] = soluOld[i];
265 for (unsigned k = 0; k < _RK; k++) {
266 solu[j][i] += this->_dt * _A[ j * _RK + k] * solk[k][i]; // global extraction and local storage for the solution
267 }
268 }
269 }

288 template <class Base>
289 void ImplicitRungeKuttaSystem<Base>::SetImplicitRungeKuttaScheme(const ImplicitRKScheme & RKscheme){
290 _RKScheme = RKscheme;
291 _RK = nRK[RKscheme];
292 _c = cIRK[RKscheme];
293 _b = bIRK[RKscheme];
294 _A = aIRK[RKscheme];
295 _Ai = aiIRK[RKscheme];
296 GetbAi();
297 }
298
299 // This idea is used to update the algebraic variable using RK in DAE problems.
300 // Refer to http://www.ams.org/journals/mcom/2006-75-254/S0025-5718-05-01809-0/S0025-5718-05-01809-0.pdfS
301 template <class Base>
302 void ImplicitRungeKuttaSystem<Base>::GetbAi(){
303 _bAi.assign(_RK, 0.);
304 _bAiSum = 0.;
305 for (unsigned i = 0; i <_RK; i++ ){
306 for (unsigned j = 0; j <_RK; j++ ){
307 _bAi[i] += _b[i] * _Ai[ i *_RK + j];
308 }
309 _bAiSum += _bAi[i];
310 }

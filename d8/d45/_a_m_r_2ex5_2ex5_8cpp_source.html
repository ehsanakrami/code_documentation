184void AssemblePoisson_AD(MultiLevelProblem& ml_prob) {
185 // ml_prob is the global object from/to where get/set all the data
186 // level is the level of the PDE system to be assembled
187 // levelMax is the Maximum level of the MultiLevelProblem
188 // assembleMatrix is a flag that tells if only the residual or also the matrix should be assembled

191 adept::Stack& s = FemusInit::_adeptStack;
192
193 // extract pointers to the several objects that we are going to use
194 NonLinearImplicitSystem* mlPdeSys = &ml_prob.get_system<NonLinearImplicitSystem> ("Poisson"); // pointer to the linear implicit system named "Poisson"
195 const unsigned level = mlPdeSys->GetLevelToAssemble();
196
197 Mesh* msh = ml_prob._ml_msh->GetLevel(level); // pointer to the mesh (level) object
198 elem* el = msh->el; // pointer to the elem object in msh (level)
199
200 MultiLevelSolution* mlSol = ml_prob._ml_sol; // pointer to the multilevel solution object
201 Solution* sol = ml_prob._ml_sol->GetSolutionLevel(level); // pointer to the solution (level) object
202
203
204 LinearEquationSolver* pdeSys = mlPdeSys->_LinSolver[level]; // pointer to the equation (level) object
205 SparseMatrix* KK = pdeSys->_KK; // pointer to the global stifness matrix object in pdeSys (level)
206 NumericVector* RES = pdeSys->_RES; // pointer to the global residual vector object in pdeSys (level)
207
208 const unsigned dim = msh->GetDimension(); // get the domain dimension of the problem
209 unsigned dim2 = (3 * (dim - 1) + !(dim - 1)); // dim2 is the number of second order partial derivatives (1,3,6 depending on the dimension)
210 unsigned iproc = msh->processor_id(); // get the process_id (for parallel computation)
211
212 // reserve memory for the local standard vectors
213 const unsigned maxSize = static_cast< unsigned >(ceil(pow(3, dim))); // conservative: based on line3, quad9, hex27
214
215 //solution variable
216 unsigned solUIndex;
217 solUIndex = mlSol->GetIndex("U"); // get the position of "U" in the ml_sol object = 0
218 unsigned solUType = mlSol->GetSolutionType(solUIndex); // get the finite element type for "T"
219
220
221
222 unsigned solUPdeIndex;
223 solUPdeIndex = mlPdeSys->GetSolPdeIndex("U"); // get the position of "U" in the pdeSys object = 0
224
225 std::cout << solUIndex << " " << solUPdeIndex << std::endl;
226
227
228 vector < adept::adouble > solU; // local solution
229 vector< adept::adouble > aResU; // local redidual vector
230
231 vector < vector < double > > crdX(dim); // local coordinates
232 unsigned crdXType = 2; // get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC)
233
234 solU.reserve(maxSize);
235 aResU.reserve(maxSize);
236
237
238
239 for(unsigned k = 0; k < dim; k++) {
240 crdX[k].reserve(maxSize);
241 }
242
243 vector <double> phi; // local test function
244 vector <double> phi_x; // local test function first order partial derivatives
245 vector <double> phi_xx; // local test function second order partial derivatives
246
247 phi.reserve(maxSize);
248 phi_x.reserve(maxSize * dim);
249 phi_xx.reserve(maxSize * dim2);
250
251 double weight; // gauss point weight
252
253 vector< int > sysDof; // local to global pdeSys dofs
254 sysDof.reserve(maxSize);
255
256 vector< double > ResU; // local residual vector
257 ResU.reserve(maxSize);
258
259 vector < double > Jac;
260 Jac.reserve(maxSize * maxSize);
261
262 KK->zero(); // Set to zero all the entries of the Global Matrix
263
264 // element loop: each process loops only on the elements that owns
265 for(int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
266
267 short unsigned ielGeom = msh->GetElementType(iel); // element geometry type
268
269 unsigned nDofsU = msh->GetElementDofNumber(iel, solUType); // number of solution element dofs
270 unsigned nDofsX = msh->GetElementDofNumber(iel, crdXType); // number of solution element dofs
271
272 // resize local arrays
273 sysDof.resize(nDofsU);
274 solU.resize(nDofsU);
275
276 for(unsigned k = 0; k < dim; k++) {
277 crdX[k].resize(nDofsX);
278 }
279
280 aResU.assign(nDofsU, 0);
281
282 // local storage of global mapping and solution
283 for(unsigned i = 0; i < nDofsU; i++) {
284 unsigned solUDof = msh->GetSolutionDof(i, iel, solUType); // local to global mapping of the solution U
285 solU[i] = (*sol->_Sol[solUIndex])(solUDof); // value of the solution U in the dofs
286 sysDof[i] = pdeSys->GetSystemDof(solUIndex, solUPdeIndex, i, iel); // local to global mapping between solution U and system
287 }
288
289 // local storage of coordinates
290 for(unsigned i = 0; i < nDofsX; i++) {
291 unsigned coordXDof = msh->GetSolutionDof(i, iel, crdXType); // local to global mapping of the coordinate X[dim]
292
293 for(unsigned k = 0; k < dim; k++) {
294 crdX[k][i] = (*msh->_topology->_Sol[k])(coordXDof); // value of the solution X[dim] //
295 }
296 }
297
298 s.new_recording();
299
300 // *** Gauss point loop *** //
301 for(unsigned ig = 0; ig < msh->_finiteElement[ielGeom][solUType]->GetGaussPointNumber(); ig++) {
302 // *** get gauss point weight, test function and test function partial derivatives ***
303 msh->_finiteElement[ielGeom][solUType]->Jacobian(crdX, ig, weight, phi, phi_x, phi_xx);
304
305 //adept::adouble solUig = 0; // solution U in the gauss point
306 vector < adept::adouble > gradSolUig(dim, 0.); // gradient of solution U in the gauss point
307
308 vector < double > x_gss(dim, 0.);
309
310 for(unsigned i = 0; i < nDofsU; i++) {
311 //solUig += phi[i] * solU[i];
312
313 for(unsigned j = 0; j < dim; j++) {
314 gradSolUig[j] += phi_x[i * dim + j] * solU[i];
315 x_gss[j] += crdX[j][i] * phi[i];
316 }
317 }
318
319 // *** phiU_i loop ***
320 for(unsigned i = 0; i < nDofsU; i++) {
321 adept::adouble LaplaceU = 0.;
322
323 for(unsigned j = 0; j < dim; j++) {
324 LaplaceU += phi_x[i * dim + j] * gradSolUig[j];
325 }
326
327 double srcTerm = - GetExactSolutionLaplace(x_gss);
328 aResU[i] += (srcTerm * phi[i] - LaplaceU) * weight;
329
330 } // end phiU_i loop
331 } // end gauss point loop
332
333 // } // endif single element not refined or fine grid loop
334
335 //--------------------------------------------------------------------------------------------------------
336 // Add the local Matrix/Vector into the global Matrix/Vector
337
338 //copy the value of the adept::adoube aRes in double Res and store them in RES
339 ResU.resize(nDofsU); //resize
340
341 for(int i = 0; i < nDofsU; i++) {
342 ResU[i] = -aResU[i].value();
343 }
344
345 RES->add_vector_blocked(ResU, sysDof);
346
347 //Extarct and store the Jacobian
348
349 Jac.resize(nDofsU * nDofsU);
350 // define the dependent variables
351 s.dependent(&aResU[0], nDofsU);
352
353 // define the independent variables
354 s.independent(&solU[0], nDofsU);
355
356 // get the and store jacobian matrix (row-major)
357 s.jacobian(&Jac[0] , true);
358 KK->add_matrix_blocked(Jac, sysDof, sysDof);
359
360 s.clear_independents();
361 s.clear_dependents();
362
363 } //end element loop for each process
364
365 RES->close();
366
367 KK->close();
368
369 // ***************** END ASSEMBLY *******************
370}

373void GetError(MultiLevelSolution* mlSol) {
374 unsigned level = mlSol->_mlMesh->GetNumberOfLevels() - 1u;
375 // extract pointers to the several objects that we are going to use
376 Mesh* msh = mlSol->_mlMesh->GetLevel(level); // pointer to the mesh (level) object
377 elem* el = msh->el; // pointer to the elem object in msh (level)
378 Solution* sol = mlSol->GetSolutionLevel(level); // pointer to the solution (level) object
379
380 const unsigned dim = msh->GetDimension(); // get the domain dimension of the problem
381 unsigned iproc = msh->processor_id(); // get the process_id (for parallel computation)
382
383 //solution variable
384 unsigned solUIndex;
385 solUIndex = mlSol->GetIndex("U"); // get the position of "U" in the ml_sol object
386 unsigned solUType = mlSol->GetSolutionType(solUIndex); // get the finite element type for "u"
387
388
389 unsigned errorIndex = mlSol->GetIndex("Error");
390
391 vector < double > solU; // local solution
392
393 vector < vector < double > > crdX(dim); // local coordinates
394 unsigned crdXType = 2; // get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC)
395
396 vector <double> phi; // local test function
397 vector <double> phi_x; // local test function first order partial derivatives
398 vector <double> phi_xx; // local test function second order partial derivatives
399 double weight; // gauss point weight
400
401
402 // reserve memory for the local standar vectors
403 const unsigned maxSize = static_cast< unsigned >(ceil(pow(3, dim))); // conservative: based on line3, quad9, hex27
404 solU.reserve(maxSize);
405
406 for(unsigned i = 0; i < dim; i++)
407 crdX[i].reserve(maxSize);
408
409 phi.reserve(maxSize);
410 phi_x.reserve(maxSize * dim);
411 unsigned dim2 = (3 * (dim - 1) + !(dim - 1)); // dim2 is the number of second order partial derivatives (1,3,6 depending on the dimension)
412 phi_xx.reserve(maxSize * dim2);
413
414 double seminorm2 = 0.;
415 double l2norm2 = 0.;
416
417 // element loop: each process loops only on the elements that owns
418 for(int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
419
420 short unsigned ielGeom = msh->GetElementType(iel);
421 unsigned nDofsU = msh->GetElementDofNumber(iel, solUType); // number of solution element dofs
422 unsigned nDofsX = msh->GetElementDofNumber(iel, crdXType); // number of coordinate element dofs
423
424 // resize local arrays
425 solU.resize(nDofsU);
426
427 for(int i = 0; i < dim; i++) {
428 crdX[i].resize(nDofsX);
429 }
430
431 // local storage of global mapping and solution
432 for(unsigned i = 0; i < nDofsU; i++) {
433 unsigned solDof = msh->GetSolutionDof(i, iel, solUType); // global to global mapping between solution node and solution dof
434 solU[i] = (*sol->_Sol[solUIndex])(solDof); // global extraction and local storage for the solution
435 }
436
437 // local storage of coordinates
438 for(unsigned i = 0; i < nDofsX; i++) {
439 unsigned xDof = msh->GetSolutionDof(i, iel, crdXType); // global to global mapping between coordinates node and coordinate dof
440
441 for(unsigned jdim = 0; jdim < dim; jdim++) {
442 crdX[jdim][i] = (*msh->_topology->_Sol[jdim])(xDof); // global extraction and local storage for the element coordinates
443 }
444 }
445
446 // find h_k
447 double hk = 0.;
448 for(unsigned i = 0; i < nDofsX - 1; i++) {
449 for(unsigned j = i + 1; j < nDofsX; j++) {
450 double dij = 0.;
451 for(unsigned jdim = 0; jdim < dim; jdim++) {
452 dij += (crdX[jdim][i] - crdX[jdim][j]) * (crdX[jdim][i] - crdX[jdim][j]);
453 }
454 dij = sqrt(dij);
455 if(dij > hk)
456 hk = dij;
457 }
458 }
459
460 double Rhok = 0.;
461
462 // *** Gauss point loop ***
463 for(unsigned ig = 0; ig < msh->_finiteElement[ielGeom][solUType]->GetGaussPointNumber(); ig++) {
464 // *** get gauss point weight, test function and test function partial derivatives ***
465 msh->_finiteElement[ielGeom][solUType]->Jacobian(crdX, ig, weight, phi, phi_x, phi_xx);
466
467
468 vector < double > x_gss(dim, 0.);
469 double laplaceUh = 0.;
470
471 double Uig = 0.;
472 std::vector< double > gradUig(dim,0);
473
474
475 for(unsigned i = 0; i < nDofsU; i++) {
476 Uig += phi[i] * solU[i];
477 for(unsigned j = 0; j < dim; j++) {
478 gradUig[j] += phi_x[i * dim + j] * solU[i];
479 x_gss[j] += crdX[j][i] * phi[i];
480 }
481 laplaceUh += (phi_xx[i * dim2 + 0] + phi_xx[i * dim2 + 1]) * solU[i];
482 }
483
484 //std::cout << laplaceUh << " ";
485
486 l2norm2 += Uig*Uig*weight;
487 for(unsigned j = 0; j < dim; j++) {
488 seminorm2 += gradUig[j] * gradUig[j] * weight;
489 }
490
491 double LaplaceUexact = GetExactSolutionLaplace(x_gss);
492 Rhok += (LaplaceUexact - laplaceUh) * (LaplaceUexact - laplaceUh) * weight;
493
494 } //end element loop for each process
495
496 Rhok = hk * sqrt(Rhok);
497
498 if( msh->el->GetIfElementCanBeRefined(iel) ) {
499 sol->_Sol[errorIndex]->set(iel, Rhok);
500 }
501 else {
502 sol->_Sol[errorIndex]->set(iel, 0.);
503 }
504
505 }// add the norms of all processes
506
507 sol->_Sol[errorIndex]->close();
508
509
510 // add the norms of all processes
511 NumericVector* norm_vec;
512 norm_vec = NumericVector::build().release();
513 norm_vec->init (msh->n_processors(), 1 , false, AUTOMATIC);
514
515 norm_vec->set (iproc, l2norm2);
516 norm_vec->close();
517
518 l2norm2 = norm_vec->l1_norm();
519
520 norm_vec->set (iproc, seminorm2);
521 norm_vec->close();
522
523 seminorm2 = norm_vec->l1_norm();
524
525 delete norm_vec;
526
527 double H1norm = sqrt( l2norm2 + seminorm2 );
528
529 unsigned N = msh->GetNumberOfElements();
530 *sol->_Sol[errorIndex] *= (2.*sqrt(N)) / ( 3.*H1norm );
531
532 sol->_Sol[errorIndex]->close();
533
534}

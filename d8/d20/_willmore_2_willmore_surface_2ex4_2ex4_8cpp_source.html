306// Building the P-Willmore assembly function.
307void AssembleBloodCell (MultiLevelProblem& ml_prob) {
308 // ml_prob is the global object from/to where get/set all the data
309 // level is the level of the PDE system to be assembled
310
311 // Call the adept stack object.
312 adept::Stack& s = FemusInit::_adeptStack;
313
314 // Extract pointers to the several objects that we are going to use.
315 TransientNonlinearImplicitSystem* mlPdeSys = &ml_prob.get_system<TransientNonlinearImplicitSystem> ("PWillmore"); // pointer to the linear implicit system named "Poisson"
316
317 // Define level and time variable.
318 double dt = mlPdeSys->GetIntervalTime();
319 const unsigned level = mlPdeSys->GetLevelToAssemble();
320
321 // Point to the mesh and element objects.
322 Mesh *msh = ml_prob._ml_msh->GetLevel (level);
323 elem *el = msh->el;
324
325 // Point to mlSol, solution (level), and equation (level) objects.
326 MultiLevelSolution *mlSol = ml_prob._ml_sol;
327 Solution *sol = ml_prob._ml_sol->GetSolutionLevel (level);
328 LinearEquationSolver *pdeSys = mlPdeSys->_LinSolver[level];
329
330 // Point to the global stiffness mtx and residual vectors in pdeSys (level).
331 SparseMatrix *KK = pdeSys->_KK;
332 NumericVector *RES = pdeSys->_RES;
333
334 // Convenience variables to encode the dimension.
335 const unsigned dim = 2;
336 const unsigned DIM = 3;
337
338 // Get the process_id (for parallel computation).
339 unsigned iproc = msh->processor_id();
340
341 // Extract the solution vector; get solDx positions in the ml_sol object.
342 unsigned solDxIndex[DIM];
343 solDxIndex[0] = mlSol->GetIndex ("Dx1");
344 solDxIndex[1] = mlSol->GetIndex ("Dx2");
345 solDxIndex[2] = mlSol->GetIndex ("Dx3");
346
347 // Extract the finite element type for solx.
348 unsigned solxType;
349 solxType = mlSol->GetSolutionType (solDxIndex[0]);
350
351 // Get positions of solDx in the pdeSys object.
352 unsigned solDxPdeIndex[DIM];
353 solDxPdeIndex[0] = mlPdeSys->GetSolPdeIndex ("Dx1");
354 solDxPdeIndex[1] = mlPdeSys->GetSolPdeIndex ("Dx2");
355 solDxPdeIndex[2] = mlPdeSys->GetSolPdeIndex ("Dx3");
356
357 // Define solx and solxOld.
358 std::vector < adept::adouble > solx[DIM];
359 std::vector < double > solxOld[DIM];
360
361 // Get the finite element type for "x", it is always 2 (LAGRANGE QUADRATIC).
362 unsigned xType = 2;
363
364 // Get positions of Y in the ml_sol object.
365 unsigned solYIndex[DIM];
366 solYIndex[0] = mlSol->GetIndex ("Y1");
367 solYIndex[1] = mlSol->GetIndex ("Y2");
368 solYIndex[2] = mlSol->GetIndex ("Y3");
369
370 // Extract the finite element type for Y.
371 unsigned solYType;
372 solYType = mlSol->GetSolutionType (solYIndex[0]);
373
374 // Get positions of Y in the pdeSys object.
375 unsigned solYPdeIndex[DIM];
376 solYPdeIndex[0] = mlPdeSys->GetSolPdeIndex ("Y1");
377 solYPdeIndex[1] = mlPdeSys->GetSolPdeIndex ("Y2");
378 solYPdeIndex[2] = mlPdeSys->GetSolPdeIndex ("Y3");
379
380 // Define solY and solYOld.
381 std::vector < adept::adouble > solY[DIM];
382 std::vector < double > solYOld[DIM];
383
384 // Local-to-global pdeSys dofs.
385 std::vector< unsigned > SYSDOF;
386
387 // Define local residual vectors.
388 std::vector < double > Res;
389 std::vector< adept::adouble > aResx[3];
390 std::vector< adept::adouble > aResY[3];
391
392 // Local (column-ordered) Jacobian matrix
393 std::vector < double > Jac;
394
395 //MatSetOption ( ( static_cast<PetscMatrix*> ( KK ) )->mat(), MAT_NEW_NONZERO_ALLOCATION_ERR, PETSC_FALSE );
396
397 KK->zero(); // Set to zero all the entries of the Global Matrix
398 RES->zero(); // Set to zero all the entries of the Global Residual
399
400
401 // Setting up solLambda1 (vol) and solLambda2 (area).
402 unsigned solLambaPdeIndex;
403
404 adept::adouble solLambda1 = 0.;
405 adept::adouble aResLambda1;
406 unsigned lambda1PdeDof;
407
408 adept::adouble solLambda2 = 0.;
409 adept::adouble aResLambda2;
410 unsigned lambda2PdeDof;
411
412 if (volumeConstraint || areaConstraint) {
413 unsigned solLambdaIndex;
414 solLambdaIndex = mlSol->GetIndex ("Lambda");
415 solLambaPdeIndex = mlPdeSys->GetSolPdeIndex ("Lambda");
416
417 if (volumeConstraint) {
418 double lambda1;
419 if (iproc == 0) {
420 lambda1 = (*sol->_Sol[solLambdaIndex]) (0); // global to local solution
421 lambda1PdeDof = pdeSys->GetSystemDof (solLambdaIndex, solLambaPdeIndex, 0, 0);
422 }
423 MPI_Bcast (&lambda1, 1, MPI_DOUBLE, 0, MPI_COMM_WORLD);
424 MPI_Bcast (&lambda1PdeDof, 1, MPI_UNSIGNED, 0, MPI_COMM_WORLD);
425 solLambda1 = lambda1;
426 }
427
428 if (areaConstraint) {
429 double lambda2;
430 if (iproc == 0) {
431 lambda2 = (*sol->_Sol[solLambdaIndex]) (volumeConstraint); // global to local solution
432 lambda2PdeDof = pdeSys->GetSystemDof (solLambdaIndex, solLambaPdeIndex, 0, volumeConstraint);
433 }
434 MPI_Bcast (&lambda2, 1, MPI_DOUBLE, 0, MPI_COMM_WORLD);
435 MPI_Bcast (&lambda2PdeDof, 1, MPI_UNSIGNED, 0, MPI_COMM_WORLD);
436 solLambda2 = lambda2;
437 }
438
439 std::vector < double > value (2);
440 std::vector < int > row (1);
441 std::vector < int > columns (2);
442 value[0] = 1;
443 value[1] = -1;
444 columns[1] = (volumeConstraint) ? lambda1PdeDof : lambda2PdeDof;
445
446 // For equations other than Lagrange multiplier:
447 for (int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
448 if (iel > volumeConstraint * areaConstraint) {
449 row[0] = pdeSys->GetSystemDof (solLambdaIndex, solLambaPdeIndex, 0, iel);
450 columns[0] = row[0];
451 KK->add_matrix_blocked (value, row, columns);
452 }
453 }
454 }
455
456 // Initialize area, volume, P-Willmore energy.
457 double surface = 0.;
458 double volume = 0.;
459 double energy = 0.;
460
461
462
463 // ELEMENT LOOP: each process loops only on the elements that it owns.
464 for (int iel = msh->_elementOffset[iproc]; iel < msh->_elementOffset[iproc + 1]; iel++) {
465
466 // Number of solution element dofs.
467 short unsigned ielGeom = msh->GetElementType (iel);
468 unsigned nxDofs = msh->GetElementDofNumber (iel, solxType);
469 unsigned nYDofs = msh->GetElementDofNumber (iel, solYType);
470
471 // Resize solution vectors.
472 for (unsigned K = 0; K < DIM; K++) {
473 solx[K].resize (nxDofs);
474 solxOld[K].resize (nxDofs);
475 solY[K].resize (nYDofs);
476 solYOld[K].resize (nYDofs);
477 }
478
479 // Convenience variable for keeping track of problem size.
480 unsigned sizeAll = DIM * (nxDofs + nYDofs) + volumeConstraint + areaConstraint;
481
482 // Resize local arrays.
483 SYSDOF.resize (sizeAll);
484 Res.resize (sizeAll);
485
486 for (unsigned K = 0; K < DIM; K++) {
487 aResx[K].assign (nxDofs, 0.); //resize and set to zero
488 aResY[K].assign (nYDofs, 0.); //resize and set to zero
489 }
490 aResLambda1 = 0.;
491 aResLambda2 = 0.;
492
493 // Loop which handles local storage of global mapping and solution X.
494 for (unsigned i = 0; i < nxDofs; i++) {
495
496 // Global-to-local mapping between solution node and solution dof.
497 unsigned iDDof = msh->GetSolutionDof (i, iel, solxType);
498 unsigned iXDof = msh->GetSolutionDof (i, iel, xType);
499
500 for (unsigned K = 0; K < DIM; K++) {
501 solxOld[K][i] = (*msh->GetTopology()->_Sol[K]) (iXDof) + (*sol->_SolOld[solDxIndex[K]]) (iDDof);
502 solx[K][i] = (*msh->GetTopology()->_Sol[K]) (iXDof) + (*sol->_Sol[solDxIndex[K]]) (iDDof);
503
504 // Global-to-global mapping between solution node and pdeSys dof.
505 SYSDOF[K * nxDofs + i] = pdeSys->GetSystemDof (solDxIndex[K], solDxPdeIndex[K], i, iel);
506 }
507 }
508
509 // Loop which handles local storage of global mapping and solution Y.
510 for (unsigned i = 0; i < nYDofs; i++) {
511
512 // Global-to-local mapping between solution node and solution dof.
513 unsigned iYDof = msh->GetSolutionDof (i, iel, solYType);
514 for (unsigned K = 0; K < DIM; K++) {
515
516 // Global-to-local solutions.
517 solYOld[K][i] = (*sol->_SolOld[solYIndex[K]]) (iYDof);
518 solY[K][i] = (*sol->_Sol[solYIndex[K]]) (iYDof);
519
520 // Global-to-global mapping between solution node and pdeSys dof.
521 SYSDOF[DIM * nxDofs + K * nYDofs + i] =
522 pdeSys->GetSystemDof (solYIndex[K], solYPdeIndex[K], i, iel);
523 }
524 }
525
526 // Conditions for local storage of global Lagrange multipliers.
527 if (volumeConstraint) {
528 SYSDOF[sizeAll - 1u - areaConstraint ] = lambda1PdeDof;
529 }
530
531 if (areaConstraint) {
532 SYSDOF[sizeAll - 1u ] = lambda2PdeDof;
533 }
534
535 // Start a new recording of all the operations involving adept variables.
536 s.new_recording();
537
538 // begin GAUSS POINT LOOP
539 for (unsigned ig = 0; ig < msh->_finiteElement[ielGeom][solxType]->GetGaussPointNumber(); ig++) {
540
541 const double *phix; // local test function
542 const double *phix_uv[dim]; // first order derivatives in (u,v)
543
544 const double *phiY; // local test function
545 const double *phiY_uv[dim]; // first order derivatives in (u,v)
546
547 double weight; // gauss point weight
548
549 //Extract Gauss point weight, test functions, and their partial derivatives.
550 // "0" is derivative in u, "1" is derivative in v.
551 weight = msh->_finiteElement[ielGeom][solxType]->GetGaussWeight (ig);
552
553 phix = msh->_finiteElement[ielGeom][solxType]->GetPhi (ig);
554 phix_uv[0] = msh->_finiteElement[ielGeom][solxType]->GetDPhiDXi (ig);
555 phix_uv[1] = msh->_finiteElement[ielGeom][solxType]->GetDPhiDEta (ig);
556
557 phiY = msh->_finiteElement[ielGeom][solYType]->GetPhi (ig);
558 phiY_uv[0] = msh->_finiteElement[ielGeom][solYType]->GetDPhiDXi (ig);
559 phiY_uv[1] = msh->_finiteElement[ielGeom][solYType]->GetDPhiDEta (ig);
560
561 // Initialize quantities xNew, xOld, Y, W at the Gauss points.
562 adept::adouble solxNewg[3] = {0., 0., 0.};
563 double solxOldg[3] = {0., 0., 0.};
564
565 adept::adouble solYNewg[3] = {0., 0., 0.};
566 double solYOldg[3] = {0., 0., 0.};
567
568
569 // Initialize derivatives of x and W (new, middle, old) at the Gauss points.
570 adept::adouble solxNew_uv[3][2] = {{0., 0.}, {0., 0.}, {0., 0.}};
571 adept::adouble solYNew_uv[3][2] = {{0., 0.}, {0., 0.}, {0., 0.}};
572
573 adept::adouble solx_uv[3][2] = {{0., 0.}, {0., 0.}, {0., 0.}};
574
575 double solxOld_uv[3][2] = {{0., 0.}, {0., 0.}, {0., 0.}};
576 double solYOld_uv[3][2] = {{0., 0.}, {0., 0.}, {0., 0.}};
577
578
579
580 for (unsigned K = 0; K < DIM; K++) {
581
582 for (unsigned i = 0; i < nxDofs; i++) {
583 solxNewg[K] += phix[i] * solx[K][i];
584 solxOldg[K] += phix[i] * solxOld[K][i];
585 }
586
587 for (unsigned i = 0; i < nYDofs; i++) {
588 solYNewg[K] += phiY[i] * solY[K][i];
589 solYOldg[K] += phiY[i] * solYOld[K][i];
590 }
591
592 for (int j = 0; j < dim; j++) {
593 for (unsigned i = 0; i < nxDofs; i++) {
594 solxNew_uv[K][j] += phix_uv[j][i] * solx[K][i];
595 solx_uv[K][j] += phix_uv[j][i] * 0.5 * (solx[K][i] + solxOld[K][i]);
596 solxOld_uv[K][j] += phix_uv[j][i] * solxOld[K][i];
597 }
598 }
599
600 for (int j = 0; j < dim; j++) {
601 for (unsigned i = 0; i < nYDofs; i++) {
602 solYNew_uv[K][j] += phiY_uv[j][i] * solY[K][i];
603 solYOld_uv[K][j] += phiY_uv[j][i] * solYOld[K][i];
604 }
605 }
606
607 }
608
609 // Computing the metric, metric determinant, and area element.
610 adept::adouble g[dim][dim] = {{0., 0.}, {0., 0.}};
611 for (unsigned i = 0; i < dim; i++) {
612 for (unsigned j = 0; j < dim; j++) {
613 for (unsigned K = 0; K < DIM; K++) {
614 g[i][j] += solx_uv[K][i] * solx_uv[K][j];
615 }
616 }
617 }
618 adept::adouble detg = g[0][0] * g[1][1] - g[0][1] * g[1][0];
619 adept::adouble Area = weight * sqrt (detg);
620
621 // Computing the unit normal vector N.
622 adept::adouble normal[DIM];
623 normal[0] = normalSign * (solx_uv[1][0] * solx_uv[2][1]
624 - solx_uv[2][0] * solx_uv[1][1]) / sqrt (detg);
625 normal[1] = normalSign * (solx_uv[2][0] * solx_uv[0][1]
626 - solx_uv[0][0] * solx_uv[2][1]) / sqrt (detg);
627 normal[2] = normalSign * (solx_uv[0][0] * solx_uv[1][1]
628 - solx_uv[1][0] * solx_uv[0][1]) / sqrt (detg);
629
630 // Computing Y.N and |Y|^2, which are essentially 2H and 4H^2.
631 adept::adouble YdotN = 0.;
632 adept::adouble YdotY = 0.;
633 adept::adouble Hminc0sqrd = 0;
634 for (unsigned K = 0; K < DIM; K++) {
635 YdotN += solYOldg[K] * normal[K];
636 //YdotY += solYOldg[K] * solYOldg[K];
637 YdotY += solYNewg[K] * solYNewg[K];
638 Hminc0sqrd += (solYOldg[K] - 2 * c0 * normal[K]) * (solYOldg[K] - 2 * c0 * normal[K]);
639 }
640 // double signYdotN = (YdotN.value() >= 0.) ? 1. : -1.;
641 adept::adouble multiplier = (YdotN - 2 * c0) / (2 * YdotN);
642
643
644 // Computing the metric inverse
645 adept::adouble gi[dim][dim];
646 gi[0][0] = g[1][1] / detg;
647 gi[0][1] = -g[0][1] / detg;
648 gi[1][0] = -g[1][0] / detg;
649 gi[1][1] = g[0][0] / detg;
650
651 // Computing the "reduced Jacobian" g^{ij}X_j .
652 adept::adouble Jir[dim][DIM] = {{0., 0., 0.}, {0., 0., 0.}};
653 for (unsigned i = 0; i < dim; i++) {
654 for (unsigned J = 0; J < DIM; J++) {
655 for (unsigned k = 0; k < dim; k++) {
656 Jir[i][J] += gi[i][k] * solx_uv[J][k];
657 }
658 }
659 }
660
661 // Initializing tangential gradients of X and W (new, middle, old).
662 adept::adouble solxNew_Xtan[DIM][DIM] = {{0., 0., 0.}, {0., 0., 0.}, {0., 0., 0.}};
663 adept::adouble solx_Xtan[DIM][DIM] = {{0., 0., 0.}, {0., 0., 0.}, {0., 0., 0.}};
664 adept::adouble solxOld_Xtan[DIM][DIM] = {{0., 0., 0.}, {0., 0., 0.}, {0., 0., 0.}};
665
666 adept::adouble solYNew_Xtan[DIM][DIM] = {{0., 0., 0.}, {0., 0., 0.}, {0., 0., 0.}};
667 adept::adouble solYOld_Xtan[DIM][DIM] = {{0., 0., 0.}, {0., 0., 0.}, {0., 0., 0.}};
668
669
670 // Computing tangential gradients defined above.
671 for (unsigned I = 0; I < DIM; I++) {
672 for (unsigned J = 0; J < DIM; J++) {
673 for (unsigned k = 0; k < dim; k++) {
674 solxNew_Xtan[I][J] += solxNew_uv[I][k] * Jir[k][J];
675 solx_Xtan[I][J] += solx_uv[I][k] * Jir[k][J];
676 solxOld_Xtan[I][J] += solxOld_uv[I][k] * Jir[k][J];
677
678 solYNew_Xtan[I][J] += solYNew_uv[I][k] * Jir[k][J];
679 solYOld_Xtan[I][J] += solYOld_uv[I][k] * Jir[k][J];
680 }
681 }
682 }
683
684 // Define and compute gradients of test functions for X and W.
685 std::vector < adept::adouble > phix_Xtan[DIM];
686 std::vector < adept::adouble > phiY_Xtan[DIM];
687
688 for (unsigned J = 0; J < DIM; J++) {
689 phix_Xtan[J].assign (nxDofs, 0.);
690 phiY_Xtan[J].assign (nYDofs, 0.);
691
692 for (unsigned inode = 0; inode < nxDofs; inode++) {
693 for (unsigned k = 0; k < dim; k++) {
694 phix_Xtan[J][inode] += phix_uv[k][inode] * Jir[k][J];
695 }
696 }
697
698 for (unsigned inode = 0; inode < nYDofs; inode++) {
699 for (unsigned k = 0; k < dim; k++) {
700 phiY_Xtan[J][inode] += phiY_uv[k][inode] * Jir[k][J];
701 }
702 }
703 }
704
705 // Implement the curvature equation Y = \Delta X .
706 for (unsigned K = 0; K < DIM; K++) {
707 for (unsigned i = 0; i < nxDofs; i++) {
708 adept::adouble term1 = 0.;
709 // adept::adouble term2 = 0.;
710
711 for (unsigned J = 0; J < DIM; J++) {
712 term1 += solxNew_Xtan[K][J] * phix_Xtan[J][i]; // the field x is new (i + 1) but differentiated on the surface at (i+1/2)
713 }
714 aResx[K][i] += (solYNewg[K] * phix[i] + term1) * Area;
715 }
716
717
718 // Implement the main P-Willmore equation.
719 for (unsigned i = 0; i < nYDofs; i++) {
720 adept::adouble term0 = 0.;
721 adept::adouble term1 = 0.;
722 adept::adouble term2 = 0.;
723 adept::adouble term3 = 0.;
724
725 // adept::adouble term5 = 0.;
726
727 for (unsigned J = 0; J < DIM; J++) {
728 term0 += solYNew_Xtan[K][J] * phiY_Xtan[J][i]; // the field W is new (i + 1) but differentiated on the surface at (i + 1/2)
729 term1 += solxNew_Xtan[K][J] * phiY_Xtan[J][i];
730 term2 += solYNew_Xtan[J][J];
731
732 adept::adouble term4 = 0.;
733 for (unsigned L = 0; L < DIM; L++) { // the fields W and x are old (i) but differentiated on the surface at (i + 1/2)
734 term4 += solxOld_Xtan[L][J] * solYOld_Xtan[L][K] + solxOld_Xtan[L][K] * solYOld_Xtan[L][J];
735 }
736 term3 += phiY_Xtan[J][i] * term4;
737 /* this is the trick we learned from Dzuik: basically in magnitude term3 = 2 term0, so -term0 + term3 = + term0 = 1/2 term3,
738 but the stability sign comes from -term0, for this reason term0 is taken more implicitly (i + 1), and term3/term4 is semiexplicit (i),
739 It is shockingly how everything works and any small change causes the solver to crash */
740 /* unless we show it numerically that this time scheme is second order (and I am not sure it is) we cannot claim it in the paper.
741 The fact that we have a lot of non linearities involved makes a the proof very difficult.
742 We may try to assume that the surface is known exactly in time and see what comes out of this integration scheme*/
743 }
744
745 // Blood cell equation
746 aResY[K][i] += ( ( (solLambda1 /*- YdotN * solLambda2*/) * normal[K] + (solxNewg[K] - solxOldg[K]) / dt) * phiY[i]
747 + solLambda2 * term1
748 + ( (kc / 4) * Hminc0sqrd + gamma1 ) * term1
749 + kc * multiplier * term3
750 - kc * multiplier * term0
751 // - kc * multiplier * ( YdotY + term2 ) * phiY_Xtan[K][i]
752 - kc * multiplier * ( YdotY + term2 ) * term1
753 ) * Area;
754 }
755
756 // Lagrange multiplier (volume) equation Dx.N = 0.
757 if (volumeConstraint) {
758 aResLambda1 += ( (solxNewg[K] - solxOldg[K]) * normal[K]) * Area;
759 }
760
761 // Lagrange multiplier (area) equation.
762 if (areaConstraint) {
763 //aResLambda2 += ( -YdotN * (solxNewg[K] - solxOldg[K]) * normal[K]) * Area;
764 adept::adouble term1t = 0.;
765 for (unsigned J = 0; J < DIM; J++) {
766 term1t += solx_Xtan[K][J] * (solxNew_Xtan[K][J] - solxOld_Xtan[K][J]) ;
767 }
768 aResLambda2 += term1t * Area;
769
770 }
771 }
772
773 // Compute new surface area, volume, and P-Willmore energy.
774
775 surface += Area.value();
776
777 for (unsigned K = 0; K < DIM; K++) {
778 volume += normalSign * (solxNewg[K].value() * normal[K].value()) * Area.value();
779 }
780 energy += ( (kc / 4) * Hminc0sqrd.value() + gamma1 ) * Area.value();
781
782 } // end GAUSS POINT LOOP.
783
784 //------------------------------------------------------------------------
785 // Add the local Matrix/Vector into the global Matrix/Vector
786 //copy the value of the adept::adoube aRes in double Res and store
787
788 for (int K = 0; K < DIM; K++) {
789 for (int i = 0; i < nxDofs; i++) {
790 Res[ K * nxDofs + i] = -aResx[K][i].value();
791 }
792 }
793
794 for (int K = 0; K < DIM; K++) {
795 for (int i = 0; i < nYDofs; i++) {
796 Res[DIM * nxDofs + K * nYDofs + i] = -aResY[K][i].value();
797 }
798 }
799
800 if (volumeConstraint) {
801 Res[sizeAll - 1u - areaConstraint] = - aResLambda1.value();
802 }
803
804 if (areaConstraint) {
805 Res[sizeAll - 1u] = - aResLambda2.value();
806 }
807
808 RES->add_vector_blocked (Res, SYSDOF);
809
810 // Resize Jacobian.
811 Jac.resize (sizeAll * sizeAll);
812
813 // Define the dependent variables.
814 for (int K = 0; K < DIM; K++) {
815 s.dependent (&aResx[K][0], nxDofs);
816 }
817 for (int K = 0; K < DIM; K++) {
818 s.dependent (&aResY[K][0], nYDofs);
819 }
820 if (volumeConstraint) {
821 s.dependent (&aResLambda1, 1);
822 }
823 if (areaConstraint) {
824 s.dependent (&aResLambda2, 1);
825 }
826
827 // Define the independent variables.
828 for (int K = 0; K < DIM; K++) {
829 s.independent (&solx[K][0], nxDofs);
830 }
831 for (int K = 0; K < DIM; K++) {
832 s.independent (&solY[K][0], nYDofs);
833 }
834 if (volumeConstraint) {
835 s.independent (&solLambda1, 1);
836 }
837 if (areaConstraint) {
838 s.independent (&solLambda2, 1);
839 }
840
841 // Get the Jacobian matrix (ordered by row).
842 s.jacobian (&Jac[0], true);
843
844 KK->add_matrix_blocked (Jac, SYSDOF, SYSDOF);
845
846 s.clear_independents();
847 s.clear_dependents();
848
849 } // End ELEMENT LOOP for each process.
850
851 RES->close();
852 KK->close();
853

855 double surfaceAll;
856 MPI_Reduce (&surface, &surfaceAll, 1, MPI_DOUBLE, MPI_SUM, 0, MPI_COMM_WORLD);

867 std::cout << "ENERGY = " << energyAll << std::endl;

46 void NonLinearImplicitSystemWithPrimalDualActiveSetMethod::MGsolve (const MgSmootherType& mgSmootherType) {
47
48 _bitFlipCounter = 0;
49
50 unsigned AMRCounter = 0;
51
52 clock_t start_mg_time = total_mg_time_begin();
53
54 double totalAssemblyTime = 0.;
55
56 //---------------------
57 unsigned grid0;
58
59 if (_mg_type == F_CYCLE) {
60 std::cout << std::endl << " *** Start Nonlinear Full-Cycle ***" << std::endl;
61 grid0 = 0;
62 }
63 else if (_mg_type == V_CYCLE) {
64 std::cout << std::endl << " *** Start Nonlinear V-Cycle ***" << std::endl;
65 grid0 = _gridn - 1;
66 }
67 else {
68 std::cout << "wrong CYCLE type for this solver " << std::endl;
69 abort();
70 }
71 //---------------------
72
73
74 for (unsigned igridn = grid0; igridn < _gridn; igridn++) { //_igridn
75
76 std::cout << std::endl << " ****** Start Level Max " << igridn + 1 << " ******" << std::endl;
77
78
79//***************
80 clock_t start_nl_time = nonlinear_time_begin();
81
82
83 bool ThisIsAMR = (_mg_type == F_CYCLE && _AMRtest && AMRCounter < _maxAMRlevels && igridn == _gridn - 1u) ? 1 : 0;
84
85 //---------------------
86 restart:
87 if (ThisIsAMR) _solution[igridn]->InitAMREps();
88
89
90 nonlinear_solve_single_level(mgSmootherType, totalAssemblyTime, grid0, igridn);
91
92
93 if (_bitFlipOccurred && _bitFlipCounter == 1) {
94 goto restart;
95 }
96 //---------------------
97
98 if (igridn + 1 < _gridn) ProlongatorSol (igridn + 1);
99
100 if (ThisIsAMR) AddAMRLevel (AMRCounter);
101
102
103 nonlinear_time_end(start_nl_time);
104//***************
105
106
107 std::cout << std::endl << " ****** End Level Max " << igridn + 1 << " ******" << std::endl;
108
109
110 }
111
112
113 double totalSolverTime = total_mg_time_end(start_mg_time);
114
115
116 compute_assembly_vs_net_solver_times(totalSolverTime, totalAssemblyTime);
117
118
119 }

123 void NonLinearImplicitSystemWithPrimalDualActiveSetMethod::nonlinear_solve_single_level(const MgSmootherType& mgSmootherType, double & totalAssemblyTime, const unsigned int grid0, const unsigned int igridn) {
124
125
126 for (unsigned nonLinearIterator = 0; nonLinearIterator < _n_max_nonlinear_iterations; nonLinearIterator++) {
127
128 _nonliniteration = nonLinearIterator;
129
130 if (_debug_nonlinear) {
131 _eps_fine.push_back (NumericVector::build().release());
132 _eps_fine[_eps_fine.size() - 1]->init (*_LinSolver[_gridn - 1]->_EPS);
133 }
134
135 std::cout << std::endl << " ********* Nonlinear iteration " << nonLinearIterator + 1 << " *********" << std::endl;
136
137//**** PREPARATION BEGIN***********
138 clock_t start_preparation_time = clock();
139
140 //---------------------
141 clock_t start_assembly_time = clock();
142 _levelToAssemble = igridn;
143 _LinSolver[igridn]->SetResZero();
144 _assembleMatrix = _buildSolver;
145 _assemble_system_function (_equation_systems);
146 std::cout << " ********* Level Max " << igridn + 1 << " ASSEMBLY TIME:\t" << \
147 static_cast<double> ( (clock() - start_assembly_time)) / CLOCKS_PER_SEC << std::endl;
148 //---------------------
149
150 if (!_ml_msh->GetLevel (igridn)->GetIfHomogeneous()) {
151 if (!_RRamr[igridn]) {
152 (_LinSolver[igridn]->_RESC)->matrix_mult_transpose (*_LinSolver[igridn]->_RES, *_PPamr[igridn]);
153 }
154 else {
155 (_LinSolver[igridn]->_RESC)->matrix_mult (*_LinSolver[igridn]->_RES, *_RRamr[igridn]);
156 }
157 * (_LinSolver[igridn]->_RES) = * (_LinSolver[igridn]->_RESC);
158 }
159
160 if (_buildSolver) {
161
162 _MGmatrixFineReuse = (0 == nonLinearIterator) ? false : true;
163 _MGmatrixCoarseReuse = (igridn - grid0 > 0) ? true : _MGmatrixFineReuse;
164
165 if (!_ml_msh->GetLevel (igridn)->GetIfHomogeneous()) {
166 _LinSolver[igridn]->SwapMatrices();
167 if (!_RRamr[igridn]) {
168 _LinSolver[igridn]->_KK->matrix_PtAP (*_PPamr[igridn], *_LinSolver[igridn]->_KKamr, _MGmatrixFineReuse);
169 }
170 else {
171 _LinSolver[igridn]->_KK->matrix_ABC (*_RRamr[igridn], *_LinSolver[igridn]->_KKamr, *_PPamr[igridn], _MGmatrixFineReuse);
172 }
173 }
174
175 clock_t mg_proj_mat_time = clock();
176 for (unsigned i = igridn; i > 0; i--) {
177 if (_RR[i]) {
178 if (i == igridn)
179 _LinSolver[i - 1u]->_KK->matrix_ABC (*_RR[i], *_LinSolver[i]->_KK, *_PP[i], _MGmatrixFineReuse);
180 else {
181 _LinSolver[i - 1u]->_KK->matrix_ABC (*_RR[i], *_LinSolver[i]->_KK, *_PP[i], _MGmatrixCoarseReuse);
182 if (_LinSolver[i - 1u]->_KKamr) {
183 delete _LinSolver[i - 1u]->_KKamr;
184 _LinSolver[i - 1u]->_KKamr = NULL;
185 }
186 }
187 }
188 else {
189 if (i == igridn)
190 _LinSolver[i - 1u]->_KK->matrix_PtAP (*_PP[i], *_LinSolver[i]->_KK, _MGmatrixFineReuse);
191 else {
192 _LinSolver[i - 1u]->_KK->matrix_PtAP (*_PP[i], *_LinSolver[i]->_KK, _MGmatrixCoarseReuse);
193 if (_LinSolver[i - 1u]->_KKamr) {
194 delete _LinSolver[i - 1u]->_KKamr;
195 _LinSolver[i - 1u]->_KKamr = NULL;
196 }
197 }
198 }
199 }
200 std::cout << " ********* Level Max " << igridn + 1 << " MG PROJECTION MATRICES TIME:\t" \
201 << static_cast<double> ( (clock() - mg_proj_mat_time)) / CLOCKS_PER_SEC << std::endl;
202
203 clock_t mg_init_time = clock();
204
205 _LinSolver[igridn]->MGInit (mgSmootherType, igridn + 1, _mgOuterSolver);
206
207 for (unsigned i = 0; i <= igridn; i++) {
208 if (_RR[i])
209 _LinSolver[i]->MGSetLevel (_LinSolver[igridn], igridn, _VariablesToBeSolvedIndex, _PP[i], _RR[i], _npre, _npost);
210 else
211 _LinSolver[i]->MGSetLevel (_LinSolver[igridn], igridn, _VariablesToBeSolvedIndex, _PP[i], _PP[i], _npre, _npost);
212 }
213
214 std::cout << " ********* Level Max " << igridn + 1 << " MGINIT TIME:\t" \
215 << static_cast<double> ( (clock() - mg_init_time)) / CLOCKS_PER_SEC << std::endl;
216 }
217
218
219 totalAssemblyTime += static_cast<double> ( (clock() - start_assembly_time)) / CLOCKS_PER_SEC;
220 std::cout << " ********* Level Max " << igridn + 1 << " PREPARATION TIME:\t" << \
221 static_cast<double> ( (clock() - start_preparation_time)) / CLOCKS_PER_SEC << std::endl;
222//**** PREPARATION END***********
223
224
225 clock_t startUpdateResidualTime = clock();
226
227
228 for (unsigned updateResidualIterator = 0; updateResidualIterator < _maxNumberOfResidualUpdateIterations; updateResidualIterator++) {
229
230 std::cout << " ********* Linear Cycle + Residual Update iteration " << updateResidualIterator + 1 << std::endl;
231
232 bool thisHasConverged;
233
234 thisHasConverged = Vcycle (igridn, mgSmootherType);
235
236 if (thisHasConverged || updateResidualIterator == _maxNumberOfResidualUpdateIterations - 1) break;
237
238 _LinSolver[igridn]->SetResZero();
239 _assembleMatrix = false;
240 _assemble_system_function (_equation_systems);
241
242
243 if (!_ml_msh->GetLevel (igridn)->GetIfHomogeneous()) {
244 if (!_RRamr[igridn]) {
245 (_LinSolver[igridn]->_RESC)->matrix_mult_transpose (*_LinSolver[igridn]->_RES, *_PPamr[igridn]);
246 }
247 else {
248 (_LinSolver[igridn]->_RESC)->matrix_mult (*_LinSolver[igridn]->_RES, *_RRamr[igridn]);
249 }
250 * (_LinSolver[igridn]->_RES) = * (_LinSolver[igridn]->_RESC);
251 }
252 if (_bitFlipOccurred) break;
253 }
254
255
256
257 if (_buildSolver) {
258 if (!_ml_msh->GetLevel (igridn)->GetIfHomogeneous()) {
259 _LinSolver[igridn]->SwapMatrices();
260 }
261 _LinSolver[igridn]->MGClear();
262
263 }
264
265 double nonLinearEps;
266 bool nonLinearHasConverged = HasNonLinearConverged (igridn, nonLinearEps);
267
268 std::cout << " ********* Linear Cycle + Residual Update-Cycle TIME:\t" << std::setw (11) << std::setprecision (6) << std::fixed
269 << static_cast<double> ( (clock() - startUpdateResidualTime)) / CLOCKS_PER_SEC << std::endl;
270
271
272 // *****************
273 print_iteration_and_do_additional_computations_with_given_function_level(nonLinearIterator, _levelToAssemble, get_state_vars(), get_ctrl_vars() );
274
275
276 // ***************** check active flag sets - BEGIN *******************
277
278 bool are_all_active_flag_components_the_same;
279
280 if (_nonliniteration == 0) { //do not do any active flag check in the first iteration
281 are_all_active_flag_components_the_same = true;
282 }
283 else if (_nonliniteration > 0) {
284 Solution* sol = this->GetMLProb()._ml_sol->GetSolutionLevel (_levelToAssemble); // pointer to the solution (level) object
285
286 std::vector< bool > compare_bool(_active_flag_name.size(), false);
287
288
289 for (unsigned int c = 0; c < _active_flag_name.size(); c++) {
290 unsigned int solIndex_act_flag = this->GetMLProb()._ml_sol->GetIndex (_active_flag_name[c].c_str());
291
292 int compare_return = ( (sol->_SolOld[solIndex_act_flag])->compare (* (sol->_Sol[solIndex_act_flag])));
293
294 std::cout << "At iteration " << _nonliniteration << ", active set for variable " << _active_flag_name[c];
295 if (compare_return == -1) {
296 compare_bool[c] = true;
297 std::cout << " did not change" << std::endl;
298 }
299 else {
300 std::cout << " did change" << std::endl;
301 }
302 }
303
304 //turn compare_bool into a scalar boolean
305 are_all_active_flag_components_the_same = false;
306 unsigned count_unchanged_active_flag_components = 0;
307 for (unsigned int c = 0; c < _active_flag_name.size(); c++) {
308 if (compare_bool[c] == true) count_unchanged_active_flag_components++;
309 }
310
311 if ( count_unchanged_active_flag_components == _active_flag_name.size() ) { are_all_active_flag_components_the_same = true; }
312
313 if (are_all_active_flag_components_the_same ) {
314 std::cout << "At iteration " << _nonliniteration << ", active set for all variables " << " did not change" << std::endl;
315 break;
316 }
317
318 }
319
320 // ***************** check active flag sets - END *******************
321
322
323 if (are_all_active_flag_components_the_same && (nonLinearHasConverged || _bitFlipOccurred)) break;
327
328 }

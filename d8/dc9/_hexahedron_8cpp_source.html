168 const double ix=I[0]-1.,jx=I[1]-1.,kx=I[2]-1.;
169 return (fabs(ix*jx*kx)==0)?
170 line_lag::lagQuadratic(x[0],I[0]) * line_lag::lagQuadratic(x[1],I[1]) * line_lag::lagQuadratic(x[2],I[2])
171 :
172 (-2.+ix*x[0]+jx*x[1]+kx*x[2]) * line_lag::lagQuadratic(x[0],I[0]) * line_lag::lagQuadratic(x[1],I[1]) * line_lag::lagQuadratic(x[2],I[2]);
173 }

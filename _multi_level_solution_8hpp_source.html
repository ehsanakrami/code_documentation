573

575 unsigned sol_type = _solType[i];
576
577 for(unsigned ig = 0; ig < _gridn; ig++) {
578
579 _solution[ig]->ResizeSolutionVector(_solName[i]);
580 _solution[ig]->_Sol[i]->zero();
581
582 if(func || funcMLProb) {
583 double value = 0.;
584
585 if(sol_type < 3) {
586 abort();

594// unsigned icoord_Metis = _mlMesh->GetLevel(ig)->GetSolutionDof(j, iel, 2);

597// xx[1] = (*_mlMesh->GetLevel(ig)->_topology->_Sol[1])(icoord_Metis);
598// xx[2] = (*_mlMesh->GetLevel(ig)->_topology->_Sol[2])(icoord_Metis);

600// value = (func) ? func(xx) : funcMLProb(ml_prob, xx, name);

602// _solution[ig]->_Sol[i]->set(inode_Metis, value);

605// _solution[ig]->_SolOld[i]->set(inode_Metis, value);
606// }
607// }
608// }
609// }
610 }
611 else if(sol_type < 5) {
612
613 const double offset_to_include_line = 1.e-8;
614
615 const unsigned solType_coords = BIQUADR_FE;
616
617 const unsigned dim = _mlMesh->GetDimension();
618
619 CurrentElem < double > geom_element_iel(dim, _mlMesh->GetLevel(ig) );
620
621 for(int iel = _mlMesh->GetLevel(ig)->_elementOffset[_iproc];
622 iel < _mlMesh->GetLevel(ig)->_elementOffset[_iproc + 1]; iel++) {
623 value = 0.;
624
625// ------- - BEGIN
626 geom_element_iel.set_coords_at_dofs_and_geom_type(iel, solType_coords);
627
628 const short unsigned ielGeom = geom_element_iel.geom_type();
629
630 geom_element_iel.set_elem_center_3d(iel, solType_coords);
631// ------- - END
632
633
634 for(unsigned iface = 0; iface < _mlMesh->GetLevel(ig)->GetElementFaceNumber(iel); iface++) {
635
636
637 const int bdry_index = _mlMesh->GetLevel(ig)->el->GetFaceElementIndex(iel, iface);
638
639 if (bdry_index < 0) {
640 const unsigned int face_index_in_domain = - ( bdry_index + 1);
641
642 //compute face element center - BEGIN
643 geom_element_iel.set_coords_at_dofs_bdry_3d(iel, iface, solType_coords);
644
645 geom_element_iel.set_elem_center_bdry_3d();
646
647 const unsigned ielGeom_bdry = _mlMesh->GetLevel(ig)->GetElementFaceType(iel, iface);
648 //compute face element center - END
649
650
651 for(unsigned f = 0; f < LIST_OF_CTRL_FACES ::_face_with_extremes_index_size; f++) {
652
653 if (face_index_in_domain == /*ctrl::*/LIST_OF_CTRL_FACES :: _face_with_extremes_index[f]) {
654 const unsigned number_of_tangential_direction_components = LIST_OF_CTRL_FACES :: _num_of_tang_components_per_face ;
655
656 unsigned cond_for_tang_component_all = 1;
657
658 for(unsigned t = 0; t < number_of_tangential_direction_components; t++) {
659
660 const bool cond_for_tang_component_t = (geom_element_iel.get_elem_center_bdry_3d()[ /*ctrl::*/LIST_OF_CTRL_FACES ::tangential_direction_to_Gamma_control(face_index_in_domain, number_of_tangential_direction_components)[t] ] >
661 /*ctrl::*/LIST_OF_CTRL_FACES ::_face_with_extremes_extremes_on_tang_surface[f][t][0] + offset_to_include_line &&
662 geom_element_iel.get_elem_center_bdry_3d()[ /*ctrl::*/LIST_OF_CTRL_FACES ::tangential_direction_to_Gamma_control(face_index_in_domain, number_of_tangential_direction_components)[t] ] <
663 /*ctrl::*/LIST_OF_CTRL_FACES ::_face_with_extremes_extremes_on_tang_surface[f][t][1] - offset_to_include_line
664 );
665
666 if ( cond_for_tang_component_t == false ) { cond_for_tang_component_all *= 0; }
667
668 }
669
670 if (cond_for_tang_component_all == true) { value = 1.; }
671
672 }
673
674
675 }//end face_contol_index loop
676
677
678// value = (func) ? func(xx) : funcMLProb(ml_prob, xx, name);
679
680 unsigned placeholder_index = 0/*2*/;
681
682 unsigned solDof = _mlMesh->GetLevel(ig)->GetSolutionDof(placeholder_index, iel, sol_type);
683
684 _solution[ig]->_Sol[i]->set(solDof, value);
685
686 if(_solTimeOrder[i] == 2) {
687 _solution[ig]->_SolOld[i]->set(solDof, value);
688 }

690 }
691 } //end iface
692
693 } //end iel
694
695 }
696
697 _solution[ig]->_Sol[i]->close();
698
699 if(_solTimeOrder[i] == 2) {
700 _solution[ig]->_SolOld[i]->close();
701 }
702 }
703
704 }
705 }
706
707 return;
